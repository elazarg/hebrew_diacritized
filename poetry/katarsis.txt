הִרְהוּרִים בְּמִדְבַּר יְהוּדָה
צָף כִּמְכֻשָּׁף בְּפָנִים לוֹהֲטוֹת עַל מַיִם שַׁמְנוּנִיִּים.
יַם הַמֶּלַח, יַם בְּרֵאשִׁית.
צוּקֵי מוֹאָב מְלֵאֵי הוֹד, מַעֲבִירִים בִּי רֶטֶט מִקְרָאִי.
וַאֲנִי נִזְכָּר בִּמְעָרַת קוּמְרָאן,
שָׁם נָזִיר עוֹטֵה לָבָן אוֹחֵז קֻלְמוּס,
גָּנַז בְּחֶרְדַת קֹדֶשׁ מְגִלַּת קְלָף בְּכַד חֶרֶס.
בֵּין אוֹר לְחֹשֶׁךְ, אֲנִי חוֹשֵׂף מַלְתָּעוֹת בְּחִיּוּךְ שֶׁל כָּרִישׁ.
מִתְרַפֵּק עַל הָרֶגַע.
שׁוֹכֵחַ מִמָּה שֶׁהָיָה, מִתְעַלֵּם מִמָּה שֶׁיִּהְיֶה.
טַבְּעוֹת מַיִם מִתְפַּשְּׁטוֹת סָבִיב.
צְלִילֵי מֵיתָר רַכִּים,
סִימְפוֹנְיַת הָרוּחַ הַחַמָּה, מִצְּפוֹן מַעֲרָב.
הַשֶּׁבֶר הַנָּמוּךְ בָּעוֹלָם,
מְרַפֵּא אֶת שִׁבְרֵי נִשְׁמָתִי.
אֲנוּסִים
דּוֹמֶה בַּעֲלִי לְפָר, מְעוֹרֵר בִּי קֶבֶס וְתִעוּב.
עַל מִשְׁכָּבִי בַּלֵּילוֹת, חוֹלֶמֶת עָלַיִךְ יָפָתִי.
לִבִּי יוֹצֵא אֵלַיִךְ כִּרְסִיסֵי טַל זוֹהֲרִים.
הַגִּידִי לִי, אֵיכָה נִחְיֶה, כְּזוּג נָשִׁים אוֹהֲבוֹת.
מַיִם רַבִּים לֹא יוּכְלוּ לְכַבּוֹת אֶת אַהֲבָתֵנוּ.
עַזָּה כַּמָּוֶת אַהֲבָתִי לָךְ וְקָשָׁה כִּשְׁאוֹל קִנְאָה.
רֵיחַ גּוּפֵךְ בְּנַפְשִׁי, בֵּין שָׁדַיִךְ אָלִין.
הַפַּרְדֵּס נָעוּל, הָבָה נֵלֵךְ, וְאֵרַשְׂתִּיךְ לִי לְעוֹלָם.
וְהַחוֹתֶנֶת מְצִיצָה מִן הַחֲרַכִּים, וּמְגַלָּה אֶת סוֹדָן.
כּוֹכַב הַשַּׁחַר זוֹהֵר וְאוֹר נוֹצֵץ מֵחַלּוֹן בֵּית הַמִּדְרָשׁ.
לוֹמֵד הָאַבְרֵךְ, "וַיִּרְקֹד הַשָּׂטָן לְפָנָיו, וַיְפַתֵּהוּ לַחְטֹא".
בַּסֵּתֶר שׁוֹמֵר בְּחֵיקוֹ תְּמוּנַת עַלְמָה.
מֵלִיט פָּנָיו כְּשָׁקוּעַ בְּדַרְכָּהּ שֶׁל תּוֹרָה.
בֵּין דַּפֵּי הַתַּלְמוּד קָנְתָה לִילִית אֶת מְקוֹמָהּ.
מַבְעִירָה חֲלָצָיו, וְאֵינָהּ מַרְפָּה.
בְּחֶרְדַת קֹדֶשׁ, מַרְעִיד אֶת כַּף יָדוֹ, וְגוֹמֵר.
חֶרֶשׁ מֵגִיחַ רַב מַשְׁגִּיחַ.
וְקֶרִי בְּיַד הַמַּתְמִיד.
שׁוּבִי אֵלַי שׁוּלַמִּית, שׁוּבִי וְאֶחֱזֶה בָּךְ.
קוּמִי יָפָתִי וְנָרוּץ הַרְחֵק מִכָּאן.
לְסוּסָתִי בְּרִכְבֵי פַּרְעֹה, דִּמִּיתִיךְ.
גַּע בְּחַמּוּקַי, בִּצְרוֹר הַמֹּר שֶׁלִּי, אֲהוּבִי.
וְאָז מְצָאוּם שׁוֹמְרֵי הַצְּנִיעוּת.
גִּדּוּל
אֵין מַדְרִיךְ לַמָּוֶת.
סְבִיבוֹ כֻּלָּם בְּרִיאִים.
הוּא מְשַׁיֵּט סַהֲרוּרִי וְסוֹעֵר.
תָּאִים מְחַבְּלִים בְּגוּפוֹ.
הוֹרְסֵי תָּכְנִיּוֹת.
יוֹצֵא לְמַסָּע, מֵכִין אֶת מוֹתוֹ.
קוֹל פְּנִימִי מִתּוֹכוֹ, מַעֲמִיד פָּנִים כְּמַאֲמִין.
גִּדּוּל פִּרְאִי.
חוֹלֶה בָּא חוֹלֶה הוֹלֵךְ וְהַסַּרְטָן נִשְׁאָר.
הַמְּטֻפָּל מְמַקֵּד זַעֲמוֹ בָּעוֹלָם.
רֹאשׁ הַמַּחְלָקָה מֵכִין אוֹתוֹ כְּפַג תֹּקֶף.
בַּמִּשְׂרָד תִּיק חָדָשׁ נִפְתָּח.
אוֹנְקוֹלוֹג, מוֹדֵד אֶת הַגֹּדֶל.
הֵמָטוֹלוֹג, בּוֹחֵן אֶת תְּנוּעַת הַפֶּרֶא.
הַקַּרְדְּיוֹלוֹג פּוֹלֵשׁ בְּהַדְמָיָה.
הַפָּתוֹלוֹג, מְאַבְחֵן אֶת מְסֻכָּנוּתוֹ.
הַתָּאִים נֶחְלָקִים לְטוֹבִים וְרָעִים.
"סַרְטָן אָרוּר, רוֹצֵחַ.
נַעֲלִים אוֹתְךָ גִּדּוּל.
נְכַוֵּץ אוֹתְךָ בִּפְצָצָה כִימִית.
נְשַׁבֵּשׁ חַיֶּיךָ בְּטִיל בִּיּוֹלוֹגִי.
אַךְ אַתָּה לֹא בַּבַּיִת.
מִתְגּוֹרֵר בְּדִירָה חֲדָשָׁה.
הַיּוֹם בְּרֵאָתוֹ שֶׁל מַר עָשָׁן.
מָחָר בְּרִקְמַת עוֹר שֶׁל גְּבֶרֶת שֶׁמֶשׁ."
קְרִינָה, זִהוּם אֲוִיר.
עוֹבֵר בְּתוֹרָשָׁה, מִבַּיִת לְבַיִת.
אַךְ מִישֶׁהוּ שָׁם, רוֹצֶה שֶׁאֵעָלֵם.
אַרְמוֹנוֹתַי וַחֲלוֹמוֹתַי נִמְחָקִים.
צִלּוּם אַחֲרוֹן שֶׁל זִכָּרוֹן.
מַשֶּׁהוּ שֶׁיִּשָּׁאֵר אַחֲרַי.
מִבְחַר גִּדּוּלִים.
גַּל עֵד
צֶמֶד אָהוּב, גַּל עֵד.
בָּבַת עַיִן שְׁלִישִׁית.
שְׁתֵּי דִּמְעוֹת דָּם, לוֹחֲשׁוֹת, כְּגַלֵּי יָם.
דִּמְעָה אַחַת מְגַלָּה, דִּמְעָה שְׁנִיָּה עֵדָה.
עַמּוּד אֵשׁ בְּנִשְׁמָתִי, גַּלְגַּל הַחַיּוּת בְּגוּפִי.
חֲוָיַת קִיּוּמִי הָאַחֵר, הָרוֹאֶה אֶת הַנּוֹלָד,
מַעְבֹּרֶת מַחְשַׁבְתִּי, מְצַוָּה: עַד כָּאן.
מְלַוָּה אוֹתִי.
כְּשֶׁהַכֹּל קוֹרֵס, הַמֵּת חַי.
עֵין גַּל עֵד, שְׁלִישִׁית,
מִקְלַעַת שֶׁמֶשׁ, הַכּוֹחַ,
מִשְׁמַר גְּבוּלוֹתַי,
מַזְהִירָה, תִּהְיֶה מְאֻזָּן.
לְעוֹלָם לֹא תִּהְיֶה לְבַד.
גִּלְגּוּלִים
עוֹמֵס עַל גַּבִּי חוֹתָם מְקֻלָּל.
נֶאֱבָק בַּיִּסּוּרִים.
מַחְשִׁיךְ עֵינַי מִלִּרְאוֹת פְּצָעַי.
מְזַהֶה מַגֵּפָה, לֹא מְחַסֵּן אֶת עַצְמִי.
מַחְבִּיא מַרְבְּצֵי דֹּמֶן, וְנוֹפֵל,
בּוֹדֵד עַל בַּקְבּוּק קוֹנְיָאק.
בַּמִּטָּה, מִתְכַּרְבֵּל כְּקִפּוֹד, כְּעֻבָּר בְּעַרְסָל.
חוֹלֵם, בּוֹהֶה, מוֹשֵׁךְ רַגְלַיִם.
סְגֹל עַפְעַפַּיִם בְּמַבָּט מְצֹעָף,
זִכְרוֹנִי שׁוֹקֵעַ, וַאֲנִי עוֹטֶה שִׁרְיוֹן.
מִתְרַחֵק לְבוּעַת שַׁלְוָה.
דֵּמֶנְצְיָה מַחְזִירָה אוֹתִי בְּגִלְגּוּל מְחֻדָּשׁ.
נוֹשֵׁף בּוּעוֹת סַבּוֹן.
כֹּה שָׂמֵחַ, עַל כָּל נִפּוּחַ בָּלוֹן.
גְּשָׁרִים וּשְׁקָרִים
רוֹבֵץ עַל גֶּשֶׁר רָעוּעַ מִיְּסוֹדוֹ.
זוֹחֵל כְּנָחָשׁ בִּסְבַךְ שְׁקָרִים.
גּוֹלֵשׁ כְּעִוֵּר, לְשִׂיחַ חֵרְשִׁים.
הַגֶּשֶׁר בְּשִׁפּוּעַ חַד. אֵימְתָנִי.
לֹא נוֹרָא, אֲנִי מִתְנַחֵם, רַק עוֹד שֶׁקֶר אֶחָד.
וּבוֹדֶה מִתּוֹכִי סִפּוּר דִּמְיוֹנִי.
נִתְפַּסְתִּי בְּשֶׁקֶר אֶחָד, סִדְרָתִי.
"צֵא מֵחַיַּי שַׁקְרָן".
הִיא אוֹמֶרֶת.
וַאֲנִי לֹא מוֹדֶה.
וְהִיא לֹא מַאֲמִינָה.
הַגֶּשֶׁר עָלָיו רָבַצְתִּי נִשְׁבָּר.
"אוֹתִי מְשַׁקְּרִים רַק פַּעַם אַחַת, שָׂרַפְתָּ הַכֹּל".
הִיא חוֹזֶרֶת.
"לֵךְ תָּקִים קֶשֶׁר חָדָשׁ, מֵעֵבֶר לַנָּהָר עִם אַחֶרֶת".
וַאֲנִי נִשְׁבָּע.
"אֵלֵךְ יָשָׁר, אֶקְבֹּר אֶת שְׁקָרַי, אִוָּלֵד מֵחָדָשׁ".
הִיא נוֹעֶצֶת עֵינַיִם.
"אַף לְעַצְמְךָ אַתָּה מְשַׁקֵּר, נִכְלֵאתָ בְּמַעְגַּל שְׁקָרֶיךָ".
וְאָז מְסַכֶּמֶת.
"רַכֶּבֶת הַכַּחַשׁ, דָּרְסָה לָנֶצַח אֶת אַהֲבָתִי".
גַּשָּׁשִׁית אִינְטִימִית
הוֹלְכֵי הָרָכִיל כִּנּוּ אוֹתָהּ נַעֲרַת מַקְהֵלָה זְנוּתִית.
בְּעֵינַי תְּמוּנָתָהּ מוּאֶרֶת.
לֹא שָׁמַעְתִּי. לֹא רָאִיתִי. מָחַקְתִּי.
שְׁטוּף מְזִמָּה וְזִמָּה, עֵינַי צָבְעוּ אוֹתָהּ כִּקְדוֹשָׁה.
עָטַפְתִּי אֶת כֻּלָּהּ בְּהִתְרַגְּשׁוּת.
בְּזִכְרוֹנִי נוֹתְרָה רַק תַּאֲוָתִי.
וְשִׁיר אֶחָד שֶׁאָהֲבָה.
בִּפְגִישָׁתֵנוּ הָאַחֲרוֹנָה הִיא בָּרְחָה.
חָתְרָה לַחוֹף בִּדְמָעוֹת.
לְלֹא מִלָּה, לְלֹא הֶסְבֵּר, נֶעֶלְמָה מֵחַיַּי.
הַחְלָפַת מְסִלָּה
בְּמִדְרוֹן שִׁירֵי הָעַצְבוּת,
עֵינַי מְחַפְּשׂוֹת נְקֻדּוֹת אוֹר.
מִתְבּוֹנֵן בְּסֵפֶל הַקָּפֶה, בְּפִסְלִי הָאַחֲרוֹן,
בְּצִיּוּר נְדִידַת הַצִּפּוֹרִים, שֶׁצִּיַּרְתִּי מִזְּמַן,
מִסְתַּכֵּל בַּחֹרֶשׁ מִבַּעַד לַחַלּוֹן.
פְּרֶלוּד לִפְסַנְתֵּר שֶׁל שׁוֹפֶּן.
מֵאֵט אֶת הַצְּלִילָה לְיַם תּוֹדַעְתִּי.
הָרַכֶּבֶת שֶׁל תּוֹכִי בּוֹחֶרֶת מַסְלוּל חָדָשׁ.
בּוֹקַעַת מִתַּחְתִּית נִשְׁמָתִי לִמְסִלָּה עִלִּית.
לְפָנַי נִפְרָשׂ עוֹלָם אֲמִתִּי.
מַרְאֶה מוּחָשִׁי וּמְחֻדָּד, מֵפִיחַ בִּי כּוֹחַ.
חַיִּים מִחוּץ לָעַצְמִי הַמְּיֻסָּר.

הַלֵּב אֵינוֹ רַק שְׁרִיר, עוֹרֵק כְּלִילִי סָתוּם.
אַדְמָתִי בּוֹעֶרֶת בְּאַרְגָּמָן לוֹהֵט.
גַּעְגּוּעַ רוֹמַנְטִי מִשְׁתַּחְרֵר מִדַּם לִבִּי.
שִׁיר.
מִשְׂחָקִים
יֶלֶד וְאָבִיו בַּחֲנוּת צַעֲצוּעִים.
"אֲנִי רוֹצֶה רוֹבֶה".
"נִרְאֶה כְּמוֹ רוֹבֶה אֲמִתִּי".
יֶלֶד מְשַׂחֵק בְּרוֹבֵה צַעֲצוּעַ, יוֹרֶה בְּחָבֵר.
הֶחָבֵר נוֹפֵל, מֵנִיעַ אֶת רַגְלָיו וְרֹאשׁוֹ.
הַיּוֹרֶה: "אַתָּה מֵת, אָסוּר לְךָ לָזוּז".
הָאֵם: "לְהַפְסִיק לְשַׂחֵק, בּוֹאוּ לֶאֱכֹל".
יֶלֶד חַי וְיֶלֶד מֵת, אוֹכְלִים לֶחֶם, שׁוֹתִים שׁוֹקוֹ.
לְחַיָּל נִפְלַט כַּדּוּר.
חֲדָשׁוֹת: "חַיַּל צַהַ"ל פָּלַט כַּדּוּר וְהָרַג אֶת חֲבֵרוֹ".
הָאֵם הַבּוֹכָה: "לִבֵּנוּ עִם מִשְׁפַּחַת הֶחָלָל".
הָאָב: "טְרָגֶדְיָה נוֹרָאָה לִשְׁתֵּי הַמִּשְׁפָּחוֹת".
מ.צ.ח: "כַּדּוּר נִפְלַט כְּתוֹצָאָה מִמִּשְׂחָק בְּנֶשֶׁק".
אַרְצוֹת הַבְּרִית:
"יֶלֶד בֶּן 4 נָטַל רוֹבֶה וְיָרָה לַמָּוֶת בַּחֲבֵרוֹ".
הָאָב: "לֹא מֵבִין אֵיךְ זֶה קָרָה? הוּא יֶלֶד טוֹב.
הַנֶּשֶׁק הָיָה תָּלוּי עַל קִיר".
הָאֵם: "הִזְעַקְתִּי מִיָּד אֶת כּוֹחוֹת הַבִּטָּחוֹן".
חֲדָשׁוֹת:
"יָרָה בְּגָמָל".
"יָרָה בְּכֶלֶב".
"יָרָה בְּאָחִיו".
"יָרָה בְּעַצְמוֹ".
"יָרָה בָּרֹאשׁ".
"יָרָה בָּרֶגֶל".
"יָרָה בְּנַהַג מוֹנִית".
"יָרָה בַּטֶּסְטֶר".
הַשֶּׁמֶשׁ זוֹרַחַת וְשׁוֹקַעַת.
הַמִּטְוָח מַמְשִׁיךְ.
פּוֹרְטְרֵטִים
בַּגָּלֶרְיָה בְּלוֹנְדוֹן חַשְׁתִּי בְּחִילָה.
רָאִיתִי צִיּוּרִים שֶׁל מְפֻרְסָמִים.
וִיקְטוֹרְיָה, מַלְכָּה שֶׁהִיא תְּמוּנָה.
לֹא אָדָם.
פּוֹרְטְרֵט לְלֹא הַבָּעָה.
מַקִּיפִים אוֹתִי רָאשִׁים, חֶלְקָם לֹא אֱנוֹשִׁיִּים.
גָּלֶרְיַת הַחַיִּים.
חֲתֻנָּה, עוּגָה עִם חָתָן וְכַלָּה.
מִתַּחַת לַחֻפָּה, דְּיוֹקָן כָּפוּל.
מוֹרָן וּשְׂנִיר נִשָּׂאִים.
תְּמוּנוֹת מִטַּיֶּלֶת, שְׁבִירַת הַכּוֹס.
גָּלֶרְיָה הַמֻּצֶּגֶת בָּאַלְבּוֹם, פּוֹרְטְרֵטִים נוֹסָפִים.
וְהֵם מִתְמוֹגְגִים בַּהֲנָאָה.
בַּדֶּרֶךְ עָבַרְתִּי דֶּרֶךְ כְּפָר נִדָּח.
תְּמוּנַת פַּרְצוּף עֲנָק, קִדְּמָה אֶת פָּנַי.
רֹאשׁ הַכְּפָר מְעֻנָּב וּמְמֻשְׁקָף.
מַבִּיט בִּי.
"תָּמִיד אֶהְיֶה רֹאשׁ".
"חָשׁוּב לִי לָדַעַת מִי אַתָּה".
"שֶׁתִּבְחַר בַּפּוֹרְטְרֵט שֶׁלִּי".
בְּבֵית קָפֶה אֲנִי מֻקָּף בְּפָנִים.
"קָפֶה פּוֹרְטְרֵט".
צַיָּדִים מִטִּבְעָם
צִפּוֹר מְשַׁיֵּט בְּשַׁלְוָה בַּשָּׁמַיִם.
צַיַּד הָעוֹפוֹת, שָׁם לְמַטָּה בַּשָּׂדֶה, דָּרוּךְ.
בְּצִמְאוֹן דָּם מְכַוֵּן אֶת קְנֵה הָרוֹבֶה.
לוּ הָיָה בְּיָדִי עָצְמַת הַכִירוּרְג.
לְנַתֵּחַ רֹאשׁ צְמֵא דָּם.
לַחְקֹר כְּמֻמְחֶה, כָּל תָּא,
עָנָף, גֶּזַע, אֻנּוֹת, וּקְלִפָּה.
שֶׁל אוֹתוֹ הַלּוֹחֵץ עַל הַהֶדֶק.
אוֹתוֹ מְקַפֵּד הַחַיִּים.
דּוֹאֶה מַטָּה כְּצִפּוֹר, אֶל הַלֹּא נוֹדַע.
יְדֵי הָאָדָם מְגֹאָלוֹת בְּדָם.
צָדִים מִטִּבְעָם, צִפּוֹר וְאָדָם.
צֵרוּפִים
הַיּוֹם וְאֶתְמוֹל הָרוּחַ מְעַנֶּגֶת.
אֶתְמוֹל וְהַיּוֹם בְּרוּחַ נְכָאִים.
לֹא מוֹחֵק אֶת הֶעָבָר, לֹא שׁוֹכֵחַ.
מְאֹהָב בַּמּוּעָקָה, בַּחֲרָדָה.
שׁוֹקֵעַ בְּמַעֲמַקֵּי תּוֹדַעְתִּי.
מְפַלֵּס לִי דֶּרֶךְ וּמַתְחִיל מֵחָדָשׁ.
לִבִּי נוֹטֶה לִתְחוּשׁוֹת אַשְׁמָה.
מְסָרֵב לַהֲדֹף אֶת רִגְשׁוֹתַי.
לֹא אַדְחִיק מוּעָקָה אוֹ כְּאֵב.
אַף לֹא אֹמַר כִּי אֵין שְׁלִיטָה.
מְטַפֵּל בְּעַצְמִי.
מְחַפֵּשׂ אֶת הַדֶּרֶךְ וּמַכְבִּיד עַל עַצְמִי.
מַשְׁלִים עִם דְּחָפִים וִיצָרִים.
אֶתְמוֹל הָיְתָה בִּי בַּצֹּרֶת בִּשְׂדוֹת הַחֶמְלָה.
הַיּוֹם מְלַבְלֶבֶת הַבּוּשָׁה וּפוֹרַחַת.
אֶתְמוֹל, יְצִירָתִי נָפְחָה אֶת רוּחָהּ.
הַיּוֹם שָׁבָה הָרוּחַ לִנְפֹּחַ בְּמִפְרָשַׂי.
צֵל מֵעִיב
אֵין בְּלִבִּי גַּם לֹא צֵל צִלּוֹ שֶׁל סָפֵק.
הָאֻמְלָל, הַחַי בְּצִלּוֹ הַמֵּעִיב שֶׁל אַחֵר.
נוֹפֵל לְמַלְכֹּדֶת קוּרֵי הָעַכָּבִישׁ.
מַעֲלֶה עֹבֶשׁ וְסוֹבֵל.
אֶכָּנֵס לְבוּעָה, אַסְתִּיר עֵינַי מִלִּרְאוֹת.
רוֹאֶה עֵץ שֶׁאֵינוֹ סוֹלֵחַ מֵטִיל צֵל קָדוּם.
שָׁרָשָׁיו שָׁרְשֵׁי נָקָם.
אֵין כָּל דֶּרֶךְ לֶאֱחֹז בּוֹ, אִישִׁיּוּתוֹ מְפֻצֶּלֶת, חֲמַקְמַקָּה.
מְכַסֶּה אֶת כֻּלִּי, מֵגִיחַ כְּמַפֹּלֶת הַמְּכַסָּה כָּל חַי.
וַאֲנִי אָטוּס בֵּין הַצְּלָלִים, עַד תַּעֲבֹר נִקְמָתוֹ.
אֲגַיֵּס מִתּוֹכִי אֶת כִּשְׁרוֹנִי לִקְרַב בְּלִימָה.
אֶצֹּר לִי מָצוֹף, מֵעַל עָמְקֵי דִּכָּאוֹן.
אֶגְלֹשׁ עַל מִלִּים מֵחַדְרֵי לְבָבִי.
אֶנְדֹּד לִי בֵּין טִפּוֹת שִׁיר, לְמַעֲמַקֵּי עִיר מִקְלָט.
צַר מִלְּהָכִיל
מְכָלִי רוֹתֵחַ וְגוֹלֵשׁ, צַר מִלְּהָכִיל.
לֹא מוּכָן לִסְפֹּג יוֹתֵר.
מֵיתָרַי לֹא גְּמִישִׁים יוֹתֵר.
אֶנְעַל אֶת עַצְמִי בִּפְנֵי כָּל נֶזֶק עֲתִידִי.
דַּי לְהָכִיל אֶת חוֹרְשֵׁי רָעָתִי.
גְּבוּל אֲשַׂרְטֵט לְכָל רוֹמֵס.
לֹא אֶהְיֶה עוֹד הַקָּרְבָּן הַתְּמִידִי.
דַּי לְהַעֲמָדַת הַפָּנִים.
אֵינִי יָכוֹל יוֹתֵר לַהֲפֹךְ אֶת עוֹרִי.
גַּם אִם תֵּעָצֵר זְרִימָתִי וּפְרִיחָתִי,
לֹא אַסְכִּים יוֹתֵר לָשֵׂאת צְלָב שֶׁל אַחֵר.
גַּם אִם אֶחְזֹר לַחַיִּים, כְּקַרְנַף מְכֻנָּף.
לֹא אֶעֱבֹר יוֹתֵר דֶּרֶךְ יִסּוּרִים.
בָּלַעְתִּי מַיִם מְלוּחִים בַּמְּעַרְבֹּלֶת,
הוּצַפְתִּי בְּיַם רִגְשׁוֹתַי.
נִשְׁמָתִי נֶאֶבְקָה בְּקַרְנַף מְכֻנָּף.
עַל עֶרֶשׂ דְּוַי הֵנַפְתִּי דֶּגֶל לָבָן.
הֱיוֹת וְאֵינֶנִּי נָמֵר, אֵין בְּכוֹחִי לַהֲפֹךְ חֲבַרְבּוּרוֹת.
לֹא אוּכַל לִהְיוֹת הַקַּרְנַף.
אֶסְתַּפֵּק בְּלִהְיוֹת מְכֻנָּף.
אָעוּף כְּפַרְפַּר, הַשּׁוֹמֵעַ קוֹל עֲטַלֵּף.
אֶתְחַמֵּק מִלִּהְיוֹת פְּרִי בָּשֵׁל לִמְכַרְסְמֵי הַלַּיְלָה.
רֵיחוֹת חַיִּים וּמָוֶת
אָבִי,
שָׁקַעְתָּ בְּעִרְפּוּל, בְּחֹסֶר הַכָּרָה.
נִקַּזְתִּי הַפְרָשׁוֹתֶיךָ בְּקָטֵטֶר וְזוֹנְדָּה.
הַשַּׁחַר עָלָה, וְהָאָח הֶחְלִיף אֶת בָּלוֹן הַחַמְצָן.
רֵיחוֹת חָמְרֵי חִטּוּי וְשֶׁתֶן, חַדְרֵי חוֹלִים.
בָּרַחְתִּי לְקָפֶה עֲטָרָה.
זֶה עַתָּה יָצְאוּ עוּגוֹת הַשְּׁמָרִים.
נִיחוֹחוֹת שׁוֹשַׁנֵּי שׁוֹקוֹלָד וְקִנָּמוֹן.
הִזְמַנְתִּי קָפֶה וְעוּגַת צִמּוּקִים.
צְעִירָה נִכְנְסָה וַאֲנִי מְחַיֵּךְ.
רִבְקָה מִבְּצַלְאֵל.
"דְּמָעוֹת וְגֶשֶׁם חַד הֵם".
דֵמִיס רוּסוֹס, וּבְנָהּ שֶׁל אַפְרוֹדִיטֶה.
"גַּם לִי הִקְדִּישׁ דֵמִיס שִׁיר: רֵבֵּקָה", חִיְּכָה.
צָהֳרַיִם, אֲנִי מִתְנַתֵּק מֵאֶתְנַחְתָּא רוֹמַנְטִית.
אָבִיךָ נִפְטַר לִפְנֵי שָׁעָה, אָמַר הָרוֹפֵא.
אִמָּא, גַּם מִמֵּךְ לֹא נִפְרַדְתִּי.
בְּאוֹתוֹ יוֹם שִׁשִּׁי רָקַדְתִּי עַל שֻׁלְחָן.
בְּפַּאבּ מִסְעָדָה.
וְאַתְּ בְּבֵיתֵךְ גָּוַעְתְּ.
הִגַּעְתִּי, מַדִּיף בֹּשֶׂם וְרֵיחַ בִּירָה בִּנְשִׁימָתִי.
שָׁקַע בִּי רֵיחַ מוֹתֵךְ, כַּעַס וְאַשְׁמָה.
אִמָּא, אַבָּא,
הֶעֱדַפְתִּי אֶת חַיֵּי הָרֶגַע.
מַדּוּעַ נָתַתִּי לָכֶם לְהֵעָלֵם מֵחַיַּי?
רְעָשִׁים
הַמּוֹחוֹן בּוֹגֵד בִּי.
לֹא בּוֹלֵם אֶת זַעֲזוּעַי.
הֶחָלָל מֵעִיק וְרַע, אַךְ כָּל צְעָדַי רְצוֹנִיִּים.
מַאֲשִׁים אֶת יְצָרַי וּמֵנִיחַ יָדִי עַל מַפְתֵּחַ לִבִּי.
רְעָשִׁים מַפְרִיעִים.
אֲנִי מֵצֵר, וּמְיַצֵּר אֶת הָרַע.
אֲנִי שׁוֹאֵל.
לָמָּה טוֹב לְמַלֵּא חָלָל אַחֵר?
לְמִי שֶׁמַּגִּיעַ לְפָנַי.
מִתְבּוֹנֵן פְּנִימָה וּמְצַמְצֵם אֶת הַמַּחְשָׁבָה.
הַמּוֹחוֹן שֶׁבִּי רוֹעֵם בִּתְזָזִית.
אֲנִי שׁוֹאֵל.
לָמָּה רַע? לָמָּה כְּשֶׁטּוֹב לִי, רַע לְךָ?
וּמָה כֹּה רַע בְּכָךְ?
אֲנִי, אָדָם טוֹב, הוּא אָדָם רַע.
אֲנִי נוֹתֵן, וְהוּא רַק מְקַבֵּל.
הַמּוֹחוֹן שֶׁבִּי רוֹעֵשׁ.
רַק לְעַצְמְךָ תִּתֵּן.
אֲבָל אַתָּה לֹא לְבַד.
מִתְבּוֹנֵן מִסָּבִיב.
תֵּן מֵעַצְמְךָ לְאַחֵר.
תְּעַגֵּל עַצְמְךָ בְּטוֹב.
שִׁבְעָה עֲשׂוֹרִים
בּוֹדֵדָה עֲלִיצוּתִי כְּדוּגִית.
הַשָּׁטָה בַּאֲגַם גּוֹעֵשׁ, דּוֹמֵעַ.
שׁוֹאֵל מִי יַצִּילֵנִי. וּלְאָן?
הַאִם, לְמִלְחָמָה, אוֹ לְקַבָּלָה?
מַהֲלָךְ יָבֵשׁ בַּעֲרוּץ נַחַל אַכְזָב.
שִׁבְעָה עֲשׂוֹרִים שֶׁל יְמֵי כִּפּוּר.
שִׁבְעִים יְמֵי זִכָּרוֹן, וְלֵילוֹת סֵדֶר.
עֲשׂוֹרִים שֶׁל יִזְכֹּר,
לֵילוֹת עַצְמָאוּת, לֹא שֶׁלִּי הֵם.
נְגִיעוֹת שְׁקִיעָה קְסוּמוֹת.
עַנְנֵי כְּבָשִׂים, צִפּוֹרִים נוֹדְדוֹת.
יוֹשֵׁב בִּשְׁתִיקָה בְּאָהֳלִי.
לֹא עֵת חֲגִיגָה הִיא לִי.
מְגוֹלֵל אֶבֶן, מִבּוֹר בְּתוֹכִי.
נִפְרָד מִתְּמוּנוֹתַי הַמְּסַפְּרוֹת.
וַאֲנִי מְכַבֶּה בִּנְשִׁיפָה שִׁבְעִים נֵרוֹת.
עֵינַי הָעֲיֵפוֹת, דּוֹמְעוֹת.
מִצְּחוֹק.
שְׁרִיקָה
מַשְׁרוֹקִית קְטַנָּה מְצֻפָּה זָהָב,
תְּלוּיָה בְּחוּט שָׁנִי עַל צַוָּארִי.
בַּיּוֹם וּבַלַּיִל.
"הַיֶּלֶד הַשּׁוֹרֵק" הָיָה שְׁמִי.
שְׁרִיקָה שֶׁיָּצְאָה מִקְּרָבַי.
גֹּבַהּ הַצְּלִיל, מְעַרְבֹּלֶת בַּחֲלַל הַפֶּה.
דָּמִי רָתַח.
"הַשְּׁרִיקָה" שֶׁל י.ל פֶּרֶץ,
כִּשְּׁפָה אוֹתִי כִּתְפִלָּה.
הַשּׁוֹרֵק, לְיַד הָרַכֶּבֶת.
הַשּׁוֹטֵר הַמְּכַוֵּן תְּנוּעָה.
שׁוֹפֵט הַכַּדּוּרֶגֶל.
הַמּוֹרֶה לִסְפּוֹרְט.
שְׁרִיקַת הָאֲנָפוֹת לְיַד בְּרֵכוֹת הַדָּגִים.
מְרַחֲפוֹת וְצוֹלְלוֹת, פְּרוּשׂוֹת כְּנָפַיִם.
מְצִלּוֹת עַל הַמַּיִם.
צָדוֹת דְּגִיגִים.
אֵין לִי שְׁרִיקַת פְּתִיחָה אוֹ סִיּוּם.
שׁוֹרֵק לְלֹא הֲפוּגָה.
צְלִיל חָדָשׁ, מְמַלֵּא חָלָל.
רוּחַ שׁוֹרֶקֶת בְּלִבִּי עַד לְאֵינְסוֹף.
קוֹלִי הַפְּנִימִי שׁוֹרֵק לְמַעֲנִי.
בַּחֲלוֹמִי, בִּתְחִיַּת הַמֵּתִים אֶשְׁרֹק שׁוּב.
תְּפִלָּה
הֵם מַרְעִיפִים עַל עַצְמָם אַהֲבָה.
וֶאֱלוֹהִים מֵעָלַי קוֹרֵא, "אֵין עוֹד מִלְּבַדִּי".
וְהָאָדָם עַל פְּנֵי הָאֲדָמָה שׁוֹאֵג, "רַק אֲנִי קַיָּם בָּעוֹלָם".
אָנָּא, אֲנִי מִתְחַנֵּן, הַכֵּה אֶת הָאֶגוֹ בְּמַבּוּל.
שְׁלַח אֵלַי מַצִּיל, תֵּבַת נֹחַ וְעוֹרֵב.
כְּלִי שֶׁיְּגָרֵד מִמּוֹחִי אֶת הָאֹפֶל.
אֶשְׁתַּמֵּשׁ בּוֹ כְּמַבְרֵג.
אוֹתוֹ הַשּׁוֹלֵף בֹּץ מֵחֲרִיצֵי סוּלְיוֹת גּוּמִי.
הֲיִשְׁמַע קוֹלִי, צְלִילֵי מַחְשְׁבוֹתַי?
הוּא הַצּוֹוֵחַ בְּאָזְנַי, "רַק אֲנִי קַיָּם".
תַּפְנִית חַדָּה
כּוֹחַ דִּמְיוֹנִי נוֹגֵעַ בְּסִיגֵי לִבִּי.
מְלַקֵּט כָּל אוֹת וּמִלָּה.
כּוֹתֵב, מוֹחֵק וּבוֹרֵר.
רוֹקֵחַ תַּרְכִּיזִים.
הֲגִיגַי זוֹחֲלִים אַחֲרֵי שְׁבָרִים.
מְנַסֶּה לְאַגֵּף מַהוּת.
פּוֹרֵק מִתּוֹכִי כַּעַס וּבוּז עַל עוֹלָם וּמְלוֹאוֹ.
רוֹאֶה אֲנָשִׁים כִּרְפֵי שֵׂכֶל, כִּנְכֵי רוּחַ.
אַךְ לֹא לְכֻלָּם מוֹחַ וְעִפָּרוֹן.
וְגַם עַל שׁוֹטֵה הַכְּפָר כְּבָר נִכְתַּב חָרוּז.
מָה כִּי תִּשְׁפֹּךְ זַעַמְךָ עַל אֲנָשִׁים?
צֹר שִׁיר.
נְשֹׁם, נְדֹד הַרְחֵק מִכָּאן.
עֲזֹב.
קִבַּלְתִּי עַל עַצְמִי לְוַתֵּר עַל דַּחַף הַרְסָנִי.
מַקְשִׁיב לְקוֹל אַחֵר, מְחַבֵּר לְמַעְגָּל אֱנוֹשִׁי.
לְמָשָׁל:
כֻּלָּם מְקַבְּלִים אֶת הַשַּׁבָּת הַנִּכְנֶסֶת.
מַבְדִּילִים בֵּין קֹדֶשׁ לְחֹל, בְּצֵאתָהּ
כָּל אִשָּׁה וְאִישׁ בְּדַרְכָּם.
וַאֲנִי אֵלֵךְ בְּדַרְכִּי שֶׁלִּי.
יֵשׁ חַיִּים מֵעֵבֶר לְמוּעָקָה מְיַסֶּרֶת.
שִׁיר שַׁבָּת תָּמִים, נוֹסֵךְ עַל פָּנַי חִיּוּךְ.
