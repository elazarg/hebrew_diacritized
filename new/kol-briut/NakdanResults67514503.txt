יְלָדִים וּמִתְבַּגְּרִים
תִּינוֹקוֹת יְלָדִים וּמִתְבַּגְּרִים
הַמֵּידָע בָּאֲתָר נוֹעַד לְסַיֵּיעַ, לָכֶם הַהוֹרִים, בְּטִיפּוּל בַּיֶּלֶד וּבַמִּתְבַּגֵּר. הַמֵּידָע מִתְיַיחֵס לַנּוֹשְׂאִים: גְּדִילָה וְהִתְפַּתְּחוּת, חִיסּוּנִים, בְּטִיחוּת, תְּזוּנָה נְכוֹנָה, בְּרִיאוּת הַתַּלְמִיד בְּבֵית הַסֵּפֶר, קַיְיטָנוֹת, דִּיאֵטָה בְּקֶרֶב מִתְבַּגְּרִים, עִישּׁוּן וְעוֹד. כְּמוֹ כֵן נִיתָּן לִמְצוֹא מֵידָע עַל יְלָדִים עִם צְרָכִים מְיוּחָדִים, טִיפַּת חָלָב, מְכוֹנִים לְהִתְפַּתְּחוּת הַיֶּלֶד, וְהַמַּעֲרָךְ הַהִתְפַּתְּחוּתִי בְּיִשְׂרָאֵל.  

בְּרִיאוּת הַתַּלְמִיד - שֵׁירוּת הַבְּרִיאוּת בְּבֵית הַסֵּפֶר נוֹעֲדוּ לְאַפְשֵׁר לַתַּלְמִידִים לְמַמֵּשׁ אֶת מְלוֹא הַפּוֹטֶנְצְיָאל הַבְּרִיאוּתִי שֶׁלָּהֶם, מִתּוֹךְ הִתְבַּסְּסוּת עַל רְפוּאָה מוֹנַעַת וְקִידּוּם בְּרִיאוּת; לְאַתֵּר תַּלְמִידִים עִם בְּעָיוֹת בְּרִיאוּת וּלְהַפְנוֹתָם מוּקְדָּם כְּכָל הָאֶפְשָׁר לְהֶמְשֵׁךְ בֵּירוּר אוֹ טִיפּוּל; לְיַיעֵץ וּלְהַדְרִיךְ אֶת הַתַּלְמִידִים, הַצֶּוֶות הַחִינּוּכִי וְהַהוֹרִים בְּנוֹשְׂאֵי בְּרִיאוּת סְפֵּצִיפִיִּים.

יְלָדִים עִם עִיכּוּב הִתְפַּתְּחוּתִי - כְּ-10% מִכְּלַל הַיְּלָדִים יֵשׁ אִיחוּר מַשְׁמָעוּתִי בְּתַהֲלִיכֵי הַהִתְפַּתְּחוּת הַשּׁוֹנִים. יְלָדִים אֵלּוּ נִכְנָסִים לְקָטֵגוֹרְיָה שֶׁל יְלָדִים עִם חֲשָׁשׁ לְעִיכּוּב (אִיחוּר) הִתְפַּתְּחוּתִי. סִימָנִים רִאשׁוֹנִים לְעִיכּוּב הִתְפַּתְּחוּתִי, זְכוּיוֹת וְשֵׁירוּתִים לַיֶּלֶד הַמְּיוּחָד.

עֲקוּמּוֹת גְּדִילָה -  הַעֲרָכַת הַגְּדִילָה הִיא אֶחָד הַמַּדָּדִים הַחֲשׁוּבִים לְהַעֲרָכַת מַצַּב הַבְּרִיאוּת וְהַתְּזוּנָה שֶׁל הַיֶּלֶד/ה. בְּדַף זֶה מוּצָּגוֹת עֲקוּמּוֹת גְּדִילָה לְפִי גִּילָאֵי הַיְּלָדִים, לְשִׁימּוּשׁ אַנְשֵׁי הַמִּקְצוֹעַ וְהַהוֹרִים.
תְּזוּנַת יְלָדִים וּמִתְבַּגְּרִים - תְּקוּפַת הַיַּלְדוּת וְהַהִתְבַּגְּרוּת מִתְאַפְיֶינֶת בִּגְדִילָה, בַּהִתְפַּתְּחוּת וּבִרְכִישַׁת הָרַגְלִים לַחַיִּים. תְּזוּנָה הוֹלֶמֶת בְּגִילָאִים אֵלּוּ מְאַפְשֶׁרֶת גְּדִילָה וְהִתְפַּתְּחוּת תַּקִּינוּת וּמְסַיַּיעַת לִשְׁמוֹר עַל בְּרִיאוּת הַיְּלָדִים בַּהוֹוֶה וּבְחַיֵּיהֶם הַבּוֹגְרִים בֶּעָתִיד.
חִיסּוּנִים מוּמְלָצִים לִילָדִים - חִיסּוּנִים נֶחֱשָׁבִים לִפְרִיצַת דֶּרֶךְ בִּרְפוּאָה הַמּוֹנַעַת. הַחִיסּוּנִים מְגִנִּים עַל בְּרִיאוּת הַיְּלָדִים, עַל-יְדֵי מְנִיעַת הַדְבָּקָה בְּמַחֲלוֹת קָשׁוֹת. חִיסּוּנִים לִילָדִים וְנוֹעַר נִיתָּנִים בְּבֵית הַסֵּפֶר בְּהֶתְאֵם לְתָכְנִית שִׁגְרַת הַחִיסּוּנִים שֶׁל מִשְׂרַד הַבְּרִיאוּת. הַשְׁלָמַת חִיסּוּנִים נִיתֶּנֶת לְפִי הַצּוֹרֶךְ.
סְבִיבָה בְּטוּחָה וְהִתְנַהֲגוּת בְּרִיאָה  - הַקְנָיַית הֶרְגֵּלֵי הִתְנַהֲגוּת בְּרִיאָה לִילָדִים וִיצִירַת סְבִיבָה בְּטוּחָה לְמֶרְחַב פְּעוּלָּה עֲבוּרָם, מְאַפְשְׁרוֹת לָהֶם לִגְדּוֹל וּלְהִתְפַּתֵּחַ בִּבְרִיאוּת וּבַבִּטְחָה.
בְּרִיאוּת הַשִּׁינַּיִים - חָשׁוּב לִשְׁמוֹר עַל בְּרִיאוּת הַשִּׁינַּיִים כְּבָר עִם בְּקִיעַת שִׁינֵּי הֶחָלָב הָרִאשׁוֹנוֹת וּלְאוֹרֶךְ כָּל הַחַיִּים. אֲתָר זֶה מֶרְכַּז מֵידָע הַסְבָּרָתִי וּמֵידָע עַל זְכוּיוֹת וְשֵׁירוּתִים הַנִּיתָּנִים לִילָדִים בִּתְחוּם בְּרִיאוּת הַשֵּׁן.
פִיזְיוֹתֵרַפְּיָה לִילָדִים - תְּחוּם מִקְצוֹעִי רָחָב וּמְגוּוָּן וְלוֹ זִיקָּה לְכָל תְּחוּמֵי הָרְפוּאָה. הַשֵּׁירוּת נִיתַּן לִילָדִים בְּכָל גִּיל וּבְכָל מַצַּב בְּרִיאוּת, וּמִתְמַקֵּד בְּשִׁיקּוּם תִּפְקוּד וּתְנוּעָה לְמַעַן קִידּוּם בְּרִיאוּת וְעַצְמָאוּת הַיֶּלֶד. 
הַמְלָצוֹת לְהוֹרִים בְּנוֹשֵׂא הִתְמוֹדְדוּת עִם תְּגוּבוֹת חֲרָדָה אֵצֶל יְלָדִים בְּמַצְּבֵי לַחַץ - בְּמִידָּה וְאַתֶּם מְזַהִים כִּי יַלְדְּכֶם סוֹבֵל מֵהַסִּימָנִים הַמְּפוֹרָטִים בְּמֶשֶׁךְ תְּקוּפָה מְמוּשֶּׁכֶת, אוֹ כִּי סִימָנִים אֵלֶּה מַחְמִירִים, אָנָּא פָּנוּ לְיִיעוּץ פְּסִיכוֹלוֹגִי.
בְּרִיאוּת מִתְבַּגְּרִים - תְּקוּפַת גִּיל הַהִתְבַּגְּרוּת הִיא תְּקוּפָה שֶׁבָּהּ לְצַד גְּדִילָה, שִׁינּוּיִים גּוּפָנִיִּים וְהִתְבַּגְּרוּת מִינִית, חָלִים אֵצֶל הַמִּתְבַּגְּרִים גַּם שִׁינּוּיִים רִגְשִׁיִּים וְהִתְנַהֲגוּתִיִּים. מִתְבַּגְּרִים רַבִּים חוֹוִים קְשָׁיִים מִסּוּגִים שׁוֹנִים, שֶׁלְּעִיתִּים עֲלוּלִים לְהוֹבִיל לְהִתְנַהֲגוּת מְסַכֶּנֶת בְּרִיאוּתִית.
אֲתַר בְּרִיאוּת מִתְבַּגְּרִים מְאַגֵּד מֵידָע רֵלֵוַונְטִי לְמִתְבַּגְּרִים וּלְהוֹרֵיהֶם, כְּגוֹן: תְּזוּנָה נְבוֹנָה בְּגִיל הַהִתְבַּגְּרוּת וּבְעָיוֹת תְּזוּנָה, הִתְנַהֲגוּת בְּסִיכּוּן וְהִתְמַכְּרוּת, קַעֲקוּעִים וּפִירְסִינְגּ, זְהִירוּת וּמְנִיעָה שֶׁל מַחֲלוֹת הַמּוֹעֳבָרוֹת בְּאֶמְצָעוּת מַגָּע מִינִי, אִשְׁפּוּז פְּסִיכִיאַטְרִי שֶׁל קְטִינִים, מֵידָע לִנְעָרוֹת בְּהֵרָיוֹן לֹא מְתוּכְנָן וְעוֹד.
טִיפּוֹת חָלָב  -  טִיפַּת חָלָב הִיא תַּחֲנָה הַמְּסַפֶּקֶת שֵׁירוּתֵי בְּרִיאוּת וּרְפוּאָה בִּתְחוּם קִידּוּם בְּרִיאוּת וּמְנִיעָה לְנָשִׁים הָרוֹת, תִּינוֹקוֹת וִילָדִים (גִּילָאֵי לֵידָה עַד 6 שָׁנִים) וּמִשְׁפְּחוֹתֵיהֶם. תַּחֲנוֹת טִיפַּת חָלָב פְּזוּרוֹת בְּכָל רַחֲבֵי הָאָרֶץ וּמוּפְעָלוֹת עַל-יְדֵי לִשְׁכוֹת הַבְּרִיאוּת.
