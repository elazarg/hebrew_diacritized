הַפְרָיָה חוּץ גּוּפִית - IVF

הַפְרָיָה חוּץ גּוּפִית (In Vitro Fertilization - IVF) אֲשֶׁר יְדוּעָה גַּם בַּשֵּׁמוֹת: "הַפְרָיָה מְלָאכוּתִית" וְ"הַפְרָיַית מִבְחָנָהּ", הִינָּהּ אֶחָד מִן הַטִּיפּוּלִים הַשְּׁכִיחִים בְּיוֹתֵר בְּקֶרֶב טִיפּוּלֵי הַפּוֹרִיּוּת הַמִּתְקַדְּמִים. הַהַפְרָיָה הַחוּץ גּוּפִית הִיא טֶכְנִיקַת טִיפּוּלֵי הַפִּרְיוֹן הַשְּׁכִיחָה בְּיוֹתֵר בָּעִידָּן הַמּוֹדֶרְנִי.
 
בְּתַהֲלִיךְ הַהַפְרָיָה הַחוּץ גּוּפִית מִתְקַיֶּימֶת פְּגִישָׁה יְזוּמָה בֵּין הַבֵּיצִית לְזֵרְעוֹן בִּתְנָאֵי מַעְבָּדָה, כְּאַלְטֶרְנָטִיבָה לַמָּקוֹם הַמִּפְגָּשׁ הַטִּבְעִי שֶׁלָּהֶם – הַחֲצוֹצְרָה בְּגוּף הָאִישָּׁה. בְּמִסְגֶּרֶת טִיפּוּל הַפּוֹרִיּוּת מְבַצְּעִים שְׁאִיבַת בֵּיצִיּוֹת מִגּוּף הָאִישָּׁה, אִיחוּד עִם תָּאֵי הַזֶּרַע וְהַפְרָיָה בִּתְנָאֵי מַעְבָּדָה. לְאַחַר מִכֵּן, מַחֲדִירִים אֶת הָעוּבָּר אֶל הָרֶחֶם, תּוֹךְ מַתַּן טִיפּוּל הוֹרְמוֹנָאלִי תּוֹמֵךְ. 
 
בַּמָּקוֹר, טִיפּוּל זֶה הוּמְצָא כְּדֵי לַעֲקוֹף אֶת הַצּוֹרֶךְ בַּחֲצוֹצְרוֹת, אֵצֶל נָשִׁים שֶׁהַחֲצוֹצְרוֹת שֶׁלָּהֶן הָיוּ חֲסֵרוֹת אוֹ חֲסוּמוֹת, אוּלָם בַּשָּׁנִים הָאַחֲרוֹנוֹת, טִיפּוּל זֶה נִמְצָא בְּשִׁימּוּשׁ כְּדֵי לִפְתּוֹר עוֹד מִסְפַּר בְּעָיוֹת, כּוֹלֵל בְּעָיוֹת זֶרַע קָשׁוֹת, אַנְדּוֹמֶטְרִיּוֹזִיס קָשֶׁה (גְּדִילָה שֶׁל רִקְמַת הָרֶחֶם בַּאֲזוֹרִים שֶׁמִּחוּץ לָרֶחֶם), הַפְרָעוֹת בִּיּוּץ וְהוֹרְמוֹנִים מוּרְכָּבוֹת וּבְעָיוֹת הַנּוֹבְעוֹת מִשִּׁילּוּב מִסְפַּר הַפְּרָעוֹת בְּקֶרֶב בְּנֵי הַזּוּג. 
 
עַל-פִּי חוֹק בִּיטּוּחַ בְּרִיאוּת מַמְלַכְתִּי, טִיפּוּלֵי הַהַפְרָיָה הַחוּץ גּוּפִית יְמוּמְּנוּ עַל קוּפּוֹת הַחוֹלִים לִבְנֵי זוּג שֶׁלָּהֶם אֵין יְלָדִים בְּנִישּׂוּאֵיהֶם הַנּוֹכְחִיִּים (עַד שְׁנֵי יְלָדִים) וְכֵן, לְאִשָּׁה לְלֹא יְלָדִים הַמְּעוּנְיֶינֶת לְהָקִים מִשְׁפָּחָה שֶׁבְּרֹאשָׁהּ עוֹמֵד הוֹרֶה עַצְמָאִי ("מִשְׁפָּחָה חַד הוֹרִית") לְהוּלֶּדֶת יֶלֶד רִאשׁוֹן וְשֵׁנִי;
טִיפּוּלֵי IVF נִיתָּנִים לְנָשִׁים הֶחָל מִגִּיל 18 וְאִילּוּ הַגִּיל הַמֵּירַבִּי לְקַבָּלַת טִיפּוּלִים הוּא 45 (עַד לְיוֹם הַהוּלֶּדֶת 45).
(תְּרוּמַת בֵּיצִית נִיתֶּנֶת לְנָשִׁים בְּגִילָאֵי 18 עַד 54).
 
מַחֲזוֹרֵי הַטִּיפּוּל
הַנְחָיוֹת חֲדָשׁוֹת לְגַבֵּי מִסְפַּר מַחֲזוֹרֵי הַטִּיפּוּל מְפוֹרָטִים בְּחוֹזֵר מִינְהַל רְפוּאָה "מַחֲזוֹרֵי טִיפּוּלֵי הַפְרָיָה חוּץ גּוּפִית (IVF) בְּסַל הַשֵּׁירוּתִים" (6/2014 ) מִיּוֹם 20.1.2014.
עִיקְּרֵי הַהַנְחָיוֹת:

נָשִׁים שֶׁמָּלְאוּ לָהֶן 39, יוּכְלוּ לְבַצֵּעַ מִשִּׁיקּוּלִים רְפוּאִיִּים, כְּאֶפְשָׁרוּת רִאשׁוֹנָה, טִיפּוּלֵי IVF. 
נָשִׁים שֶׁמָּלְאוּ לָהֶן 42, לֹא יְבַצְּעוּ יוֹתֵר מִשְּׁלוֹשָׁה מַחֲזוֹרֵי IVF עוֹקְבִים, בָּהֶם לֹא הִגִּיעוּ בְּטִיפּוּל לְשָׁלָב שֶׁל הַחְזָרַת עוּבָּרִים. (בְּמִידָּה וְתִהְיֶה הַחְזָרַת עוּבָּרִים וַאֲפִילּוּ הַמַּחְזוֹר הַשְּׁלִישִׁי, הֲרֵי שֶׁהַסְּפִירָה מַתְחִילָה מֵחָדָשׁ). 
בְּכָל גִּיל, לְאַחַר 4 מַחֲזוֹרֵי טִיפּוּל עוֹקְבִים, בָּהֶם לֹא הִגִּיעוּ לִשְׁלַב הַחְזָרַת עוֹבְרִים, אוֹ לְאַחַר 8 מַחֲזוֹרֵי IVF, יֵיעָרֵךְ דִּיּוּן שֶׁל הַצֶּוֶות הַמְּטַפֵּל בְּאַחְרָיוּת הַיְּחִידָה בָּהּ בּוּצַּע מַחְזוֹר הַטִּיפּוּל הָאַחֲרוֹן, בְּהִשְׁתַּתְּפוּת עוֹבֶדֶת סוֹצְיָאלִית, בְּנוֹגֵעַ לַסִּיבּוֹת לְחוֹסֶר הַהַצְלָחָה וְהַצָּעוֹת לְהֶמְשֵׁךְ הַטִּיפּוּל. הַתָּכְנִית לְהֶמְשֵׁךְ טִיפּוּל לְרַבּוֹת טִיפּוּלֵי IVF נוֹסָפִים, תִּקָּבַע בְּהֶתְאֵם לְמַסְקְנוֹת הַדִּיּוּן.
שְׁאֵלוֹת וּתְשׁוּבוֹת לְגַבֵּי עִדְכּוּן הַהַנְחָיוֹת.
 
הַהֲכָנוֹת לִקְרַאת הַטִּיפּוּל 
בְּדִיקוֹת מַעְבָּדָה שׁוֹנוֹת לִשְׁלִילַת סִיבּוֹת שֶׁעֲלוּלוֹת לִפְגּוֹעַ בְּהֵרָיוֹן, בִּבְרִיאוּת הָאֵם אוֹ הָעוּבָּר. הַבְּדִיקוֹת כּוֹלְלוֹת נוֹגְדָנֵי אַדֶּמֶת, נוֹגְדָנֵי כֶּשֶׁל חִיסּוּנִי, אַנְטִיגֵן צַהֶבֶת נְגִיפִית מִסּוּג B וְ-C וּמִשְׁטָח מִצַּוָּואר הָרֶחֶם לִשְׁלִילַת הִתְפַּתְּחוּת סַרְטַן צַוָּואר הָרֶחֶם, אוֹ זִיהוּמִים, אִם קַיָּים חֲשָׁד.
שְׁלִילַת מַחֲלוֹת תּוֹרַשְׁתִּיּוֹת עַל יְדֵי הַעֲרָכַת הֶעָבָר הַמִּשְׁפַּחְתִּי.
הַדְגָּמַת רֶחֶם תַּקִּין לְפִי צִילּוּם רֶחֶם עַדְכָּנִי אוֹ הִיסְטְרוֹסְקוֹפִּיָּה.
בְּדִיקַת זֶרַע עַדְכָּנִית. 
נִלְקַחַת דְּגִימַת דָּם לְשֵׁם קְבִיעַת רָמוֹת הַהוֹרְמוֹנִים הַהַתְחָלָתִיּוֹת, עַל מְנַת לִשְׁלוֹל בְּעָיוֹת אֶפְשָׁרִיּוֹת נוֹסָפוֹת (כְּגוֹן זִיהוּמִים).
 
בִּיצּוּעַ הַטִּיפּוּל
גֵּירוּי הַשְּׁחָלוֹת - יֵשׁ מִסְפַּר דְּרָכִים אֶפְשָׁרִיּוֹת לִגְרוֹם לְגֵירוּי הַשְּׁחָלוֹת, וְעַל-יְדֵי כָּךְ לַיְּצוּר מִסְפָּר רַב שֶׁל בֵּיצִיּוֹת הַמַּתְאִימוֹת לַהַפְרָיָה.  
שְׁאִיבַת הַבֵּיצִיּוֹת – לְאַחַר גֵּירוּי הַשְּׁחָלוֹת, הָאִישָּׁה מַגִּיעָה לִשְׁאִיבַת הַבֵּיצִיּוֹת. הַשְּׁאִיבָה נַעֲשֵׂית בַּחֲדַר נִיתּוּחַ סָמוּךְ לְמַעְבֶּדֶת הַהַפְרָיָה הַחוּץ גּוּפִית. הִיא מִתְבַּצַּעַת תַּחַת הַרְדָּמָה מְקוֹמִית בְּצַוָּואר הָרֶחֶם אוֹ בְּהַרְדָּמָה כְּלָלִית קִצְרַת טְוָוח. הַפְּעוּלָּה כְּרוּכָה בְּאִי נוֹחוֹת עַד כְּאֵב, וּלְאַחֲרֶיהָ תִּידָּרֵשׁ מְנוּחָה שֶׁל מִסְפַּר שָׁעוֹת.
הַפְרָיַית הַבֵּיצִיּוֹת וּגְדִילַת הָעוֹבְרִים – לְאַחַר הַשְּׁאִיבָה, הַבֵּיצִיּוֹת מוֹעֳבָרוֹת לַמַּעְבָּדָה וּמוּנָּחוֹת בִּצְלוֹחִית הַמְּכִילָה מַצַּע הַדְגָּרָה. בְּמַקְבִּיל, מוֹסֵר הַגֶּבֶר דְּגִימַת זֶרַע, שֶׁעוֹבֶרֶת הֲכָנָה מְדוּקְדֶּקֶת וְרַק אָז מַפְגִּישִׁים אֶת תָּאֵי הַזֶּרַע וְהַבֵּיצִיּוֹת בְּתוֹךְ נוֹזֵל הַהַדְגָּרָה.
הַחְזָרַת הָעוּבָּר לָרֶחֶם - זֶהוּ הַשָּׁלָב הָאַחֲרוֹן. הַתַּהֲלִיךְ פָּשׁוּט, אֵינוֹ כָּרוּךְ בִּכְאֵבִים וְלֹא מַצְרִיךְ הַרְדָּמָה. בְּמַהֲלָכוֹ מוּחְדָּר צִינּוֹר אוֹ קַטֶטֵר דַּק מְאוֹד לָרֶחֶם דֶּרֶךְ צַוָּואר הָרֶחֶם, וְהָעוּבָּרִים מוּזְרָקִים לַחֲלַל הָרֶחֶם. לְאַחַר הַהַחְזָרָה, מוּמְלֶצֶת מְנוּחָה שֶׁל שְׁעָתַיִים בִּשְׁכִיבָה עַל הַגַּב. בְּתוֹם הַמְּנוּחָה, הָאִישָּׁה מִשְׁתַּחְרֶרֶת לְבֵיתָהּ וּבְמֶשֶׁךְ הַיְּמָמָה הַבָּאָה עָלֶיהָ לְהַרְבּוֹת בִּמְנוּחָה וּבִשְׁכִיבָה בַּמִּיטָּה כִּמְעַט לַחֲלוּטִין. כְּמוֹ כֵן, יֵשׁ לְהִימָּנַע מִיַּחֲסֵי מִין לִתְקוּפָה שֶׁל שְׁבוּעַיִים.

סִיבּוּכִים וְסִיכּוּנִים
בְּעִקְבוֹת גֵּירוּי הַשְּׁחָלוֹת יָכוֹל לְהִתְרַחֵשׁ גֵּירוּי יֶתֶר, שֶׁיּוֹפִיעַ הָחֵל מִיּוֹם-יוֹמַיִים לְאַחַר שְׁאִיבַת הַבֵּיצִיּוֹת וְעַד לִתְקוּפוֹת שֶׁל כְּ- 3-4 שָׁבוּעוֹת אַחֲרֵי שְׁאִיבַת הַבֵּיצִיּוֹת. יֵשׁ לָנוּחַ וּלְהַרְבּוֹת בִּשְׁתִיָּיה כְּטִיפּוּל. גֵּירוּי יֶתֶר, בֵּינוֹנִי אוֹ קָשֶׁה - הוּא נָדִיר יוֹתֵר, וְלִפְעָמִים מַצְרִיךְ אִשְׁפּוּז.
 
 
מַעֲרֶכֶת מִיקְרוֹמָנִיפּוּלַצְיָה
מִיקְרוֹמָנִיפּוּלַצְיָה הִינָּהּ פְּעוּלָּה שֶׁנִּיתָּן לְבַצֵּעַ בְּמַעְבָּדָה, תַּחַת הַגְדָּלָה שֶׁל מִיקְרוֹסְקוֹפּ, בְּתָאִים בּוֹדְדִים. נִיתָּן לְהָבִיא לַהַפְרָיָה, יְצִירַת עוּבָּר וְהַשָּׂגַת הֵרָיוֹן, עִם בֵּיצִית וְזֵרְעוֹן בּוֹדְדִים. הַשִּׁיטָה מְאַפְשֶׁרֶת הַחְדָּרַת זֶרַע יְשִׁירוֹת לְתוֹךְ הַבֵּיצִית גַּם בְּזוּגוֹת בָּהֶם בֶּן הַזּוּג סוֹבֵל מִלִּיקּוּיֵי זֶרַע חֲמוּרִים. 
הַפְּעוּלָּה נִקְרֵאת: ICSI -  Intra Insemination Sperm Injection . נָהוּג לְבַצֵּעַ מִיקְרוֹמָנִיפּוּלַצְיָה בְּכָל אוֹתָם מִקְרִים שֶׁעָבְרוּ הַפְרָיָה חוּץ-גּוּפִית, וְהוּשְּׂגוּ בָּהֶם אֲחוּזֵי הַפְרָיָה נְמוּכִים מִ-20%.
