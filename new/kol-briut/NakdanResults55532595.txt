בְּרִיאוּת הַסְּבִיבָה
בְּרִיאוּת הַסְּבִיבָה מִתְיַיחֶסֶת לְמַעֲרֶכֶת יַחֲסֵי גּוֹמְלִין שֶׁבֵּין הָאָדָם לַסְּבִיבָה, מִתּוֹךְ דְּאָגָה לִבְרִיאוּתוֹ וּלְאֵיכוּת חַיָּיו שֶׁל הָאָדָם, תּוֹךְ הִתְחַשְּׁבוּת בְּהֶיבֵּטִים תַּבְרוּאָתִיִּים וּסְבִיבָתִיִּים.
 
הַמַּחְלָקָה לִבְרִיאוּת הַסְּבִיבָה בְּשֵׁירוּתֵי בְּרִיאוּת הַצִּיבּוּר בְּמִשְׂרַד הַבְּרִיאוּת מַעֲרִיכָה אֶת הַגּוֹרְמִים הַתַּבְרוּאִיִּים, הַסְּבִיבָתִיִּים, הַפִיזִיִּים, הַכִּימִיִּים, הַבִּיּוֹלוֹגִיִּים, הַחֶבְרָתִיִּים וְהַסּוֹצְיוֹ-אֶקוֹנוֹמִיִּים, הָעֲלוּלִים לִגְרוֹם לִמְפַגֵּעַ פּוֹטֶנְצְיָאלִי וּלְנֶזֶק בְּרִיאוּתִי שֶׁל הַדּוֹר הַנּוֹכְחִי וְהַדּוֹרוֹת הַבָּאִים וּפוֹעֶלֶת לִמְנִיעָתָם וּלְתִיקּוּנָם. 

הַמַּחְלָקָה קוֹבַעַת אֶת הַמְּדִינִיּוּת שֶׁל הַמִּשְׂרָד בְּנוֹשְׂאֵי בְּרִיאוּת הַסְּבִיבָה, יוֹזֶמֶת תַּקָּנוֹת, שׁוּתָּפָהּ בַּהֲלִיכֵי תִּכְנוּן אֲזוֹרִיִּים וְעוֹסֶקֶת בְּפִיקּוּחַ תַּבְרוּאָתִי  שׁוֹטֵף וּבְבַקָּרָה שֶׁל גּוֹרְמִים שׁוֹנִים. 
 
 
הַנּוֹשְׂאִים שֶׁעֲלֵיהֶם אַחְרָאִית הַמַּחְלָקָה

מַתַּן חַוַּות דַּעַת מִקְדָּמִית עַל-פִּי חוֹק רִישּׁוּי עֲסָקִים 
פִּיקּוּחַ תַּבְרוּאִי בְּקַיְיטָנוֹת וּמַחֲנוֹת נוֹעַר  
מַתַּן חַוַּות דַּעַת לְתָכְנִיּוֹת מִתְאָר אַרְצִיּוֹת, מְחוֹזִיּוֹת, מְקוֹמִיּוֹת, מְפוֹרָטוֹת (בְּמִסְגֶּרֶת חֲבֵרוֹת בְּוַועֲדוֹת הַתִּכְנוּן)
טִיפּוּל בַּתְּלוּנוֹת וּפְנִיּוֹת בְּנוֹשְׂאִים שֶׁפּוֹרְטוּ לְעֵיל

מִשְׂרַד הַבְּרִיאוּת רוֹאָה חֲשִׁיבוּת רַבָּה בִּשְׁתִיַּית מַיִם וּמֵנִיעַ לָאַחֲרוֹנָה מֵיזָם הָעוֹסֵק בְּעִידוּד שְׁתִיַּית מֵי בֶּרֶז - תַּחַת הַסִּיסְמָה "אִם צְמֵאִים - רַק מַיִם אֲמִיתִּיִּים
בְּרֵיכַת שְׂחִייָּה הִינָּהּ מִבְנֶה הַמֵּכִיל מַיִם לְמַטְּרוֹת שְׂחִיָּיה אוֹ נוֹפֶשׁ, וְהִיא חֲשׂוּפָה לַזִּיהוּמִים שֶׁמְּקוֹרָם מִן הַמִּתְרַחֲצִים וּמֵהַסְּבִיבָה. 


עַל מְנַת  לִשְׁמוֹר עַל בְּרִיאוּת הַצִּיבּוּר תִּיקֵּן מִשְׂרַד הַבְּרִיאוּת תַּקָּנוֹת לִבְרֵיכוֹת שְׂחִייָּה צִיבּוּרִיּוֹת.  

תַּקָּנוֹת אֵלֶּה עוֹסְקוֹת בְּאֵיכוּתָם הַמִּיקְרוֹבְיָאלִית שֶׁל מֵי הָרַחֲצָה, בִּשְׁמִירָה עַל תְּנַאי תַּבְרוּאָה נְאוֹתִים בַּחֲצֵרִי הַבְּרֵיכָה וּבְמִבְנֵי הַשֵּׁירוּתִים וְהַמֶּלְתָּחוֹת, בְּהַצְלָלָה וְעוֹד. 

 

בַּעֲלֵי בְּרֵיכַת הַשְּׂחִייָּה אַחְרָאִים עַל עֲמִידָה בִּדְרִישׁוֹת הַתַּקָּנוֹת, וּבֵין הַיֶּתֶר: תְּנָאֵי הַתַּבְרוּאָה בְּחַצְרֵי הַבְּרֵיכָה, אֵיכוּת הַמַּיִם וְנִיטּוּרָם, הַצְלָלָה מֵעַל הַמַּיִם וּבַחֲצֵרִים וְהַעֲסָקַת מַפְעִיל בְּרֵיכָה שֶׁעָבַר הַכְשָׁרָה.


 

חִיטּוּי הַמַּיִם

בְּרֵיכַת שְׂחִייָּה מְחוּיֶּבֶת בְּמַעֲרֶכֶת טִיפּוּל בְּמַיִם הַכּוֹלֶלֶת מַעֲרֶכֶת סִינּוּן וְחִיטּוּי וּמַעֲרֶכֶת בַּקָּרָה מְמוּחְשֶׁבֶת לְצוֹרֶךְ בַּקָּרָה וְנִיטּוּר שֶׁל מְדַדֵּי אֵיכוּת פִיזִיקָלִיִּים וְכִימִיִּים שֶׁל הַמַּיִם (כְּגוֹן עֲכִירוּת וּשְׁאֵרִית חוֹמֶר חִיטּוּי).

הַתַּקָּנוֹת מְחַיְּיבוֹת חִיטּוּי רָצִיף שֶׁל מֵי הַבְּרֵיכָה בִּכְלוֹר, בִּבְרוֹם אוֹ בְּחוֹמֶר אַחֵר שֶׁאִישֵּׁר הַמְּנַהֵל. בְּמִי בְּרֵיכָה שֶׁנַּעֲשָׂה בָּהֶם חִיטּוּי בַּחוֹמֶר אוֹ בְּתַהֲלִיךְ שֶׁאֵינוֹ מְבוּסָּס עַל כְּלוֹר, מְחוּיֶּבֶת עָצְמַת הַחִיטּוּי לִהְיוֹת  שְׁוַות עֵרֶךְ לַעֲרָכִים הַמְּפוֹרָטִים בַּתַּקָּנוֹת.

כַּיּוֹם אֵין חוֹבָה לְהַתְקִין מַעֲרָכוֹת חִיטּוּי נוֹסָפוֹת (כְּדוּגְמַת חִיטּוּי בְּ-UV), מֵעֵבֶר לְאֵלֶּה הַנִּדְרָשׁוֹת בַּתַּקָּנוֹת.  הוֹסָפַת חַסָּמֵי טִיפּוּל נוֹסָפִים עֲשׂוּיָה לְסַיֵּיעַ בִּשְׁמִירָה עַל אֵיכוּת הַמַּיִם וְעַל בְּרִיאוּת הַצִּיבּוּר. מַעַרְכוֹת חִיטּוּי נוֹסָפוֹת אֵינָן מְהַוּוֹת תַּחֲלִיף לִדְרִישַׁת שְׁאֵרִית חוֹמֶר חִיטּוּי בְּמַיִם אוֹ דְּרִישׁוֹת אֵיכוּת אֲחֵרוֹת בְּהֶתְאֵם לַנִּדְרָשׁ בַּתַּקָּנוֹת, וְנִיתָּן לְשַׁלְּבָן עִם מַעַרְכוֹת הַחִיטּוּי בִּכְלוֹר. מַעַרְכוֹת חִיטּוּי נוֹסָפוֹת בַּבְּרֵיכָה נִדְרָשׁוֹת בְּאִישּׁוּר מִשְׂרַד הַבְּרִיאוּת.

 

בְּדִיקוֹת הַמַּיִם

בַּעֲלֵי הַבְּרֵיכָה נִדְרָשִׁים לְבַצֵּעַ בְּדִיקָה שֶׁל אֵיכוּתָם הַפִיזִית וְהַכִּימִית שֶׁל הַמַּיִם, בְּכָל שָׁלוֹשׁ שָׁעוֹת, וּבְנוֹסָף לְבַצֵּעַ בְּדִיקוֹת מִיקְרוֹבְּיָאלִיּוֹת בְּמַעְבָּדָה מוּכֶּרֶת עַל יְדֵי מִשְׂרַד הַבְּרִיאוּת, לְגִילּוּי וְזִיהוּי  שֶׁל חַיְידַּקִּים בְּמַיִם. תְּדִירוּתָן שֶׁל בְּדִיקוֹת אֵלֶּה הִינָּהּ אַחַת לְחוֹדֶשׁ בְּמַהֲלַךְ כָּל הַשָּׁנָה וּפַעֲמַיִים בְּחוֹדֶשׁ בְּיוּלִי - אוֹגוּסְט. תּוֹצָאוֹת חֲרִיגוֹת מֵהָעֲרָכִים הַקְּבוּעִים בַּתֶּקֶן, מְדַוְּוחוֹת לְמִשְׂרַד הַבְּרִיאוּת.

בַּעֲלֵי הַבְּרֵיכָה נִדְרָשִׁים לְהַצִּיג אֶת תּוֹצְאוֹת הַבְּדִיקוֹת הַיּוֹמִיּוֹת וְהַחוֹדְשִׁיּוֹת לִקְהַל הַמִּתְרַחֲצִים עַל גַּבֵּי לוּחַ מוֹדָעוֹת בַּבְּרֵיכָה.

 

הַתַּקָּנוֹת מְפָרְטוֹת דְּרִישׁוֹת נוֹסָפוֹת הַמִּתְיַיחֲסוֹת לְהַצְלָלָה, מִבְנֵי הַשֵּׁירוּתִים, הַמֶּלְתָּחוֹת וַחֲצַר הַבְּרֵיכָה.

 

בְּהֶתְאֵם לְצַו רִישּׁוּי עֲסָקִים, בְּרֵיכַת שְׂחִייָּה צִיבּוּרִית מְחוּיֶּבֶת בְּרִישְׁיוֹן עֵסֶק, הַמַּבְטִיחַ שֶׁהַבְּרֵיכָה נִמְצֵאת תַּחַת פִּיקּוּחָהּ שֶׁל הָרָשׁוּת הַמְּקוֹמִית וּמִשְׂרַד הַבְּרִיאוּת.

מוּמְלָץ לַצִּיבּוּר לְוַודֵּא אֶת קִיּוּמוֹ שֶׁל הָרִישָּׁיוֹן טֶרֶם הָרַחֲצָה.

הַבַּקָּשָׁה תִּשְׁלַח בְּצֵירוּף הַמִּסְמָכִים הַמְּפוֹרָטִים בְּתַקָּנוֹת רִישּׁוּי עֲסָקִים: תְּנָאֵי תַּבְרוּאָה נְאוֹתִים לִבְרֵיכוֹת שְׂחִיָּה, תשנ"ד 1994
הָרָשׁוּת הַמְּקוֹמִית תַּעֲבִיר אֶת הַבַּקָּשָׁה וְהַמִּסְמָכִים הַנִּלְוִוים לַמַּחְלָקָה לִבְרִיאוּת הַסְּבִיבָה בְּלִשְׁכַּת הַבְּרִיאוּת, לְפִי הָאֵזוֹר בּוֹ נִמְצֵאת הַבְּרֵיכָה 
לְאַחַר קַבָּלַת הַבַּקָּשָׁה מֵהָרָשׁוּת הַמְּקוֹמִית, יִקָּבַע מוֹעֵד לְבִיקּוֹרֶת שֶׁל הַמַּחְלָקָה לִבְרִיאוּת הַסְּבִיבָה בְּשֶׁטַח הַבְּרֵיכָה, לִבְדִיקַת הַבְּרֵיכָה וְהַמִּתְקָנִים, בְּהֶתְאֵם לִקְרִיטֶרְיוֹנִים הַנִּדְרָשִׁים בַּתַּקָּנוֹת
 לְפִי תּוֹצְאוֹת הַבְּדִיקָה יִינָּתֵן לָרָשׁוּת הַמְּקוֹמִית אֶחָד מִן הַבָּאִים:

אִישּׁוּר בִּתְנָאִים
סֵירוּב
מִקְוָואוֹת
​קְהַל הַטּוֹבְלוֹת וְהַטּוֹבְלִים  מִתְבַּקֵּשׁ לִנְהוֹג בְּהֶתְאֵם לְתַקָּנוֹת לִשְׁעַת חֵירוּם (נְגִיף קוֹרוֹנָה הֶחָדָשׁ) התש"ף  2020, וּלְהוֹרָאוֹת הַמְּנַהֵל הַמִּתְפַּרְסְמִים בָּאֲתָר הַמִּשְׂרָד.   לְשֵׁם שְׁמִירָה עַל בְּרִיאוּת הַצִּיבּוּר תִּיקֵּן מִשְׂרַד הַבְּרִיאוּת תַּקָּנוֹת לִתְנָאֵי תַּבְרוּאָה נְאוֹתִים לְמִקְוְואוֹת טַהֲרָה שֶׁיִּמְנְעוּ פְּגִיעָה אֶפְשָׁרִית בִּבְרִיאוּת הַצִּיבּוּר. 

בַּעֲלֵי הַמִּקְוָואוֹת נִדְרָשִׁים לְבַצֵּעַ בְּדִיקוֹת מַעְבָּדָה לְאֵיכוּת הַמַּיִם וְלִבְדִיקַת הִימָּצְאוּתָם שֶׁל חַיְידַּקִּים מַזִּיקִים. 
עַל בַּעֲלֵי הַמִּקְוָואוֹת לְדַוֵּוחַ לְמִשְׂרַד הַבְּרִיאוּת עַל תּוֹצְאוֹת הַבְּדִיקוֹת. 
שְׁאֵלוֹת וּתְשׁוּבוֹת
 	

​כֵּיצַד נִיתָּן לָדַעַת שֶׁהַמִּקְוֶוה בְּפִיקּוּחַ?
​ מִקְוֶוה שֶׁלּוֹ רִישָּׁיוֹן עֵסֶק נִמְצָא תַּחַת פִּיקּוּחַ הָרָשׁוּת הַמְּקוֹמִית וּמִשְׂרַד הַבְּרִיאוּת.​
סָגוּר מֵידָע נוֹסָף
​אִם לְמִקְוֵוה אֵין רִישְׁיוֹן עֵסֶק, הָאֵם מוּתָּר לִרְחוֹץ בּוֹ?
רַק בַּמִּקְוֶוה שֶׁלּוֹ יֵשׁ רִישְׁיוֹן עֵסֶק, נִיתָּן לָדַעַת שֶׁהַתְּנָאִים הַתַּבְרוּאָתִיִּים וְהָאֲחֵרִים תַּקִּינִים
אִילּוּ פָּרָמֶטְרִים יְכוֹלִים לְהָעִיד לִמְבַקֵּר/ת בַּמִּקְוֶוה עַל תְּקִינוּת הַמַּיִם?
​צְלִילוּת הַמַּיִם – הַמַּיִם צְרִיכִים לִהְיוֹת נְקִיִּים וּצְלוּלִים, כָּךְ שֶׁנִּיתָּן לִרְאוֹת בְּקַלּוּת אֶת קַרְקָעִית הַמִּקְוֶוה.
כְּלוֹר - הִימָּצְאוּת כְּלוֹר בְּמַיִם.
