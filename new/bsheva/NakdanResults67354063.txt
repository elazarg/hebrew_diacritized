הַמְּכִינָה הַקְּדַם צְבָאִית ''עֹצֶם'' שֶׁבַּיִּישּׁוּב נָוֶוה הִגִּישָׁה הַיּוֹם (חֲמִישִׁי) לְבֵית הַמִּשְׁפָּט הַשָּׁלוֹם בְּתֵל אָבִיב תְּבִיעַת דִּיבָּה בְּסַךְ מִילְיוֹן שְׁקָלִים נֶגֶד חֲדָשׁוֹת 13 וְהַכַּתָּב עָמְרִי מֵנִיב.

הַתְּבִיעָה מַגִּיעָה, כַּזָּכוּר, בְּעִקְבוֹת פִּרְסוּם מְגַמָּתִי וְשִׁקְרִי נֶגֶד הַמְּכִינָה שֶׁשּׁוּדַּר בַּמַּהֲדוּרָה הַמֶּרְכָּזִית שֶׁל 'חֲדָשׁוֹת 13' וְעָסְקָה בְּהִתְבַּטְּאוּיוֹתָיו לִכְאוֹרָה שֶׁל הָרַב אָסָף נָאוּמְבּוּרְג נֶגֶד צַהַ"ל.

בְּטֶרֶם נָקְטָה הֲלִיכִים מִשְׁפָּטִיִּים, דָּרְשָׁה הַמְּכִינָה פִּיצּוּי שֶׁל 300,000 שְׁקָלִים עַל עוֹגְמַת הַנֶּפֶשׁ. בִּכְתַב הַהַתְרָאָה נִטְעַן כִּי הַכַּתָּב מֵנִיב שִׁידֵּר כַּתָּבָה שִׁקְרִית וּמָנִיפּוּלָטִיבִית בְּמַטָּרָה לְהַכְפִּישׁ אֶת שֵׁם הַמְּכִינָה. מִשֶּׁלֹּא נֶעֶנְתָה הַמְּכִינָה, הֶחְלִיטָה לְהַגִּישׁ אֶת תְּבִיעַת הַדִּיבָּה בְּסַךְ מִילְיוֹן שְׁקָלִים.


בִּתְבִיעַת הַדִּיבָּה נִכְתַּב: "עִנְיָינָהּ שֶׁל תְּבִיעָה זוֹ בְּפִרְסוּם שִׁקְרִי וְאַף זְדוֹנִי כְּנֶגֶד הַתּוֹבְעִים בֶּעָרוּץ חֲדָשׁוֹת מֶרְכָּזִי אֲשֶׁר קִיבֵּל תְּהוּדָה יוֹצֵאת דּוֹפֶן בַּתִּקְשׁוֹרֶת הַיִּשְׂרְאֵלִית וּבַצִּיבּוּרִיּוּת הַיִּשְׂרְאֵלִית עַל כָּל אֲגַפֶּיהָ וְזֹאת בָּרֹאשׁ וּבָרִאשׁוֹנָה בְּשֶׁל עִנְיָנוֹ שֶׁל הֶעָרוּץ בְּעַצְמוֹ לְקַדֵּם אֶת הַחֲדָשׁוֹת 'הַמַּסְעִירוֹת' שֶׁחָשַׂף, וּבְשֶׁל עִנְיָינָם שֶׁל בַּעֲלֵי אִינְטֵרֵס אֲחֵרִים לִתְקוֹף אֶת הַתּוֹבְעִים כְּחֵלֶק מֵאָגֶ'נְדַּת שִׂנְאָה סְדוּרָה שֶׁהֵם מְתַחְזְקִים כְּלַפֵּי הַתּוֹבַעַת 1, כְּלַפֵּי רַבָּנֶיהָ וּכְלַפֵּי בּוֹגְרֶיהָ".

"אָכֵן בְּדִבְרֵי הַיָּמִים שֶׁל עוֹלָם דִּינֵי לָשׁוֹן הָרַע בִּמְדִינַת יִשְׂרָאֵל, מְעַטּוֹת הֵן הַפָּרָשׁוֹת בָּהֶן יָצְאָה לַאֲוִויר הָעוֹלָם דִּיבָּה כָּל כָּךְ חֲמוּרָה כְּפִי שֶׁיָּצְאָה בַּמִּקְרֶה דְּנַן כְּנֶגֶד הַתּוֹבַעַת 1 וּכְנֶגֶד הָרַב נֶאֶמְבּוּרְג אִישִׁית אֲשֶׁר כָּל חֶטְאוֹ בְּכָךְ שֶׁלִּימֵּד בֵּין חֲצֵרֶיהָ שֶׁל הַתּוֹבַעַת 1 וְנִקְלָע (שֶׁלֹּא בְּטוֹבָתוֹ) לְקַו הָאֵשׁ שֶׁל חוֹרְשֵׁי רָעָתָהּ שֶׁל הַתּוֹבַעַת", נִכְתַּב.

הֵם הוֹסִיפוּ: "זוֹ לֹא רַק הַקַּלּוּת הַבִּלְתִּי נִסְבֶּלֶת שֶׁל פִּרְסוּם הַדִּיבָּה, אוֹפֶן פִּרְסוּמָהּ וְהֶמְשֵׁךְ פִּרְסוּמָהּ גַּם לְאַחַר שֶׁהוּבַן שֶׁמְּדוּבָּר בַּלָּשׁוֹן הָרַע מִתְּחִילָּה וְעַד הַסּוֹף, אֶלָּא גַּם הָאוֹפֶן שֶׁבּוֹ הִשְׁתַּמְּשׁוּ בַּפִּרְסוּם הַשִּׁקְרִי עֶשְׂרוֹת וּמֵאוֹת מְעַצְּבֵי דֵּעָה וּפוֹלִיטִיקָאִים כְּדֵי לְהוֹסִיף לְשִׂיחַ הַשִּׂנְאָה וְהַמַּשְׂטֵמָה כְּנֶגֶד הַתּוֹבַעַת וּכְנֶגֶד הָרַב נֶאֶמְבּוּרְג אִישִׁית. זֹאת מִבְּלִי לָקַחַת בְּחֶשְׁבּוֹן כִּי מֵאֲחוֹרֵי הַפִּרְסוּמִים וְהַמִּילִּים יֵשׁ אֲנָשִׁים שֶׁנִּפְגָּעִים קָשׁוֹת מֵהַפִּרְסוּמִים הַזְּדוֹנִיִּים הַנָּ"ל".

מְכִינַת 'עָצֶם' (עַצְמוֹנָה), שֶׁהוּקְמָה עַל יְדֵי שַׂר יְרוּשָׁלַיִם הָרַב רָפִי פֶּרֶץ, הָעֲמִידָה לְאוֹרֶךְ הַשָּׁנִים אַלְפֵי בּוֹגְרִים שֶׁהִתְגַּיְּיסוּ לְשֵׁירוּת מַשְׁמָעוּתִי בְּצַהַ"ל. רַבִּים מִתַּלְמִידֵי הַמְּכִינָה הִמְשִׁיכוּ לְתַפְקִידִים פִּיקּוּדִיִּים.

מִמְּכִינַת עֹצֶם נִמְסַר לֶעָרוּץ 7: "מְכִינַת עֹצֶם פּוֹעֶלֶת כְּבָר קָרוֹב לִשְׁלוֹשִׁים שָׁנִים וְהוֹצִיאָה תַּחַת יָדָהּ אַלְפֵי בּוֹגְרִים שֶׁשֵּׁירְתוּ שִׁירוֹת מַשְׁמָעוּתִי בְּיוֹתֵר בִּצְבָא הַהֲגָנָה לְיִשְׂרָאֵל וּבְכָל מַעַרְכוֹת הַבִּיטָּחוֹן. רַבִּים מִתַּלְמִידֵי הַמְּכִינָה יָצְאוּ לְפִיקּוּד וְלִקְצוּנָּה וְחֶלְקָם תּוֹפְסִים הַיּוֹם עֲמָדוֹת מַשְׁמָעוּתִיּוֹת בְּשִׁדְרַת הַפִּיקּוּד שֶׁל צַהַ"ל. לְאַחַר שֵׁירוּתָם, הִמְשִׁיכוּ רַבִּים מֵהֶם לְלֵב הָעֲשִׂיָּיה הַיִּשְׂרְאֵלִית וּמַשְׁפִּיעִים בְּעֶמְדוֹת מַפְתֵּחַ בַּמִּגְזָר הַצִּיבּוּרִי וְהַפְּרָטִי, בַּהַנְהָגָה, בַּהִתְיַישְּׁבוּת בְּפִיתּוּחַ הָאָרֶץ וְשִׂיגְשׂוּגָהּ".


עוֹד נִמְסַר: "בְּשֵׁם אַלְפֵי תַּלְמִידֵנוּ וּבוֹגְרֵינוּ, בְּשֵׁם הַצֶּדֶק וּבְשֵׁם הָרָצוֹן לְהֵיטִיב לַחֶבְרָה הַיִּשְׂרְאֵלִית אָנוּ תּוֹבְעִים בַּחֲרִיפוּת אֶת כָּל מִי שֶׁבְּרֶגֶל גַּסָּה, בְּחִיתּוּכֵי קְטָעִים וּבְכַוָּונַת מְכַוֵּון נִיסָּה לְהַכְפִּישׁ בְּיוֹדְעִין אֶת שְׁמָם וּלְחַבֵּל בַּמִּפְעָל הָאַדִּיר הַזֶּה שֶׁהֶעֱמִיד וּמַמְשִׁיךְ לְהַעֲמִיד דּוֹרוֹת שֶׁל חֲלוּצִים בְּעֵת הַזּוֹ. בֵּית הַמִּדְרָשׁ שֶׁלָּנוּ יַמְשִׁיךְ לְהַצְמִיחַ עוֹד אַלְפֵי אֲנָשִׁים שֶׁמַּאֲמִינִים בִּמְדִינַת יִשְׂרָאֵל, בִּצְבָא הַהֲגָנָה לְיִשְׂרָאֵל וּבְעַם יִשְׂרָאֵל וּמְשַׁלְּבִים בְּהִצְטַיְּינוּת בֵּין סָפְרָא לְסַיָּיפָא. הִשְׁתַּלְּבוּת זוֹ בַּחֶבְרָה הַיִּשְׂרְאֵלִית הַיּוֹנֶקֶת מֵחִיבּוּר עָמוֹק לְבֵית הַמִּדְרָשׁ לֹא נִיתָּן לַעֲצִירָה זוֹ. מִי שֶׁמְּנַסֶּה לְפַרְסֵם מֵידָע שִׁקְרִי וְשָׁגוּי עָלָיו לָשֵׂאת בַּתּוֹצָאוֹת".

עו"ד מִיכָאֵל דְּבוֹרִין, הַמְּיַיצֵּג אֶת מְכִינַת עַצְמוֹנָה וְאֶת הָרַב אָסָף נֶאַמְבּוּרְג, מָסַר: "מְדוּבָּר בְּאֶחָד הַמִּקְרִים הַחֲמוּרִים בְּיוֹתֵר שֶׁל הוֹצָאַת לָשׁוֹן הָרַע הַזְּכוּרִים לִי בַּשָּׁנִים הָאַחֲרוֹנוֹת. לָבוֹא וְלָקַחַת אִישׁ מָתוּן כְּמוֹ הָרַב אָסָף נֶאֶמְבּוּרְג וּלְצַיֵּיר אוֹתוֹ כְּאִישׁ קִיצוֹנִי אַנְטִי צִיּוֹנִי שֶׁמַּטִּיף לְאִי גִּיּוּס לְצַהַ"ל זֶה הַרְבֵּה יוֹתֵר מֵרַשְׁלָנוּת וּדְרוּשָׁה כַּוָּונַת זָדוֹן שֶׁל מַמָּשׁ וּמַשְׂטֵמָה אֲמִיתִּית לְמִפְעַל הַמְּכִינוֹת כְּדֵי לַחֲשׁוֹב בִּכְלָל עַל אֶפְשָׁרוּת כְּזוֹ. תְּבִיעָה זוֹ בָּאָה לְהַבְהִיר כִּי שְׁמָם הַטּוֹב שֶׁל הַתּוֹבְעִים לֹא יִהְיֶה לְמִרְמָס".


כַּזָּכוּר, לִפְנֵי לְמַעְלָה מִשָּׁנָה פִּרְסֵם הָעִיתּוֹנַאי עָמְרִי מֵנִיב כָּתְבָה בַּ'חֲדָשׁוֹת 13' בְּמִסְגַּרְתָּהּ חָשַׂף קִטְעֵי הַקְלָטוֹת מִשִּׁיעוּרָיו שֶׁל הָרַב אָסָף נֶאֶמְבּוּרְג בִּמְכִינַת "עֹצֶם". בַּכַּתָּבָה הוּצַּג הָרַב נָאוּמְבּוּרְג כְּרַב קִיצוֹנִי שֶׁמַּטִּיף לְהִתְבַּדְּלוּת מוּחְלֶטֶת מֵחִילּוּנִים וְאַף לְהִימָּנְעוּת מִשֵּׁירוּת בַּצָּבָא.

מִסְפַּר יָמִים לְאַחַר מִכֵּן הָעִיתּוֹנַאי חָנָן עַמִּיאוֹר מְאַתֵּר "פֶּרְסְפֶּקְטִיבָה" חָשַׂף אֶת הַהַקְלָטוֹת הַמְּלֵאוֹת שֶׁל אוֹתָם שִׁיעוּרִים שֶׁל הָרַב אָסָף נָאוּמְבּוּרְג, בְּאֶמְצָעוּתָן הוֹכִיחַ כִּי הַקְלָטַת דִּבְרֵי הָרַב נֶעֶרְכָה בְּאוֹפֶן מְגַמָּתִי וְהַמִּשְׁפָּטִים שֶׁפּוּרְסְמוּ הוּצְאוּ מֵהֶקְשֵׁרָם.

מֵנִיב הִתְנַצֵּל מְאוּחָר יוֹתֵר כָּל הַכַּתָּבָה הַמַּכְפִּישָׁה. "אֲנִי מְבַקֵּשׁ לְהִתְנַצֵּל", כָּתַב בְּחֶשְׁבּוֹן הַטְוִויטֶר שֶׁלּוֹ. "'אֵין דָּבָר שֶׁחָשׁוּב לִי יוֹתֵר מִלְּדַיֵּיק. כְּשֶׁאֲנִי חָתוּם עַל אַיְיטֵם, זֶה יִהְיֶה בַּרְזֶל. חֲבָל לִי שֶׁזֶּה הוֹבִיל לְשִׂמְחָה לְאֵיד כָּאן מִפּוֹלִיטִיקָאִים שֶׁאֲנִי מְסַקֵּר. מִתְנַצֵּל".

מְאוּחָר יוֹתֵר מֵנִיב הוּצָא לְחוּפְשָׁה עַד לְסִיּוּם הַבֵּירוּר בְּנוֹגֵעַ לַכַּתָּבָה שֶׁלּוֹ. כַּיּוֹם הוּא מְשַׁמֵּשׁ מֵנִיב כִּכְתַב מִשְׁטָרָה וְתַחְקִירִים בַּחֲדָשׁוֹת 13.

מֵחֲדָשׁוֹת 13 נִמְסַר: ''נִלְמַד אֶת כְּתַב הַתְּבִיעָה וְנָגִיב, כַּמְּקוּבָּל, בְּבֵית מִשְׁפָּט'
