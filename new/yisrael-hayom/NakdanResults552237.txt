שָׁלוֹם לֹא - לְהִתְרָאוֹת כֵּן: הָאִינְטֵרֵסִים שֶׁל בֶּן סַלְמָאן

עוֹדֵד גְּרָנוֹת

דִּגְלֵי יִשְׂרָאֵל לֹא הוּנְפוּ אֶל רֹאשׁ הַתּוֹרֶן בְּנֵיאוֹם, עַיִר הֶעָתִיד הַסָּעוּדִית הַנִּמְצֵאת לְחוֹף הַיָּם הָאָדוֹם. גַּם הִמְנוֹנִים לֹא נוּגְּנוּ שָׁם. תְּמוּנוֹת מְשׁוּתָּפוֹת שֶׁל יוֹרֵשׁ הָעֶצֶר הַסָּעוּדִי וְרֹאשׁ הַמֶּמְשָׁלָה נְתַנְיָהוּ לֹא הוּפְצוּ לִכְלֵי הַתִּקְשׁוֹרֶת, וּבְהוֹדָעוֹת רִשְׁמִיּוֹת שֶׁיָּצְאוּ מֵרִיאָד דּוּבַּר רַק עַל פְּגִישָׁה דּוּ-צְדָדִית בֵּין הַמַּנְהִיג הַסָּעוּדִי לְבֵין שַׂר הַחוּץ הָאָמֵרִיקָנִי פּוֹמְפְּאוֹ.

גַּם שַׂר הַחוּץ הַסָּעוּדִי, שֶׁרַק לִפְנֵי יָמִים אֲחָדִים הִכְרִיז שֶׁלֹּא תִּהְיֶה נוֹרְמָלִיזַצְיָה עִם יִשְׂרָאֵל לִפְנֵי שֶׁזּוֹ תַּגִּיעַ לְהֶסְדֵּר עִם הַפָלַשְׂתִּינִים, מִיהֵר אֶמֶשׁ לְצַיֵּיץ בִּתְגוּבָה עַל פִּרְסוּם דְּבַר הַפְּגִישָׁה לָאָמוּר: לֹא הָיוּ דְּבָרִים מֵעוֹלָם.


הַמַּסְקָנָה הַיְּחִידָה שֶׁעוֹלָה מִן הַהַכְחָשׁוֹת בְּרִיאָד הִיא שֶׁהַמַּנְהִיג הַסָּעוּדִי הַצָּעִיר, שֶׁמֵּיזַם הָעֲנָק שֶׁל עִיר הֶעָתִיד הִיא חֵלֶק מֵחֲזוֹנוֹ לְהַצְעִיד אֶת אַרְצוֹ אֶל עִידָּן חָדָשׁ אַחֲרֵי שֶׁיַּאַזְלוּ בָּהּ מַאַגְרֵי הַנֵּפְט, עֲדַיִין חוֹשֵׁשׁ לָתֵת  פּוּמְבִּי לְיַחֲסֵי אַרְצוֹ עִם יִשְׂרָאֵל. אֵלֶּה, הַמִּתְנַהֲלִים זֶה שָׁנִים רַבּוֹת מִתַּחַת לַשּׁוּלְחָן, וְהִתְהַדְּקוּ בְּאוֹרַח מַשְׁמָעוּתִי בְּאַרְבַּע שְׁנוֹתָיו שֶׁל הַנָּשִׂיא טְרַאמְפּ בַּבַּיִת הַלָּבָן.

חֲשָׁשׁוֹ שֶׁל יוֹרֵשׁ הָעֶצֶר נוֹבֵעַ הֵן  מֵהִתְנַגְּדוּת אָבִיו, הַמֶּלֶךְ סַלְמָאן, לְכָל שִׁדְרוּג שֶׁל הַיְּחָסִים עִם יִשְׂרָאֵל שֶׁלֹּא עַל פִּי תְּנָאֵי הַיּוֹזְמָה הָעַרְבִית שֶׁהוֹבִילָה סָעוּדְיָה לִפְנֵי כִּמְעַט 20 שָׁנָה, וְהֵן מִן הַהַעֲרָכָה שֶׁהַכְרָזָה בְּטֶרֶם עֵת עַל נוֹרְמָלִיזַצְיָה עֲלוּלָה לִיצוֹר קֶרַע בְּתוֹךְ בֵּית הַמְּלוּכָה, דָּבָר שֶׁעָלוּל לְחַבֵּל בְּסִיכּוּיָיו לַהֲפוֹךְ לְמֶלֶךְ.

כָּל זֶה לֹא מָנַע מִמּוּחַמַּד בֶּן סַלְמָאן לִפְעוֹל בְּמֶרֶץ בָּחוֹדָשִׁים הָאַחֲרוֹנִים, כְּדֵי לְעוֹדֵד שׁוּרָה שֶׁל מְדִינוֹת עַרְבִיּוֹת לְכוֹנֵן יְחָסִים דִּיפְּלוֹמָטִיִּים עִם יִשְׂרָאֵל - וּבְרֹאשָׁן אִיחוּד הָאֵמִירוּיוֹת וּבַחְרֵיין שֶׁבַּמִּפְרָץ הַפַּרְסִי וְסוּדָאן  שֶׁלְּחוֹף הַיָּם הָאָדוֹם. לְסוּדָאן, אַגַּב, הִמְרִיאָה אֶתְמוֹל מִשְׁלַחַת יִשְׂרְאֵלִית כְּדֵי לְקַדֵּם אֶת הַחֲתִימָה עַל הֶסְכֵּם שָׁלוֹם וְשִׁיתּוּף פְּעוּלָּה בֵּין שְׁתֵּי הַמְּדִינוֹת (רָאוּ יְדִיעָה בְּעַמּוּד זֶה).

בֶּן סַלְמָאן פָּתַח אֶת שְׁמֵי אַרְצוֹ בִּפְנֵי טִיסוֹת יִשְׂרְאֵלִיּוֹת וְאוֹתֵת, יוֹתֵר מִפַּעַם אַחַת, שֶׁלֹּא יְשַׁעְבֵּד אֶת הָאִינְטֵרֵסִים שֶׁל סָעוּדְיָה לְטוֹבַת פִּתְרוֹן  הַסִּכְסוּךְ הַיִּשְׂרְאֵלִי-פָלַשְׂתִּינִי.

כָּל אֵלֶּה לֹא עָצְרוּ בְּעַד יוֹרֵשׁ הָעֶצֶר הַסָּעוּדִי אֶת קִיּוּם הַפְּגִישָׁה עִם נְתַנְיָהוּ וּפוֹמְפֵּאּוֹ לְחוֹף הַיָּם הָאָדוֹם, תּוֹךְ נְטִילַת הַסִּיכּוּן שֶׁדְּבַר הַבִּיקּוּר יִדְלוֹף  לַתִּקְשׁוֹרֶת. הַסִּיבָּה: בֶּן סַלְמָאן מוּדְאָג מְאוֹד מִן הֶעָתִיד, וּבְעִיקָּר מֵהָאֶפְשָׁרוּת שֶׁחִילּוּפֵי הַשִּׁלְטוֹן בַּבַּיִת הַלָּבָן יִפְגְּעוּ בָּאִינְטֵרֵסִים הַחִיּוּנִיִּים שֶׁל אַרְצוֹ. הוּא מוּדְאָג מִן הָאֶפְשָׁרוּת שֶׁמִּמְשַׁל בַּיְידֶּן יַחֲזוֹר לְהֶסְכֵּם הַגַּרְעִין עִם אִירָאן בִּנְּסִיבּוֹת שֶׁרַק יַגְבִּירוּ אֶת הַתֵּיאָבוֹן בְּטֵהֵרָן לְהַמְשִׁיךְ לַחְתּוֹר תַּחַת בֵּית הַמְּלוּכָה הַסָּעוּדִי, וּלְסַיֵּיעַ לַמּוֹרְדִים בְּתֵימָן לִפְגּוֹעַ בְּמִתְקְנֵי הַנֵּפְט שֶׁל הַמַּמְלָכָה.

בֶּן סַלְמָאן מוּדְאָג גַּם מֵהָאֶפְשָׁרוּת שֶׁהַמִּמְשָׁל הָאָמֵרִיקָנִי הֶחָדָשׁ יִהְיֶה הַרְבֵּה פָּחוֹת סוֹבְלָנִי מִזֶּה שֶׁל טְרַאמְפּ לַהֲפָרוֹת שֶׁל זְכוּיוֹת אָדָם בְּסָעוּדְיָה.

זֶה הָרֶקַע לַפְּגִישָׁה שֶׁלִּכְאוֹרָה לֹא הִתְקַיְּימָה בָּעִיר הֶעָתִיד שֶׁלְּחוֹף הַיָּם הָאָדוֹם. יוֹרֵשׁ הָעֶצֶר הַסָּעוּדִי מְעוּנְיָין לְיַישֵּׁר קַו עִם יִשְׂרָאֵל בִּנְקִיטַת עֶמְדָּה אֲחִידָה מוּל הַמִּמְשָׁל הָאָמֶרִיקָנִי הֶחָדָשׁ בְּסוּגְיַית הַגַּרְעִין הָאִירָאנִי, וְהוּא מִסְתַּמֵּךְ עַל קִשְׁרֵי הֶעָבָר הַחַמִּים בֵּין יִשְׂרָאֵל לְבֵין אַנְשֵׁי בַּיְידֶּן, שֶׁיּוֹעִילוּ גַּם לִפְתִיחַת דַּף חָדָשׁ בֵּין רִיאָד לְווֹשִׁינְגְּטוֹן.

כָּל זֹאת, בְּלִי שֶׁיִּידָּרֵשׁ, בֵּינְתַיִים, לַחְתּוֹם עַל חוֹזֵה שָׁלוֹם אִיתָּנוּ וְלִפְתּוֹחַ שַׁגְרִירוּת בְּתֵל אָבִיב אוֹ בִּירוּשָׁלַיִם.

"הַדְּאָגָה לָאִינְטֵרֵסִים הַסָּעוּדִיִּים הֵבִיאָה אֶת יוֹרֵשׁ הָעֶצֶר לִפְעוֹל כְּפִי שֶׁלֹּא פָּעַל בֶּעָבָר, וּלְהִסְתַּכֵּן בִּדְלִיפַת הַמִּפְגָּשׁ עִם נְתַנְיָהוּ
