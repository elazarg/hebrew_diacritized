לֹא רַק לִבְנֵי אָדָם: רְפוּאָה אַלְטֶרְנָטִיבִית לבע"ח
הָרְפוּאָה הַמַּשְׁלִימָה וְהַטִּיפּוּלִים הָאַלְטֶרְנָטִיבִיִּים הֵחֵלּוּ לִתְפּוֹס מָקוֹם חָשׁוּב בְּשִׁיפּוּר אֵיכוּת הַחַיִּים שֶׁל בַּעֲלֵי הַחַיִּים שֶׁלָּנוּ • מוּמְחִית: "אֵין רְפוּאָה אַחַת שֶׁטּוֹבָה לְכוּלָּם"

דִּין שְׁמוּאֵל אַלְמָס

טִיפּוּל בִּרְפוּאָה מַשְׁלִימָה וְטִיפּוּלִים אַלְטֶרְנָטִיבִיִּים אֲחֵרִים מְאוֹד נְפוֹצִים לָנוּ בְּנֵי הָאָדָם, כְּבָר לֹא מְעַט שָׁנִים. בַּשָּׁנִים הָאַחֲרוֹנוֹת חֵלֶק מֵהַטִּיפּוּלִים הָאֵלֶּה, הַמְּהַוִּוים אֶמְצָעִים לְהַשְׁלָמַת הַטִּיפּוּל הַוֵּוטֵרִינָרִי הַקְּלָאסִי, תּוֹפְסִים מָקוֹם חָשׁוּב גַּם בְּטִיפּוּל וְשִׁיפּוּר אֵיכוּת הַחַיִּים שֶׁל הַכְּלָבִים וְהַחֲתוּלִים שֶׁלָּנוּ. אוֹרְלִי זַכַּאי, מוּמְחִית בִּרְפוּאָה סִינִית לְבַעֲלֵי חַיִּים מִטַּעַם חֶבְרַת בִּיוֹפֶּט מַצִּיגָה אֶת כָּל הָאֶפְשָׁרוּיוֹת בַּתְּחוּם.
תְּזוּנָה טִבְעִית
תְּזוּנָה טִבְעִית עַל בְּסִיס בָּשָׂר, עוֹף וְדָגִים, מוּתְאֶמֶת לִכְלָבִים וַחֲתוּלִים, הָפְכָה לִמְאוֹד נְפוֹצָה בַּשָּׁנִים הָאַחֲרוֹנוֹת. כְּמוֹ מִשְׁלוֹחַ שֶׁל אוֹכֶל שֶׁאֲנַחְנוּ מַזְמִינִים לְעַצְמֵנוּ, כָּךְ תּוּכְלוּ לְהַזְמִין גַּם אֶת הַמָּזוֹן הַזֶּה יְשִׁירוֹת אֲלֵיכֶם הַבַּיְתָה מִמִּסְפַּר חֲבֵרוֹת שֶׁקַּיָּימוֹת הַיּוֹם בָּאָרֶץ. תּוּכְלוּ לְהוֹסִיף אֶת הַמָּזוֹן לְתַפְרִיט הַיּוֹמִי, כְּדֵי לְהַזְכִּיר לַכֶּלֶב וְלֶחָתוּל שֶׁלָּכֶם אֶת הַתְּזוּנָה הַמְּקוֹרִית שֶׁלָּהֶם בְּמִידָּה מְסוּיֶּמֶת.

 
הַמָּזוֹן מַגִּיעַ בְּמָנוֹת קְפוּאוֹת שֶׁל בָּשָׂר יַחַד עִם יְרָקוֹת וּמַרְכִּיבִים חֲשׁוּבִים אֲחֵרִים שֶׁרוּבָּם מַשְׁלִימִים תַּפְרִיט מְאוּזָּן. כָּל מָה שֶׁצָּרִיךְ זֶה לְהוֹצִיא מֵהַפְרִיזֵר יוֹם קוֹדֶם, לְהַפְשִׁיר לְגַמְרֵי לְטֶמְפֵּרָטוּרַת הַחֶדֶר וְלֶאֱכוֹל. אֶפְשָׁרוּת נוֹסֶפֶת הִיא מַמָּשׁ לְהָכִין אֶת הָאוֹכֶל בְּעַצְמָם עֲבוּר הַכֶּלֶב אוֹ הֶחָתוּל- לְבַשֵּׁל בַּבַּיִת אֶת הַבָּשָׂר הָאָהוּב עַל הַכֶּלֶב וּלְהוֹסִיף לוֹ אֶת הַמַּרְכִּיבִים הַנִּדְרָשִׁים, כְּדֵי לְיַיצֵּר תַּפְרִיט מְאוּזָּן. בְּכָל מִקְרֶה, חָשׁוּב בְּיוֹתֵר לִקְרוֹא וּלְהָבִין טוֹב מָה הַמַּרְכִּיבִים הַנִּדְרָשִׁים כְּדֵי לְיַיצֵּר תַּפְרִיט בָּרִיא וּלְהַחֲלִיט מָה הַדֶּרֶךְ הַנּוֹחָה לָכֶם בְּתִפְעוּל שֶׁל הַתּוֹסֶפֶת הַזּוֹ לְיוֹם יוֹם כְּדֵי שֶׁתּוּכְלוּ לְהַתְמִיד וְתִשְׁמְרוּ עַל הַבְּרִיאוּת שֶׁלָּהֶם.


דָּגֵשׁ עַל בְּרִיאוּת מַעֲרֶכֶת הָעִיכּוּל
פְּרוֹבְיוֹטִיקָה הִיא מוּצָר מוּכָּר מְאוֹד וְאֶחָד הַנִּצְרָכִים בְּיוֹתֵר לִבְנֵי אָדָם, לְחִיזּוּק הַפְּעִילוּת הַתַּקִּינָה שֶׁל מַעֲרֶכֶת הָעִיכּוּל שֶׁלָּנוּ. עַכְשָׁיו יֵשׁ גַּם לֹא מְעַט מוּצָרִים כָּאֵלֶּה לַכְּלָבִים וַחֲתוּלִים. כְּבָר שָׁנִים יוֹדְעִים שֶׁמַּעֲרֶכֶת עִיכּוּל בְּרִיאָה וַחֲזָקָה מַשְׁפִּיעָה יְשִׁירוֹת עַל מַעֲרֶכֶת חִיסּוּן חֲזָקָה וּמְשַׁפֶּרֶת אֶת הַבְּרִיאוּת שֶׁל הַכֶּלֶב וְהֶחָתוּל. אֶפְשָׁר לִמְצוֹא מוּצְרֵי פְּרוֹבְיוֹטִיקָה בִּמְיוּחָד לַכְּלָבִים וַחֲתוּלִים וְאֶפְשָׁר לִמְצוֹא הַיּוֹם גַּם מְזוֹנוֹת שֶׁהוֹסִיפוּ לָהֶם פְּרוֹבִּיּוֹטִיקָה מֵרֹאשׁ, כָּךְ שֶׁהַכֶּלֶב מְקַבֵּל בְּכָל אֲרוּחָה עוֹד קְצָת בְּרִיאוּת לַצַּלַּחַת
צִמְחֵי מַרְפֵּא
צִמְחֵי מַרְפֵּא וְתוֹסָפִים טִבְעִיִּים, כְּמוֹ לְמָשָׁל כּוּרְכּוּם, יוֹתֵר וְיוֹתֵר נְפוֹצִים הַיּוֹם גַּם לַכְּלָבִים וְלֹא רַק לַאֲנָשִׁים. חָשׁוּב לְהִתְיַיעֵץ עִם אִישׁ מִקְצוֹעַ לִפְנֵי שֶׁנּוֹתְנִים צִמְחֵי מַרְפֵּא, בֶּטַח אִם הַכֶּלֶב אוֹ הֶחָתוּל סוֹבְלִים מִבְּעָיָה בְּרִיאוּתִית וְנוֹטְלִים תְּרוּפוֹת. אֶפְשָׁר לָתֵת צִמְחֵי מַרְפֵּא לְכֶלֶב בָּרִיא, כְּדֵי לִשְׁמוֹר עַל הַבְּרִיאוּת שֶׁלּוֹ וְלִמְנוֹעַ מַחֲלוֹת וְאֶפְשָׁר כַּמּוּבָן לְטַפֵּל בִּבְעָיוֹת בְּרִיאוּת קַיָּימוֹת.
דִּיקּוּר סִינִי
שִׁיטַת הַטִּיפּוּל הַזּוֹ נִמְצֵאת כָּאן כְּבָר שָׁנִים רַבּוֹת לִבְנֵי אָדָם, אֲבָל גַּם כְּלָבִים וַחֲתוּלִים כְּבָר מְטוּפָּלִים בְּדִיקּוּר סִינִי. אֶפְשָׁר לְהַגִּיד שֶׁדִּיקּוּר סִינִי הָפַךְ לְטִיפּוּל לְגַמְרֵי מְקוּבָּל בָּעוֹלָם וְגַם בָּאָרֶץ. 
אֵקוֹפֶּרְסוּרָה לֹא מַצְרִיכָה מְחָטִים לְטִיפּוּל וּמְאַפְשֶׁרֶת לָכֶם בְּצוּרָה קַלָּה לַעֲשׂוֹת מִין טִיפּוּל בֵּייתִי כְּזֶה, רַק קַבְּלוּ הַנְחָיוֹת מֵהַמְּטַפֵּל שֶׁלָּכֶם בִּרְפוּאָה סִינִית וְתַעֲשׂוּ הַשְׁלָמָה שֶׁל הַטִּיפּוּל בַּיּוֹם יוֹם, כְּדֵי לְחַזֵּק אֶת הַטִּיפּוּל. כְּמוֹ כֵן, מַסָּאז', רֵיְיקִי, טִיפּוּלִים טִבְעִיִּים אֲחֵרִים, נַעֲשׂוּ יוֹתֵר פּוֹפּוּלָרִיִּים וּמְאַפְשְׁרִים עוֹד דְּרָכִים לְטַפֵּל בַּכֶּלֶב.

"אֵין רְפוּאָה אַחַת שֶׁטּוֹבָה לְכוּלָּם", מַסְבִּירָה אוֹרְלִי זַכַּאי, "חָשׁוּב בְּכָל מִקְרֶה כְּצַעַד רִאשׁוֹן לְהַגִּיעַ לְוֵוטֵרִינָר, לִבְדּוֹק אֶת הַמַּצָּב הַבְּרִיאוּתִי שֶׁל הַכֶּלֶב וְהֶחָתוּל, כְּדֵי לְקַבֵּל תְּמוּנָה מְלֵאָה וְאָז תּוּכְלוּ לְהַחֲלִיט מָה הַטִּיפּוּל שֶׁאַתֶּם מַעֲדִיפִים לַעֲשׂוֹת. תִּתְיַיעֲצוּ עִם מְטַפֵּל בִּרְפוּאָה סִינִית לְגַבֵּי הָאֶפְשָׁרוּיוֹת, כֵּיוֹוֹן שֶׁיֵּשׁ מַצָּבִים שֶׁנִּיתָּן וְרָצוּי לְטַפֵּל בְּשִׁילּוּב עִם הָרְפוּאָה הַקּוֹנְבֶנְצְיוֹנָלִית וְיֵשׁ מַצָּבִים שֶׁנִּיתָּן לְטַפֵּל בְּדִיקּוּר סִינִי בִּלְבַד כְּדֵי לִפְתּוֹר אֶת הַבְּעָיָה".
