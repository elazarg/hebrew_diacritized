נָעִים לְהַכִּיר

"גִּיל וָיֵין, 26, נָשׂוּי, מִתְגּוֹרֵר בְּתֵל אָבִיב, מוּזִיקַאי".

נִיצוֹץ שֶׁל נוֹסְטַלְגִּיָּה

"אַחַד הַדְּבָרִים שֶׁאֲנִי זוֹכֵר בָּאוֹפֶן הֲכִי מַשְׁמָעוּתִי מִגִּיל צָעִיר הוּא הַחִיבּוּר שֶׁלִּי לְמוּזִיקָה. אֲנִי מַרְגִּישׁ שֶׁכָּל הַחַיִּים הָיָה לִי חִיבּוּר לְמוּזִיקָה. כְּבָר בְּגִיל שְׁמוֹנֶה אָמַרְתִּי לַהוֹרִים וְלַחֲבֵרִים לִי שֶׁאֲנִי רוֹצָה לִהְיוֹת מוּזִיקַאי כְּשֶׁאֶהְיֶה גָּדוֹל. אוּלַי זֶה נִשְׁמָע הָזוּי, אֲבָל כְּשֶׁהָיִיתִי בֶּן עֶשֶׂר הִקְלַטְתִּי דִּיסְק בַּבַּיִת, עִם שִׁירִים שֶׁכָּתַבְתִּי, הִלְחַנְתִּי וְהֵפַקְתִּי. הָיְיתָה אָז תּוֹכְנַת מַחְשֵׁב מְאוֹד פְּשׁוּטָה שֶׁאִפְשְׁרָה אֶת זֶה, וּבְעֶזְרָתָהּ הֵפַקְתִּי אֶת הַדִּיסְק. בַּשָּׁנִים הָאַחֲרוֹנוֹת חִיפַּשְׂתִּי אוֹתוֹ בְּטֵירוּף, וְלֹא מָצָאתִי. אֲבָל אוּלַי בְּעֶצֶם טוֹב שֶׁכָּךְ, לֵךְ תֵּדַע כַּמָּה זֶה יִהְיֶה מֵבִיךְ".

נִיצוֹץ שֶׁל שׁוֹרָשִׁים

"גָּדַלְתִּי בְּבַיִת חִילּוֹנִי, כָּזֶה שֶׁחוֹגְגִים בּוֹ אֶת רֹאשׁ הַשָּׁנָה וּפֶסַח, אֲבָל לֹא יוֹתֵר מִזֶּה. כְּשֶׁגָּדַלְתִּי, הִתְעַנְיַינְתִּי בַּמָּסוֹרֶת, הִתְחַלְתִּי לַחֲקוֹר קְצָת יוֹתֵר לָעוֹמֶק אֶת הָעִנְיָין, בֵּירַרְתִּי לָמָּה עוֹשִׂים דְּבָרִים מְסוּיָּמִים, וְאָז הֶחְלַטְתִּי לְקַיֵּים מִצְווֹת שׁוֹנוֹת, לַעֲשׂוֹת מֵעֵבֶר לְמָה שֶׁעוֹשִׂים אֶצְלֵנוּ בַּבַּיִת. מֵאָז לְמָשָׁל אֲנִי צָם בְּכָל כִּיפּוּר, מַקְפִּיד בְּפֶסַח עַל אִיסּוּר אֲכִילַת חָמֵץ, וְעוֹשֶׂה כָּל שַׁבָּת קִידּוּשׁ אֵצֶל הַהוֹרִים שֶׁל אִשְׁתִּי. הָרָצוֹן שֶׁלִּי לַעֲשׂוֹת מֵעֵבֶר לְמָה שֶׁהִכַּרְתִּי בַּבַּיִת אוּלַי גַּם נוֹבֵעַ מִזֶּה שֶׁהָיִיתִי בְּגַן דָּתִי, וַאֲנִי רוֹאֶה שֶׁזֶּה מַשְׁפִּיעַ עָלַי עַד הַיּוֹם. לְדַעְתִּי זֶה גַּם מַשְׁפִּיעַ עַל הַמּוּזִיקָה שֶׁאֲנִי עוֹשֶׂה.

"הַשִּׁירִים שֶׁלָּמַדְתִּי בַּגַּן מַשְׁפִּיעִים עָלַי עַד הַיּוֹם בְּכַמָּה אֳופָנִים. זֶה גַּם מוֹפִיעַ דֶּרֶךְ הַכְּתִיבָה, בְּכָךְ שֶׁאֲנִי מַרְגִּישׁ שֶׁיֵּשׁ מַשֶּׁהוּ רוּחָנִי בַּכְּתִיבָה שֶׁלִּי. אִי אֶפְשָׁר בֶּאֱמֶת לְהַסְבִּיר אֶת זֶה בַּמִּילִּים, אֲבָל אֶפְשָׁר  לְהַרְגִּישׁ. מֵעֵבֶר לְכָךְ, אֲנִי אֲפִילּוּ מַרְגִּישׁ אֶת זֶה יוֹתֵר חָזָק בַּלְּחָנִים. יֵשׁ לִי שִׁירִים, חֶלְקָם יָצְאוּ וְחֶלְקָם עֲדַיִין לֹא, שֶׁאֲנִי מַמָּשׁ שׁוֹמֵעַ אֶת הַלַּחַן שֶׁלָּהֶם וּמְדַמְיֵין תְּפִילָּה, שַׁבָּת, בֵּית כְּנֶסֶת.

"בְּנוֹסָף, כְּשֶׁיִּהְיוּ לִי יְלָדִים – בָּרוּר לִי שֶׁאֶרְצֶה שֶׁהֵם יַכִּירוּ אֶת הַדְּבָרִים הַבְּסִיסִיִּים בַּיַּהֲדוּת וְיִשְׁמְרוּ עֲלֵיהֶם. בְּסוֹפוֹ שֶׁל דָּבָר מְדוּבָּר בִּדְבָרִים הֲכִי יָפִים שֶׁיֵּשׁ לָנוּ בְּתוֹר עַם. מְרַגֵּשׁ אוֹתִי לִרְאוֹת אֵיךְ בְּכִיפּוּר לְמָשָׁל רוֹב הָאֲנָשִׁים לֹא נוֹסְעִים, וְזֶה מַדְהִים בְּעֵינַיי. וַאֲנִי מְדַבֵּר עַל מֶרְכַּז תֵּל אָבִיב, בָּאֵזוֹר שֶׁל הַבַּיִת שֶׁלִּי, לֹא עַל עִיר חֲרֵדִית. וּבִכְלָל, יֵשׁ הַרְבֵּה יוֹתֵר מָסוֹרֶת בְּתֵל אָבִיב מִמָּה שֶׁאֲנָשִׁים חוֹשְׁבִים. לְמָשָׁל, בְּשַׁבָּתוֹת רוֹב הַחֲנוּיוֹת בָּרְחוֹבוֹת הֲכִי מֶרְכָּזִיִּים סְגוּרוֹת. שָׁנָה אַחַת הָיִיתִי בְּרֹאשׁ הַשָּׁנָה בַּדִּירָה שֶׁלִּי בְּמֶרְכַּז תֵּל אָבִיב, וְהָיִיתִי בְּשׁוֹק מִזֶּה שֶׁכִּמְעַט הַכֹּל סָגוּר. בֵּינֵינוּ, מָה יִישָּׁאֵר לָנוּ בַּחַיִּים אִם לֹא עֶרְכֵי הַיַּהֲדוּת?".

נִיצוֹץ שֶׁל יַהֲדוּת

"אֲנִי מַרְגִּישׁ שֶׁיֵּשׁ בִּי מַשֶּׁהוּ שֶׁמּוּשְׁפָּע מֵהַמּוּזִיקָה הַיְּהוּדִית הָעַתִּיקָה, מֵהַפִּיּוּטִים. כְּמוֹ שֶׁאָמַרְתִּי, כְּיֶלֶד הָיִיתִי בְּגַן דָּתִי, פָּשׁוּט כִּי הוּא הָיָה מְאוֹד קָרוֹב לַבַּיִת שֶׁלָּנוּ. כְּשֶׁאַתָּה מִתְחַנֵּךְ בְּגַן דָּתִי, וְכַמּוּבָן מַכִּיר וְשַׂר אֶת הַשִּׁירִים וְהַתְּפִילּוֹת שֶׁשָּׁרִים בַּגַּן, מַשֶּׁהוּ בָּזֶה דָּבֵק בְּךָ, וַאֲנִי מַרְגִּישׁ וְחוֹשֵׁב שֶׁזֶּה מַשְׁפִּיעַ עַד הַיּוֹם עַל הַמּוּזִיקָה שֶׁלִּי".

נִיצוֹץ שֶׁל עֲשִׂיָּיה

"אֲנִי מַרְגִּישׁ בְּאוֹפֶן מְסוּיָּם שֶׁאֱלוֹהִים קָשׁוּר לַיְּצִירָה שֶׁלִּי. הַהַרְגָּשָׁה הִיא שֶׁאֲנִי מְקַבֵּל אֶת הַשִּׁירִים שֶׁלִּי יוֹתֵר מֵאֲשֶׁר יוֹצֵר אוֹתָם בְּעַצְמִי. יֵשׁ שִׁירִים שֶׁמְּאוֹד עָזְרוּ לִי בַּחַיִּים, וַאֲנִי רוֹצָה לָתֵת אֶת הַמַּתָּנָה הַזּוֹ, בִּזְכוּת מָה שֶׁקִּיבַּלְתִּי, גַּם לַאֲחֵרִים. בָּעִנְיָין הַזֶּה אֲנִי יָכוֹל לוֹמַר שֶׁהַדָּבָר שֶׁהֲכִי מְרַגֵּשׁ אוֹתִי זֶה כְּשֶׁאֲנָשִׁים אוֹמְרִים לִי: 'הַשִּׁיר שֶׁלָּךְ רִיפֵּא אוֹתִי'. וּבִכְלָל הַקֶּשֶׁר שֶׁלִּי לַה' מִתְבַּטֵּא גַּם עַל יְדֵי כָּךְ שֶׁוִּויתַּרְתִּי עַל הַרְבֵּה דְּבָרִים יִקְרֵי עֵרֶךְ בִּשְׁבִיל שֶׁיִּקְרוּ דְּבָרִים טוֹבִים בָּעוֹלָם".

נִיצוֹץ שֶׁל שַׁבָּת

"שַׁבָּת מִבְּחִינָתִי הִיא הַמָּקוֹם שֶׁבּוֹ בֶּאֱמֶת יֵשׁ לִי זְמַן לְעַצְמִי. בְּכָל הַשָּׁבוּעַ אֲנִי מַקְדִּישׁ זְמַן לְגִיל הַמּוּזִיקַאי, וּבְשַׁבָּת לְגִיל הָאָדָם. אֲנִי גַּם לֹא עוֹבֵד בָּאוּלְפָּן בְּשַׁבָּת, וְגַם לֹא מוֹפִיעַ בְּשַׁבָּת. לִפְנֵי כַּמָּה שָׁנִים הָיָה בִּי מַשֶּׁהוּ פְּנִימִי שֶׁדָּרַשׁ מִמֶּנִּי לְהַפְסִיק לַעֲבוֹד בְּשַׁבָּת, וְזֶה מָה שֶׁאָכֵן קָרָה. הַיּוֹם אֲנִי יָכוֹל לוֹמַר שֶׁזֶּה מָה שֶׁמַּשְׁאִיר אוֹתִי שָׁפוּי. בְּלִי הַשַּׁבָּת הָיִיתִי חַי אֶת הָעֲבוֹדָה 24/7".

נִיצוֹץ שֶׁל תְּפִילָּה

"אֲנִי מְדַבֵּר עִם אֱלוֹקִים מִדֵּי יוֹם. מֵעֵבֶר לְזֶה, אֶחָד מֵהַשִּׁירִים בָּאַלְבּוֹם הָאַחֲרוֹן שֶׁלִּי נִקְרָא 'סְצֵנוֹת', וּמִבְּחִינָתִי כּוּלּוֹ תְּפִילָּה. הַדֶּרֶךְ שֶׁל אָמָּן רְצוּפָה קְשָׁיִים וּמִכְשׁוֹלִים, אָז לֹא פַּעַם אֲנִי מִתְפַּלֵּל לֶאֱלוֹהִים וּפָשׁוּט מְבַקֵּשׁ אֶת עֶזְרָתוֹ, מְבַקֵּשׁ מִמֶּנּוּ שֶׁיִּרְאֶה לִי מָה אֲנִי צָרִיךְ לַעֲשׂוֹת, שֶׁיָּאִיר לִי אֶת הַדֶּרֶךְ.

"אֶצְלִי הַתְּפִילָּה יְכוֹלָה לִהְיוֹת עַל הַדְּבָרִים הֲכִי קְטַנִּים בַּחַיִּים, וְזֶה קוֹרֶה מַמָּשׁ בְּאוֹפֶן יוֹמְיוֹמִי, וְלִפְעָמִים הִיא גַּם עַל הַדְּבָרִים הֲכִי חֲשׁוּבִים וּגְדוֹלִים בַּחַיִּים. מֵעֵבֶר לְכָךְ שֶׁעֶצֶם הַתְּפִילָּה עֲלֵיהֶם בָּטוּחַ עוֹזֶרֶת, הַתְּפִילָּה גַּם עוֹזֶרֶת לִי לְזַהוֹת מָה בֶּאֱמֶת חָשׁוּב לִי. אַתָּה רוֹאֶה כַּמָּה אַתָּה רוֹצֶה דְּבָרִים לְפִי אֵיךְ וְכַמָּה שֶׁאַתָּה מִתְפַּלֵּל עֲלֵיהֶם. הַתְּפִילָּה מַבְהִירָה לִי דְּבָרִים, מְמַקֶּדֶת אוֹתִי, וְשָׂמָה דְּבָרִים בָּאִיזּוּן הַנָּכוֹן שֶׁלָּהֶם".

נִיצוֹץ שֶׁל לִימּוּד

"לִפְנֵי הַחֲתוּנָּה, בַּפְּגִישׁוֹת שֶׁעָשִׂינוּ עִם הָרַב שֶׁחִיתֵּן אוֹתָנוּ, מְאוֹד נֶהֱנֵיתִי לְהַכִּיר אֶת הַפָּן הָרוּחָנִי הֶעָמוֹק שֶׁל הַיַּהֲדוּת, שֶׁאֲנִי מוֹדֶה שֶׁלֹּא כָּל כָּךְ הִכַּרְתִּי עַד אָז בְּאוֹתָהּ רָמָה שֶׁל עוֹמֶק. זֶה מְאוֹד עִנְיֵין אוֹתִי, כִּי הַרְבֵּה שָׁנִים אֲנִי מְאוֹד מְחוּבָּר לֶאֱמוּנָה וּלְיַהֲדוּת, לוֹמֵד הֲמוֹן דְּבָרִים שֶׁקְּשׁוּרִים לְרוּחָנִיּוּת. אֲנִי בְּהֶחְלֵט רוֹצֶה לְהַמְשִׁיךְ עִם זֶה וְלִלְמוֹד עוֹד וְעוֹד עַל הַיַּהֲדוּת".

נִיצוֹץ שֶׁל מִצְווֹה

"אֲנִי מְאוֹד מְחוּבָּר לְמִצְוַות כִּיבּוּד הוֹרִים בִּפְרָט, וּלְ'וְאַהֲבַת לְרֵעֲךָ כָּמוֹךְ' בִּכְלָל. בְּקֶשֶׁר לַהוֹרִים שֶׁלִּי אֲנִי יָכוֹל לוֹמַר שֶׁאֲנִי מְאוֹד מִשְׁתַּדֵּל לְכַבֵּד אוֹתָם, וְכַיּוֹם, כְּשֶׁאֲנִי יוֹתֵר רָחוֹק מֵהֶם פִיזִית - אֲנִי גָּר בְּתֵל אָבִיב וְהֵם בְּכַרְמִיאֵל - אֲנִי אֲפִילּוּ מַרְגִּישׁ יוֹתֵר אֶת הַצּוֹרֶךְ לְכַבֵּד אוֹתָם, לְהָרִים טֵלֵפוֹן, לְהִתְעַנְיֵין בִּשְׁלוֹמָם וְלַעֲזוֹר לָהֶם בְּמָה שֶׁאֲנִי יָכוֹל.

"בְּקֶשֶׁר לַכָּבוֹד הַכְּלָלִי לְכָל אָדָם, אֲנִי יָכוֹל לוֹמַר שֶׁאֲנִי מִשְׁתַּדֵּל מְאוֹד שֶׁלֹּא לִפְגּוֹעַ בַּאֲנָשִׁים, וְשֶׁאֲנִי מִשְׁתַּדֵּל תָּמִיד לָשִׂים אֶת עַצְמִי בַּצַּד הַשֵּׁנִי, בְּמָקוֹם שֶׁל מִי שֶׁאֲנִי מְדַבֵּר אִיתּוֹ, כּוֹלֵל לְדַמְיֵין אֵיךְ הוּא יְקַבֵּל אֶת הַדְּבָרִים. לָכֵן, לְמָשָׁל, אֲנִי מִשְׁתַּדֵּל לְדַבֵּר לַאֲנָשִׁים בָּאוֹפֶן שֶׁבּוֹ הָיִיתִי רוֹצֶה שֶׁאֲנָשִׁים יְדַבְּרוּ אֵלִי. וְלַעֲשׂוֹת אוֹ לֹא לַעֲשׂוֹת לַאֲנָשִׁים דְּבָרִים שֶׁהָיִיתִי אוֹ לֹא הָיִיתִי רוֹצֶה שֶׁיַּעֲשׂוּ לִי. זֶה עֵרֶךְ בְּסִיסִי בְּעֵינַי, וַאֲנִי מְאוֹד מִשְׁתַּדֵּל לְיַישֵּׂם אוֹתוֹ בַּחַיִּים".

נִיצוֹץ שֶׁל סִיּוּם

"מָה שֶׁאֶרְצֶה לְהַשְׁאִיר אַחֲרֵי בְּעוֹלָם זוֹ שְׁאֵלָה מְאוֹד גְּדוֹלָה וְקָשֶׁה. אֲנִי בֶּן אָדָם שֶׁחוֹשֵׁב לְמֶרְחַקִּים שֶׁל חָמֵשׁ שָׁנִים קָדִימָה, גַּג עֶשֶׂר, אֲבָל אַף פַּעַם לֹא חָשַׁבְתִּי מָה יִישָּׁאֵר מִמֶּנִּי אַחֲרֵי שֶׁאֵלֵךְ. אֲנִי מְקַוּוֹה שֶׁהַשִּׁירִים שֶׁלִּי יַמְשִׁיכוּ לְהַשְׁפִּיעַ לְטוֹבָה עַל אֲנָשִׁים, וַאֲנִי רוֹצֶה שֶׁתִּהְיֶה לִי תְּרוּמָה מַשְׁמָעוּתִית לַתְּחוּם שֶׁאֲנִי עוֹסֵק בּוֹ, הַמּוּזִיקָה. וְעִם זֹאת, כַּמּוּבָן, אֲנִי רוֹצֶה לְהַשְׁאִיר יְלָדִים טוֹבִים שֶׁיַּמְשִׁיכוּ אֶת  הַדֶּרֶךְ הַנְּכוֹנָה בָּעוֹלָם הַזֶּה".
