"הָרַב הַמַּטִּיף נֶגֶד הַחִיסּוּן הִגִּיעַ לְהִתְחַסֵּן בַּחֲשַׁאי"
יוֹעֵץ רְפוּאִי בָּכִיר בַּחֶבְרָה הַחֲרֵדִית, הַנֶּחֱשֶׁבֶת כַּר פּוֹרֶה שֶׁל חַדְשׁוֹת כָּזָב וְתֵיאוֹרְיוֹת קוֹנְסְפִּירַצְיָה בְּנוֹשֵׂא הַחִיסּוּנִים, מְגַלֶּה בָּאוּלְפָּן ynet כִּי קִיבֵּל לָאַחֲרוֹנָה פְּנִיּוֹת מִכַּמָּה פְּעִילִים - כּוֹלֵל אֶחָד הָרַבָּנִים הַבּוֹלְטִים הַמַּטִּיפִים נֶגֶד הַחִיסּוּן - בְּבַקָּשָׁה כִּי יְחַסֵּן אוֹתָם בַּחֲשַׁאי: "מְדוּבָּר כְּבָר בַּאֲלָפִים"
קוֹבִּי נַחְשׁוֹנֵי
פּוּרְסַם: 18.02.21 , 15:17


מִתְנַגְּדֵי חִיסּוּנִי הַקּוֹרוֹנָה הִצְלִיחוּ לִסְחוֹף אַחֲרֵיהֶם הֲמוֹנֵי יִשְׂרְאֵלִים, וְאִילְּצוּ אֶת מִשְׂרַד הַבְּרִיאוּת לָצֵאת בְּקַמְפֵּיְין הַסְבָּרָה נֶגֶד הַ"פֵיְיק נְיוּז", אֲבָל נִרְאֶה כִּי חֶלְקָם לֹא מַאֲמִינִים אֲפִילּוּ לְעַצְמָם. יוֹעֵץ רְפוּאִי בָּכִיר בַּחֶבְרָה הַחֲרֵדִית, הַנֶּחֱשֶׁבֶת כַּר פּוֹרֶה שֶׁל חַדְשׁוֹת כָּזָב וְתֵיאוֹרְיוֹת קוֹנְסְפִּירַצְיָה בְּנוֹשֵׂא זֶה, מְגַלֶּה כִּי קִיבֵּל לָאַחֲרוֹנָה פְּנִיּוֹת מִכַּמָּה פְּעִילִים כָּאֵלֶּה - כּוֹלֵל אֶחָד הָרַבָּנִים הַבּוֹלְטִים הַמַּטִּיפִים נֶגֶד הַחִיסּוּן - בְּבַקָּשָׁה כִּי יַחֲסָן אוֹתָם בַּחֲשַׁאי.

הָרַב יוֹסֵי עֶרְבְּלִיך, יוֹ"ר אִרְגּוּן "לְמַעַנְכֶם", יָצָא בַּשָּׁבוּעַ שֶׁעָבַר בְּמִבְצָע לְהַגְדָּלַת מִסְפַּר הַמִּתְחַסְּנִים הַחֲרֵדִים, וּבְשִׁיתּוּף עִם הַמֶּרְכָּזִים הָרְפוּאִיִּים שִׁיבָּא וְאִיכִילוֹב פֶּתַח מוֹקְדֵי חִיסּוּן בְּרִיכּוּזֵי הַמִּגְזָר וְאַף בְּתוֹךְ הַיְּשִׁיבוֹת. כָּעֵת הוּא מְסַפֵּר, כָּאָמוּר, כִּי בֵּין הַמִּשְׁתַּכְנְעִים "לָתֵת כָּתֵף" הָיוּ לֹא רַק אֶזְרָחִים מִן הַשּׁוּרָה - אֶלָּא אַף כָּאֵלֶּה הַפּוֹעֲלִים יוֹם-יוֹם בְּנִיסָּיוֹן לְחַבֵּל בַּמַּאֲמָץ הַלְּאוּמִּי לְמִיגּוּר הַמַּגֵּפָה.

"הָיְיתָה אֵלַיי פְּנִיָּיה מֵאַחַד הָאֲנָשִׁים שֶׁקּוֹרְאִים לְעַצְמָם רַבָּנִים", אָמַר עֶרְבְּלִיך בְּרֵאָיוֹן לָאוּלְפָּן ynet. "רָאִיתִי כָּל מִינֵי סִרְטוֹנִים שֶׁבָּהֶם הוּא מַסְבִּיר לַצִּיבּוּר כַּמָּה גְּדוֹלָה הַסַּכָּנָה שֶׁכְּרוּכָה בַּחִיסּוּן הַזֶּה. וְאָז יוֹם אֶחָד אֲנִי מְקַבֵּל מִמֶּנּוּ טֵלֵפוֹן וְהוּא מַתְחִיל לְהִתְיַיעֵץ אִיתִּי 'בְּשׁוּשׁוּ' וְלִשְׁאוֹל אוֹתִי מָה דַּעְתִּי עַל הַחִיסּוּן, מָה אֲנִי יוֹדֵעַ לְסַפֵּר וְלוֹמַר לוֹ עַל כָּךְ.
"אֲנִי כְּבָר הֵבַנְתִּי לְאָן הַשִּׂיחָה הַזּוֹ הוֹלֶכֶת, וְאָכֵן כְּשֶׁסִּייַּמְנוּ, אַחֲרֵי 10 דַּקּוֹת אוֹ רֶבַע שָׁעָה, הוּא שׁוֹאֵל אוֹתִי: אֵיךְ אַתָּה יָכוֹל לִדְאוֹג לִי לְחִיסּוּן שֶׁיַּעֲבוֹר בְּשֶׁקֶט? כְּשֶׁרָאִיתִי שֶׁאָדָם כָּזֶה, שֶׁמְּסַפֵּר וְאוֹמֵר כַּמָּה חֲשָׁשׁוֹת יֵשׁ בַּחִיסּוּן הַזֶּה, מְבַקֵּשׁ לְקַבֵּל אוֹתוֹ - זֶה נָתַן תַּמְרִיץ לָצֵאת עִם הַפַּיְילוֹט הַזֶּה, וּלְשִׂמְחָתִי הַהֵיעָנוּת הָיְיתָה בִּלְתִּי רְגִילָה".
"מְבִיאִים אֶת הַחִיסּוּנִים אֲלֵיהֶם"
עֶרְבְּלִיך מְסַפֵּר עַל אַלְפֵי "מִתְחַסְּנִים בְּשׁוּשׁוּ", וּמוֹסִיף כִּי מֵעֵבֶר לְאֵלֶּה שֶׁהוּא מְטַפֵּל בָּהֶם, יֶשְׁנָם אִרְגּוּנֵי בְּרִיאוּת אֲחֵרִים שֶׁאִמְּצוּ אֶת הַמּוֹדֵל. "לְצַעֲרִי הָרַב, כְּבָר תְּקוּפָה אֲרוּכָּה שֶׁהַפֵיְיק נְיוּז מְחַלְחֵל לַצִּיבּוּר הַחֲרֵדִי, דֶּרֶךְ מַטִּיפִים נֶגֶד הַחִיסּוּנִים. לָכֵן הִתְחַלְנוּ, בְּסִיּוּעַ בָּתֵּי הַחוֹלִים אִיכִילוֹב וְשִׁיבָּא, בַּשָּׁבוּעַ שֶׁעָבַר, בְּמִבְצַע הַנְגׇּשַׁת הַחִיסּוּנִים עַד אֲלֵיהֶם הַבַּיְתָה, כִּי אֲנִי לֹא רוֹאֶה דֶּרֶךְ אַחֶרֶת לְהִילָּחֵם בְּפֵיְיק נְיוּז. אַךְ וְרַק לְהַגִּיעַ אֲלֵיהֶם וּלְחַסֵּן אוֹתָם. 
הָרַב יוֹסֵף עֶרְבְּלִיך, יוֹר לְמַעַנְכֶן
 
"אֲנַחְנוּ נִמְצָאִים בִּתְקוּפָה שֶׁל מִלְחָמָה", עֶרְבְּלִיך מוֹסִיף, "אָז אֲנַחְנוּ צְרִיכִים לַעֲשׂוֹת אֶת הַכֹּל עַל מְנַת לְחַסֵּן כַּמָּה שֶׁיּוֹתֵר. בָּרוּךְ ה' שֶׁבִּבְנֵי בְּרָק אֲנַחְנוּ בְּ65% צִיבּוּר מַחֲלִימִים וּמִתְחַסְּנִים, וַאֲנִי מַאֲמִין שֶׁנַּגִּיעַ בַּיּוֹמַיִים הַקְּרוֹבִים לְ70%... יֵשׁ פֶּלֶג קִיצוֹנִי מְאוֹד בִּירוּשָׁלַיִם שֶׁהִצְלַחְנוּ לְחַסֵּן שָׁם 400 אַבְרֵכִים וְזֶה מְחַלְחֵל. אִי אֶפְשָׁר לְהִתְעַלֵּם מִקְּרִיאָתָם הַבְּרוּרָה שֶׁל הָרַבָּנִים לָלֶכֶת לְהִתְחַסֵּן, וְגַם סִיפּוּר הַיּוֹלְדוֹת הַצְּעִירוֹת, שֶׁאֲנִי מְלַוֶּוה אִישִׁית, שֶׁלֹּא הִתְחַסְּנוּ וְהִגִּיעוּ לְבָתֵּי הַחוֹלִים בְּמַצָּב קְרִיטִי כְּשֶׁהֵן מוּחְבָּרוֹת בְּחֶלְקָן לְמַכְשִׁיר אֶקְּמוֹ, חִלְחֵל לַצִּיבּוּר".

הָרַב עֶרְבְּלִיך מְסַפֵּר כִּי הַחֲשָׁשׁ מִפְּנֵי פְּגִיעָה בְּפוֹרִיּוּת הָיְיתָה הַכַּר הַנּוֹחַ בְּיוֹתֵר לִצְמִיחַת הַקּוֹנְסְפִּירַצְיוֹת בַּמִּגְזָר. "הָיִינוּ אֵצֶל כָּל מִינֵי מְנַהֲלֵי מַחְלָקוֹת בְּגֵנִיקוֹלוֹגְיָה וּמְיַילְּדוֹת, וְיָצָא מִכְתָּב בָּרוּר שֶׁאֵין חָשַׁשׁ לָלֶכֶת לְהִתְחַסֵּן". לְדִבְרֵי עֶרְבְּלִיך, "זֶה עָזַר וְהֶעֱלָה אֶת הַמּוֹטִיבַצְיָה בְּקֶרֶב נָשִׁים לְהִתְחַסֵּן... אֲבָל בַּתְּקוּפָה הָאַחֲרוֹנָה – בַּשְּׁבוּעַיִים-שְׁלוֹשָׁה הָאַחֲרוֹנִים - רָאִינוּ יְרִידָה מְסוּיֶּמֶת אֵצֶל הַצִּיבּוּר הַחֲרֵדִי, וִידִידִי הַפְּרוֹפ' בַּרְבָּשׁ הִצִּיעַ לָקַחַת אֶת הַיּוֹזְמָהּ וּלְהִיכָּנֵס לְמוֹסְדוֹת הַחִינּוּךְ לְחַסֵּן שָׁם".
