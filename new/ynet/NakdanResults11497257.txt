טְרַאמְפּ, נַזְכִּיר, יָצָא וְהִכְרִיז עַל נִיצְחוֹנוֹ – לְלֹא שׁוּם בָּסִיס – כְּבָר בְּלֵיל הַבְּחִירוֹת, כְּשֶׁהַסְּפִירָה הָיְיתָה בְּעִיצּוּמָהּ וְהִסְתַּמֵּן לוֹ יִתְרוֹן גָּדוֹל בְּכַמָּה מְדִינוֹת. מְדוּבָּר בְּיִתְרוֹן שֶׁלַּכּוֹל הָיָה בָּרוּר שֶׁיִּצְטַמֵּק וּבְחֵלֶק מֵהַמִּקְרִים יִימָּחֵק, מִשּׁוּם שֶׁבִּמְדִינוֹת רַבּוֹת נִסְפָּרִים תְּחִילָּה הַקּוֹלוֹת מֵהַקַּלְפִּיּוֹת שֶׁל יוֹם הַבְּחִירוֹת, שֶׁרוֹב הַמַּצְבִּיעִים בָּהֶן הָיוּ רֶפּוּבְּלִיקָנִים, וְרַק אַחַר כָּךְ הַקּוֹלוֹת שֶׁנִּשְׁלְחוּ בַּדּוֹאַר בַּהַצְבָּעָה הַמּוּקְדֶּמֶת, שִׁיטָה שֶׁרוֹב הַמַּעֲדִיפִים אוֹתָהּ הַשָּׁנָה הָיוּ דֵּמוֹקְרָטִים

מֵאָז מְקַדֵּם טְרַאמְפּ אֶת הַטַּעֲנָה שֶׁהַבְּחִירוֹת מוּשְׁחָתוֹת, וְשֶׁהַדֵּמוֹקְרָטִים מְזַיְּיפִים קוֹלוֹת וְגוֹנְבִים מִמֶּנּוּ אֶת הַנִּיצָּחוֹן. לַטְּעָנוֹת שֶׁל טְרַאמְפּ אֵין שׁוּם הוֹכָחוֹת, יְתֵרָה מִכָּךְ: עִיתּוֹנָאִים וּפַרְשָׁנִים הִזְהִירוּ בְּמֶשֶׁךְ חֳודָשִׁים כִּי הוּא יְנַצֵּל אֶת הַפַּעַר הָרִאשׁוֹנִי וְהַצָּפוּי כְּדֵי לִזְרוֹעַ סְפֵקוֹת בְּנוֹגֵעַ לְטוֹהַר הַבְּחִירוֹת. מֵאָז תְּחִילַּת הַסְּפִירָה הִסְפִּיק קַמְפֵּיְין טְרַאמְפּ לְהַגִּישׁ כַּמָּה עֲתִירוֹת לְבָתֵּי מִשְׁפָּט נֶגֶד נִיהוּל הַסְּפִירָה בִּמְדִינוֹת שׁוֹנוֹת, וְהוּא כְּבָר בִּיקֵּשׁ סְפִירָה חוֹזֶרֶת בְּוִויסְקוֹנְסִין, שָׁם הִפְסִיד בְּפַעַר שֶׁל שֶׁל 0.6%. בִּמְדִינָה זוֹ רַשַּׁאי כָּל אֶחָד מֵהַמּוּעֲמָדִים לְבַקֵּשׁ סְפִירָה חוֹזֶרֶת אִם הִפְסִיד בְּפַעַר שֶׁל פָּחוֹת מֵאָחוּז. סְפִירָה חוֹזֶרֶת צְפוּיָה גַּם בְּג'וֹרְגִ'יָּה, שָׁם צְפוּיִים לְהַפְרִיד בֵּין בַּיְידֶּן לִטְרַאמְפּ לְכָל הַיּוֹתֵר אֲלָפִים בּוֹדְדִים שֶׁל קוֹלוֹת.
אֶמֶשׁ הִבְהִיר טְרַאמְפּ בַּהוֹדָעָה דֶּרֶךְ הַקַּמְפֵּיְין שֶׁלּוֹ שֶׁהוּא מִתְכַּוֵּון לְהַמְשִׁיךְ לְהִילָּחֵם: "אֲנַחְנוּ מַאֲמִינִים שֶׁהָעָם הָאָמֵרִיקָנִי רָאוּי לִשְׁקִיפוּת מְלֵאָה בְּנוֹגֵעַ לְכָל סְפִירַת הַקּוֹלוֹת, וְשֶׁזֶּה כְּבָר לֹא נוֹגֵעַ לְשׁוּם מַעֲרֶכֶת בְּחִירוֹת סְפֵּצִיפִית. הָעִנְיָין כָּאן הוּא טוֹהַר תַּהֲלִיךְ הַבְּחִירוֹת שֶׁלָּנוּ כּוּלּוֹ. מֵהַהַתְחָלָה אָמַרְנוּ שֶׁכָּל הַקּוֹלוֹת הַחוּקִּיִּים חַיָּיבִים לְהִיסָּפֵר וְכָל הַקּוֹלוֹת הַלֹּא חוּקִּיִּים חַיָּיבִים שֶׁלֹּא לְהִיסָּפֵר, וּבְכָל זֹאת נִתְקַלְנוּ בַּהִתְנַגְּדוּת לָעִיקָּרוֹן הַבְּסִיסִי הַזֶּה מִצַּד דֵּמוֹקְרָטִים בְּכָל פִּינָּה. נַמְשִׁיךְ בַּתַּהֲלִיךְ הַזֶּה בְּכָל הַדְּרָכִים שֶׁהַחוֹק מְאַפְשֵׁר לָנוּ כְּדֵי לְהַבְטִיחַ שֶׁלָּעַם הָאָמֵרִיקָנִי יִהְיֶה אֵמוּן בַּמִּמְשָׁל שֶׁלָּנוּ. לְעוֹלָם לֹא אֲוַותֵּר עַל הַמִּלְחָמָה עֲבוּרְכֶם וַעֲבוּר אַרְצֵנוּ".

תְּגוּבַת הָאֲנָשִׁים לַנִּיצָּחוֹן שֶׁל ג'וֹ בַּיְידֶּן
"אַתָּה מְפוּטָּר, דּוֹנִי
קוֹדֶם לָכֶן דִּיוְּוחוּ בְּ-CNN שֶׁבִּסְבִיבַת טְרַאמְפּ מְסַפְּרִים כִּי הוּא מְרוּגָּז מְאוֹד מֵהַהִתְרַחֲשׁוּיוֹת וְחָשׁ שֶׁאֵין דֵּי אֲנָשִׁים הַמִּתְיַיצְּבִים לִימִינוֹ. לְדִבְרֵי הָרֶשֶׁת, בְּקֶרֶב בַּעֲלֵי בְּרִיתוֹ שֶׁל טְרַאמְפּ גּוֹבֶרֶת הַתְּחוּשָׁה שֶׁמִּישֶׁהוּ יִצְטָרֵךְ לְהַסְבִּיר לַנָּשִׂיא שֶׁיִּיתָּכֵן שֶׁזְּמַנּוֹ בַּתַּפְקִיד מַגִּיעַ לְסִיּוּמוֹ. לְפִי הַדִּיוּוּחַ, הֵם טֶרֶם הֶחֱלִיטוּ מִי צָרִיךְ לַעֲשׂוֹת זֹאת, אַךְ הוֹעֳלָה הָרַעֲיוֹן שֶׁבִּתּוֹ אִיוַונְקָה וּבַעֲלָהּ גַּ'ארד קוּשְׁנֶר יְקַבְּלוּ עַל עַצְמָם אֶת הַמְּשִׂימָה. רֶשֶׁת "פוֹקְס נְיוּז" דִּיוְּוחָה אֶתְמוֹל כִּי לִטְרַאמְפּ אֵין לְפִי שָׁעָה כַּוּוֹנָה לְהַכִּיר בְּהֶפְסֵד, וְכִי הוּא מְקַבֵּל מִיּוֹעֲצָיו עֵצוֹת סוֹתְרוֹת בַּנּוֹשֵׂא. חֶלְקָם סְבוּרִים שֶׁהַדָּבָר הַנָּכוֹן כָּעֵת הוּא לְהוֹדוֹת בַּהֶפְסֵד, וְאִילּוּ אֲחֵרִים סְבוּרִים שֶׁעָלָיו לְהֵיאָבֵק מִשְׁפָּטִית.

אַף שֶׁטְּרַאמְפּ מַמְשִׁיךְ לְהַבְטִיחַ לְהֵיאָבֵק מִשְׁפָּטִית בַּתּוֹצָאוֹת וְלִטְעוֹן שֶׁהֵן זוּיְּפוּ וְהוּטּוּ בְּדַרְכֵי מִרְמָה לְטוֹבַת יְרִיבוֹ הַדֵּמוֹקְרָטִי, מֵהַמַּחֲנֶה הָרֶפּוּבְּלִיקָנִי נִשְׁמְעוּ בְּסוֹף הַשָּׁבוּעַ גַּם קְרִיאוֹת הַרְגָּעָה. מַנְהִיג הָרוֹב הָרֶפּוּבְּלִיקָנִי בַּסֵּנָאט מִיץ' מֶקוֹנֶל, שֶׁנִּרְאֶה כִּי הוּא בַּדֶּרֶךְ לִשְׁמוֹר עַל תּוֹאֳרוֹ, אָמַר כִּי הוּא מַאֲמִין שֶׁתִּהְיֶה הַעֲבָרָה מְסוּדֶּרֶת וְלֹא-אַלִּימָה שֶׁל הַשִּׁלְטוֹן אִם טְרַאמְפּ יוּכְרַז כְּמַפְסִיד. כְּשֶׁנִּשְׁאַל אִם תִּהְיֶה הַעֲבָרָה כַּזּוֹ הֵשִׁיב: "כַּמּוּבָן. הָיְיתָה לָנוּ הַעֲבָרַת שִׁלְטוֹן מְסוּדֶּרֶת מֵאָז 1792 מִדֵּי אַרְבַּע שָׁנִים, עָבַרְנוּ הָלְאָה, לְמִמְשָׁל חָדָשׁ". עִם זֹאת, יְצוּיַּן כִּי מֶקוֹנֶל, יְרִיבָם הַמַּר שֶׁל הַדֵּמוֹקְרָטִים, מְנַפֵּק בַּיָּמִים הָאַחֲרוֹנִים גַּם הַצְהָרוֹת הָעוֹלוֹת בְּקָנֶה אֶחָד עִם טַעֲנוֹתָיו שֶׁל טְרַאמְפּ לִשְׁחִיתוּת, כְּגוֹן "כָּל קוֹל חוּקִּי חַיָּיב לְהִיסָּפֵר, וְכָל כָּל לֹא-חוּקִּי חַיָּיב שֶׁלֹּא לְהִיסָּפֵר".
עַל רֶקַע הַבְטָחָתוֹ שֶׁל טְרַאמְפּ לְהַמְשִׁיךְ "לְהִילָּחֵם וְלֹא לְוַותֵּר" נִזְכָּרִים כָּעֵת בְּאַרְהָ"ב בְּסֵירוּבוֹ קוֹדֶם לָכֶן הַשָּׁנָה לְהִתְחַיֵּיב שֶׁהַשִּׁלְטוֹן יַעֲבוֹר בְּצוּרָה מְסוּדֶּרֶת אִם יַפְסִיד. בַּיְידֶּן הִתְיַיחֵס בַּקַּיִץ לָאֶפְשָׁרוּת שֶׁטְּרַאמְפּ יְסָרֵב לְהִתְפַּנּוֹת מֵהַבַּיִת הַלָּבָן, וְרָמַז שֶׁבְּמִקְרֶה כָּזֶה יִהְיֶה צוֹרֶךְ לִשְׁלוֹחַ אֶת הַצָּבָא כְּדֵי לְהוֹצִיאוֹ מִשָּׁם. הֲתַרְחִישׁ הַזֶּה נִשְׁמַע קִיצוֹנִי בְּיוֹתֵר, וְהוּא מְאוֹד לֹא סָבִיר גַּם עַכְשָׁיו, אֲבָל הִתְבַּצְּרוּתוֹ שֶׁל טְרַאמְפּ בְּעֶמְדָּתוֹ וְסֵירוּבוֹ לְהַכִּיר בַּהֶפְסֵד גּוֹרֶמֶת לְאָמֵרִיקָנִים לְכָל הַפָּחוֹת לְהַרְהֵר בָּאֶפְשָׁרוּת שֶׁיִּתְמַמֵּשׁ

כָּךְ אוֹ כָּךְ, לְפִי הַדִּיוּוּחִים, בְּאַרְהָ"ב כְּבָר נֶעֱרָכִים לַבַּיִת הַלָּבָן הַדֵּמוֹקְרָטִי. בַּשָּׁעוֹת הָאַחֲרוֹנוֹת, עוֹד לִפְנֵי שֶׁהוּכְרַז נִיצְחוֹנוֹ, הוּרְחֲבוּ הַגְבָּלוֹת הַטִּיסָה בְּאֵזוֹר בֵּיתוֹ שֶׁל בַּיְידֶּן בְּוִוילְמִינְגְטוֹן שֶׁבִּמְדִינַת דֵּלָאוֵור, לְבַקָּשַׁת הַשֵּׁירוּת הַחֲשָׁאִי. בְּמַקְבִּיל דִּיוֵּוחַ "נְיוּ יוֹרְק טַיְימְס" כִּי צִוְּותוֹ שֶׁל בַּיְידֶּן כְּבָר מֵאִיץ אֶת הַתּוֹכְנִיּוֹת שֶׁלּוֹ לִתְקוּפַת הַעֲבָרַת הַשִּׁלְטוֹן לְיָדָיו מִידֵי הַמִּמְשָׁל הָרֶפּוּבְּלִיקָנִי. לְפִי הַדִּיוּוּחַ הַזֶּה, יִיתָּכֵן שֶׁרִאשׁוֹנֵי הַבְּכִירִים שֶׁיְּאַיְּישׁוּ אֶת הַבַּיִת הַלָּבָן יוּכְרְזוּ כְּבָר הַשָּׁבוּעַ.

בַּ"נְיוּ יוֹרְק טַיְימְס" מְדַוְּוחִים כִּי הַקַּבִּינֵט שֶׁל בַּיְידֶּן עָשׂוּי לִהְיוֹת מֵהַמְּגוּוָּנִים בַּהִיסְטוֹרְיָה, וּמַזְכִּירִים שֶׁבַּיְידֶּן עַצְמוֹ הִצְהִיר כִּי קַבִּינֵט כְּזֶה צָרִיךְ לִכְלוֹל גְּבָרִים, נָשִׁים, להט"בים, סְטְרֵיְיטִים, אַנְשֵׁי מֶרְכָּז, אֲנָשִׁים מֵהַצַּד הַשֵּׁנִי שֶׁל הַמַּפָּה הַפּוֹלִיטִית, שְׁחוֹרִים, לְבָנִים וְאַסְיָיתִים. בַּיְידֶּן הִסְבִּיר זֹאת בְּכָךְ שֶׁ"יֵשׁ מַשְׁמָעוּת לְכָךְ שֶׁאַתָּה נִרְאֶה כְּמוֹ כָּל הַמְּדִינָה, כִּי כָּל אֶחָד מֵבִיא נְקוּדַּת הַשְׁקָפָה מְעַט שׁוֹנָה"
הָרְשָׁתוֹת בְּאַרְהָ"ב הִכְרִיזוּ עַל נִיצְחוֹנוֹ שֶׁל בַּיְידֶּן אַחֲרֵי שֶׁהִגִּיעוּ לַמַּסְקָנָה שֶׁהַנָּשִׂיא הַיּוֹצֵא טְרַאמְפּ לֹא יוּכַל עוֹד לְהַדְבִּיק אֶת הַפַּעַר מִמֶּנּוּ בְּפֶנְסִילְבֶנְיָה, שָׁם לִקְרַאת תּוֹם סְפִירַת הַקּוֹלוֹת מוֹבִיל בַּיְידֶּן בְּיִתְרוֹן שֶׁל 0.5%. פֶּנְסִילְבֶנְיָה, שֶׁמְּחַלֶּקֶת 20 אֵלֶקְטוֹרִים, הָיְיתָה הַמְּדִינָה הַחֲשׁוּבָה בְּיוֹתֵר עֲבוּר קַמְפֵּיְין בַּיְידֶּן מִבֵּין מְדִינוֹת הַמַּפְתֵּחַ, וּבְעֶצֶם הַזְּכִיָּיה בָּהּ הוּא הִבְטִיחַ לְמַעֲשֶׂה עֲמִידָה בְּכָל הַמַּטָּרוֹת הַגְּדוֹלוֹת שֶׁהִצִּיב לוֹ לְמַעֲרֶכֶת הַבְּחִירוֹת הַזּוֹ, וּבְרֹאשָׁן הֲקָמָתָהּ מֵחָדָשׁ שֶׁל "הַחוֹמָה הַכְּחוּלָּה" בַּמַּעֲרָב הַתִּיכוֹן שֶׁל אַרְהָ"ב: הוּא הִצְלִיחַ לְנַצֵּחַ בְּפֶנְסִילְבֶנְיָה, וִויסְקוֹנְסִין וּמִישִׁיגֶן, שָׁלוֹשׁ מְדִינוֹת שֶׁמְּחַלְּקוֹת 46 אֵלֶקְטוֹרִים, וְשֶׁטְּרַאמְפּ הִפְתִּיעַ כְּשֶׁנִּיצֵּחַ בָּהֶן בְּ-2016. לִפְנֵי כֵן לֹא נִיצֵּחַ בָּהֶן מוּעֲמָד רֶפּוּבְּלִיקָנִי בְּמֶשֶׁךְ 28 שָׁנָה, וְלָכֵן הֵן נֶחְשְׁבוּ לְמִבְצָרָם שֶׁל הַדֵּמוֹקְרָטִים

הַנִּיצָּחוֹן בְּפֶנְסִילְבֶנְיָה הֵבִיא אֶת בַּיְידֶּן לְפִי כָּל הָרְשָׁתוֹת בְּאַרְהָ"ב לְ-273 אֵלֶקְטוֹרִים לְפָחוֹת, אֲבָל לְפִי חֶלְקָן מַאֲזָנוֹ גָּבוֹהַּ יוֹתֵר כְּבָר עַכְשָׁיו. "פוֹקְס נְיוּז" הִכְרִיזָה כְּבָר עַל נִיצְחוֹנוֹ גַּם בְּאָרִיזוֹנָה 11 אֵלֶקְטוֹרִים וּבְנֶבָאדָה 6, וּלְפִיכָךְ אֶצְלָהּ עוֹמֵד מַאֲזָנוֹ עַל 290. לִטְרָמאפּ יֵשׁ כָּרֶגַע 214. לִקְרַאת סִיּוּם הַסְּפִירָה בְּאָרִיזוֹנָה מוֹבִיל בַּיְידֶּן בְּ-0.6%, וְלִקְרַאת סִיּוּם הַסְּפִירָה בְּנֶבָאדָה הוּא מוֹבִיל בְּ-2%.
