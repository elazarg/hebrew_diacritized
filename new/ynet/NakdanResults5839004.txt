לֹא רַק טְרַאמְפּ: הַכִּירוּ אֶת הַמִּשְׁפָּחוֹת הַיְּהוּדִיּוֹת שֶׁל בַּיְידֶּן וְהָאָרִיס

פּוּרְסַם: 15.02.21 , 12:07

רַבּוֹת דּוּבַּר עַל הַמִּשְׁפָּחָה הַיְּהוּדִית שֶׁל נְשִׂיא אַרְהָ"ב לְשֶׁעָבַר, דוֹנַלְד טְרַאמְפּ - מִי שֶׁהָיָה לַנָּשִׂיא הַמְּכַהֵן הָרִאשׁוֹן שֶׁבִּתּוֹ יְהוּדִיָּיה וּנְכָדָיו יְהוּדִים. מָה שֶׁפָּחוֹת מוֹכֵר הוּא הַמִּשְׁפָּחוֹלוֹגְיָה הַיְּהוּדִית שֶׁל הַנָּשִׂיא גֵּ'ו בַּיְידֶּן  וּסְגָנִיתוֹ קָמֶלָה הָארִיס: לְבַיְידֶּן שְׁלוֹשָׁה יְלָדִים שֶׁכּוּלָּם נְשׂוּאִים לַיְּהוּדִים, וְהוּא סָב לִנְכָדִים יְהוּדִים (רַעְיָיתוֹ הָרִאשׁוֹנָה שֶׁל בַּיְידֶּן וְיַלְדָּתוֹ הַקְּטַנָּה נֶהֶרְגוּ בִּשְׁנַת 1972 בִּתְאוּנַת דְּרָכִים).
<< הַכּוֹל עַל הָעוֹלָם הַיְּהוּדִי - בְּפֶיְיסְבּוּק שֶׁל עֲרוּץ הַיַּהֲדוּת. הִיכָּנְסוּ >>
בַּעֲלָהּ שֶׁל הָארִיס, דָּאג אַמהוֹף, הוּא יְהוּדִי, וְהַשְּׁנַיִים מְצַיְּינִים בְּיַחַד אֶת הַחַגִּים הַיְּהוּדִיִּים. בַּחֲתוּנָּתָם בְּ-2014 שָׁבַר אַמהוֹף אֶת הַכּוֹס, כְּמִנְהַג הַחוּפָּה הַיְּהוּדִית. שְׁנֵי יְלָדָיו שֶׁל אַמהוֹף מִנִּישּׂוּאָיו הָרִאשׁוֹנִים מְכַנִּים אֶת הָארִיס "מֵאמָאלֶה" – שִׁילּוּב שֶׁל קָמֶלָה וּמָאמׇאלֶע בְּיִידִישׁ.

בְּחַג הַחֲנוּכָּה הָאַחֲרוֹן הֶעֱלוּ הָארִיס וְאַמְהוֹף סִרְטוֹן לַטְוִויטֶר שֶׁבּוֹ הִדְלִיקוּ אֶת נֵרוֹת הַחֲנוּכָּה. "אֲנִי אוֹהֶבֶת אֶת חֲנוּכָּה כִּי הוּא עוֹסֵק בַּהֲבָאַת אוֹר לַמָּקוֹם שֶׁבּוֹ יֵשׁ חוֹשֶׁךְ", אָמְרָה הָארִיס בַּסִּרְטוֹן, כְּשֶׁהִיא מְבַטֵּאת נָכוֹן אֶת הַמִּילָּה חֲנוּכָּה, עִם הַדְגָּשַׁת הָאוֹת חֵי"ת. "וְזֶה גַּם מֵבִיא תִּיקּוּן עוֹלָם, כְּלוֹמַר מַאֲבָק לְצֶדֶק וְלִכְבוֹד הָאָדָם בַּאֲשֶׁר הוּא, וְגַם עוֹסֵק בְּ'חֲנוּכָּה' שֶׁל מַשֶּׁהוּ מֵחָדָשׁ".


עִם הַשְׁבָּעָתָהּ כִּסְגָנִית הַנָּשִׂיא, נֶעֶשְׂתָה הִיסְטוֹרְיָה קְטַנָּה כַּאֲשֶׁר יְהוּדִי – בַּעֲלָהּ - מִתְגּוֹרֵר בְּאֶחָד מִשְּׁנֵי הַמְּעוֹנוֹת הָרִשְׁמִיִּים שֶׁל בְּכִירֵי הַמִּמְשָׁל בְּאַרְהָ"ב, הַU.S. Naval Observatory, הַמָּעוֹן הָרִשְׁמִי שֶׁל סְגָנִית נְשִׂיא אַרְהָ"ב. הָארִיס וְאַמְהוּף הֵם אֶחָד מִנִּי זוּגוֹת רַבִּים וְהוֹלְכִים שֶׁל יְהוּדִים אָמֵרִיקָאִים הַנִּישָּׂאִים לִבְנֵי זוּג שֶׁלֹּא מֵאוֹתוֹ גֶּזַע, לְרַבּוֹת יְהוּדִים לְבָנִים וְלֹא-לְבָנִים, הַמְּהַוִּוים כְּ-6% עַד 15% מִכְּלַל יְהוּדִי אַרְהָ"ב.

כְּבָר לֹא מִתְרַגְּשִׁים מִנִּישּׂוּאֵי תַּעֲרוֹבֶת
הַמִּשְׁפָּחוֹת הַמְּעוֹרָבוֹת שֶׁיּוֹבִילוּ אֶת הַמִּמְשָׁל הָאָמֵרִיקָנִי הֶחָדָשׁ אֵינָן בִּבְחִינַת חָרִיג – וּלְמַעֲשֶׂה, הֵן הִתְגַּלְּמוּת סִיפּוּרָם שֶׁל יִהוּדֵי אַרְהָ"ב כַּיּוֹם. רוֹב יְהוּדִי אַרְהָ"ב, 58% – אֵינָם נִישָּׂאִים לִבְנֵי זוּג יְהוּדִים, לְפִי הַסֶּקֶר הָאַרְצִי הָאַחֲרוֹן שֶׁנֶּעֱרַךְ בִּשְׁנַת 2013 בִּידֵי מְכוֹן הַמֶּחְקָר "פְּיוּ". 
רַבּוֹת מִמִּשְׁפְּחוֹת מְעוֹרָבוֹת אֵלּוּ אֵינָן נוֹטְשׁוֹת אֶת הַיַּהֲדוּת, אֶלָּא דַּוְוקָא מְגַדְּלוֹת אֶת יַלְדֵיהֶן כִּיהוּדִים וּמְצַיְּינוֹת אֶת הַחַגִּים הַיְּהוּדִיִּים, לְעִיתִּים לְצַד הַחַגִּים שֶׁל דָּתוֹת אֲחֵרוֹת. לְמַעֲשֶׂה, לְפִי הַסֶּקֶר שֶׁל "פִּיו", כְּ-45% מֵהַיְּהוּדִים שֶׁנִּישְּׂאוּ לְלֹא יְהוּדִי מְגַדְּלִים אֶת יַלְדֵיהֶם כִּיהוּדִים, לְעוּמַּת 28% בְּ-1990.

עַד לִפְנֵי כַּחֲמִישִּׁים שָׁנָה נָהֲגוּ הַקְּהִילּוֹת אוֹ הַמִּשְׁפָּחוֹת שֶׁל הַיְּהוּדִים שֶׁנִּישְּׂאוּ בְּנִישּׂוּאִים מְעוֹרָבִים, לְנַדּוֹתָם. כַּיּוֹם הֵם חֵלֶק בִּלְתִּי נִפְרָד מֵהַקְּהִילָּה הַיְּהוּדִית בְּאַרְהָ"ב. בַּתְּנוּעָה הָרֵפוֹרְמִית, הַנֶּחֱשֶׁבֶת לַתְּנוּעָה הַיְּהוּדִית הַדָּתִית הַגְּדוֹלָה בְּאַרְהָ"ב, רַבָּנִים מַשִּׂיאִים זוּגוֹת מְעוֹרָבִים, בָּתֵּי כְּנֶסֶת רַבִּים מְקַבְּלִים לְשׁוּרוֹתֵיהֶם חֲבֵרִים לֹא יְהוּדִים, וְתַפְקִידִים מְסוּיָּמִים בִּתְפִילּוֹת בְּבָתֵּי הַכְּנֶסֶת פְּתוּחִים בִּפְנֵי בְּנֵי מִשְׁפָּחָה שֶׁאֵינָם יְהוּדִים.
בַּזֶּרֶם הָרְבִיעִי - הַתְּנוּעָה הִרְקוֹנְסְטְרוּקְטִיבִית, הַמּוּכֶּרֶת גַּם כִּ"תְּנוּעָה לַיַּהֲדוּת מִתְחַדֶּשֶׁת" (הַקְּטַנָּה וְהַשְּׂמָאלִית בְּיוֹתֵר מִבֵּין הַזְּרָמִים הַלִּיבֵּרָלִיִּים הַיְּהוּדִיִּים-דָּתִיִּים בְּאַרְהָ"ב, שֶׁחֲבֵרִים בָּהּ כְּמֵאָה בָּתֵּי כְּנֶסֶת) - נִרְשַׁם בְּ-2015 צַעַד הִיסְטוֹרִי כַּאֲשֶׁר בּוּטַּל הָאִיסּוּר עַל קַבָּלַת סְטוּדֶנְטִים הַנְּשׂוּאִים לְלֹא-יְהוּדִים לְבֵית הַסֵּפֶר לְהַסְמָכַת רַבָּנִים שֶׁל הַתְּנוּעָה.

אֵצֶל הַקּוֹנְסֶרְבָטִיבִים, שֶׁנּוֹטִים לִהְיוֹת מְעַט שַׁמְרָנִים יוֹתֵר, כְּרֶבַע מִבָּתֵּי הָאָב כּוֹלְלִים בֶּן מִשְׁפָּחָה לֹא יְהוּדִי, לְפִי הַסֶּקֶר שֶׁל "פְּיוּ". אֲפִילּוּ בְּקֶרֶב הַיְּהוּדִים הָאוֹרְתּוֹדוֹכְּסִים בְּאַרְהָ"ב, נִיכָּר שִׁינּוּי בַּטּוֹן כְּלַפֵּי נִישּׂוּאִים מְעוֹרָבִים, וּבִמְקוֹם נִידּוּי שֶׁל חַבְרֵי קְהִילָּה שֶׁנִּישְּׂאוּ לְלֹא-יְהוּדִים, נִיכֶּרֶת נְכוֹנוּת רַבָּה יוֹתֵר לְקַבֵּל בִּבְרָכָה בְּנֵי זוּג שֶׁאֶחָד מֵהֶם אֵינוֹ יְהוּדִי, בְּתִקְוָוה שֶׁבֶּן הַזּוּג הַלֹּא יְהוּדִי יִתְגַּיֵּיר בְּסוֹפוֹ שֶׁל דָּבָר.
גַּם יַחֲסָהּ שֶׁל הַחֶבְרָה הָאָמֵרִיקָנִית כְּלַפֵּי הַמִּיעוּט הַיְּהוּדִי הִשְׁתַּנָּה לְטוֹבָה עִם הַזְּמַן. בִּשְׁנוֹת הַחֲמִישִּׁים וְהַשִּׁישִּׁים שֶׁל הַמֵּאָה הַקּוֹדֶמֶת חֲלָקִים נִרְחָבִים מֵהָאוּכְלוּסִייָּה בְּאַרְהָ"ב סָלְדוּ מִיְּהוּדִים: בְּ-1958 אָמְרוּ רַק 62% מִכְּלַל הָאָמֵרִיקָאִים כִּי יִהְיוּ מוּכָנִים לְהַצְבִּיעַ עֲבוּר מוּעֲמָד יְהוּדִי שֶׁמַּתְאִים הֵיטֵב לַתַּפְקִיד, לְעוּמַּת  91% בְּ-2015. בְּסֶקֶר שֶׁנֶּעֱרַךְ בְּ-1964 נִמְצָא כִּי 43% מִכְּלַל הָאָמֵרִיקָאִים רוֹאִים בִּיהוּדִים אֶת הָאֲשֵׁמִים בְּמוֹתוֹ שֶׁל יֵשׁוּ, לְעוּמַּת 26% בְּ-2004. לְצַד זֶה נְצַיֵּין כִּי בְּ-2019 נִרְשַׁם שִׂיא שֶׁל 40 שָׁנָה בְּמִסְפַּר הַתַּקְרִיּוֹת הָאַנְטִישֵׁמִיּוֹת בְּאַרְהָ"ב, אוּלָם אַנְטִישֵׁמִיּוּת כִּכְלָל נִשְׁאֲרָה בְּמִידָּה רַבָּה תּוֹפָעָה שׁוּלִית.

"אֲנִי הַקָּתוֹלִי הַיָּחִיד מִמּוֹצָא אִירִי שֶׁמִּשְּׁאֵלָתוֹ הִתְגַּשְּׁמָה - בִּתִּי נִישְּׂאָה לִמְנַתֵּחַ יְהוּדִי", הִתְבַּדֵּחַ גֵּ'ו בַּיְידֶּן בְּ-2016 בְּאֵירוּעַ פּוֹלִיטִי בְּאוֹהָיוֹ, כְּשֶׁדִּיבֵּר עַל חֲתָנוֹ הַיְּהוּדִי, הָוָוארְד קְרֵיין. קְרֵיין וְאַשְׁלִי, בִּתּוֹ הַצְּעִירָה שֶׁל בַּיְידֶּן, נִישְּׂאוּ בְּ-2012 בַּטֶּקֶס שֶׁעָרְכוּ כּוֹמֶר קָתוֹלִי וְרַב רֵפוֹרְמִי, ג'וֹזֶף פוֹרְמָן. גֵּ'ו בַּיְידֶּן הִצְטָרֵף לְרִיקּוּד הוֹרָה סוֹעֵר בַּחֲתוּנָּה, כְּפִי שֶׁדִּיוֵּוחַ אָז הַ"קַּוֶּולֶר".
"הַכְּתוּבָּה נֶחְתְּמָה. בְּנֵי הַזּוּג נִישְּׂאוּ זֶה לְזוֹ תַּחַת חוּפָּה מַרְהִיבָה שֶׁהָיְיתָה עֲשׂוּיָה מֵעֲנָפִים מְכוּסֵּי אָרִיג", סִיפֵּר פוֹרְמָן לַאֲתָר "פוֹרְוֵורְד". פוֹרְמָן מְשַׁמֵּשׁ כְּרַב קְהִילַּת "אוֹר חָדָשׁ" בִּנְיוּ-גֶ'רְזִי. "טֶקֶס הַחוּפָּה הֵחֵל בְּבִרְכַּת בָּרוּךְ הַבָּא הַמָּסוֹרְתִּית, וְכָלַל בְּרָכָה שֶׁנָּשָׁא הַכּוֹמֶר וְאֶת טֶקֶס שֶׁבַע הַבְּרָכוֹת הַיְּהוּדִי, שֶׁבְּסוֹפוֹ דָּרַךְ הֶחָתָן עַל הַכּוֹס".
שְׁלוֹשָׁה נְכָדִים יְהוּדִים
בְּנוֹ שֶׁל בַּיְידֶּן, בּוֹ, שֶׁנִּפְטַר מִסַּרְטָן בְּ-2015, נָשָׂא אַף הוּא יְהוּדִיָּיה לְאִישָּׁה – הָאלִי אוֹלִיבֶרָה, שֶׁאֶת אִימָּהּ הַיְּהוּדִיָּיה הִכִּיר בַּיְידֶּן מִיַּלְדוּת. בַּיְידֶּן סִיפֵּר בְּחִיּוּךְ שֶׁהָיָה מְאוֹהָב בְּאוֹלִיבֶרָה בְּיַלְדוּתוֹ. "אֲנִי הָיִיתִי הַיֶּלֶד הַקָּתוֹלִי, וְהִיא הָיְיתָה הַיַּלְדָּה הַיְּהוּדִיָּיה. לַמְרוֹת זֹאת, נִיסִּיתִי, אֲבָל לֹא הִגַּעְתִּי אִיתָּהּ לְשׁוּם מָקוֹם", אָמַר.

בְּנוֹ הַשֵּׁנִי שֶׁל בַּיְידֶּן, הַאנטֵר, הִתְחַתֵּן לֹא מִכְּבָר בַּשֵּׁנִית – הַפַּעַם נָשָׂא לְאִישָּׁה אֶת מְלִיסָה כֹּהֵן, יוֹצֶרֶת סִרְטֵי תְּעוּדָה מִדְּרוֹם אַפְרִיקָה. יָמִים סְפוּרִים אַחֲרֵי פְּגִישָׁתָם, קִעְקֵעַ הַאנטֵר בַּיְידֶּן אֶת הַמִּילָּה "שָׁלוֹם" כְּדֵי שֶׁתִּהְיֶה תּוֹאֶמֶת לַמִּילָּה הַמְּקוּעְקַעַת שֶׁל כֹּהֵן. בַּחוֹדֶשׁ מֵרְץ הָאַחֲרוֹן נוֹלַד לִבְנֵי הַזּוּג הַמִּתְגּוֹרְרִים בְּלוֹס-אַנְגֵּ'לֶס בְּנָם הָרִאשׁוֹן. בְּכָךְ עָלָה מִסְפַּר הַנְּכָדִים הַיְּהוּדִים שֶׁל בַּיְידֶּן לִשְׁלוֹשָׁה, בְּנוֹסָף לִשְׁנֵי הַיְּלָדִים שֶׁל בּוֹ וְהָאלִי.

בַּיְידֶּן, כַּיָּדוּעַ, אֵינוֹ נְשִׂיא אַרְהָ"ב הָרִאשׁוֹן שֶׁהוּא סַב גֵּאֶה לִנְכָדִים יְהוּדִים. דוֹנַלְד טְרַאמְפּ עָשָׂה הִיסְטוֹרְיָה כְּשֶׁהָיָה לְנָשִׂיא הַמְּכַהֵן הָרִאשׁוֹן שֶׁלּוֹ יֶלֶד יְהוּדִי: בִּתּוֹ, אִיוַונְקָה טְרַאמְפּ, הִתְגַּיְּירָה בְּגִיּוּר אוֹרְתּוֹדוֹכְּסִי לִפְנֵי שֶׁנִּישְּׂאָה לְבַעֲלָהּ, גַּ'ארְד קוּשְׁנֶר, בְּ-2009. הַזּוּג מְגַדֵּל אֶת שְׁלוֹשֶׁת יַלְדֵיהֶם כִּיהוּדִים אוֹרְתּוֹדוֹכְּסִים, שׁוֹמֵר שַׁבָּת בְּמִידָּה מְסוּיֶּמֶת, מִתְפַּלֵּל בְּבָתֵּי כְּנֶסֶת אוֹרְתּוֹדוֹכְּסִיִּים וְיַלְדֵיהֶם לוֹמְדִים בְּבָתֵּי סֵפֶר יְהוּדִיִּים. אֶחָד מֵהֶם תּוֹעַד לִפְנֵי מִסְפַּר שָׁבוּעוֹת בְּטֶקֶס קַבָּלַת שַׁבָּת בְּגַן חַבָּ"ד.
הַלּוֹחֶמֶת נֶגֶד אַנְטִישֵׁמִיּוּת בָּרֶשֶׁת
צֶ'לְסִי קְלִינְטוֹן, בִּתָּם הַיְּחִידָה שֶׁל נְשִׂיא אַרְהָ"ב לְשֶׁעָבַר, בִּיל קְלִינְטוֹן וְשֶׁל הִילָארִי רוֹדְהָאם קְלִינְטוֹן, נִישְּׂאָה לְמַארְק מֶזְוִוינְסְקִי הַיְּהוּדִי בִּשְׁנַת 2010, בַּטֶּקֶס שֶׁאוֹתוֹ עָרְכוּ בִּמְשׁוּתָּף רַב וְכוֹמֶר וְשֶׁכְּלָל חוּפָּה וְשֶׁבַע הַבְּרָכוֹת. לַזּוּג שְׁלוֹשָׁה יְלָדִים. 
קְלִינְטוֹן אָמְנָם נוֹצְרִיָּיה מַתּוֹדִיסְטִית, אַךְ הִשְׁמִיעָה אֶת קוֹלָהּ לֹא פַּעַם בְּתַקִּיפוּת נֶגֶד גִּילּוּיֵי אַנְטִישֵׁמִיּוּת בְּאָמֵרִיקָה. בִּתְגוּבָה לַפּוֹסְט שֶׁכָּתַב מִישֶׁהוּ בְּרֶשֶׁת עַל כָּךְ שֶׁהִיא "אֲפִילּוּ לֹא יְהוּדִיָּיה, הִיא סְתָם מְכוֹעֶרֶת", עָנְתָה קְלִינְטוֹן: "הַיי אָדָם – אַתָּה צוֹדֵק. אֲנִי לֹא יְהוּדִיָּיה. אֲבָל אִם אַתָּה חוֹשֵׁב שֶׁאֲנִי מְכוֹעֶרֶת, הַרְגֵּשׁ חוֹפְשִׁי לְעוֹלָם לֹא לְהִתְבּוֹנֵן בִּי. הַיְּהוּדִי הַמְּכוֹעָר הוּא תְּפִיסָה אַנְטִישֵׁמִית דּוֹחָה בַּת מֵאוֹת בַּשָּׁנִים, אָז בְּבַקָּשָׁה בַּפַּעַם הַבָּאָה פָּשׁוּט דַּבֵּר רַק עַל 'מְכוֹעָר', וְאֶל תְּעָרֵב אֶת הַשְּׁאָר בְּכָךְ. תּוֹדָה".

אַחֲרֵי שֶׁחַבְרַת הַקּוֹנְגְּרֵס אִילְהָאן עוֹמַאר, אַחַת מִשָּׁלוֹשׁ חַבְרוֹת הַקּוֹנְגְּרֵס הַמּוּסְלְמִיּוֹת, טָעֲנָה בְּפֶבְּרוּאָר 2019 כִּי אִרְגּוּן אַיְיפָּא"ק - הַשְּׁדוּלָּה הַפְּרוֹ-יִשְׂרְאֵלִית בְּאַרְהָ"ב - מְשַׁלֶּמֶת לַפּוֹלִיטִיקָאִים כְּדֵי לִתְמוֹךְ בְּיִשְׂרָאֵל, צִייְּצָה קְלִינְטוֹן: "אֲנַחְנוּ מְצַפִּים מִכָּל הַפְּקִידִים הַנִּבְחָרִים מִכָּל מִפְלָגָה, וּמִכָּל הַדְּמוּיוֹת הַצִּיבּוּרִיּוֹת, לֹא לִסְחוֹר בָּאַנְטִישֵׁמִיּוּת".
וְכַמּוּבָן, יֶשְׁנָם גַּם כָּל אֵלֶּה שֶׁכִּמְעַט הִגִּיעוּ לַמִּשְׂרָה הַנֶּחֱשֶׁקֶת: גֵּ'ו לִיבֶּרְמָן, יְהוּדִי אוֹרְתּוֹדוֹכְּסִי, שֶׁהָיָה זוֹכֶה בְּתַפְקִיד סְגַן נְשִׂיא אַרְהָ"ב אִילּוּ אֶל גּוּר הָיָה נִבְחָר לְנָשִׂיא. גּ'וֹן קֵרִי, שֶׁהִפְסִיד לְבוּשׁ בַּבְּחִירוֹת לַנְּשִׂיאוּת בִּשְׁנַת 2004, הָיָה נֶכֶד לְסָבִים יְהוּדִים מִצַּד אָבִיו, אֲשֶׁר בִּרְבוֹת הַיָּמִים הִתְנַצְּרוּ. אָחִיו שֶׁל קֵרִי, קָאמֵרוֹן קֵרִי, הִתְגַּיֵּיר.
לִיאוֹר זַלְצְמָן הִשְׁתַּתְּפָה בַּהֲכָנַת הַכַּתָּבָה.
