 לִפְנֵי אֶלֶף שָׁנָה וְיוֹתֵר, אֵי-שָׁם, מַמָּשׁ בִּקְצֵה הָעוֹלָם, הָיְתָה אֶרֶץ שֶׁאֲנָשֶׁיהָ נַעֲשׂוּ כֹּה מְנֻמָּסִים, עַד שֶׁבְּקשִׁי דִּבְּרוּ זֶה עִם זֶה. הֵם מֵעוֹלָם לֹא אָמְרוּ יוֹתֵר מִן הַהֶכְרֵחִי, מִלִּים כְּמוֹ "בְּהֶחְלֵט", וְ"אָכֵן כֵּן", וְ"חֵן-חֵן" וּ"בִרְשׁוּתְךָ". כָּל גִּלּוּי שֶׁל חִבָּה אוֹ סְלִידָה נֶחְשַׁב לְגַסּוּת רוּחַ שֶׁלֹּא תֵּאָמֵן. אִישׁ לֹא אָמַר שֶׁהוּא אוֹהֵב אוֹ שׂוֹנֵא, מְאֻשָּׁר אוֹ אֻמְלָל. אִישׁ מֵעוֹלָם לֹא פָּרַץ בִּצְחוֹק, וְאִם מִישֶׁהוּ פָּרַץ בְּבֶכִי, מִיָּד הִתְרַחֲקוּ מִמֶּנּוּ כָּל חֲבֵרָיו.
הַמֶּלֶךְ שֶׁל אוֹתָהּ אֶרֶץ נָשָׂא אִשָּׁה טוֹבַת לֵב וְיָפָה מֵאֶרֶץ שְׁכֵנָה. אֶלָּא שֶׁהָאֲנָשִׁים בְּאַרְצָהּ שֶׁל הַכַּלָּה הָיוּ שׁוֹנִים בְּתַכְלִית מִן הָאֲנָשִׁים בְּאַרְצוֹ שֶׁל בַּעֲלָהּ. הֵם צָחֲקוּ וּפִטְפְּטוּ, הֵם הִצְטַהֲלוּ מֵרֹב שִׂמְחָה וְהִזְדַּעֲקוּ כְּשֶׁלִּבָּם נִשְׁבַּר. כָּל רֶגֶשׁ קָטָן נִרְשַׁם עַל פְּנֵיהֶם וְגַם עַל פָּנֶיהָ שֶׁלָּהּ.
כְּשֶׁהִגִּיעָה לְמוֹלַדְתָּהּ הַחֲדָשָׁה הֻפְתְּעָה הַמַּלְכָּה מֵהִתְנַהֲגוּתָם שֶׁל נְתִינֶיהָ; הִיא לֹא הֵבִינָה מַדּוּעַ הִקְפִּידוּ כָּל כָּךְ עַל רִשְׁמִיּוּת וְאִפּוּק בִּמְקוֹם לְקַדֵּם אֶת פָּנֶיהָ בִּתְרוּעוֹת שִׂמְחָה. וּמִכֵּיוָן שֶׁגַּם בַּהֶמְשֵׁךְ הֵם לֹא הִשְׁתַּנּוּ וְנוֹתְרוּ דּוֹמְמִים וְנֻקְשִׁים, הִיא בָּכְתָה וְהִתְחִילָה לְהִתְגַּעְגֵּעַ לְבֵיתָהּ הַקּוֹדֵם.
מִיּוֹם לְיוֹם כָּחֲשָׁה וְהֶחְוִירָה. אַנְשֵׁי הֶחָצֵר הָיוּ אָמְנָם מְנֻמָּסִים מִכְּדֵי לְהַבְחִין בִּמְצוּקָתָהּ, אַךְ הַמַּלְכָּה הַצְּעִירָה יָדְעָה שֶׁהִיא חוֹלָה וְהֶאֱמִינָה שֶׁסּוֹפָהּ קָרֵב.
וְהָיְתָה לָהּ, לַמַּלְכָּה, פֵיָה סַנְדָּקִית בְּשֵׁם טַבּוֹרֶט, שֶׁאוֹתָהּ אָהֲבָה בְּכָל לִבָּהּ. קְצָת לִפְנֵי מוֹתָהּ הִיא שָׁלְחָה לִקְרֹא לָהּ, וּכְשֶׁהַפֵיָה אָמְנָם הִגִּיעָה, הֵן יָצְאוּ לְטִיּוּל אָרֹךְ, רַק שְׁתֵּיהֶן לְבַדָּן. אִישׁ לֹא יָדַע מָה אָמְרוּ זוֹ לָזוֹ, וְכַעֲבֹר זְמַן-מָה, אַחֲרֵי שֶׁיָּלְדָה נְסִיכָה קְטַנָּה, מֵתָה הַמַּלְכָּה. מוּבָן שֶׁכָּל אַנְשֵׁי הֶחָצֵר הִצְטַעֲרוּ עַל מוֹתָהּ שֶׁל הַמַּלְכָּה הַמִּסְכֵּנָה, אֲבָל כָּל גִּלּוּי שֶׁל רֶגֶשׁ נֶחְשַב הֲרֵי לְגַסּוּת רוּחַ, כָּךְ שֶׁלְּמַעֵט הַלְוָיָה מְפֹאֶרֶת וּשְׁאַר גִּנּוּנֵי אֲבֵלוּת, נִמְשְׁכוּ הַחַיִּים כְּהֶרְגֵּלָם.
הַתִּינֹקֶת הֻטְבְּלָה בַּשֵּׁם אוֹרְסוּלָה וְנִמְסְרָה לְטִפּוּלָן שֶׁל כַּמָּה מִנְּשׁוֹת הֶחָצֵר. נְסִיכֹנֶת מִסְכֵּנָה! הִיא לֹא הִפְסִיקָה לִבְכּוֹת, שׁוּם דָּבָר לֹא הִרְגִּיעַ אוֹתָהּ.
כָּל נְשׁוֹת הֶחָצֵר נֶחְרְדוּ מִן הָרַעַשׁ הַנּוֹרָא שֶׁכְּמוֹתוֹ לֹא שָׁמְעוּ זֶה מִכְּבָר, אֲבָל עַד שֶׁמָּלְאוּ לָהּ שְׁנָתַיִם בְּעֵרֶךְ, הִמְשִׁיכָה הַנְּסִיכָה לְיַלֵּל בְּכָל פַּעַם שֶׁהָיָה לָהּ קַר אוֹ כְּשֶׁהָיְתָה רְעֵבָה, וּלְהִצְטַהֵל כְּשֶׁהָיְתָה מְרֻצָּה.
אֲבָל בְּמֶשֶׁךְ הַזְּמַן הִיא בְּכָל זֹאת הֵבִינָה לְמָה הִתְכַּוְּנוּ אוֹמְנוֹתֶיהָ כְּשֶׁאָמְרוּ לָהּ בְּנִימָה קָרָה וּמְנֻמֶּסֶת שֶׁהִיא יַלְדָּה רָעָה, וּלְאַט-לְאַט הִשְׁתַּתְּקָה. הִיא הָיְתָה פָּעוֹטָה נֶחְמָדָה עִם פְּנֵי תִּינֹקֶת עֲגֻלִּים וְעֵינַיִם גְּדוֹלוֹת וּכְחֻלּוֹת, עַלִּיזוֹת וּמַבְרִיקוֹת, שֶׁאוֹרָן דָּעַךְ אָמְנָם כְּכָל שֶׁהִתְבַּגְּרָה, וְגַם פָּנֶיהָ נָפְלוּ וְהֶחְוִירוּ. הִיא לֹא הֻרְשְׁתָה לְהִתְרוֹעֵעַ עִם יְלָדִים אֲחֵרִים שֶׁמָּא יִדְבְּקוּ בָּהּ נִימוּסֵיהֶם הָרָעִים. לֹא לִמְּדוּ אוֹתָהּ מִשְׂחָקִים, וְלֹא נָתְנוּ לָהּ צַעֲצוּעִים. אֶת רֹב זְמַנָּהּ, לְמַעֵט שְׁעוֹת הַלִּמּוּדִים, הִיא הֶעֱבִירָה לְיַד הַחַלּוֹן כְּשֶׁהִיא צוֹפָה בַּצִּפּוֹרִים הַמְּרַחֲפוֹת בַּשָּׁמַיִם הַכְּחֻלִּים הַצְּלוּלִים; פֹּה וָשָׁם פָּלְטָה אֲנָחָה קְטַנָּה וַעֲגוּמָה, הַרְחֵק מֵאָזְנֵיהֶן שֶׁל הָאוֹמְנוֹת.
יוֹם אֶחָד הָפְכָה הַפֵיָה הַזְּקֵנָה טַבּוֹרֶט אֶת עַצְמָהּ לְבִלְתִּי נִרְאֵית וְעָפָה אֶל אַרְמוֹן הַמֶּלֶךְ כְּדֵי לִבְדֹּק אֶת מַצַּב הָעִנְיָנִים. הִיא נָחֲתָה הַיְשֵׁר בַּחֲדַר הַיְּלָדִים וְרָאֲתָה אֶת אוֹרְסוּלָה הַקְּטַנָּה וְהַמִּסְכֵּנָה יוֹשֶׁבֶת לְיַד הַחַלּוֹן וְרֹאשָׁהּ שָׁעוּן עַל יָדָהּ.
זֶה הָיָה חֶדֶר יָפֶה וְהָדוּר, אֲבָל לֹא הָיוּ בּוֹ צַעֲצוּעִים אוֹ בֻּבּוֹת, וְהַפֵיָה קִמְּטָה אֶת מִצְחָהּ כְּשֶׁהִבְחִינָה בְּחֶסְרוֹנָם וְטִלְטְלָה אֶת רֹאשָׁהּ.
"אֲרוּחַת הָעֶרֶב שֶׁל הוֹד מַעֲלָתֵךְ מוּכָנָה," אָמְרָה הָאוֹמֶנֶת הָרָאשִׁית לְאוֹרְסוּלָה.
"אֲנִי לֹא רְעֵבָה," אָמְרָה אוֹרְסוּלָה בְּלִי לְהָרִים אֶת רֹאשָׁהּ.
"נִדְמֶה לִי שֶׁכְּבָר אָמַרְתִּי לְהוֹד מַעֲלָתֵךְ שֶׁאֵין זֶה מְנֻמָּס לוֹמַר שֶׁאֵינֵךְ רְעֵבָה אוֹ שֶׁאֵינֵךְ רוֹצָה לֶאֱכֹל," אָמְרָה הָאוֹמֶנֶת. "אָנוּ מַמְתִּינוֹת לְהוֹד מַעֲלָתֵךְ."
וְהַנְּסִיכָה אָמְנָם קָמָה וְנִגְּשָׁה אֶל הַשֻּׁלְחָן, וטַבּוֹרֶט לֹא גָּרְעָה עַיִן מִן הַמִּתְרַחֵשׁ וְלֹא הֶחְמִיצָה שׁוּם פְּרָט. וּכְשֶׁהִבְחִינָה בְּחִוְרוֹנָהּ שֶׁל אוֹרְסוּלָה הַקְּטַנָּה וְרָאֲתָה שֶׁכִּמְעַט לֹא אָכְלָה וְשֶׁלֹּא הֻרְשְׁתָה לְדַבֵּר אוֹ לִצְחֹק בִּשְׁעַת הַסְּעוּדָה, נֶאֶנְחָה וּמִצְחָהּ הִתְקַמֵּט אֲפִלּוּ יוֹתֵר מִקֹּדֶם. הִיא חָזְרָה בְּטִיסָה אֶל בֵּיתָהּ הַפֵיָתִי וְשָׁם יָשְׁבָה שָׁעָה אֲרֻכָּה שְׁקוּעָה בְּמַחְשָׁבוֹת.
לְבַסּוֹף הִתְנַעֲרָה וּפָנְתָה אֶל הַחֲנוּת הַגְּדוֹלָה בְּיוֹתֵר בְּאֶרֶץ הַפֵיוֹת.
זוֹ הָיְתָה חֲנוּת מוּזָרָה לְהַפְלִיא; זֹאת לֹא הָיְתָה מַכֹּלֶת וְגַם לֹא חֲנוּת בַּדִּים אוֹ כּוֹבָעִים, וּבְכָל זֹאת נִמְכְּרוּ בָּהּ סֻכָּר וּשְׂמָלוֹת וְכוֹבָעִים. אֶלָּא שֶׁהַסֻּכָּר הָיָה סֻכַּר פְּלָאִים שֶׁכִּשֵּׁף כָּל נוֹזֵל שֶׁלְּתוֹכוֹ נִבְחַשׁ, לְכָל שִׂמְלָה הָיָה קֶסֶם מִשֶּׁלָּהּ, וְהַכּוֹבָעִים הָיוּ כּוֹבְעֵי מִשְׁאָלָה. זוֹ הָיְתָה מִין חֲנוּת שֶׁבָּהּ נִסְחֲרוּ כָּל סוּגֵי הַכְּשָׁפִים וְהַלְּחָשִׁים.
אֶל תּוֹךְ הַחֲנוּת הַזֹּאת עוֹפְפָה לָהּ טַבּוֹרֶט, וּבִהְיוֹתָהּ לָקוֹחָה וָתִיקָה וְטוֹבָה, מִהֵר בַּעַל הַחֲנוּת וְנִגַּשׁ אֵלֶיהָ. הוּא הִשְׁתַּחֲוָה וּבִקֵּשׁ לָדַעַת אֵיךְ יוּכַל לַעֲזֹר לָהּ.
"אֲנִי," אָמְרָה טַבּוֹרֶט, "זְקוּקָה לִנְסִיכָה."
"נְסִיכָה!" אָמַר הַחֶנְוָנִי שֶׁלֹּא הָיָה אֶלָּא קוֹסֵם זָקֵן. "בְּאֵיזוֹ מִדָּה אַתְּ רוֹצֶה אוֹתָהּ? יֵשׁ לִי כַּמָּה בַּמְּלַאי."
"הִיא צְרִיכָה לְהֵרָאוֹת בַּת שֵׁשׁ בְּעֵרֶךְ, אֲבָל עָלֶיהָ לִגְדֹּל בַּהֶמְשֵׁךְ."
"אֲנִי יָכוֹל לְהָכִין לָךְ אַחַת," אָמַר הַקּוֹסֵם, "אֲבָל זֶה לֹא יִהְיֶה זוֹל."
"הַמְּחִיר לֹא מְשַׁנֶּה," אָמְרָה טַבּוֹרֶט, "בִּתְנַאי שֶׁתֵּרָאֶה בְּדִיּוּק כָּךְ." הִיא שָׁלְפָה דְּיוֹקָן שֶׁל אוֹרְסוּלָה מִכִּיס נִסְתָּר שֶׁל שִׂמְלָתָהּ וּמָסְרָה אוֹתוֹ לַזָּקֵן. הוּא בָּחַן אֶת הַדְּיוֹקָן בִּקְפִידָה.
"אֲנִי יָכוֹל לְהַשִּׂיג אַחַת כָּזֹאת. לְמָתַי אַתְּ צְרִיכָה אוֹתָהּ?" שָׁאַל.
"לְעַכְשָׁו," אָמְרָה טַבּוֹרֶט. "עַד מָחָר בָּעֶרֶב, אִם תּוּכַל. כַּמָּה זֶה יַעֲלֶה לִי?"
"זֶה יַעֲלֶה לֹא מְעַט," אָמַר הַקּוֹסֵם בְּכֹבֶד רֹאשׁ. "לֹא קַל לְיַצֵּר דְּבָרִים כָּאֵלֶּה כְּמוֹ שֶׁצָּרִיךְ. אֵיזֶה מִין קוֹל אָמוּר לִהְיוֹת לָהּ?"
"הִיא לֹא צְרִיכָה לִהְיוֹת פַּטְפְּטָנִית," אָמְרָה טַבּוֹרֶט. "זֶה לֹא אָמוּר לְיַקֵּר אֶת הָעֵסֶק. דַּי שֶׁתַּגִּיד, 'בִּרְשׁוּתְךָ', 'לֹא תּוֹדָה', 'בְּהֶחְלֵט' וְ'אָכֵן כֵּן'."
"אִם כָּךְ," אָמַר הַקּוֹסֵם, "אֲנִי יָכוֹל לְבַצֵּעַ אֶת הָעֲבוֹדָה תְּמוּרַת אַרְבַּע טְפִיפוֹת שֶׁל חָתוּל, שְׁתֵּי צְרָחוֹת שֶׁל דָּגִים וּשְׁתֵּי שִׁירוֹת בַּרְבּוּר."
"זֶה יוֹתֵר מִדַּי," קָרְאָה טַבּוֹרֶט. "אֲנִי אֶתֵּן לְךָ אֶת הַטְּפִיפוֹת וְהַצְּרָחוֹת, אֲבָל לְבַקֵּשׁ שִׁירַת בַּרְבּוּר!"
הִיא לֹא בֶּאֱמֶת חָשְׁבָה שֶׁזֶּה יָקָר. הִיא פָּשׁוּט נָהֲגָה לַעֲמֹד עַל הַמִּקָּח, זֶה הָיָה עִנְיָן שֶׁל עִקָּרוֹן.
"אֲנִי לֹא יָכוֹל לְהָכִין אוֹתָהּ בְּפָחוֹת," אָמַר הַקּוֹסֵם. "אִם אַתְּ חוֹשֶׁבֶת שֶׁזֶּה מֻגְזָם, אוּלַי תְּנַסִּי אֶת מַזָּלֵךְ בַּחֲנוּת אַחֶרֶת."
"מִכֵּיוָן שֶׁאֲנִי בֶּאֱמֶת מְמַהֶרֶת וְאֵין לִי זְמַן לְבַזְבֵּז בְּחִפּוּשִׂים, אֵין לִי בְּרֵרָה אֶלָּא לְהַסְכִּים. אֲבָל אֲנִי עֲדַיִן חוֹשֶׁבֶת שֶׁזֶּה יָקָר. מָתַי זֶה יִהְיֶה מוּכָן?"
"מָחָר בָּעֶרֶב."
"בְּסֵדֶר גָּמוּר, אֲנִי מְצַפָּה שֶׁתַּעֲמֹד בְּמִלָּתְךָ. וְלֹא מְשַׁנֶּה מָה יִקְרֶה – רַק אַל תַּעֲשֶׂה אוֹתָהּ רַעֲשָׁנִית אוֹ הֲמוֹנִית," אָמְרָה טַבּוֹרֶט, וּכְבָר הִסְתַּלְּקָה מִן הַחֲנוּת וְעָפָה לְבֵיתָהּ.
לְמָחֳרָת בָּעֶרֶב חָזְרָה וְשָׁאֲלָה אִם הָעֲבוֹדָה מוּכָנָה.
"מִיָּד אָבִיא אוֹתָהּ, אֲנִי בָּטוּחַ שֶׁהִיא תִּמְצָא חֵן בְּעֵינַיִךְ," אָמַר הַקּוֹסֵם וְיָצָא בִּסְעָרָה, וְעַד מְהֵרָה אָמְנָם חָזַר כְּשֶׁהוּא מוֹלִיךְ אַחֲרָיו יַלְדֹּנֶת נָאָה כְּבַת שֵׁשׁ, יַלְדֹּנֶת שֶׁדָּמְתָה לַנְּסִיכָה אוֹרְסוּלָה עַד כְּדֵי כָּךְ שֶׁאִישׁ לֹא יָכוֹל הָיָה לְהַבְחִין בֵּינֵיהֶן.
"וּבְכֵן," אָמְרָה טַבּוֹרֶט, "זֶה נִרְאֶה לֹא רַע. אֲבָל אַתָּה בָּטוּחַ שֶׁזֶּה יַחְזִיק מַעֲמָד? רַק שֶׁלֹּא תִּתְפָּרֵק לִי פִּתְאוֹם."
"זֹאת מְלֶאכֶת מַחֲשֶׁבֶת מִן הַמַּעֲלָה הָרִאשׁוֹנָה," אָמַר הַקּוֹסֵם בְּגַאֲוָה וְטָפַח עַל גַּבָּהּ שֶׁל הַיַּלְדָּה. "הַבִּיטִי בָּהּ! מִכַּף רֶגֶל וְעַד רֹאשׁ לֹא תִּמְצְאִי בָּהּ שׁוּם פְּגָם. גַּם פֵיָה אַחַת מִתּוֹךְ עֶשְׂרִים לֹא תּוּכַל לְהַבְחִין בֵּינָהּ לְבֵין הַדָּבָר הָאֲמִתִּי, וּלְבֶן תְּמוּתָה אֵין שׁוּם סִכּוּי."
"הִיא בֶּאֱמֶת נֶחְמָדָה," הִסְכִּימָה טַבּוֹרֶט בִּזְמַן שֶׁסּוֹבְבָה אֶת הַיַּלְדֹּנֶת מִצַּד לְצַד. "וְעַכְשָׁו אֲשַׁלֵּם לְךָ וְאָעוּף מִכָּאן," אָמְרָה, וּבְעוֹדָהּ מְדַבֶּרֶת שָׁלְפָה אֶת שַׁרְבִיטָהּ וְנוֹפְפָה בּוֹ שָׁלוֹשׁ פְּעָמִים, וְסִדְרָה שֶׁל צְלִילִים מְשֻׁנִּים נִפְלְטָה מִמֶּנּוּ.
קֹדֶם רַחַשׁ טְפִיפוֹת, וְאַחֲרֵיהֶן צְוָחוֹת חַדּוֹת וְנוֹקְבוֹת וּלְבַסּוֹף צְלִילִים יְפֵהפִיִּים פּוֹלְחֵי לֵב שֶׁל שִׁירָה עֲצוּבָה.
הַקּוֹסֵם תָּפַס אֶת הַצְּלִילִים וְטָמַן אוֹתָם בְּכִיסוֹ, וּבֵינְתַיִם נָטְלָה טַבּוֹרֶט אֶת הַיַּלְדָּה וּבְלִי שׁוּם גִּנּוּנִים תָּחֲבָה אוֹתָהּ מִתַּחַת לִזְרוֹעָהּ, רֹאשׁ לְמַטָּה וְרַגְלַיִם לְמַעְלָה, וְעָפָה לְדַרְכָּהּ.
בְּאוֹתוֹ לַיְלָה בָּאַרְמוֹן הִקְשְׁתָה הַנְּסִיכָה הַקְּטַנָּה אֶת עָרְפָּהּ וְסֵרְבָה לָלֶכֶת לִישֹׁן. הָאוֹמְנוֹת עָמְלוּ קָשׁוֹת עַד שֶׁנִּכְנְסָה לְמִטָּתָהּ, וְגַם אָז הִיא לֹא בֶּאֱמֶת נִרְדְּמָה, רַק שָׁכְבָה לְלֹא נִיעַ וְהֶעֱמִידָה פְּנֵי יְשֵׁנָה. אַחֲרֵי שֶׁנֶּעֶזְבָה לְנַפְשָׁהּ הִיא הִתְרוֹמְמָה בַּלָּאט וְטִפְּסָה אֶל אֶדֶן הַחַלּוֹן, וְשָׁם הִצְטַנְּפָה לִצְרוֹר קָטָן וְנָשְׂאָה אֶת עֵינֶיהָ הָעֲצוּבוֹת לַיָּרֵחַ. הִיא נִרְאֲתָה כֹּה יָפָה וְרַכָּה בְּשַׂעֲרוֹתֶיהָ הַחֲמִימוֹת וְהַמַּבְהִיקוֹת הַגּוֹלְשׁוֹת עַל כְּתֵפֶיהָ, שֶׁרֹב הָאֲנָשִׁים הָיוּ מִתְקַשִּׁים לִכְעֹס עָלֶיהָ. הִיא הִשְׁעִינָה אֶת סַנְטֵרָהּ עַל יָדֶיהָ הַלְּבָנוֹת הַקְּטַנּוֹת, וּבְעוֹדָהּ בּוֹהָה בַּיָּרֵחַ נִקְווּ דְּמָעוֹת בְּעֵינֶיהָ הַכְּחֻלּוֹת הַגְּדוֹלוֹת, וְהִיא מִהֲרָה לְנַגֵּב אוֹתָן בְּשַׁרְווּל כֻּתָּנְתָּהּ מִפַּחַד אוֹמְנוֹתֶיהָ.
"הָה, יָרֵחַ, יָרֵחַ יָפֶה וְזוֹהֵר!" הִיא אָמְרָה לְעַצְמָהּ, "הַאִם מַרְשִׁים לְךָ לִבְכּוֹת כְּשֶׁאַתָּה עָצוּב? הָיִיתִי רוֹצָה לַעֲלוֹת אֵלֶיךָ וְלָגוּר אִתְּךָ, אֲנִי בְּטוּחָה שֶׁיּוֹתֵר נֶחְמָד שָׁם לְמַעְלָה מֵאֲשֶׁר כָּאן."
"הַאִם תִּרְצִי אוּלַי לָבוֹא אִתִּי?" אָמַר קוֹל לְיָדָהּ. וּכְשֶׁנָּשְׂאָה אֶת עֵינֶיהָ רָאֲתָה יְשִׁישֹׁנֶת מוּזָרָה בִּגְלִימָה אֲדֻמָּה. הִיא לֹא חָשְׁשָׁה, כִּי הַזְּקֵנָה חִיְּכָה אֵלֶיהָ בַּחֲבִיבוּת וְעֵינֶיהָ הַשְּׁחֹרוֹת הִבְרִיקוּ, אַף שֶׁאַפָּהּ הִתְעַקֵּל מֵעַל סַנְטֵרָהּ הָאָרֹךְ.
"לְאָן אַתְּ רוֹצָה לָקַחַת אוֹתִי?" שָׁאֲלָה הַנְּסִיכָה הַקְּטַנָּה כְּשֶׁהִיא מוֹצֶצֶת אֶת אֲגוּדָלָהּ וְעֵינֶיהָ כִּמְעַט יוֹצְאוֹת מֵחֹרֵיהֶן.
"אֶל חוֹף הַיָּם, אֶל מָקוֹם שֶׁבּוֹ תּוּכְלִי לְשַׂחֵק בַּחוֹל, מָקוֹם שֶׁבּוֹ יִהְיוּ לָךְ חֲבֵרִים קְטַנִּים וַחֲבֵרוֹת לַמִּשְׂחָק, וְאִישׁ לֹא יִמְנַע מִמְּךָ לְהַרְעִישׁ."
"אֲנִי בָּאָה," אָמְרָה אוֹרְסוּלָה וְזִנְּקָה מִמְּקוֹמָהּ.
"קָדִימָה," אָמְרָה הַזְּקֵנָה בַּעֲדִינוּת, וְהִיא אָסְפָה אֶת אוֹרְסוּלָה בִּזְרוֹעוֹתֶיהָ וְעָטְפָה אוֹתָהּ בִּגְלִימָתָהּ הָאֲדֻמָּה, וְעַד מְהֵרָה הִתְרוֹמְמוּ לָהֶן וּפָרְחוּ מֵעַל גַּגּוֹת הַבָּתִּים.
אֲוִיר הַלַּיְלָה הָיָה רַעֲנָן. אוֹרְסוּלָה עָצְמָה אֶת עֵינֶיהָ וְנִרְדְּמָה, וְגַם בִּשְׁנָתָהּ הֵן הִמְשִׁיכוּ לָעוּף מֵעַל גְּבָעוֹת וַעֲמָקִים, הָלְאָה וְהָלְאָה, הַרְחֵק מִן הָאַרְמוֹן וּלְעֵבֶר חוֹף הַיָּם.
אֵי-שָׁם, מֵעֵבֶר לֶחָצֵר וְלָאַרְמוֹן, בַּעֲיָרַת דַּיָּגִים זְעִירָה עַל שְׂפַת הַיָּם, נִצְּבָה בִּקְתָּה קְטַנָּה שֶׁבָּהּ חַי דַּיָּג בְּשֵׁם מַרְק עִם אִשְׁתּוֹ וּשְׁלֹשֶׁת יְלָדָיו. מַרְק הָיָה עָנִי. הַדָּגִים שֶׁלָּכַד בְּעֶזְרַת סִירָתוֹ הַקְּטַנָּה הִסְפִּיקוּ בְּקשִׁי לְמִחְיָתוֹ, אֲבָל לִשְׁלשֶׁת יְלָדָיו, אוֹלִיבֵר, פִילִיפּ וּבֵּל הַקְּטַנָּה הָיוּ לְחָיַיִם וְרֻדּוֹת וְעֵינַיִם נוֹצְצוֹת. כָּל הַיּוֹם שִׂחֲקוּ בַּחוֹל וְצָעֲקוּ עַד שֶׁהִצְטָרְדוּ. אֶל הָעֲיָרָה הַזֹּאת נָשְׂאָה הַפֵיָה אֶת אוֹרְסוּלָה הַנָּמָה וְהֵנִיחָה אוֹתָהּ בַּעֲדִינוּת עַל סַף בֵּיתוֹ שֶׁל מַרְק. הִיא נָשְׁקָה לְלֶחְיָהּ, הָדְפָה אֶת הַדֶּלֶת בְּפֶרֶץ רוּחַ וְנֶעֶלְמָה לִפְנֵי שֶׁמִּישֶׁהוּ הִבְחִין בָּהּ.
הַדַּיָּג וְאִשְׁתּוֹ בִּלּוּ עֶרֶב שָׁקֵט יַחַד. הָאִשָּׁה תָּפְרָה בֶּגֶד לְאֶחָד מִיְּלָדֶיהָ, וְהַבַּעַל תִּקֵּן אֶת רִשְׁתּוֹ בִּזְמַן שֶׁהַדֶּלֶת נִפְתְּחָה לְלֹא קוֹל וַאֲוִיר הַלַּיְלָה הַקַּר נָשַׁב לְתוֹךְ הַבִּקְתָּה.
"אִשְׁתִּי," אָמַר הַדַּיָּג, "גְּשִׁי וּרְאִי אִם יֵשׁ מִישֶׁהוּ בַּדֶּלֶת."
הָאִשָּׁה קָמָה וְנִגְּשָׁה אֶל הַדֶּלֶת, וְשָׁם, עַל הַסַּף, שָׁכְבָה לָהּ אוֹרְסוּלָה הַקְּטַנָּה, עֲטוּפָה בִּכְתֹנֶת הַלַּיְלָה הַלְּבָנָה שֶׁלָּהּ וּשְׁקוּעָה בְּשֵׁנָה עֲמֻקָּה.
הָאִשָּׁה פָּלְטָה צְעָקָה קְטַנָּה לְמַרְאֵה הַיַּלְדָּה.
"בַּעֲלִי, בְּחַיַּי, יֵשׁ פֹּה יַלְדָּה קְטַנָּה!" קָרְאָה, וּכְבָר הֵרִימָה אֶת הַיַּלְדָּה בִּזְרוֹעוֹתֶיהָ וְנָשְׂאָה אוֹתָהּ אֶל תּוֹךְ הַבִּקְתָּה. וְשָׁם, בַּחֲמִימוּת וּבָאוֹר, פָּקְחָה אוֹרְסוּלָה אֶת עֵינֶיהָ. הִיא הִתְיַשְּׁבָה בְּבֶהָלָה וְהִבִּיטָה סְבִיבָהּ. יֶלֶד אַחֵר בִּמְקוֹמָהּ הָיָה פּוֹרֵץ אוּלַי בְּבֶכִי, אֲבָל הִיא רַק רָעֲדָה בְּכָל גּוּפָהּ, כִּמְעַט מְבֹעֶתֶת מִכְּדֵי לְדַבֵּר.
לְמַרְבֵּה הַפֶּלֶא, שָׁכְחָה אֶת הַטִּיסָה הַמּוּזָרָה שֶׁהֵבִיאָה אוֹתָהּ לְשָׁם, וְלֹא יָכְלָה לַעֲנוֹת עַל שְׁאֵלוֹתֵיהֶם שֶׁל הַדַּיָּג וְאִשְׁתּוֹ. הִיא זָכְרָה רַק שֶׁהִיא אוֹרְסוּלָה הַנְּסִיכָה, וּכְשֶׁשָּׁמְעוּ זֹאת הָאִישׁ הַטּוֹב וְאִשְׁתּוֹ, הֵם חָשְׁבוּ שֶׁיָּצְאָה מִדַּעְתָּהּ. וְעִם זֹאת, כְּשֶׁבָּחֲנוּ אֶת כְּתֹנֶת הַלַּיְלָה שֶׁלָּהּ הָעֲשׂוּיָה פִּשְׁתָּן לָבָן מְשֻׁבָּח וְרָקוּם לְהַפְלִיא וְרָאוּ אֶת הַכֶּתֶר הַקָּטָן בְּשׁוּלֶיהָ, הִסְכִּימוּ שֶׁהִיא בַּת לְמִשְׁפָּחָה מְיֻחֶסֶת. וְכָךְ אוֹ אַחֶרֶת, הֲרֵי לֹא יְשַׁלְּחוּ אֶת הַיְּצוּר הַקָּטָן וְהַמִּסְכֵּן מִבֵּיתָם בְּלַיְלָה קַר כָּל כָּךְ. מוּטָב שֶׁיִּדְאֲגוּ לָהּ בֵּינְתַיִם, עַד שֶׁמִּישֶׁהוּ יָבוֹא לָקַחַת אוֹתָהּ. וְכָךְ הֶאֱכִילָה הָאִשָּׁה אֶת אוֹרְסוּלָה בִּקְצָת לֶחֶם וְחָלָב חָמִים וְהִשְׁכִּיבָה אוֹתָהּ בְּמִטַּת בִּתָּהּ הַקְּטַנָּה.
לְמָחֳרָת בַּבֹּקֶר, כְּשֶׁהִגִּיעוּ נְשׁוֹת הֶחָצֵר לְהָעִיר אֶת הַנְּסִיכָה, הֵן מָצְאוּ אוֹתָהּ יְשֵׁנָה כְּמוֹ תָּמִיד בְּמִטָּתָהּ הַקְּטַנָּה, וְלֹא עָלָה עַל דַּעְתָּן שֶׁזֹּאת לֹא הִיא, אֶלָּא נְסִיכַת צַעֲצוּעַ שֶׁתָּפְסָה אֶת מְקוֹמָהּ. לְמַעֲשֶׂה, הֵן הָיוּ מְאוֹד מְרֻצּוֹת, כִּי כְּשֶׁאָמְרוּ, "הִגִּיעַ הַזְּמַן שֶׁהוֹד מַעֲלָתֵךְ תָּקוּם," הִיא עָנְתָה רַק, "בְּהֶחְלֵט," וְהֵנִיחָה לָהֶן לְהַלְבִּישׁ אוֹתָהּ בְּלִי אַף מִלָּה נוֹסֶפֶת. וּכְשֶׁגַּם בַּיָּמִים הַבָּאִים הִיא לֹא הִשְׁתּוֹבְבָה כְּלָל וְרַק לְעִתִּים נְדִירוֹת פָּצְתָה אֶת פִּיהָ, הִתְרַשְּׁמוּ כֻּלָּן מִן הַשִּׁפּוּר שֶׁחָל בָּהּ, וְהַחִבָּה שֶׁרָחֲשׁוּ לָהּ הָלְכָה וְגָדְלָה.
כָּל נְשׁוֹת הֶחָצֵר סָבְרוּ שֶׁהַנְּסִיכָה הַצְּעִירָה בֹּרְכָה בַּגִּנּוּנִים הַמְּעֻדָּנִים בְּיוֹתֵר בַּמַּמְלָכָה, וַאֲפִלּוּ הַמֶּלֶךְ חִיֵּךְ וְהִבִּיט בָּהּ בְּעַיִן טוֹבָה.
וְאֵי-שָׁם בְּבִקְתַּת הַדַּיָּגִים גָּדְלָה לָהּ בֵּינְתַיִם אוֹרְסוּלָה הָאֲמִתִּית, תְּמִירָה וּזְקוּפָה כְּאַלְמוֹן וְקַלָּה וּמִתְרוֹנֶנֶת כְּצִפּוֹר.
מִכֵּיוָן שֶׁאִישׁ לֹא בָּא לְקַחְתָּהּ, אִמְּצוּ אוֹתָהּ הַדַּיָּג הַטּוֹב וְאִשְׁתּוֹ, וְהִיא גָּדְלָה בֵּין יַלְדֵיהֶם. הִיא שִׂחֲקָה אִתָּם עַל חוֹף הַיָּם וְהָלְכָה אִתָּם לְבֵית הַסֵּפֶר. חַיֶּיהָ הַקּוֹדְמִים הִתְפּוֹגְגוּ לְמִין חֲלוֹם שֶׁבְּקשִׁי זָכְרָה.
לִפְעָמִים הָיְתָה הָאֵם מוֹצִיאָה אֶת כְּתֹנֶת הַלַּיְלָה הַקְּטַנָּה הָרְקוּמָה וּמַרְאָה לָהּ אוֹתָהּ וְתוֹהָה מֵאַיִן בָּאָה וּלְמִי הִיא שַׁיֶּכֶת.
"לֹא אִכְפַּת לִי לְמִי אֲנִי שַׁיֶּכֶת," אָמְרָה אוֹרְסוּלָה. "רַק שֶׁלֹּא יָבוֹאוּ וְיִקְּחוּ אוֹתִי מִכֶּם, זֶה כָּל מָה שֶׁאִכְפַּת לִי." וּבִזְמַן שֶׁגָּדְלָה וְיָפְתָה, גָּדְלָה גַּם נְסִיכַת הַצַּעֲצוּעַ שֶׁהֶחְלִיפָה אוֹתָהּ בַּחֲצַר הַמַּלְכוּת וְנִרְאֲתָה בְּדִיּוּק כְּמוֹתָהּ, בְּהֶבְדֵּל אֶחָד; בְּעוֹד שֶׁפְּנֵי אוֹרְסוּלָה הִשְׁחִימוּ וּלְחָיֶיהָ הֶאֱדִימוּ, נוֹתְרוּ פְּנֵי הַכְּפִילָה בְּחִוְרוֹנָן, וְרַק סֹמֶק קַל הִסְתַּמֵּן בִּלְחָיֶיהָ.
הַשָּׁנִים חָלְפוּ. אוֹרְסוּלָה שֶׁבַּבִּקְתָּה הָיְתָה לְאִשָּׁה צְעִירָה וּתְמִירָה. אוֹרְסוּלָה שֶׁבָּאַרְמוֹן נֶחְשְׁבָה לַיָּפָה בַּנָּשִׁים, וְכֻלָּם הֶעֱרִיצוּ אֶת נִימוּסֶיהָ, אַף שֶׁמֵּעוֹלָם לֹא אָמְרָה דָּבָר מִלְּבַד "בִּרְשׁוּתְךָ" "לֹא תּוֹדָה," "בְּהֶחְלֵט" וְ"אָכֵן כֵּן".
הַמֶּלֶךְ הִזְדַּקֵּן בֵּינְתַיִם, וְגַם שְׂעָרָם שֶׁל מַרְק וְאִשְׁתּוֹ הֶאֱפִיר. אוֹלִיבֵר בְּכוֹרָם וּמְקוֹר גַּאֲוָתָם הוּא שֶׁיָּצָא עַכְשָׁו לַדַּיִג, ואוֹרְסוּלָה טִפְּלָה בַּזְּקֵנִים. הִיא נִקְּתָה אֶת הַבַּיִת וְהִטְלִיאָה וְתָפְרָה וְהָיְתָה כֹּה מוֹעִילָה עַד שֶׁלֹּא יָכְלוּ לְהִסְתַּדֵּר בִּלְעָדֶיהָ. הַפֵיָה טַבּוֹרֶט קָפְצָה מִדֵּי פַּעַם לְבִקּוּר מִבְּלִי שֶׁיַּבְחִינוּ בָּהּ לִבְדֹּק מָה שְׁלוֹם אוֹרְסוּלָה, וּבְכָל פַּעַם שֶׁרָאֲתָה כַּמָּה הִיא בְּרִיאָה וּמְאֻשֶּׁרֶת, חָשְׁבָה בְּסִפּוּק עַל הַגּוֹרָל הַנּוֹרָא שֶׁמִּמֶּנּוּ חִלְּצָה אוֹתָהּ. אֲבָל עֶרֶב אֶחָד, כְּשֶׁהִגִּיעָה אַחֲרֵי שֶׁנֶּעֶדְרָה זְמַן-מָה, הִבְחִינָה בְּמַשֶּׁהוּ שֶׁגָּרַם לָהּ לַעֲצֹר וְלַחְשֹׁב. אוֹלִיבֵר וְאוֹרְסוּלָה עָמְדוּ יַחַד וְהִשְׁקִיפוּ עַל הַגַּלִּים, וְטַבּוֹרֶט הִתְקָרְבָה וְצוֹתְתָה לְשִׂיחָתָם:
"כְּשֶׁנִּתְחַתֵּן," אָמַר אוֹלִיבֵר בְּרֹךְ, "נָגוּר בַּבִּקְתָּה הַקְּטַנָּה לְיָדָם כְּדֵי שֶׁנּוּכַל לְבַקֵּר אוֹתָם כָּל יוֹם. אֲבָל זֶה לֹא יִקְרֶה עַד שֶׁבֵּל הַקְּטַנָּה תִּגְדַּל וְתִתְפֹּס אֶת מְקוֹמֵךְ, כִּי אֵיךְ יוּכְלוּ אָבִי וְאִמִּי לְהִסְתַּדֵּר בִּלְעָדַיִךְ?"
"וּמוּטָב שֶׁלֹּא נַגִּיד לָהֶם שֶׁאֲנַחְנוּ רוֹצִים לְהִתְחַתֵּן," אָמְרָה אוֹרְסוּלָה, "כְּדֵי שֶׁלֹּא יִצְטַעֲרוּ שֶׁהֵם מְעַכְּבִים אוֹתָנוּ."
לְשֵׁמַע הַדְּבָרִים הָאֵלֶּה הִרְצִינָה טַבּוֹרֶט וְשָׁקְעָה בְּמַחְשָׁבוֹת עֲמֻקּוֹת. לְבַסּוֹף עָפָה בַּחֲזָרָה לָאַרְמוֹן לִבְדֹּק אֵיךְ מִתְנַהֲלִים הָעִנְיָנִים בַּצַּד הָאַחֵר שֶׁל הָעוֹלָם. הַמֶּלֶךְ הָיָה בְּעִצּוּמָהּ שֶׁל יְשִׁיבַת מוֹעֶצֶת הַמְּדִינָה. טַבּוֹרֶט מִהֲרָה לַהֲפֹךְ אֶת עַצְמָהּ לְנִרְאֵית, וְהַמֶּלֶךְ, שֶׁתָּמִיד שָׂמַח לְהֵעָזֵר בַּעֲצָתָהּ, הִזְמִין אוֹתָהּ לָשֶׁבֶת לְצִדּוֹ.
"בִּרְגָעִים אֵלֶּה מַמָּשׁ," אָמַר הוֹד מַלְכוּתוֹ, "אָנוּ עוֹמְדִים לְהַפְקִיד אֶת שַׁרְבִיטֵנוּ בְּיָדַיִם צְעִירוֹת וְנִמְרָצוֹת מִשֶּׁלָּנוּ; לְמַעֲשֶׂה, אֲנַחְנוּ חוֹשְׁבִים שֶׁאָנוּ זְקֵנִים מִכְּדֵי לִמְלֹךְ וּמִתְכַּוְּנִים לְוַתֵּר עַל כִּסְּאֵנוּ לְטוֹבַת בִּתֵּנוּ הַיְּקָרָה, שֶׁתִּמְלֹךְ בִּמְקוֹמֵנוּ." (מְלָכִים מְדַבְּרִים כָּךְ לִפְעָמִים; הֵם אוֹמְרִים "אֲנַחְנוּ" בִּמְקוֹם "אֲנִי" כְּדֵי לְהַעֲצִים אֶת חֲשִׁיבוּתָם, אֲבָל טַבּוֹרֶט לֹא הִתְבַּלְבְּלָה.)
"לִפְנֵי שֶׁאַתֶּם נוֹקְטִים צַעַד כָּזֶה, עָלֵינוּ לְנַהֵל שִׂיחָה קְטַנָּה בִּפְרָטִיּוּת," אָמְרָה וּמָשְׁכָה אֶת הַמֶּלֶךְ הַמֻּפְתָּע וְהַמֻּדְאָג לְפִנָּה שְׁקֵטָה.
אַחֲרֵי כְּמַחֲצִית הַשָּׁעָה הוּא חָזַר אֶל הַמּוֹעָצָה חִוֵּר וְנִרְעָשׁ עַד עִמְקֵי נִשְׁמָתוֹ, וּמִמְחָטָה צְמוּדָה לְעֵינוֹ.
"נִכְבָּדַי," מִלְמֵל, "אָנָּא סִלְחוּ לָנוּ עַל הִתְנַהֲגוּתֵנוּ הַבִּלְתִּי רְגִילָה. זֶה עַתָּה סָפַגְנוּ מַהֲלוּמָה אֲנוּשָׁה. הִתְבָּרֵר לָנוּ מִמָּקוֹר מֻסְמָךְ, שֶׁאֵין לְהָטִיל סָפֵק בִּמְהֵימָנוּתוֹ, שֶׁבִּתֵּנוּ, בִּתֵּנוּ הַיְּקָרָה…" וְכָאן נִשְׁנַק בְּהִתְיַפְּחוּת, וְכִמְעַט שֶׁלֹּא הִצְלִיחַ לַחְזֹר וְלִמְצֹא אֶת קוֹלוֹ – "הִיא-הִיא-בְּעֶצֶם לֹא בִּתֵּנוּ כְּלָל וּכְלָל, אֶלָּא רַק חֶזְיוֹן תַּעְתּוּעִים." וְכָאן צָנַח הַמֶּלֶךְ לְכִסְּאוֹ הֲלוּם צַעַר, וְהַפֵיָה טַבּוֹרֶט קָמָה עַל רַגְלֶיהָ וְסִפְּרָה לְאַנְשֵׁי הֶחָצֵר אֶת הַסִּפּוּר מִתְּחִלָּתוֹ; כֵּיצַד גָּנְבָה אֶת הַנְּסִיכָה הָאֲמִתִּית כְּדֵי לְהָגֵן עָלֶיהָ וְהֵנִיחָה נְסִיכַת צַעֲצוּעַ בִּמְקוֹמָהּ. אַנְשֵׁי הֶחָצֵר הֶחְלִיפוּ מַבְּטֵי פְּלִיאָה. הָיָה בָּרוּר שֶׁאֵינָם מַאֲמִינִים לְאַף מִלָּה.
"הַנְּסִיכָה הִיא בֶּאֱמֶת גְּבֶרֶת צְעִירָה וּמַקְסִימָה," אָמַר רֹאשׁ הַמֶּמְשָׁלָה.
"הַאִם יֵשׁ לְהוֹד מַלְכוּתְךָ אֵיזוֹ סִבָּה לְהִתְלוֹנֵן עַל הִתְנַהֲגוּתָהּ שֶׁל הוֹד מַעֲלָתָהּ?" שָׁאַל זְקַן הַשָּׂרִים.
"שׁוּם סִבָּה," הִתְיַפֵּחַ הַמֶּלֶךְ. "הִיא תָּמִיד הָיְתָה בַּת לְמוֹפֵת."
"אִם כָּךְ," אָמַר זְקַן הַשָּׂרִים, "אֵינִי רוֹאֶה אֵיזוֹ סִבָּה יֵשׁ לְהוֹד מַלְכוּתְךָ לָשִׂים לֵב לְמָה שֶׂהָ… אִישִׁיּוּת הַזֹּאת אוֹמֶרֶת."
"אֵינְכֶם מַאֲמִינִים לִי, שׁוֹטִים זְקֵנִים שֶׁכְּמוֹתְכֶם?" רָגְזָה טַבּוֹרֶט, "קִרְאוּ לַנְּסִיכָה, אֲנִי אוֹכִיחַ לָכֶם."
"בְּבַקָּשָׁה," הֵם קָרְאוּ בִּתְשׁוּבָה.
וְהַמֶּלֶךְ אָמְנָם זִמֵּן אֶת הוֹד מַעֲלָתָהּ.
לֹא עָבְרוּ כַּמָּה דַּקּוֹת, וְהִיא בָּאָה מֻקֶּפֶת בִּבְנוֹת לְוָיָתָהּ. הִיא לֹא אָמְרָה מִלָּה, וּבְעֶצֶם מֵעוֹלָם לֹא פָּצְתָה אֶת פִּיהָ לִפְנֵי שֶׁפָּנוּ אֵלֶיהָ. הִיא רַק נִכְנְסָה וְנֶעֶמְדָה בִּדְמָמָה בְּאֶמְצַע הַחֶדֶר.
"עָלָה עַל דַּעְתֵּנוּ שֶׁנּוֹכְחוּתֵךְ מִתְבַּקֶּשֶׁת," הִתְחִיל הַמֶּלֶךְ. אֲבָל טַבּוֹרֶט כְּבָר הִתְקַדְּמָה לְעֶבְרָהּ וּלְלֹא כָּל גִּנּוּנֵי טֶקֶס שָׁלְפָה אֶת שַׁרְבִיטָהּ וְטָפְחָה בִּקְלִילוּת עַל רֹאשָׁהּ. בִּן רֶגַע הִתְגַּלְגֵּל הָרֹאשׁ עַל הָרִצְפָּה בִּזְמַן שֶׁהַגּוּף נוֹתַר לַעֲמֹד כְּמוֹ קְלִפָּה רֵיקָה. "אָכֵן כֵּן," אָמַר הָרֹאשׁ בִּזְמַן שֶׁהִתְגַּלְגֵּל לְעֵבֶר הַמֶּלֶךְ, וְהוּא וְכָל שְׁאַר אֲנָשָׁיו כִּמְעַט הִתְעַלְּפוּ מֵרֹב פַּחַד.
"הַפֵיָה סִפְּרָה לָנוּ," הוּא אָמַר, "שֶׁאֵי-שָׁם חַיָּה נְסִיכָה אֲמִתִּית שֶׁאוֹתָהּ הִיא רוֹצָה שֶׁנְּאַמֵּץ לָנוּ כְּבַת. וּבֵינְתַיִם, הָבָה נָנִיחַ אֶת הוֹד מַעֲלָתָהּ בִּזְהִירוּת בָּאָרוֹן וְנַכְרִיז עַל אֵבֶל כְּלָלִי בְּעִקְבוֹת הָאֵרוּעַ הַמַּחְרִיד."
וּבְעוֹדוֹ מְדַבֵּר הִגְנִיב הַמֶּלֶךְ מַבָּט מְלֵא רֹךְ אֶל הַגּוּף וְאֶל הָרֹאשׁ וְעָזַב אֶת הַחֶדֶר בְּבֶכִי.
עַל פִּי הַסִּכּוּם, נִשְׁלְחָה טַבּוֹרֶט לְהָבִיא אֶת הַנְּסִיכָה אוֹרְסוּלָה בִּזְמַן שֶׁהַמֶּלֶךְ וְהַמּוֹעָצָה נֶעֶרְכוּ לְקַבֵּל אֶת פָּנֶיהָ.
עוֹד בְּאוֹתוֹ עֶרֶב עָפָה הַפֵיָה אֶל בִּקְתַּת הַדַּיָּג וְשָׁם חָשְׂפָה אֶת הָאֱמֶת עַל מוֹצָאָהּ שֶׁל אוֹרְסוּלָה וְעַל הַפְּרֵדָה הַצְּפוּיָה.
בְּצַעַר עָמֹק וּבִיבָבוֹת רָמוֹת הִתְקַבְּלוּ הַחֲדָשׁוֹת. גַּם אוֹרְסוּלָה הַמִּסְכֵּנָה עַצְמָהּ מֵרְרָה בְּבֶכִי.
"לֹא נוֹרָא," קָרְאָה לְבַסּוֹף, "אִם אֲנִי בֶּאֱמֶת כָּזֹאת נְסִיכָה חֲשׁוּבָה, אֲנִי אַזְמִין אֶת כֻּלְּכֶם לִחְיוֹת אִתִּי. אֲנִי בְּטוּחָה שֶׁאָבִי הַמֶּלֶךְ יַסְכִּים כְּשֶׁיִּשְׁמַע כַּמָּה יָפֶה דְּאַגְתֶּם לִי."
בַּמּוֹעֵד שֶׁנִּקְבַּע הִגִּיעָה טַבּוֹרֶט בְּמֶרְכָּבָה מְפֹאֶרֶת רְתוּמָה לְאַרְבָּעָה סוּסִים לְהוֹבִיל אֶת אוֹרְסוּלָה לְבֵיתָהּ הֶחָדָשׁ. זוֹ הָיְתָה נְסִיעָה אֲרֻכָּה-אֲרֻכָּה, עִם הַפְסָקָה שֶׁבָּהּ הִלְבִּישָׁה טַבּוֹרֶט אֶת הַנְּסִיכָה בְּשִׂמְלָה מַרְהִיבָה מִמֶּשִׁי לָבָן עִם סִלְסוּלֵי זָהָב וְעִטְּרָה אֶת צַוָּארָהּ וְאֶת שַׂעֲרוֹתֶיהָ בִּפְנִינִים כְּדֵי לְהַתְאִים אֶת הוֹפָעָתָהּ לַחֲצַר הַמַּלְכוּת.
בְּרֹב פְּאֵר וְהָדָר נֶאֶסְפוּ הַמֶּלֶךְ וְשָׂרָיו לְקַדֵּם אֶת פְּנֵי הַנְּסִיכָה הַחֲדָשָׁה. כָּל הַפַּרְצוּפִים עָטוּ חָמְרָה וַחֲרָדָה. לְבַסּוֹף נִפְתְּחָה הַדֶּלֶת, וטַבּוֹרֶט נִכְנְסָה כְּשֶׁהִיא מוֹלִיכָה בְּיָדָהּ אֶת הָאִשָּׁה הַצְּעִירָה.
"זֶהוּ אָבִיךְ!" אָמְרָה לְאוֹרְסוּלָה וְהוֹרְתָה עַל הַמֶּלֶךְ; וְאוֹרְסוּלָה לֹא נִזְקְקָה לְיוֹתֵר מִזֶּה כְּדֵי לָרוּץ אֵלָיו, לִכְרֹךְ אֶת זְרוֹעוֹתֶיהָ סְבִיב צַוָּארוֹ וּלְהַדְבִּיק לוֹ נְשִׁיקָה מְצַלְצֶלֶת.
הוֹד מַלְכוּתוֹ כִּמְעַט הִתְעַלֵּף, וְכָל אַנְשֵׁי הֶחָצֵר עָצְמוּ אֶת עֵינֵיהֶם בִּרְעָדָה.
"לֹא יִתָּכֵן!" אָמַר אֶחָד.
"לֹא יֵאָמֵן!" אָמַר אַחֵר.
"מָה עָשִׂיתִי?" קָרְאָה אוֹרְסוּלָה וְהֶעֱבִירָה אֶת עֵינֶיהָ מִן הָאֶחָד לַשֵּׁנִי. הָיָה בָּרוּר שֶׁטָּעֲתָה, אֲבָל בְּמָה? "הַאִם נִשַּׁקְתִּי אֶת הָאָדָם הַלֹּא נָכוֹן?" שָׁאֲלָה, וְכֻלָּם גָּנְחוּ בִּתְשׁוּבָה.
"בְּחַיַּי," קָרְאָה טַבּוֹרֶט, "אִם הִיא לֹא מוֹצֵאת חֵן בְּעֵינֵיכֶם, אֲנִי יְכוֹלָה לְהַחְזִיר אוֹתָהּ לְאֵלֶּה שֶׁמַּמָּשׁ רוֹצִים בָּהּ. אֲנִי אֶתֵּן לָכֶם שָׁבוּעַ לְהִתְעַשֵּׁת לִפְנֵי שֶׁאֶבְחַן אֶת הִתְנַהֲגוּתְכֶם. הִיא הַרְבֵּה יוֹתֵר מִדַּי טוֹבָה בִּשְׁבִיל כָּל אֶחָד וְאֶחָד מִכֶּם." וּמִיָּד פָּרְחָה לְדַרְכָּהּ בִּרְכִיבָה עַל שַׁרְבִיטָהּ וְהִשְׁאִירָה אֶת אוֹרְסוּלָה לְהִתְחַבֵּב עַל חֲבֵרֶיהָ הַחֲדָשִׁים כְּמֵיטַב יְכָלְתָּהּ. אֶלָּא שֶׁאוֹרְסוּלָה לֹא יָכְלָה לְהִתְחַבֵּב עֲלֵיהֶם כְּלָל וּכְלָל, כְּפִי שֶׁגִּלְּתָה עַד מְהֵרָה.
בְּכָל פַּעַם שֶׁפָּצְתָה אֶת פִּיהָ אוֹ זָזָה, הֵם נִרְאוּ מְזֻעְזָעִים, וְעַד מְהֵרָה נִהְיְתָה כָּל כָּךְ מְבֹהֶלֶת וּמֻטְרֶדֶת מֵהִתְנַהֲגוּתָם, שֶׁפָּרְצָה בְּבֶכִי, מָה שֶׁגָּרַם לָהֶם לְהֵרָאוֹת אֲפִלּוּ יוֹתֵר מְזֻעְזָעִים.
"זֶה בְּהֶחְלֵט שִׁנּוּי אַחֲרֵי נְסִיכָתֵנוּ הַמְּתוּקָה," סָחָה גְּבִירָה אַחַת לַחֲבֶרְתָּהּ.
"בְּהֶחְלֵט," הָיְתָה הַתְּשׁוּבָה. "הֲלוֹא אֲפִלּוּ כְּשֶׁרֹאשָׁהּ הֻפַּל, הִיא לֹא אִבְּדָה אֶת עֶשְׁתּוֹנוֹתֶיהָ וְרַק אָמְרָה, 'אָכֵן כֵּן.'"
כָּל נְשׁוֹת הֶחָצֵר סָלְדוּ מֵאוֹרְסוּלָה הַמִּסְכֵּנָה, וְהֵן לֹא אִחֲרוּ לְהַפְגִּין אֶת סְלִידָתָן. עוֹד לִפְנֵי תֹּם הַשָּׁבוּעַ שֶׁקָּצְבָה הַפֵיָה כְּבָר רָזְתָה הַנְּסִיכָה וְהֶחְוִירָה וְלֹא הֵעֵזָּה לְהָרִים אֶת קוֹלָהּ מֵעֵבֶר לִלְחִישָׁה.
"אֲבוֹי, מָה קָרָה?" קָרְאָה טַבּוֹרֶט בְּשׁוּבָהּ לְמַרְאֵה הַשִּׁנּוּי שֶׁחָל בְּאוֹרְסוּלָה הַמִּסְכֵּנָה. "לֹא טוֹב לָךְ פֹּה? הֵם לֹא נֶחְמָדִים אֵלַיִךְ?"
"קְחִי אוֹתִי הַבַּיְתָה, טַבּוֹרֶט הַיְּקָרָה," קָרְאָה אוֹרְסוּלָה בְּבֶכִי. "הַחְזִירִי אוֹתִי לְאוֹלִיבֵר וּלְפִילִיפּ וּלְבֵּל. אֲנִי שׂוֹנֵאת אֶת הָאֲנָשִׁים הָאֵלֶּה."
וּמִיָּד הִתְחַדְּשׁוּ דִּמְעוֹתֶיהָ.
טַבּוֹרֶט רַק חִיְּכָה וְלִטְּפָה אֶת רֹאשָׁהּ וְאַחַר כָּךְ הָלְכָה לְדַבֵּר עִם הַמֶּלֶךְ וַאֲנָשָׁיו.
"אוּלַי תַּסְבִּירוּ לִי," קָרְאָה, "מַדּוּעַ מְמָרֶרֶת הַנְּסִיכָה אוֹרְסוּלָה בְּבֶכִי? בָּרוּר לִי שֶׁזֹּאת אַשְׁמַתְכֶם. אֶל חֲתִיכַת עֵץ וְעוֹר בְּצוּרַת נְסִיכָה לֹא הִתְקַשֵּׁיתֶם לִהְיוֹת נֶחְמָדִים, אֲבָל כְּלַפֵּי אִשָּׁה אֲמִתִּית בָּשָׂר וָדָם אֵין לָכֶם שׁוּם חֶמְלָה."
"בִּתֵּנוּ הַיְּקָרָה עָלֶיהָ הַשָּׁלוֹם –" פָּתַח הַמֶּלֶךְ וְאָמַר, וְהַפֵיָה קָטְעָה אוֹתוֹ.
"אֲנִי בֶּאֱמֶת מַאֲמִינָה," אָמְרָה, "שֶׁאַתָּה רוֹצֶה אֶת הַבֻּבָּה שֶׁלְּךָ בַּחֲזָרָה, וְלָכֵן אֶתֵּן לְךָ לִבְחֹר. אֶת מִי אַתָּה מַעֲדִיף: אֶת אוֹרְסוּלָה שֶׁלִּי, הַנְּסִיכָה הָאֲמִתִּית, אוֹ אֶת אוֹרְסוּלָה שֶׁלְּךָ, הַמְּזֻיֶּפֶת?"
הַמֶּלֶךְ שָׁקַע בְּכִסְּאוֹ. "זֶה יוֹתֵר מִדַּי בִּשְׁבִילִי. קִרְאִי לְחַבְרֵי הַמּוֹעָצָה, שֶׁהֵם יַצְבִּיעוּ עַל זֶה."
וְהַפֵיָה אָמְנָם זִמְּנָה אֶת חַבְרֵי הַמּוֹעָצָה וְהִסְבִּירָה לָהֶם בְּמָה מְדֻבָּר.
"הָבִיאוּ לְכָאן אֶת שְׁתֵּי הַנְּסִיכוֹת," אָמְרָה. נְסִיכַת הַצַּעֲצוּעַ נִשְׁלְפָה בִּזְהִירוּת רַבָּה מִן הָאָרוֹן שֶׁבּוֹ אֻחְסְנָה, וְעַל הַשֻּׁלְחָן לְצִדָּהּ הֻנַּח רֹאשָׁהּ. גַּם הַנְּסִיכָה הָאֲמִתִּית הִגִּיעָה בְּעֵינַיִם אֲדֻמּוֹת מִבֶּכִי וְבְחָזֶה עוֹלֶה וְיוֹרֵד.
"נִדְמֶה לִי שֶׁאֵין סָפֵק מִי מֵהֶן עֲדִיפָה," אָמַר רֹאשׁ הַמֶּמְשָׁלָה לִזְקַן הַשָּׂרִים.
"נִדְמֶה לִי שֶׁדַּעְתִּי כְּדַעְתְּךָ," אָמַר זְקַן הַשָּׂרִים.
"אָז אֶפְשָׁר לָגֶשֶׁת לַהַצְבָּעָה," אָמְרָה טַבּוֹרֶט. וְכֻלָּם הִצְבִּיעוּ בְּעַד אוֹרְסוּלָה הַמְּזֻיֶּפֶת, אִישׁ לֹא בָּחַר בָּאֲמִתִּית. טַבּוֹרֶט פָּרְצָה בִּצְחוֹק.
"אַתֶּם חֲבוּרָה שֶׁל שׁוֹטִים רְפֵי שֵׂכֶל," אָמְרָה, "אֲבָל אַתֶּם תְּקַבְּלוּ אֶת מְבֻקַּשְׁכֶם." הִיא הֵרִימָה אֶת רֹאשָׁהּ שֶׁל הַנְּסִיכָה הַמְּזֻיֶּפֶת וּבְהֶנֵּף שַׁרְבִיט חִבְּרָה אוֹתוֹ לְגוּפָהּ. הוּא סָב לְאִטּוֹ וְאָמַר, "בְּהֶחְלֵט." וּלְשֵׁמַע הַקּוֹל הַיָּשָׁן וְהַמֻּכָּר פָּרְצוּ אַנְשֵׁי הֶחָצֵר בְּמִין תְּרוּעָה, כְּכָל שֶׁהִתִּירוּ לָהֶם כְּלָלֵי הַנִּימוּס, וְהַמֶּלֶךְ נֶאֱלַם דֹּם מֵרֹב אשֶׁר.
"מִיָּד נִפְתַּח בַּהֲכָנוֹת," קָרָא, "לְוַתֵּר עַל כִּתְרֵינוּ­ וּלְהַעֲבִיר אֶת הַשִּׁלְטוֹן לִידֵי בִּתֵּנוּ הַיְּקָרָה."
לְשֵׁמַע הַדְּבָרִים הַלָּלוּ חָזְרוּ אַנְשֵׁי הֶחָצֵר וְהֵרִיעוּ.
טַבּוֹרֶט לְבַדָּהּ הִצְטַחֲקָה בְּבוּז. הִיא אָסְפָה אֶת אוֹרְסוּלָה הָאֲמִתִּית בִּזְרוֹעוֹתֶיהָ וְעָפָה בַּחֲזָרָה אֶל הַבִּקְתָּה שֶׁל מַרְק.
בָּעֶרֶב הוּאֲרָה הָעִיר, וַחֲגִיגוֹת גְּדוֹלוֹת נֶעֶרְכוּ לִכְבוֹד הַחְלָמָתָהּ שֶׁל הַנְּסִיכָה, וְאִלּוּ אוֹרְסוּלָה נִשְׁאֲרָה בַּבִּקְתָּה. הִיא נִשְּׂאָה לְאוֹלִיבֵר וְחַיָּה אִתּוֹ בְּאשֶׁר עַד סוֹף יָמֶיהָ.
