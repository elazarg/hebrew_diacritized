 , אֲנִי פּוֹצֵחַ בְּשִׁיר וְיוֹצֵא לַדֶּרֶךְ.
, פֶּטְקָה נִגָּשׁ אֵלַי וְאוֹמֵר, "אֲנִי אֵלֵךְ אִתְּךָ," וּשְׁנֵינוּ הוֹלְכִים וְשָׁרִים שִׁירִים.
, כְּשֶׁאֲנַחְנוּ צוֹעֲדִים, אֲנַחְנוּ רוֹאִים אִישׁ לֹא גָּבוֹהַּ יוֹתֵר מִדְּלִי עוֹמֵד בְּאֶמְצַע הַדֶּרֶךְ.
"מִי אַתָּה?" אֲנַחְנוּ שׁוֹאֲלִים.
"אֲנִי הָאִישׁ הֲכִי קָטָן בָּעוֹלָם."
"בּוֹא אִתָּנוּ."
"בּוֹאוּ נֵלֵךְ."
וַאֲנַחְנוּ הוֹלְכִים לָנוּ הָלְאָה. אֲבָל הָאִישׁ הַקָּטָן לֹא מַצְלִיחַ לְהַדְבִּיק אוֹתָנוּ. הוּא רָץ בַּמְּהִירוּת הֲכִי גְּדוֹלָה שֶׁהוּא מְסֻגָּל לָהּ, וּבְכָל זֹאת מְפַגֵּר אַחֲרֵינוּ. אָז אֲנַחְנוּ נוֹתְנִים לוֹ יָד: פֶּטְקָה לוֹקֵחַ אֶת יַד יָמִין שֶׁלּוֹ, וַאֲנִי אֶת יַד שְׂמֹאל שֶׁלּוֹ. הָאִישׁ הַקָּטָן נִתְלֶה בֵּינֵינוּ כְּשֶׁרַגְלָיו בְּקשִׁי נוֹגְעוֹת בָּאֲדָמָה. וְכָכָה אֲנַחְנוּ הוֹלְכִים הָלְאָה, כָּל הַשְּׁלוֹשָׁה, וּמְשַׁרְקְקִים שִׁירִים.
, אֲנַחְנוּ מַמְשִׁיכִים כָּכָה הָלְאָה, וּמָה אֲנַחְנוּ רוֹאִים? אִישׁ שׁוֹכֵב בְּצַד הַדֶּרֶךְ. רֹאשׁוֹ מֻנָּח עַל גֶּדֶם עֵץ, וְכָל כֻּלּוֹ אָרֹךְ כָּל כָּךְ, עַד שֶׁאֲנַחְנוּ לֹא יְכוֹלִים לִרְאוֹת אֵיפֹה הָרַגְלַיִם שֶׁלּוֹ נִגְמָרוֹת. אֲנַחְנוּ מִתְקָרְבִים אֵלָיו, וְאָז הוּא מְזַנֵּק וְנֶעֱמָד עַל הָרַגְלַיִם וְהוֹלֵם פַּעַם אַחַת בְּאֶגְרוֹפוֹ עַל גֶּדֶם הָעֵץ, וְהַגֶּדֶם נֶעֱלָם מִתַּחַת לָאֲדָמָה.
וְאָז הָאִישׁ הָאָרֹךְ מַבִּיט סְבִיבוֹ, רוֹאֶה אוֹתָנוּ וְאוֹמֵר, "מִי אַתֶּם שֶׁמַּפְרִיעִים לִי לִישֹׁן?"
"אֲנַחְנוּ שְׁלוֹשָׁה חֶבְרֶה לָעִנְיָן," אֲנַחְנוּ אוֹמְרִים. "רוֹצֶה לָבוֹא אִתָּנוּ?"
"בְּסֵדֶר," הָאִישׁ הָאָרֹךְ אוֹמֵר, פּוֹסֵעַ פְּסִיעָה אַחַת וּמִתְקַדֵּם בְּאֵיזֶה עֶשְׂרִים מֵטֶר בְּבַת אַחַת.
"הֵי," הָאִישׁ הַקָּטָן צוֹעֵק לוֹ, "חַכֵּה לָנוּ!"
אֲנַחְנוּ תּוֹפְסִים אֶת הָאִישׁ הַקָּטָן וְרָצִים אֶל הָאָרֹךְ.
"לֹא," אֲנַחְנוּ אוֹמְרִים, "אַתָּה לֹא יָכוֹל לָלֶכֶת כָּכָה. לֵךְ בִּצְעָדִים קְטַנִּים."
הָאִישׁ הָאָרֹךְ מַתְחִיל לָלֶכֶת בִּצְעָדִים קְטַנִּים, אֲבָל זֶה לֹא עוֹזֵר. עֲשָׂרָה צְעָדִים – וְהוּא נֶעֱלָם מֵהָעַיִן.
"אִם כָּכָה," אֲנַחְנוּ אוֹמְרִים, "תֵּן לָאִישׁ הַקָּטָן לָשֶׁבֶת לְךָ עַל הַכָּתֵף, וְאוֹתָנוּ קַח תַּחַת הַזְּרוֹעוֹת."
הָאִישׁ הָאָרֹךְ מוֹשִׁיב אֶת הָאִישׁ הַקָּטָן עַל כְּתֵפוֹ, לוֹקֵחַ אוֹתָנוּ תַּחַת זְרוֹעוֹתָיו, וְכֻלָּנוּ מַמְשִׁיכִים בַּדֶּרֶךְ.
"נוֹחַ לְךָ?" אֲנִי שׁוֹאֵל אֶת פֶּטְקָה.
"מְאוֹד. וּלְךָ?"
"גַּם לִי נוֹחַ."
אֲנַחְנוּ פּוֹצְחִים בִּשְׁרִיקַת מַנְגִּינוֹת עַלִּיזוֹת. הָאִישׁ הָאָרֹךְ הוֹלֵךְ וּמְשַׁרְקֵק מַנְגִּינוֹת, וְהָאִישׁ הַקָּטָן יוֹשֵׁב לוֹ עַל הַכָּתֵף וּמְשַׁרְקֵק בְּכָל מְאוֹדוֹ.
, אֲנַחְנוּ מַמְשִׁיכִים לָלֶכֶת, וּמָה אֲנַחְנוּ רוֹאִים? חֲמוֹר עוֹמֵד מַמָּשׁ בְּאֶמְצַע הַדֶּרֶךְ. זֶה מְשַׂמֵּחַ אוֹתָנוּ, וַאֲנַחְנוּ מַחְלִיטִים לִרְכֹּב עָלָיו. הָרִאשׁוֹן שֶׁמְּנַסֶּה הוּא הָאִישׁ הָאָרֹךְ. הוּא מֵרִים רֶגֶל מֵעַל גַּב הַחֲמוֹר, אֲבָל הַחֲמוֹר אֲפִלּוּ לֹא מַגִּיעַ לוֹ אֶל מִתַּחַת לַבִּרְכַּיִם. אֵיךְ שֶׁהָאִישׁ הָאָרֹךְ רוֹצֶה לָשֶׁבֶת עָלָיו, הַחֲמוֹר קָם וְהוֹלֵךְ, וְהָאִישׁ הָאָרֹךְ נוֹפֵל אַרְצָה בְּכָל הַכּוֹחַ. אָז אֲנַחְנוּ מְנַסִּים לְהוֹשִׁיב אֶת הָאִישׁ הַקָּטָן עַל הַחֲמוֹר. אֲבָל אֵיךְ שֶׁהַחֲמוֹר עוֹשֶׂה כַּמָּה צְעָדִים, הָאִישׁ הַקָּטָן מְאַבֵּד שִׁוּוּי מִשְׁקָל וְצוֹנֵחַ אַרְצָה. הוּא קָם וְאוֹמֵר, "שֶׁהָאִישׁ הָאָרֹךְ שׁוּב יִשָּׂא אוֹתִי עַל הַכָּתֵף, וְאַתָּה וּפֶּטְקָה תִּרְכְּבוּ עַל הַחֲמוֹר."
אֲנַחְנוּ מִתְיַשְּׁבִים כְּמוֹ שֶׁהָאִישׁ הַקָּטָן אוֹמֵר, וְרוֹכְבִים לָנוּ שְׂמֵחִים וּמְשַׁרְקְקִים מַנְגִּינוֹת.
, אֲנַחְנוּ מַגִּיעִים לַאֲגַם גָּדוֹל. עַל גְּדַת הָאֲגַם יֵשׁ סִירָה.
"בּוֹאוּ נִכָּנֵס לַסִּירָה," פֶּטְקָה אוֹמֵר. אֲנִי וּפֶּטְקָה מִתְיַשְּׁבִים בַּסִּירָה בְּנוֹחוּת, אֲבָל אֶת הָאִישׁ הָאָרֹךְ אֲנַחְנוּ מוֹשִׁיבִים בְּקשִׁי. הוּא נֶאֱלָץ לְהִשְׁתּוֹפֵף, לְהִתְכַּוֵּץ, לְהָרִים בִּרְכַּיִם מַמָּשׁ עַד לַסַּנְטֵר.
הָאִישׁ הַקָּטָן מִתְיַשֵּׁב אֵיפְשֶׁהוּ מִתַּחַת לַסַּפְסָל, וְלַחֲמוֹר לֹא נוֹתָר מָקוֹם. אִלְמָלֵא הָיִינוּ מוֹשִׁיבִים אֶת הָאִישׁ הָאָרֹךְ בַּסִּירָה, עוֹד הָיָה אֶפְשָׁר לְהוֹשִׁיב בָּהּ אֶת הַחֲמוֹר. אֲבָל שְׁנֵיהֶם יַחַד לֹא נִכְנָסִים.
"אַתֶּם יוֹדְעִים מָה?" הָאִישׁ הַקָּטָן אוֹמֵר, "אִם אַתָּה, הָאָרֹךְ, תְּפַלֵּס לְךָ דֶּרֶךְ בָּאֲגַם, נוּכַל לָקַחַת אֶת הַחֲמוֹר בַּסִּירָה, וּלְכֻלָּנוּ יִהְיֶה מָקוֹם."
אֲנַחְנוּ מַעֲלִים אֶת הַחֲמוֹר לַסִּירָה, וְהָאִישׁ הָאָרֹךְ מְפַלֵּס לוֹ דֶּרֶךְ בָּאֲגַם וְגוֹרֵר אַחֲרָיו אֶת הַסִּירָה שֶׁלָּנוּ בְּחֶבֶל. הַחֲמוֹר יוֹשֵׁב, מְפַחֵד לָזוּז. זוֹ כַּנִּרְאֶה הַפַּעַם הָרִאשׁוֹנָה שֶׁלּוֹ בְּסִירָה. לָאֲחֵרִים טוֹב, וַאֲנַחְנוּ מִתְקַדְּמִים לְאֹרֶךְ הָאֲגַם וּמְשַׁרְקְקִים מַנְגִּינוֹת. גַּם הָאִישׁ הָאָרֹךְ מְשַׁרְקֵק שִׁירִים בִּזְמַן שֶׁהוּא גּוֹרֵר אֶת הַסִּירָה שֶׁלָּנוּ. 
, אֲנַחְנוּ יוֹצְאִים לַגָּדָה הַנֶּגְדִּית, וּמוּלֵנוּ עוֹמֵד אוֹטוֹ.
"מָה זֶה יָכוֹל לִהְיוֹת?" הָאִישׁ הָאָרֹךְ שׁוֹאֵל.
"מָה זֶה?" הָאִישׁ הַקָּטָן שׁוֹאֵל.
"זֶה," אֲנִי אוֹמֵר, "אוֹטוֹ".
"אוֹטוֹ שֶׁנִּסַּע בּוֹ עַכְשָׁו", פֶּטְקָה אוֹמֵר.
אֲנַחְנוּ מִתְיַשְּׁבִים בָּאוֹטוֹ.
אֲנִי וּפֶּטְקָה מִתְיַשְּׁבִים מֵאֲחוֹרֵי הַהֶגֶה, הָאִישׁ הַקָּטָן מְטַפֵּס עַל הַפָּנָס הַקִּדְמִי, אֲבָל לָאִישׁ הָאָרֹךְ וְלַחֲמוֹר אֵין מַסְפִּיק מָקוֹם בָּאוֹטוֹ.
אֲנַחְנוּ מַכְנִיסִים אֶת הַסִּירָה לָאוֹטוֹ, מַעֲמִידִים אֶת הַחֲמוֹר בְּתוֹךְ הַסִּירָה – וְהַכֹּל טוֹב, רַק שֶׁלָּאִישׁ הָאָרֹךְ אֵין מָקוֹם.
אֲנַחְנוּ מוֹשִׁיבִים בָּאוֹטוֹ אֶת הַחֲמוֹר וְאֶת הָאִישׁ הָאָרֹךְ – אֲבָל לַסִּירָה אֵין מָקוֹם.
אֲנַחְנוּ מִתְבַּלְבְּלִים לְגַמְרֵי, כְּבָר לֹא יוֹדְעִים מָה לַעֲשׂוֹת.
וְהִנֵּה הָאִישׁ הַקָּטָן עָץ לָנוּ עֵצָה, "שֶׁהָאִישׁ הָאָרֹךְ יֵשֵׁב בָּאוֹטוֹ, אֶת הַחֲמוֹר יָנִיחַ עַל הַבִּרְכַּיִם, וְאֶת הַסִּירָה יָרִים מֵעַל הָרֹאשׁ."
אֲנַחְנוּ מוֹשִׁיבִים אֶת הָאִישׁ הָאָרֹךְ בַּסִּירָה, מְנִיחִים לוֹ אֶת הַחֲמוֹר עַל הַבִּרְכַּיִם וְנוֹתְנִים לוֹ לְהַחְזִיק אֶת הַסִּירָה בַּיָּדַיִם.
"לֹא כָּבֵד?" הָאִישׁ הַקָּטָן שׁוֹאֵל אוֹתוֹ. 
"לֹא, לֹא נוֹרָא," הָאִישׁ הָאָרֹךְ אוֹמֵר.
אֲנִי מַכְנִיס הִלּוּךְ, וְכֻלָּנוּ יוֹצְאִים לַדֶּרֶךְ. טוֹב לְכֻלָּם, רַק לָאִישׁ הַקָּטָן לֹא נוֹחַ עַל הַפָּנָס מִקָּדִימָה. הוּא מְקַפֵּץ מִצַּד לְצַד כְּמוֹ נַחוּם-תָּקוּם. לְכָל הַשְּׁאָר זֶה לֹא מַפְרִיעַ, וַאֲנַחְנוּ נוֹסְעִים לָנוּ וּמְשַׁרְקְקִים מַנְגִּינוֹת.
, אֲנַחְנוּ מַגִּיעִים לְאֵיזוֹ עִיר. בְּשָׁעָה שֶׁאֲנַחְנוּ נוֹסְעִים בָּרְחוֹבוֹת, אֲנָשִׁים מַבִּיטִים בָּנוּ וּמַצְבִּיעִים עָלֵינוּ. "תִּרְאוּ מָה זֶה," הֵם אוֹמְרִים, "יוֹשֵׁב אֵיזֶה עֲנָק בָּאוֹטוֹ, שָׂם לוֹ חֲמוֹר עַל הַבִּרְכַּיִם וּמַחְזִיק סִירָה מֵעַל הָרֹאשׁ. חָה! חָה! חָה! וְזֶה שֶׁעַל הַפָּנָס! הוּא לֹא גָּבוֹהַּ יוֹתֵר מִדְּלִי! תִּרְאוּ אֵיךְ הוּא מְקַפֵּץ מִצַּד לְצַד! חָה! חָה! חָה!"
אֲנַחְנוּ חוֹנִים מוּל מָלוֹן, מְנִיחִים אֶת הַסִּירָה עַל הָאָרֶץ, שָׂמִים אֶת הָאוֹטוֹ בַּגָּרָז', קוֹשְׁרִים אֶת הַחֲמוֹר לְעֵץ וְקוֹרְאִים לְבַעַל הַמָּלוֹן.
יוֹצֵא אֵלֵינוּ בַּעַל הַמָּלוֹן וְאוֹמֵר, "אֵיךְ אֲנִי יָכוֹל לַעֲזֹר לָכֶם?"
"תִּשְׁמַע", אֲנַחְנוּ אוֹמְרִים לוֹ, "בְּסֵדֶר מִבְּחִינָתְךָ שֶׁנִּישַׁן כָּאן הַלַּיְלָה?"
"לְגַמְרֵי בְּסֵדֶר," אוֹמֵר בַּעַל הַמָּלוֹן וּמוֹבִיל אוֹתָנוּ לְחֶדֶר וּבוֹ אַרְבַּע מִטּוֹת.
אֲנִי וּפֶּטְקָה נִשְׁכָּבִים, וְאִלּוּ הָאִישׁ הָאָרֹךְ וְהָאִישׁ הַקָּטָן לֹא מַצְלִיחִים לִשְׁכַּב בְּשׁוּם אֹפֶן. כָּל הַמִּטּוֹת קְצָרוֹת לָאִישׁ הָאָרֹךְ, וְלָאִישׁ הַקָּטָן אֵין עַל מָה לְהָנִיחַ אֶת הָרֹאשׁ, כִּי הַכָּרִית גְּבוֹהָה מִמֶּנּוּ, וְהוּא יָכוֹל לְהִשָּׁעֵן עַל הַכָּרִית רַק בַּעֲמִידָה. אֲבָל מֵאַחַר שֶׁכֻּלָּנוּ עֲיֵפִים מְאוֹד, אֵיכְשֶׁהוּ אֲנַחְנוּ נִשְׁכָּבִים וְנִרְדָּמִים. הָאִישׁ הָאָרֹךְ פָּשׁוּט נִשְׁכָּב עַל הָרִצְפָּה, וְהָאִישׁ הַקָּטָן עוֹלֶה כֻּלּוֹ עַל הַכָּרִית וְנִרְדָּם שָׁם.
, אֲנַחְנוּ מִתְעוֹרְרִים לְמָחֳרָת בַּבֹּקֶר וּמַחְלִיטִים לְהַמְשִׁיךְ בַּדֶּרֶךְ. לְפֶתַע פִּתְאוֹם הָאִישׁ הַקָּטָן אוֹמֵר, "יוֹדְעִים מָה? נִמְאַס לִי לְהִסָּחֵב עִם הַסִּירָה הַזֹּאת, וְנִמְאַס לִי מֵהָאוֹטוֹ. בּוֹאוּ נֵלֵךְ בָּרֶגֶל."
"אֲנִי לֹא הוֹלֵךְ בָּרֶגֶל," הָאִישׁ הָאָרֹךְ אוֹמֵר, "אֲנִי אֶתְעַיֵּף מִדַּי."
"אַתָּה, עֲנָק מְגֻדָּל שֶׁכְּמוֹתְךָ, תִּתְעַיֵּף?" הָאִישׁ הַקָּטָן מִצְטַחֵק.
"בֶּטַח שֶׁאֶתְעַיֵּף," הָאִישׁ הָאָרֹךְ אוֹמֵר. "הַלְוַאי שֶׁהָיִיתִי מוֹצֵא סוּס שֶׁיַּתְאִים לִי בַּגֹּדֶל."
"אֵיפֹה תִּמְצָא סוּס שֶׁיַּתְאִים לְךָ?" פֶּטְקָה מִתְעָרֵב. "אַתָּה צָרִיךְ פִּיל, לֹא סוּס."
"וְכָאן לֹא תַּשִּׂיג פִּיל," אֲנִי אוֹמֵר, "כָּאן זֶה לֹא אַפְרִיקָה."
אֲנִי רַק אוֹמֵר אֶת זֶה, וְהִנֵּה נִשְׁמָעוֹת מֵהָרְחוֹב נְבִיחוֹת וַהֲמֻלָּה וּצְעָקוֹת. אֲנַחְנוּ מַבִּיטִים מִבַּעַד לַחַלּוֹן וּלְנֶגֶד עֵינֵינוּ פִּיל צוֹעֵד לוֹ בָּרְחוֹב וּבְעִקְבוֹתָיו הֶהָמוֹן. לְרַגְלֵי הַפִּיל רָץ כְּלַבְלַב וְנוֹבֵחַ בְּכָל כּוֹחוֹ, וְאִלּוּ הַפִּיל הוֹלֵךְ לוֹ בְּרֹגַע, לֹא שָׂם לֵב לְאַף אֶחָד.
"הִנֵּה," הָאִישׁ הַקָּטָן אוֹמֵר לָאִישׁ הָאָרֹךְ, "הִנֵּה לְךָ פִּיל, בְּדִיּוּק בַּזְּמַן. עֲלֵה עָלָיו וּרְכַב."
"וְאַתָּה, רְכַב עַל הַכְּלַבְלַב. הוּא בְּדִיּוּק בַּגֹּדֶל שֶׁלְּךָ," הָאִישׁ הָאָרֹךְ אוֹמֵר.
"נָכוֹן," אֲנִי אוֹמֵר. "הָאִישׁ הָאָרֹךְ יִרְכַּב עַל הַפִּיל, הָאִישׁ הַקָּטָן יִרְכַּב עַל הַכְּלַבְלַב, וּפֶּטְקָה וַאֲנִי נִרְכַּב עַל הַחֲמוֹר." וַאֲנַחְנוּ יוֹצְאִים בְּרִיצָה לָרְחוֹב.
, אֲנַחְנוּ רָצִים אֶל הָרְחוֹב. פֶּטְקָה וַאֲנִי מִתְיַשְּׁבִים עַל הַחֲמוֹר, הָאִישׁ הַקָּטָן נִשְׁאָר לְיַד הַשַּׁעַר, וְהָאִישׁ הָאָרֹךְ רָץ אַחֲרֵי הַפִּיל. כְּשֶׁהוּא מַגִּיעַ לַפִּיל, הוּא קוֹפֵץ עָלָיו וְחוֹזֵר אֵלֵינוּ. הַכְּלַבְלַב רָץ אַחֲרֵי הַפִּיל, נוֹבֵחַ בְּכָל מְאוֹדוֹ. אֵיךְ שֶׁהֵם מַגִּיעִים לְשַׁעַר, הָאִישׁ הַקָּטָן מִזְדָּרֵז וְקוֹפֵץ עַל הַכְּלַבְלַב. וְכָכָה כֻּלָּנוּ מַמְשִׁיכִים בַּדֶּרֶךְ: מִקָּדִימָה הָאִישׁ הָאָרֹךְ עַל הַפִּיל, אַחֲרָיו פֶּטְקָה וַאֲנִי עַל הַחֲמוֹר, וְאַחֲרוֹן – הָאִישׁ הַקָּטָן עַל הַכְּלַבְלַב. טוֹב לְכֻלָּנוּ וְשָׂמֵחַ, וַאֲנַחְנוּ מְשַׁרְקְקִים שִׁירִים.
אֲנַחְנוּ יוֹצְאִים מֵהָעִיר וְנוֹסְעִים לָנוּ. לְאָן אֲנַחְנוּ מַגִּיעִים וּמָה קוֹרֶה לָנוּ בַּהֶמְשֵׁךְ – זֶה כְּבָר סִפּוּר אַחֵר.
