בַּלַּיְלָה שׁוּב קִיבַּלְתִּי אוֹמֶץ, לָקוּם לִמְרוֹד בְּמַשֶּׁהוּ
לָבוֹא חָזָק מִתּוֹךְ הַחוֹשֶׁךְ, לַעֲשׂוֹת אֶת הַשִּׁינּוּי
הִבַּטְתִּי בְּעֵינֵי הַכֶּלֶב, וּבְאוֹיֵב הֲכִי שָׂנוּא
חָשַׁבְתִּי הָעוֹלָם בֵּית כֶּלֶא, זֶה לֹא חָכְמָה לִהְיוֹת עָצוּב.
אַף פַּעַם לֹא הָיָה לִי אוֹמֶץ, לִסְחוֹב אוֹתְךָ חֲצִי חַי מֵת
כְּמוֹ בַּשִּׁיר הַמְּפוּרְסָם שֶׁל הַ"הוֹלִיס", אַתָּה אָחִי אַתָּה לֹא כָּבֵד
בּוֹדֵד בּוֹדֵד אַךְ פּוֹפּוּלָרִי, מֵחוֹף הַיָּם בְּאִיסְטַנְבּוּל
כָּתַבְתָּ לִי חוֹשֵׁב עָלֶיךָ, כָּתַבְתִּי לְךָ חוֹשֵׁב כָּפוּל.
וּבְקָרוֹב אָהֲבָה לָחַשְׁתִּי לָךְ, וּבְקָרוֹב יָבוֹא אָבִיב
הַלְּוַואי יִהְיֶה לָנוּ עוֹד יוֹם מוּצְלָח
שֶׁאֶת חַיֵּינוּ כָּאן יָעִיף
דְּמָעוֹת יוֹרְדוֹת עַל הַיְּקוּם, בֵּינָתַיִים
הַלֵּב שֶׁלִּי רוֹצֶה כַּדּוּר, בֵּינָתַיִים
וְאוֹמֶץ עוֹד פַּעַם אוֹמֶץ
כְּשֶׁכְּלוּם כְּבָר לֹא דּוֹמֶה לִכְלוּם.
בַּלַּיְלָה שׁוּב קִיבַּלְתִּי אוֹמֶץ, לְהִתְאַהֵב בָּךְ בַּמַּהוּת
חַיֵּיךְ חַיֵּיךְ לָחַשְׁתְּ בַּחוֹשֶׁךְ, רַק בִּגְלָלֵךְ חִייַּכְתִּי שׁוּב
אַחַר כָּךְ כְּשֶׁהוֹרַדְתְּ כּוּתּוֹנֶת, רַק הִתְקָרַבְתִּי כְּדֵי לוֹמַר
תּוֹדָה לָךְ שֶׁנָּתַתָּ לִי אוֹמֶץ, לִחְיוֹת כְּאִילּוּ יֵשׁ מָחָר.
וּבְקָרוֹב אָהֲבָה לָחַשְׁתִּי לָךְ, וּבְקָרוֹב יָבוֹא אָבִיב
הַלְּוַואי יִהְיֶה לָנוּ עוֹד יוֹם מוּצְלָח, שֶׁאֶת חַיֵּינוּ כָּאן יָעִיף
דְּמָעוֹת יוֹרְדוֹת עַל הַיְּקוּם, בֵּינָתַיִים
הַלֵּב שֶׁלִּי רוֹצֶה כַּדּוּר, בֵּינָתַיִים
וְאוֹמֶץ עוֹד פַּעַם אוֹמֶץ כְּשֶׁכְּלוּם כְּבָר לֹא דּוֹמֶה לִכְלוּם.
