יֵשׁ אֲנָשִׁים שֶׁרָצִים לָגַעַת
יֵשׁ אֲנָשִׁים שֶׁקּוֹרְאִים לוֹ סְתָם
אַחֲרֵי חֲצוֹת דַּרְכּוֹ יוֹדַעַת
אֶת לִיבּוֹ הֶעָצוּב הַשָּׁר

בַּדֶּרֶךְ לַבָּמָה יֵשׁ אֶלֶף
פַּרְצוּפִים זוֹרְחֵי חִיּוּךְ
מִי שֶׁנָּע כְּמוֹ אֶבֶן מִתְגַּלְגֶּלֶת
שָׁב הַבַּיְתָה עִם פַּרְצוּף מָעוּךְ

שֵׂיעָר גּוֹלֵשׁ פֵּאָה
עִיכּוּז שֶׁל הַמְּעַט
מַבָּט בָּעֲנָנִים וְצַוָּואר עָטוּף
יֵשׁ לוֹ חִיּוּךְ שֶׁאוֹמֵר אוֹתָךְ
גּוּף אָרוֹךְ נִמְתָּח
מֵאֲחוֹרֵי הַזְּמַן שָׁר כּוֹכָב עָצוּב

אִישׁ בּוֹדֵד בַּלֵּילוֹת נוֹסֵעַ
הָרַדְיוֹ מְזַמְזֵם שִׁירִים
אִישׁ בּוֹדֵד רַק הוּא שׁוֹמֵעַ
אֵיךְ חַדְרֵי לִיבּוֹ שָׁרִים

רוֹכֵב בִּמְכוֹנִית
אֶל הָאוֹרוֹת מַבִּיט
מוֹתִיר תַ'לֵּב עָצוּב מִשִּׁיר דִּמְעָה
רַק הַיָּרֵחַ מְטוֹרָף
רוֹשֵׁם לוֹ שִׁיר נוֹסָף
עַל דְּמָעוֹת כּוֹכָב נוֹדֵד שֶׁבַּמַּרְאָה

אִם אַתְּ רוֹצָה לְעִתִּים לָגַעַת
בְּאִישׁ שֶׁמְּזַמְזֵם שִׁירִים
הַרְחֵק הַרְחֵק דִּמְעָה נוֹסַעַת
צוֹרֶבֶת אֶלֶף מֵיתָרִים

רוֹכֵב בַּמְּכוֹנִית
אֶל הָאוֹרוֹת מַבִּיט
מוֹתִיר תַ'לֵּב עָצוּב מִשִּׁיר דִּמְעָה
רַק הַיָּרֵחַ מְטוֹרָף
רוֹשֵׁם לוֹ שִׁיר נוֹסָף
