"בֶּעָתִיד נַצִּיעַ אֶת בִּיטּוּחַ הַחוֹבָה הַזּוֹל בְּיוֹתֵר בָּאָרֶץ", כָּךְ אוֹמֵר קוֹבִּי גִּילוֹן, סָמַנְכַּ"ל עוֹפֶר-אַבְנִיר, יְבוּאָנִית מוּתָג הַקַּטְנוֹעִים קִימְקוֹ.
גִּילוֹן אָמַר אֶת הַדְּבָרִים בְּמַהֲלַךְ אֵירוּעַ בּוֹ נִפְגְּשׁוּ נָצִיגֵי הַתִּקְשׁוֹרֶת עִם בְּכִירֵי הַנְהָלַת הַמּוּתָג הַטַּיְיוָואנִי.
הַצַּעַד מַגִּיעַ כִּתְגוּבָה לְמִתְחָרָה הַגְּדוֹלָה שֶׁל אַבְנִיר, חֶבְרַת מֶטְרוֹ, הִצִּיעָה זֶה מִכְּבָר בִּיטּוּחַ חוֹבָה שֶׁכְּזֶה.
מֶטְרוֹ מְיַיבֵּאת אַרְצָה אֶת מוּתָג הַקַּטְנוֹעִים סָאנְיָאנְג, הַנִּמְכָּר בְּיוֹתֵר בְּיִשְׂרָאֵל.
צַחִי גִּילוֹן, מִבַּעֲלֵי עוֹפֶר-אַבְנִיר, אָמַר כִּי בְּתָכְנִיּוֹת הַחֶבְרָה לַהֲפוֹךְ לַמּוּתָג הַמּוֹבִיל בְּיִשְׂרָאֵל כְּבָר בְּ-2011.

עוֹד חָשַׂף גִּילוֹן אֶת נְתוּנֵי מְכִירוֹת הַקַּטְנוֹעִים בַּשָּׁנִים הָאַחֲרוֹנוֹת, וּמֵהֶם עוֹלֶה כִּי הַפַּעַר בֵּין קִימְקוֹ לְבֵין סָאנְיָאנְג, הוֹלֵךְ וּמִצְטַמְצֵם.
בֵּין הַשָּׁנִים 2007 לְ-2009 הִצְטַמְצֵם נֵתַח הַשּׁוּק שֶׁל סָאנְיָאנְג מִכְּ-45% לְ-40%, וְאִילּוּ נֵתַח הַשּׁוּק שֶׁל קִימְקוֹ גָּדַל מִ-26% לְ-30%.
צַ'אנְג קַו, סְגַן נָשִׂיא בָּכִיר בְּקִימְקוֹ, אָמַר בְּהֶקְשֵׁר זֶה כִּי בְּכַוָּונַת הַחֶבְרָה לְסַיֵּיעַ לַיְּבוּאָן הַמְּקוֹמִי לַהֲפוֹךְ אֶת קִימְקוֹ לַמּוּתָג הַמּוֹבִיל בְּיִשְׂרָאֵל.

כְּצַעַד לִקְרַאת הַשָּׂגַת הַיַּעַד, נוּצַּל בִּיקּוּר הָאוֹרְחִים לְהַשָּׁקַת שְׁנֵי דִּגְמֵי קִימְקוֹ חֲדָשִׁים – לְמַעֲשֶׂה דֶּגֶם אֶחָד חָדָשׁ, וְגִרְסָה נוֹסֶפֶת לְדֶגֶם קַיָּים.
הֶחָדָשׁ עוֹנָה לְשֵׁם דָּאוּנְטָאוּן 200, וּמִתְמַקֵּם בֵּין שְׁנֵי דִּגְמֵי הַדָּאוּנְטָאוּן שֶׁכְּבָר קַיָּימִים בַּשּׁוּק – 125 וְ-300.
הַ-200 צָפוּי לְהִתְמוֹדֵד מוּל קַטְנוֹעֵי 250 עִם הַסָּפֵק דּוֹמֶה (20.5 כ"ס), מִשְׁקָל מְעַט נָמוּךְ מֵהַמְּמוּצָּע - 166 ק"ג - וּמְמַדִּים דּוֹמִים לְאֵלֶּה שֶׁל הַ-300.
תָּג הַמְּחִיר צָפוּי לַעֲמוֹד עַל "פָּחוֹת מִ-25 אֶלֶף שְׁקָלִים", אִם כִּי הַיְּבוּאָן סֵירֵב לַחֲשׂוֹף אֶת הַמְּחִיר הַמְּדוּיָּק.
הַ-300 יְעוּדְכַּן מֵעַתָּה, וְיַצִּיעַ גַּם גִּרְסַת ABS, הַכּוֹלֶלֶת בְּהֶתְאֵם לִשְׁמָהּ מַעֲרֶכֶת הַמּוֹנַעַת נְעִילַת גַּלְגַּלִּים.
גַּם בְּמִקְרֶה זֶה לֹא חָשַׂף הַיְּבוּאָן אֶת הַמְּחִיר הָעֲתִידִי.
