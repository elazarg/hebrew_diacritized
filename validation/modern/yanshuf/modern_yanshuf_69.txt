דּוּ"חַ הָאוּ"ם עַל הַקִּדְמָה בִּמְדִינוֹת הָעוֹלָם קוֹבֵעַ:
יִשְׂרָאֵל בַּמָּקוֹם ה-22 מִתוֹךְ 173 מְדִינוֹת. נִמְצֵאת בֵּין סְפָרַד לְהוֹנְג קוֹנְג, יוֹרֶדֶת בְּרָמַת הַיֶּדַע שֶׁל הַתַּלְמִידִים, נִמְצֵאת בְּרָמָה גְּבוֹהָה שֶׁל הַיּי-טֶק, אוֹהֶבֶת מְאֹד לְדַבֵּר בַּטֵּלֵפוֹן וְצוֹרֶכֶת חַשְׁמַל בְּכַמֻיוֹת
אַחַת לְשָׁנָה, בּוֹדֵק הָאוּ"ם אֶת הַמַּצָב שֶׁל כָּל מְדִינוֹת הָעוֹלָם. הַבְּדִיקָה מַקִיפָה אֶת רֹב מַעַרְכוֹת הַחַיִּים, כְּמוֹ לְמָשָׁל: חִנּוּךְ, בְּרִיאוּת, יְלוּדָה, עֲבוֹדָה, אַדְרִיכָלוּת, תַקְצִיב הַמֶּמְשָׁלָה וְעוֹד.
מִתְבָּרֵר, שֶׁמְּדִינַת יִשְׂרָאֵל נִמְצֵאת בְּכַמָה נוֹשְׂאִים בְּדַרְגָה גְּבוֹהָה מְאֹד. אֶחָד הַתְּחוּמִים הַבּוֹלְטִים בְּיוֹתֵר הוּא הַהַיי-טֵק. יִשְׂרָאֵל נֶחְשֶׁבֶת לְמַעֲצֶמֶת מַחְשְׁבִים וְאִינְטֶרְנֵט. רַק אַרְצוֹת הַבְּרִית מַקְדִימָה אֶת יִשְׂרָאֵל בִּתְחוּם זֶה. 
נוֹשֵׂא אַחֵר שֶׁיִשְׂרָאֵל מוֹבִילָה בּוֹ הוּא הַטֶּלֶפוֹנִים הַנַּיָדִים. עַל פִּי מִמְצְאֵי הַסֶּקֶר, ל-80% מֵהַיִּשְׂרְאֵלִים יֵשׁ טֶלֶפוֹנִים נַיָּדִים. יִשְׂרָאֵל נִמְצֵאת בַּמָּקוֹם הָרְבִיעִי בָּעוֹלָם בִּתְחוּם זֶה. לְפָנֶיהָ נִמְצָאוֹת אִיטַלְיָה, הוֹנְג-קוֹנְג וְלוּקְסֶמְבּוּרְג עִם 96%. זֹאת אוֹמֶרֶת, שֶׁכִּמְעַט לְכָל תוֹשָׁב שָׁם יֵשׁ טֶלֶפוֹן נַיָד.
הַנְּתוּנִים שֶׁעוֹלִים מדּוּ"חַ הָאוּ"ם מְגַלִים גַם דְּבָרִים פָּחוֹת נְעִימִים, כְּמוֹ לְמָשָׁל, הָאַבְטָלָה. כִּמְעַט כָּל אָדָם שְׁבִיעִי בְּיִשְׂרָאֵל אֵינוֹ עוֹבֵד. זֶהוּ נָתוּן קָשֶׁה מְאֹד. אוּלָם, בֵּין אוֹתָם שֶׁאֵינָם עוֹבְדִים נִמְצָאִים גַם הַדָּתִיִים הַחֲרֵדִים, שֶׁמִמֵילָא לֹא עוֹבְדִים. 
חַשְׁמַל:
יִשְׂרָאֵל נִמְצֵאת בַּמָּקוֹם ה-18 בָּעוֹלָם בִּצְרִיכַת חַשְׁמַל וְזֶה נָתוּן מְעַנְיֵן מְאֹד. אָנוּ מַקְדִימִים מְדִינוֹת קָרוֹת, הַצְּרִיכוֹת לְחַמֵם אֶת הַבָּתִים וְגַם מַקְדִימִים אֶת הַמְּדִינוֹת הַחַמוֹת וְהָעֲשִׁירוֹת, כְּמוֹ עָרַב הַסָּעוּדִית.
אַגַב, עוֹבְדֵי חַבְרַת הַחַשְׁמַל צוֹרְכִים פִּי שְׁנַיִם מִכָּל אֶזְרָח רָגִיל, מִפְּנֵי שהם מְקַבְּלִים אֶת הַחַשְׁמַל שֶׁלָּהֶם בְּחִנָם.
עִשׁוּן:
הַיִּשְׂרְאֵלִים מְעַשְׁנִים הַרְבֵּה יַחֲסִית. כָּל מְעַשֵׁן צֹרֶךְ בִּמְמֻצָע 2,223 סִיגַרְיוֹת בְּשָׁנָה. יֵשׁ מֻמְחִים הַטּוֹעֲנִים שֶׁהָעִשׁוּן הָרַב נוֹצַר בִּגְלַל הַלַּחַץ שֶׁאֲנָשִׁים נִמְצָאִים בּוֹ.
בְּסַךְ הַכֹּל, יִשְׂרָאֵל נֶחְשֶׁבֶת לִמְדִינָה שֶׁנָּעִים לָגוּר בָּה, וְכִי רָמַת הַחַיִּים כָּאן הִיא טוֹבָה, בְּיַחַס לִמְדִינוֹת אֲחֵרוֹת בָּעוֹלָם. כֻּלָם מַסְכִּימִים, שֶׁאִם הַמַּצָב הַבִּטְחוֹנִי בָּאָרֶץ הָיָה טוֹב, הַנְּתוּנִים שֶׁל הָאוּ"ם עַל מְדִינַת יִשְׂרָאֵל הָיוּ טוֹבִים הַרְבֵּה יוֹתֵר.
