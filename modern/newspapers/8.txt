הַמַּפְכָּ"ל דָּנִינוֹ: מֵהַתְּקוּפוֹת הַמּוּצְלָחוֹת שֶׁלָּנוּ
רַב-נִיצָּב יוֹחָנָן דָּנִינוֹ דִּיבֵּר בְּוַועֲדַת הַפְּנִים שֶׁל הַכְּנֶסֶת וּמִתְגָּאֶה בְּהֶישֵּׂגָיו.
לִדְבָרָיו, הַמַּצָּב בְּיִשְׂרָאֵל טוֹב מִבְּקָנָדָה וּבְפִינְלַנְד: "זוֹ מֵהַתְּקוּפוֹת הַיּוֹתֵר מוּצְלָחוֹת שֶׁל מִשְׁטֶרֶת יִשְׂרָאֵל"

חִיסּוּל רוֹדֵף חִיסּוּל, פִּיצוּץ רוֹדֵף פִּיצוּץ, עוֹד רֶצַח וְעוֹד דְּקִירָה - אֲבָל מַפְכָּ"ל הַמִּשְׁטָרָה רַב נִיצָּב יוֹחָנָן דָּנִינוֹ רוֹאֶה אֶת הַמַּצָּב בְּמִשְׁקָפַיִים וְרוּדִּים.
"הַהֶישֵּׂגִים מַרְשִׁימִים.
זוֹ מֵהַתְּקוּפוֹת הַיּוֹתֵר מוּצְלָחוֹת שֶׁל מִשְׁטֶרֶת יִשְׂרָאֵל", אָמַר הַבּוֹקֶר (יוֹם ב') בְּהוֹפָעָה בִּפְנֵי חַבְרֵי וַעֲדַת הַפְּנִים שֶׁל הַכְּנֶסֶת עַל הִתְמוֹדְדוּת הַמִּשְׁטָרָה בַּנּוֹשֵׂא.
לִדְבָרָיו, הַמַּצָּב בְּיִשְׂרָאֵל טוֹב מֵאֲשֶׁר בְּקָנָדָה וּבְפִינְלַנְד.
ynet הֶעֱבִיר אֶת הַדִּיּוּן בַּוַּועֲדָה בְּשִׁידּוּר חַי.

תִּבְדְּקוּ אֶת הָעוּבְדּוֹת.
מְדַבְּרֵי הַמַּפְכָּ"ל בַּכְּנֶסֶת
לְדִבְרֵי הַמַּפְכָּ"ל, "תּוֹכְנִית הַמִּפְנֶה נוֹעֲדָה לְהַעֲלוֹת אֶת אֵמוּן הַצִּיבּוּר בַּמִּשְׁטָרָה וּלְהַעֲלוֹת אֶת רָמַת הַבִּיטָּחוֹן הָאִישִׁי.
הִגַּעְנוּ לְהֶישֵּׂגִים רַבִּים בְּמַהֲלַךְ הַשָּׁלוֹשׁ שָׁנִים הָאַחֲרוֹנוֹת.
אֲנִי לֹא מֵקֵל רֹאשׁ בָּאֵירוּעִים שֶׁל הַתְּקוּפָה הָאַחֲרוֹנָה.
נִפְגְּעָה תְּחוּשַׁת הַבִּיטָּחוֹן וְהַמַּטָּרָה הִיא לִמְצוֹא אֵיךְ לְשַׁפֵּר אֶת תְּחוּשַׁת הַבִּיטָּחוֹן".

דָּנִינוֹ אָמַר עוֹד: "לֹא נִשְׁכַּח אֶת הָעוּבְדּוֹת, אֵיךְ בּוֹדְקִים מִשְׁטָרָה בָּעוֹלָם - הַדָּבָר הָרִאשׁוֹן הוּא הַסִּיכּוּי שֶׁלְּךָ כְּאֶזְרָח לִהְיוֹת קוֹרְבַּן עֲבֵירָה וְהַדָּבָר הַשָּׁנִי, כַּמָּה מַהֵר הִצְלַחְתָּ לָשִׂים יָד עַל הָעֲבַרְיָין.
אֵלֶּה הַפָּרָמֶטְרִים שֶׁאִיתָּם בּוֹדְקִים מִשְׁטָרָה.
יָרַדְנוּ בְּמִסְפַּר אֵירוּעֵי הָרֶצַח - בְּעֵרֶךְ 50 מִקְרֵי רֶצַח בִּשְׁנָתַיִים.
בַּשְּׁנָתַיִים הָאַחֲרוֹנוֹת יָרְדוּ בְּ-500 אֵירוּעֵי יְרִי לְפָחוֹת".

"מֵבִין אֶת תּוֹשָׁבֵי פֶּתַח תִּקְוָוה".
יוֹ"ר הַוַּועֲדָה רֶגֶב אָמְרָה לַמַּפְכָּ"ל: "אֵין לָנוּ מִשְׁטָרָה אַחֶרֶת, אֲנִי מַעֲרִיכָה אֶת הָעֲבוֹדָה שֶׁלָּכֶם וַאֲנִי יוֹדַעַת שֶׁאַתֶּם פּוֹעֲלִים לִשְׁמִירַת הַבִּיטָּחוֹן הָאִישִׁי.
אֲנַחְנוּ מַזְמִינִים לֹא רַק כְּדֵי לְדַבֵּר עַל הַדְּבָרִים הַטּוֹבִים אֶלָּא לְהַרְאוֹת עַל מָה צָרִיךְ לְפַקֵּחַ.
הַפֶּשַׁע הַפְּלִילִי הֵרִים רֹאשׁ, הָאֶזְרָח שׁוֹפֵט מָה קוֹרֶה לְיַד הַבַּיִת שֶׁלּוֹ.
כְּשֶׁהַמַּפְכָּ"ל אָמַר שֶׁהַמִּטְעָנִים מַגִּיעִים מִצַּהַ"ל גִּיחַכְתִּי, אֲבָל לְאַחַר יְשִׁיבָה חֲסוּיָה אֶצְלִי הִתְבָּרֵר שֶׁאָכֵן יֵשׁ מִטְעָנִים שֶׁזּוֹלְגִים".

מַפְכָּ"ל הַמִּשְׁטָרָה צִייֵּן בַּדִּיּוּן כִּי בָּחֳודָשִׁים הָרִאשׁוֹנִים שֶׁל 2014, יָנוּאָר וּפֶבְּרוּאָר, חָלָה יְרִידָה בְּכַמּוּת הַפְּרִיצוֹת לְדִירוֹת לְעוּמַּת הַתְּקוּפָה הַמַּקְבִּילָה אֶשְׁתָּקַד: מִ-4,650 פְּרִיצוֹת בְּיָנוּאָר-פֶבְּרוּאָר 2013 לְ-3,698 בְּ-2014.
"אִם תִּמְצְאוּ יְרִידָה כַּזֹּאת בַּהִיסְטוֹרְיָה תָּבוֹאוּ אֵלַיי", אָמַר הַמַּפְכָּ"ל.
עַל רֶקַע תְּחוּשַׁת חוֹסֶר הָאוֹנִים שֶׁהִבִּיעוּ תּוֹשָׁבֵי פֶּתַח תִּקְוָוה בְּעִקְבוֹת גַּל הַפְּשִׁיעָה שֶׁפּוֹקֵד אֶת הָעִיר, אָמַר הַמַּפְכָּ"ל דָּנִינוֹ: "אֲנִי מֵבִין אֶת תּוֹשָׁבֵי פֶּתַח תִּקְוָוה".

"הִרְחַקְנוּ 9,000 עֲבַרְיָינִים מֵהָרְחוֹבוֹת" 
דָּנִינוֹ הִתְיַיחֵס לַקֶּשֶׁר שֶׁל הַמִּשְׁטָרָה עִם הָרָשׁוּת הַשּׁוֹפֶטֶת וְאָמַר כִּי "אִם תִּתְקַשֵּׁר לְשׁוֹפְטִים וּלְשֵׁירוּת בָּתֵּי הַסּוֹהַר תְּגַלֶּה אֶת מָה שֶׁקּוֹרֶה.
הֵצַפְנוּ אוֹתָם וְהֵם לֹא עוֹמְדִים בָּעוֹמֶס.
הֵם צְרִיכִים יוֹתֵר שׁוֹפְטִים וְיוֹתֵר מִתְקְנֵי כְּלִיאָה.
בַּשְּׁנָתַיִים הָאַחֲרוֹנוֹת עָשִׂינוּ 9,000 מַעֲצָרִים עַד תּוֹם הַהֲלִיכִים, וּבְכָל בָּתֵּי הַכֶּלֶא בִּמְדִינַת יִשְׂרָאֵל יֵשׁ קְצָת יוֹתֵר מִ-12,000 עֲצוּרִים.
הַמַּשְׁמָעוּת שֶׁל הַנָּתוּן הַזֶּה הִיא שֶׁהִגְבַּרְנוּ אֶת הַהַרְתָּעָה.
אִם הִכְנַסְנוּ לַכֶּלֶא 9,000 עֲבַרְיָינִים אָז הִרְחַקְנוּ אוֹתָם מֵהָרְחוֹבוֹת".

עוֹד אָמַר הַמַּפְכָּ"ל בְּנוֹגֵעַ לְגַל הַפְּשִׁיעָה: "צוֹדְקִים אֶזְרָחִים שֶׁאוֹמְרִים שֶׁהֵם לֹא מוּכָנִים שֶׁיִּהְיֶה פִּיצוּץ לְיַד הַבַּיִת שֶׁלָּהֶם, חוֹבָתֵנוּ לְטַפֵּל בָּזֶה וּלְהַחֲזִיר אֶת תְּחוּשַׁת הַבִּיטָּחוֹן אֲבָל צָרִיךְ לָדַעַת לְהַבְחִין.
יֵשׁ כָּאן מִשְׁטָרָה אַחַת וְצָרִיךְ לִרְאוֹת שֶׁאֲנַחְנוּ נוֹתְנִים לָהּ לַעֲשׂוֹת אֶת הָעֲבוֹדָה וְלֹא פּוֹגְעִים שֶׁלֹּא לְצוֹרֶךְ.
בַּשָּׁבוּעַ שֶׁעָבַר הָיָה דִּיּוּן אֵצֶל רֹאשׁ מֶמְשָׁלָה וְכוּלָּם נָתְנוּ רוּחַ גַּבִּית.
אָנוּ לֹא מְקִלִּים רֹאשׁ וְנִמְצָאִים בְּמַאֲמָץ גָּדוֹל מְאוֹד שֶׁל כָּל הַיְּחִידוֹת לָשִׂים יָד.
שָׁמַעְתִּי הַרְבֵּה אֲמִירוֹת עַל חוֹסֶר בְּפִעְנוּחַ וַאֲנִי רוֹצֶה לוֹמַר שֶׁבְּקָרוֹב מְאוֹד נָשִׁים יָד".

אָלֵכְּס לָאוֹטֵין, אַלְמָנָה שֶׁל מַרְגָּרִיטָה שֶׁנִּרְצְחָה בְּבַת יָם, אָמַר בַּדִּיּוּן כִּי "לִי זֶה קָרָה כְּבָר פַּעַם אַחַת וְכִמְעַט קָרָה לִי פַּעַם שְׁנִיָּיה כְּשֶׁהָיִיתִי בַּדֶּרֶךְ לַתַּחֲנָה.
כְּשֶׁהַמַּפְכָּ"ל אוֹמֵר 'אֵין לָנוּ שְׁלִיטָה' זֶה לֹא נָכוֹן.

בַּמִּקְרֶה שֶׁלָּנוּ הַמִּשְׁטָרָה הָיְיתָה עַל חוֹף הַיָּם.
בְּמִקְרֶה נוֹסָף בְּפֶתַח תִּקְוָוה הַמִּשְׁטָרָה הִגִּיעָה לְבֵית הַמִּשְׁפָּט וְאָמְרָה שֶׁאָדָם הִזְמִין מִטְעַן חַבָּלָה.
לָמָּה נָתְנוּ לוֹ לְהִסְתּוֹבֵב בָּרְחוֹב? לָמָּה לֹא עָצְרוּ אוֹתוֹ? לָמָּה כְּשֶׁבְּמִשְׁטָרָהּ יוֹדְעִים שֶׁאַלְפֵּרוֹן מְאוּיָּם אוֹמְרִים לוֹ 'תִּיזָּהֵר' וּכְשֶׁאֲנִי יוֹשֵׁב לְיָדוֹ אֲנִי לֹא יוֹדֵעַ שֶׁרוֹצִים לַהֲרוֹג אוֹתוֹ וְיָכוֹל לִהְיוֹת קוֹרְבָּן?"

לִפְנֵי כִּשְׁלוֹשָׁה שָׁבוּעוֹת הִצְבִּיעַ נִיצָּב מֶנִי יִצְחָקִי, רֹאשׁ אֲגַף הַחֲקִירוֹת בַּמִּשְׁטָרָה, עַל שִׁינּוּי דְּפוּס הַפְּעוּלָּה הַעֲבַרְיָינִי.
בַּדִּיּוּן בְּאוֹתָהּ וַעֲדָה הוּא פָּסַל אֶת הַהֲנָחָה כִּי מְדוּבָּר בְּפִיצוּצִים יְזוּמִים שֶׁל אִרְגּוּנֵי הַפְּשִׁיעָה וְהִדְגִּישׁ כִּי "כּוֹחוֹת הַשּׁוּק" הִכְתִּיבוּ מַעֲבָר לְשִׁימּוּשׁ תָּכוּף בְּמִטְעֲנֵי נֶפֶץ.
"עֲבַרְיָין שֶׁיּוֹצֵא לִנְקוֹם בּוֹדֵק אֵיפֹה זֶה יַעֲלֶה לוֹ זוֹל יוֹתֵר, אֵיפֹה זֶה זָמִין יוֹתֵר וּמָה רָמַת הַסִּיכּוּן.
בָּרֶגַע שֶׁיֵּשׁ זְמִינוּת הַמְּחִירִים יוֹרְדִים, כְּמוֹ כָּל דָּבָר אַחֵר בַּשּׁוּק", הִדְגִּישׁ.
