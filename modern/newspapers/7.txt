אַתֶּם קְבַעְתֶּם: דּוּאֵט הָאַהֲבָה הַיִּשְׂרְאֵלִי הַטּוֹב בְּיוֹתֵר
מִילִּים רַבּוֹת נִכְתְּבוּ עַל הָרֶגֶשׁ הַזֶּה שֶׁכּוּלָּנוּ חוֹוִים בְּמַהֲלַךְ חַיֵּינוּ - אָהֲבָה.
לְרֶגֶל ט"ו בְּאָב שֶׁצּוּיַּן בְּסוֹף הַשָּׁבוּעַ הָאַחֲרוֹן, יָצָאנוּ לִבְדּוֹק אֵיזֶה שִׁיר אַהֲבָה הֲיִיתֶם מַקְדִּישִׁים לַחֲצִי הַשָּׁנִי שֶׁלָּכֶם וְגַם נַחֲשׂוֹף אֶת תּוֹצְאוֹת הַסֶּקֶר בּוֹ אַתֶּם קְבַעְתֶּם מַהוּ דּוּאֵט הָאַהֲבָה הַיִּשְׂרְאֵלִי הַטּוֹב בְּיוֹתֵר.

כַּמָּה מִילִּים נִכְתְּבוּ עַל אַהֲבָה וּמָה זוֹ בִּכְלָל אָהֲבָה? אִם תְּחַפְּשׂוּ בָּרֶשֶׁת אֶת הַשְּׁאֵלָה יַעֲלוּ 12.3 מִילְיוֹן תּוֹצָאוֹת.
יִתָּכֵן וְאַחַת מֵהֵן תַּתְאִים לָכֶם, אוּלַי לֹא.
זֶה לֹא בֶּאֱמֶת מְשַׁנָּה, כִּי לֹא חָשׁוּב כַּמָּה הַגְּדֵרוֹת יֵשׁ, זֶה הַכּוֹל תָּלוּי ב-אֵיךְ שֶׁמַּרְגִּישִׁים בַּלֵּב פְּנִימָה.

הַחִיבּוּר בֵּין מוּזִיקָה לְאַהֲבָה הוּא, אֶפְשָׁר לוֹמַר, סוּג שֶׁל קֶסֶם, אוּלַי אֲפִילּוּ מִיסְטִי.
גַּם הַצִּינִיקָנִים שֶׁבֵּינֵינוּ יָעִידוּ שֶׁיֵּשׁ אֵיזֶה שִׁיר אֶחָד אוֹ שְׁנַיִים שֶׁ"בָּאוּ לָהֶם בְּדִיּוּק בַּזְּמַן" כְּשֶׁהֵם הִרְגִּישׁוּ מַשֶּׁהוּ בַּלֵּב, וְהַמִּילִּים חִיבְּרוּ עֲבוּרָם אֶת הַמַּשְׁמָעוּת שֶׁל מָה שֶׁהֵם חָשׁוּ.
אֲחֵרִים יַסְכִּימוּ בְּהִתְלַהֲבוּת וְיַתְחִילוּ לִמְנוֹת אֶת הַשִּׁירִים שֶׁמַּזְכִּירִים לָהֶם אֲהָבוֹת נִשְׁכָּחוֹת, אֶת הַפְּגִישָׁה הָרִאשׁוֹנָה עִם הָאַהֲבָה הַגְּדוֹלָה, הַנְּשִׁיקָה הָרִאשׁוֹנָה, הַמַּבָּט וְהַחִיּוּךְ.

זוֹכְרִים אֶת הַדֵּיְיט הַהוּא בְּהוֹפָעָה, כְּשֶׁנִּיגְּנוּ אֶת הַשִּׁיר שֶׁהָפַךְ "לְשֶׁלָּכֶם"? (צִילּוּם: shutterstock)

הַחִיבּוּר הַזֶּה עוֹמֵד בִּבְסִיסָם שֶׁל מֶחְקָרִים רַבִּים שֶׁמְּנַסִּים לִמְצוֹא מָה זֶה בְּדִיּוּק הַדָּבָר הַזֶּה שֶׁמּוּזִיקָה עוֹשָׂה לָנוּ.
בְּגָדוֹל, מוּזִיקָה מְשַׁחְרֶרֶת אֶצְלֵנוּ בַּמּוֹחַ דּוֹפָּמִין, אוֹתוֹ חוֹמֶר שֶׁאַחְרַאי, בֵּין הַיֶּתֶר, לִתְחוּשַׁת הָעוֹנֶג אַחֲרֵי אֲרוּחָה טוֹבָה אוֹ סֶקְס.
הַחִיבּוּר נַעֲשֶׂה בְּעֶזְרַת הַתְּחוּשׁוֹת שֶׁאֲנַחְנוּ מַרְגִּישִׁים בְּעֵת שֶׁאֲנַחְנוּ שׁוֹמְעִים אֶת אוֹתוֹ שִׁיר בַּפַּעַם הָרִאשׁוֹנָה.
כַּאֲשֶׁר הַשִּׁיר הַזֶּה מְקוּשָּׁר לִרְגָשׁוֹת מְסוּיָּמִים, חֲזָקִים בִּמְיוּחָד, גַּם אִם נִשְׁמָע אוֹתוֹ שָׁנִים רַבּוֹת לְאַחַר אוֹתוֹ פַּעַם רִאשׁוֹנָה, הָרְגָשׁוֹת הַלָּלוּ יַעֲלוּ שׁוּב וְנִיזָּכֵר בְּכָל מָה שֶׁהָיָה סְבִיב הַשִּׁיר.

נָסוּ לְהִיזָּכֵר בְּמִקְרֶה כָּלְשֶׁהוּ שֶׁשִּׂימֵּחַ אֶתְכֶם אוֹ הֵעִיב עַל מַצָּב רוּחֲכֶם כְּשֶׁהֲיִיתֶם בְּבֵית הַסֵּפֶר הַיְּסוֹדִי אוֹ בַּתִּיכוֹן, מַצְלִיחִים? סָבִיר לְהַנִּיחַ שֶׁתִּצְטָרְכוּ לַחֲשׁוֹב עַל זֶה, אֲבָל אִם תִּשְׁמְעוּ אֵיזֶה שִׁיר אַהֲבָה שֶׁהִתְנַגֵּן אָז שׁוּב וָשׁוּב, בַּנֶּשֶׁף שֶׁל בֵּית הַסֵּפֶר, בְּדִיּוּק כְּשֶׁהֶחָתִיךְ שֶׁל הַשִּׁכְבָה אָמַר לֹא לַהַזְמָנָה שֶׁלָּכֶם לְסֶרֶט אוֹ לִיוָּוה אֵיזוֹ סִדְרָה אֲהוּבָה – מִיָּד תִּיזְכְּרוּ בְּכָל מָה שֶׁקָּרָה בְּאוֹתוֹ יוֹם נָתוּן, וְהַכֹּל מִשּׁוּם שֶׁהַתְּחוּשׁוֹת הַלָּלוּ צְרוּבוֹת בְּרֹאשֵׁנוּ יַחַד עִם הַמּוּזִיקָה הַזּוֹ וּבָרֶגַע שֶׁאֲנַחְנוּ שׁוֹמְעִים אוֹתָהּ שׁוּב, אֲנַחְנוּ חוֹזְרִים לְאָחוֹר בַּזְּמַן.
אֶפְשָׁר לוֹמַר שֶׁכָּל שִׁיר הוּא מֵעֵין מַגְנֵט שֶׁמּוֹשֵׁךְ הַחוּצָה אֶת הַתְּחוּשׁוֹת שֶׁנִּדְבְּקוּ אֵלָיו בִּזְמַנִּים מוּקְדָּמִים יוֹתֵר.

לְרֶגֶל ט"ו בְּאָב, שֶׁלְּמַעֲשֶׂה מְסַמֵּל בַּיַּהֲדוּת יוֹם שֶׁל אַהֲבָה וּפִיּוּס בְּקֶרֶב עִם יִשְׂרָאֵל, אַךְ עָשָׂה פְּנִיָּיה קַלָּה וְהָפַךְ בִּמְרוּצַת הַזְּמַן לְיוֹם שֶׁכּוּלּוֹ אַהֲבָה יִשְׂרְאֵלִית, בְּעִיקָּר זוּגִית, בִּיקַּשְׁנוּ מִכֶּם לִבְחוֹר אֶת דּוּאֵט הָאַהֲבָה הַיִּשְׂרְאֵלִי הֲכִי רוֹמַנְטִי, הֲכִי אָהוּב, הֲכִי מַעֲלֶה זִיכְרוֹנוֹת נְעִימִים.
