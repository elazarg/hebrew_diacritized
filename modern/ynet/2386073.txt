אִיצִיק סוּדְרִי, דּוֹבֵר שָׁ"ס, מִתְרַעֵם עַל הֲרָמַת הַגַּבּוֹת בְּעִקְבוֹת הַשָּׁקַת אֲתַר הָאִינְטֶרְנֶט הָרִשְׁמִי שֶׁל הַתְּנוּעָה הַסְּפָרַדִּית-חֲרֵדִית.
לְשִׁיטָתוֹ, שָׁ"ס תָּמִיד יָדְעָה לְנַצֵּל אֶת הַטֶּכְנוֹלוֹגְיָה לְטוֹבָתָהּ וּלְטוֹבַת מַאֲמִינֶיהָ וְאֵלּוּ שֶׁאֵינָם מַכִּירִים בְּכָךְ חוֹשְׂפִים אֶת הַדֵּעָה הַקְּדוּמָה שֶׁלָּהֶם.
"תָּמִיד חָשַׁבְנוּ שֶׁמִּצְווֹת וְטֶכְנוֹלוֹגְיָה הוֹלְכוֹת יַחַד", הוּא אוֹמֵר, "הֵקַמְנוּ עָרוּץ לַוְויָין מִשֶּׁלָּנוּ בּוֹ צוֹפִים מֵאוֹת אַלְפֵי בְּנֵי אָדָם, וְנִדְמֶה לִי שֶׁכַּמָּה עֲרוּצֵי טֵלֵוִויזְיָה יִתְקַנְּאוּ בַּרֶיְיטִינְג הַזֶּה.
שָׁ"ס הִיא הַמִּפְלָגָה הַחֲרֵדִית הָרִאשׁוֹנָה שֶׁהֵבִינָה אֶת הַפּוֹטֶנְצְיָאל שֶׁל שִׁידּוּרֵי הַתַּעֲמוּלָה בַּטֵּלֵוִויזְיָה וְנִיצְּלָה אֶת מִכְסַת הַזְּמַן שֶׁלָּהּ עוֹד בְּמַעֲרֶכֶת הַבְּחִירוֹת שֶׁל 1984, כַּאֲשֶׁר מִפְלָגוֹת חֲרֵדִיּוֹת אֲחֵרוֹת נִמְנְעוּ מִכָּךְ.
"אֲנַחְנוּ הַמִּפְלָגָה הַיְּחִידָה שֶׁבִּזְכוּתָהּ קָמוּ 10 עֲרוּצֵי רַדְיוֹ.
הָרַב עוֹבַדְיָה יוֹסֵף אָמַר בְּכָל הַמִּקְרִים: קְחוּ אֶת הַטֶּכְנוֹלוֹגְיָה הַטְּמֵאָה וְהִפְכוּ אוֹתָהּ לְמַשֶּׁהוּ יָפֶה.
גַּם בֶּעָתִיד נִבְחַן כָּל חִידּוּשׁ טֶכְנוֹלוֹגִי וְנִרְאָה כֵּיצַד אֶפְשָׁר לְתַרְגֵּם אוֹתוֹ לְמַשֶּׁהוּ מִשֶּׁלָּנוּ, לְדָבָר חִיּוּבִי מִבְּחִינַת הַהַשְׁקָפָה שֶׁלָּנוּ".
שָׁ"ס הִיא הַמִּפְלָגָה הַחֲרֵדִית הָרִאשׁוֹנָה שֶׁהֶחֱלִיטָה לְהַתְחִיל לִפְעוֹל בְּאוֹפֶן רִשְׁמִי בָּאִינְטֶרְנֵט, לְאַחַר כַּמָּה שָׁנִים שֶׁל הִיסּוּסִים.
הָאֲתָר הָרִשְׁמִי הוּשַּׁק בְּעִקְבוֹת אֲחוּזֵי הַכְּנִיסוֹת הַיָּפִים לַאֲתַר הַבִּלְתִּי רִשְׁמִי שֶׁל הַתְּנוּעָה, שֶׁהוּשַּׁק לִפְנֵי זְמַן לֹא רַב עַל יְדֵי חוֹזְרִים בִּתְשׁוּבָה, חֲסִידִים שֶׁל הָרַב דָּנִיאֵל זָר, הַמְּקוֹרָב לְשָׁ"ס.
לַמְרוֹת הַדְּבָרִים שֶׁל סוּדְרִי, דּוֹמֶה כִּי בִּתְנוּעָה עֲדַיִין מְגַלִּים יַחַס דּוּ-עֶרְכִּי כְּלַפֵּי הָאִינְטֶרְנֶט.

"אֵין בְּהַשָּׁקַת הָאֲתָר הָרִשְׁמִי מִשּׁוּם הֶכְשֵׁר לָאִינְטֶרְנֵט.
הָאִינְטֶרְנֶט עֲדַיִין טֶרֶף", מַדְגִּישִׁים בְּשָׁ"ס, "עִם זֹאת, אֲנַחְנוּ מְבִינִים שֶׁיֵּשׁ מֵאוֹת אַלְפֵי אֲנָשִׁים הַקְּרוֹבִים בְּדֵעוֹתֵיהֶם לְשָׁ"ס שֶׁמְּחוּבָּרִים לָאִינְטֶרְנֵט, בְּבָתִּים אוֹ בִּמְקוֹמוֹת עֲבוֹדָה.
לָכֵן, אִם הֵם מְחוּבָּרִים בְּכָל מִקְרֶה לְכָל הַפָּחוֹת שֶׁיִּגְלְשׁוּ לַאֲתַר שֶׁל שָׁ"ס וְלֹא לִמְחוֹזוֹת אֲחֵרִים".
בְּשָׁ"ס צִיפּוּ לַכֹּל הַיּוֹתֵר לְמֵאָה אֶלֶף כְּנִיסוֹת לַאֲתַר הָרִשְׁמִי עַד מוֹעֵד הַבְּחִירוֹת וְהוּפְתְּעוּ מֵהַהַצְלָחָה.
לְטַעֲנַת אַנְשֵׁי הַתְּנוּעָה, כְּמִילְיוֹן וְ-250,000 כְּנִיסוֹת נִרְשְׁמוּ בָּאֲתָר עַד הַיּוֹם, כְּשִׁבְעָה אָחוּז מֵהַכְּנִיסוֹת הֵן שֶׁל גּוֹלְשִׁים מִחוּץ לְיִשְׂרָאֵל.
סוּדְרִי מוֹדֶה כִּי יִיתָּכֵן שֶׁנְּתוּנֵי הַכְּנִיסוֹת לַאֲתַר אֵינָם מְדוּיָּקִים וְלֹא מְעִידִים בְּהֶכְרֵחַ עַל מִסְפַּר הַגּוֹלְשִׁים בְּפוֹעַל, שֶׁעָשׂוּי לִהְיוֹת נָמוּךְ בְּהַרְבֵּה.
"יִיתָּכֵן שֶׁחֵלֶק מֵהַגּוֹלְשִׁים נִכְנְסוּ לַאֲתַר כַּמָּה פְּעָמִים אַךְ גַּם זֶה נָתוּן מְעוֹדֵד מִבְּחִינָתֵנוּ", אוֹמֵר סוּדְרִי, "אֲנָשִׁים צְמֵאִים לְמֵידָע עַל תְּנוּעַת שָׁ"ס".
נָכוֹן לְיוֹם שִׁישִּׁי בַּצָּהֳרַיִים, הַלָּהִיט הַגָּדוֹל בָּאֲתָר הָיָה הַסֶּרֶט "רֶגַע הָאֱמֶת", שֶׁהוּפַק עַל יְדֵי שָׁ"ס לִקְרַאת הַבְּחִירוֹת - 121,464 גּוֹלְשִׁים צָפוּ בַּסֶּרֶט.
גַּם מִכְתַּב הַתְּמִיכָה שֶׁל אַרְיֵה דֶּרְעִי בְּשָׁ"ס זָכָה לַפּוֹפּוּלָרִיּוּת רַבָּה (31,573 כְּנִיסוֹת), כְּמוֹ גַּם "בִּרְכַּת מָרָן הָרַב עוֹבַדְיָה וּמַחְרוֹזֶת שִׁירֵי שָׁ"ס מִכָּל הַזְּמַנִּים" (25,752 כְּנִיסוֹת), הַזַּמָּר אֲבִי בִּיטֶר עִם שִׁיר הַהַלֵּל לְאֵלִי יִשַׁי "שְׁלוּחוֹ שֶׁל מָרָן" (18,501 כְּנִיסוֹת), אַף יוֹתֵר מֵהַזֶּמֶר בֵּנִי אֶלְבָּז עִם הַשִּׁיר "סְפָרַדִּים גֵּאִים" (16,292 כְּנִיסוֹת).
בְּמַהֲלַךְ הַשִּׂיחָה עַמֵּנוּ, סוּדְרִי מְבַקֵּשׁ לְהַדְגִּישׁ, בְּעִקְבוֹת הַבִּיקּוֹרֶת עַל הָאֲתָר הָרִשְׁמִי, שֶׁפּוּרְסְמָה כָּאן בַּשָּׁבוּעַ שֶׁעָבַר, כִּי הַשִּׁיר "הָאֵלִיטוֹת שׁוֹלְטוֹת בְּכָל הַפִּינּוֹת" ("הָלוֹ חָבֵר, הָלוֹ אֲחִי, תִּקְשׁוֹרֶת מַטְעָה, אוֹתְךָ וְאוֹתִי, אֵלִיטוֹת שׁוֹלְטוֹת בְּכָל הַפִּינּוֹת, הַרְבֵּה אַפְלָיוֹת אֲנַחְנוּ יוֹדְעִים, פִּקְחוּ עֵינֵיכֶם הַכֹּל נֶגְדְּכֶם...") יוֹפִיעַ בָּאֲתָר הָרִשְׁמִי בְּקָרוֹב וְכִי בְּנִיגּוּד לְמָה שֶׁנִּכְתַּב כָּאן, הַשִּׁיר לְכָבוֹד אֵלִי יִשַׁי מוֹפִיעַ גַּם מוֹפִיעַ בָּאֲתָר הָרִשְׁמִי.
אָז הִנֵּה, בָּאוּ דְּבָרִים עַל דִּיּוּקָם.

בְּעִקְבוֹת הַהַצְלָחָה, הוֹרָה יוֹ"ר הַתְּנוּעָה, אֵלִי יִשַׁי, לְהַגְדִּיל אֶת הַתַּקְצִיבִים הַמּוּקְדָּשִׁים לִפְעִילוּת הַתְּנוּעָה בָּאִינְטֶרְנֵט.
נִרְאֶה כִּי שָׁ"ס אַט אַט מִתְנַעֶרֶת מֵהִסְתַּיְּיגוּתָהּ כְּלַפֵּי הָרֶשֶׁת וּמַתְחִילָה לְזַהוֹת אֶת הַפּוֹטֶנְצְיָאל הַטָּמוּן בָּהּ עֲבוּרָהּ.
"מִבְּחִינָתֵנוּ, הַשָּׁמַיִים הֵם הַגְּבוּל", אוֹמֵר סוּדְרִי, "בֶּעָתִיד נַעֲבִיר בָּאֲתָר אֶת שִׁיעוּרֵי הָרַב עוֹבַדְיָה יוֹסֵף בְּשִׁידּוּר חַי.
נְאַפְשֵׁר גַּם לְכָל גּוֹלֵשׁ לְהַפְנוֹת שְׁאֵלוֹת הִלְכָתִיּוֹת לְ'בֵּית הַהוֹרָאָה', קְבוּצַת תַּלְמִידֵי חֲכָמִים שֶׁמְּקוֹרָבִים לָרַב יוֹסֵף, כַּאֲשֶׁר שְׁאֵלוֹת מוּרְכָּבוֹת יוֹעָבְרוּ לָרַב בְּאוֹפֶן אִישִׁי.
נָעֲלָה אֶת כָּל הַסְּפָרִים שֶׁל הָרַב עוֹבַדְיָה יוֹסֵף לָרֶשֶׁת וּנְאַפְשֵׁר לַגּוֹלְשִׁים לְחַפֵּשׂ בָּהֶם עַל פִּי נוֹשֵׂא אוֹ מִילּוֹת מַפְתֵּחַ".
אַחַת הָאַטְרַקְצִיּוֹת בָּאֲתָר צְפוּיָה לִהְיוֹת הָאֶפְשָׁרוּת לְקַבֵּל בְּרָכָה אוֹן לַיְין מֵהָרַב עוֹבַדְיָה יוֹסֵף, שֶׁתּוּשַּׁק מַמָּשׁ בְּקָרוֹב.
הַגּוֹלְשִׁים יְמַלְּאוּ טוֹפֶס מְקוּּוָֹן וּבוֹ יְצַיְּינוּ אֶת סוּג הַבְּרָכָה הַמְּבוּקֶּשֶׁת (בִּתְחוּם הַפַּרְנָסָה, הַשִּׁידּוּךְ, הַבְּרִיאוּת וְכוּ'), וְכָל הַטְּפָסִים יוֹעָבְרוּ לָרַב עוֹבַדְיָה יוֹסֵף.
לְאַחַר שֶׁהָרַב יְבָרֵךְ, הַגּוֹלְשִׁים יְקַבְּלוּ הוֹדָעָה עַל כָּךְ בַּדּוֹאַר הָאֵלֶקְטְרוֹנִי, אוֹמֵר סוּדְרִי.
בְּהֶקְשֵׁר זֶה, יֵשׁ לְצַיֵּין, כִּי הָרַב כַּדּוּרִי מַצִּיעַ מִשְׁלוֹחַ שֶׁל קְמֵעוֹת וִירְטוּאָלִיִּים בְּאֶמְצָעוּת הַמִּפְעָל הַמְּקוּּוָֹן שֶׁלּוֹ, "כַּדּוּרִי נֶט".

לֹא רַק עַל הַשָּׁמַיִים חוֹשְׁבִים בְּשָׁ"ס אֶלָּא גַּם עַל נוֹשְׂאִים גַּשְׁמִיִּים יוֹתֵר.
הַתְּנוּעָה תְּאַפְשֵׁר הַרְשָׁמָה לְרֶשֶׁת מְעוֹנוֹת הַיּוֹם שֶׁל שָׁ"ס דֶּרֶךְ הָאֲתָר, כְּמוֹ גַּם רִישּׁוּם לְמוֹעֲדוֹנֵי נוֹעַר שֶׁל "אֶל הַמַּעְיָין".
נוֹסָף עַל כָּךְ, בְּמַטָּרָה לְקַדֵּם אֶת "בד"צ בֵּית יוֹסֵף", מוֹסַד הַהֶכְשֵׁר שֶׁל שָׁ"ס, שֶׁמַּטְּרָתוֹ לְסַפֵּק מוּצָרִים כְּשֵׁרִים בִּמְחִירִים זוֹלִים, יוּכְלוּ הַגּוֹלְשִׁים לִרְאוֹת אֶת צֶוֶות הַמַּשְׁגִּיחִים שֶׁל בד"צ בֵּית יוֹסֵף בִּפְעִילוּת בְּמִפְעַל בָּשָׂר בִּבְרָזִיל.
זֹאת, בְּאֶמְצָעוּת מַצְלֵמוֹת אִינְטֶרְנֵט שֶׁיּוּצְּבוּ בַּמָּקוֹם.
הַצּוֹפִים לֹא בְּהֶכְרֵחַ יִרְאוּ אֶת הֲלִיךְ הַשְּׁחִיטָה בַּשִּׁידּוּר אַךְ יִיצְּפוּ בָּהֶם 'בִּזְמַן אֱמֶת', מְצַיֵּין סוּדְרִי וְלֹא מְפָרֵט.
אַטְרַקְצִיּוֹת נוֹסָפוֹת שֶׁמְּתוּכְנָנוֹת מֵהָאֲתָר: תְּפִילּוֹת בַּכּוֹתֶל בְּשִׁידּוּר חַי.

לִסְגּוֹר בְּשַׁבָּת אוֹ לֹא לִסְגּוֹר בְּשַׁבָּת?

רוֹב אֲתָרֵי הָאִינְטֶרְנֶט בְּנוֹשְׂאֵי דָּת בְּיִשְׂרָאֵל אֵינָם מוֹנְעִים מֵהַגּוֹלְשִׁים לְבַקֵּר בָּהֶם בְּשַׁבָּת.
מִי שֶׁגָּלַשׁ לַאֲתַר שָׁ"ס בְּסוֹף הַשָּׁבוּעַ צָפָה בִּתְמוּנָה שֶׁל הַדְלָקַת נֵרוֹת וְהוֹדָעָה שֶׁבִּישְּׂרָה כִּי הָאֲתָר סָגוּר עַד צֵאת הַשַּׁבָּת.
"הָיְיתָה הִתְחַבְּטוּת גְּדוֹלָה בָּעִנְיָין הַזֶּה", אוֹמֵר סוּדְרִי, "מַפְעִילֵי הָאֲתָרִים הַדָּתִיִּים שֶׁנִּשְׁאָרִים פְּתוּחִים בְּסוֹף הַשָּׁבוּעַ טוֹעֲנִים כִּי חֵלֶק מֵהַגּוֹלְשִׁים בַּאֲתָרִים מִתְגּוֹרְרִים בְּחוּץ לָאָרֶץ, וּבִגְלַל הֶפְרֵשֵׁי הַזְּמַנִּים, יִיתָּכֵן וְהֵם אֵינָם מְחַלְּלִים אֶת הַשַּׁבָּת.
אֲנַחְנוּ הֶחְלַטְנוּ לְהַחֲמִיר.
לְפִי הַנְּתוּנִים שֶׁלָּנוּ, 82,000 גּוֹלְשִׁים נִיסּוּ לְהִיכָּנֵס לַאֲתַר בְּסוֹף הַשָּׁבוּעַ וְלִכְאוֹרָה ,'פִסְפַסְנוּ אוֹתָם', אַךְ מִי שֶׁשּׁוֹמֵר מִצְווֹת לֹא יָכוֹל לִרְאוֹת בְּכָךְ פִסְפוּס".

לְמַעֲשֶׂה, מָה הַהֶבְדֵּל בֵּין גְּרָפִיקָה שֶׁל נֵרוֹת מוּאָרִים לְבֵין תְּכָנֶיי הָאֲתָר הָרְגִילִים? הֲרֵי הַגּוֹלְשִׁים שֶׁנִּכְנָסִים לַאֲתַר בְּשַׁבָּת בְּכָל מִקְרֶה מְחַלְּלִים אֶת הָאֲתָר וּמוֹרִידִים קְבָצִים מֵהַשָּׁרָת שֶׁל שָׁ"ס.

"עַל פִּי הַהֲלָכָה, חִילּוּל שַׁבָּת אֵינוֹ פְּעוּלָּה חַד פַּעֲמִית אֶלָּא מִתְבַּצֵּעַ בְּכָל רֶגַע וְרֶגַע בְּמַהֲלַךְ הַגְּלִישָׁה – כָּל רִעֲנוּן שֶׁל דַּף, כָּל צְפִיָּיה בְּדַף אִינְטֶרְנֵט.
אֲנִי לֹא בָּטוּחַ שֶׁכְּדַאי לִי לְסַפֵּר לְךָ אֶת הַדְּבָרִים הָאֵלֶּה כִּי כָּל עוֹד אַתָּה לֹא יוֹדֵעַ אַתָּה כְּמוֹ תִּינוֹק שֶׁנִּשְׁבָּה.
כַּאֲשֶׁר אַתָּה מוּדָע לְמַעֲשֵׂךָ חִילּוּל הַשַּׁבָּת מִתְבַּצֵּעַ בְּמֵזִיד".

אַךְ אִם הָאֲתָר נִפְתַּח מֵחָדָשׁ עִם צֵאת הַשַּׁבָּת בְּיִשְׂרָאֵל, גּוֹלְשִׁים שֶׁיִּיכָּנְסוּ לַאֲתַר מִמְּקוֹמוֹת בָּעוֹלָם בָּהֶם הַשַּׁבָּת עֲדַיִין בְּתוֹקֶף, יַחְלְלוּ אֶת הַשַּׁבָּת.

"זוֹ נְקוּדָּה טוֹבָה, אַךְ אֲנַחְנוּ פּוֹעֲלִים לְפִי שְׁעוֹת כְּנִיסָה וִיצִיאָה שֶׁל יִשְׂרָאֵל.
מִבְּחִינַת הַשָּׁמַיִים, אֲנַחְנוּ נְקִיִּים.
אִם הָיְיתָה בְּיָדֵינוּ תּוֹכְנָה שֶׁתִּמְנַע מִגּוֹלְשִׁים בְּרַחֲבֵי הָעוֹלָם לְהִיכָּנֵס לַאֲתַר בִּזְמַן הַשַּׁבָּת אֶצְלָם, הָיִינוּ מִשְׁתַּמְּשִׁים בָּהּ".
