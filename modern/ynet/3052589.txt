וַעֲדַת הַכְּסָפִים אִישְּׁרָה הַיּוֹם (ג') אֶת הַצָּעַת הָאוֹצָר לְהַעֲלוֹת אֶת שֹׁוִוי הַשִּׁימּוּשׁ עַל רֶכֶב צָמוּד.
הַעֲלָאַת הַמִּיסּוּי עַל רֶכֶב צָמוּד תִּיכָּנֵס לְתוֹקְפָהּ בְּ-1 בְּמַארְס 2005.
הַגִּידּוּל בַּהַכְנָסוֹת הַמְּדִינָה כְּתוֹצָאָה מֵאִישּׁוּר הַהַצָּעָה נֶאֱמַד בְּ-150 מִילְיוֹן שֶׁקֶל.
בָּאוֹצָר הִדְגִּישׁוּ כִּי אֵין מְדוּבָּר בְּהַעֲלָאַת מִיסִּים חֲדָשָׁה וְכִי מְדוּבָּר בְּהַעֲלָאַת מִסִּים שֶׁאוּשְּׁרָה כְּבָר עַל יְדֵי הַמֶּמְשָׁלָה כְּחֵלֶק מִמְּקוֹרוֹת הַהַכְנָסָה לַתַּקְצִיב 2005.
 
סָמַנְכַּ"ל רָשׁוּת הַמִּיסִּים, בּוֹעַז סוֹפֵר, הִצִּיג בִּפְנֵי הַוַּעֲדָה נְתוּנִים לְפִיהֶם שֹׁוִוי הַשִּׁימּוּשׁ הָרֵיאָלִי שֶׁל רֶכֶב צָמוּד בִּקְבוּצָה 2 (מִשְׁפַּחְתִּי בְּנֶפַח 1.6 לִיטֶר) עוֹמֵד עַל כְּ-3,300 שֶׁקֶל, זֹאת בְּהַשְׁוָוֹאָה לַמַּצָּב הַקַּיָּים לְפִיו מְחוּיָּיב רֶכֶב זֶה בְּשֹׁוִוי בְּגוֹבַהּ כְּ-1,000 שֶׁקֶל.
גַּם אִם מְנַטְרְלִים אֶת הֶחְזֵר הַהוֹן עֲדַיִין שֹׁוִוי הַשִּׁימּוּשׁ עוֹמֵד עַל כְּ-2,200 שֶׁקֶל, בְּעוֹד שֶׁשֹּׁוִוי הַשִּׁימּוּשׁ הֶחָדָשׁ הוּעֲלָה רַק לִ-1,280 שֶׁקֶל.
לִדְבָרָיו, גַּם בִּקְבוּצוֹת הָרֶכֶב הָאֲחֵרוֹת קַיָּימִים פְּעָרִים נִיכָּרִים בֵּין הַשֹּׁוִוי הָרֵיאָלִי לַשֹּׁוִוי בְּפוֹעַל.
 
מֵהָאוֹצָר נִמְסַר כִּי מַהֲלָךְ זֶה מִשְׁתַּלֵּב בַּמְּדִינִיּוּת לְפִיהָ מוּפְחָתִים שִׁיעוּרֵי הַמַּס עַל הַכְנָסָה, בְּמַקְבִּיל לְהַרְחָבַת בְּסִיס הַמַּס בְּאֶמְצָעוּת בִּיטּוּל פְּטוּרִים וְצִמְצוּם עִיוּוּתִים.
הוּבְהַר לַוַּועֲדָה כִּי אָחוּז קָטָן מִכְּלַל הַשְּׂכִירִים מְקַבֵּל רֶכֶב צָמוּד מִמְּקוֹם הָעֲבוֹדָה.
לְפִיכָךְ, וּלְאוֹר הָעוּבְדָּה שֶׁשֹּׁוִוי הַשִּׁימּוּשׁ בְּפוֹעַל נָמוּךְ מִזֶּה הָרֵיאָלִי, הֲרֵי שֶׁלְּמַעֲשֶׂה יִתְרַת הַשְּׂכִירִים מְסַבְסְדִים אֶת הָעוֹבְדִים שֶׁיֵּשׁ לָהֶם רֶכֶב צָמוּד.
בְּנוֹסָף, הֲרֵי שֶׁקַּיָּים מִתְאַם חִיּוּבִי בֵּין גּוֹבַהּ הַהַכְנָסָה לְאָחוּז הַמְּקַבְּלִים רֶכֶב צָמוּד, כָּךְ שֶׁהַהַכְבָּדָה בַּמִּיסּוּי עַל רֶכֶב צָמוּד נוֹפֶלֶת בְּעִיקָּרָהּ עַל בַּעֲלֵי הַהַכְנָסוֹת הַגְּבוֹהוֹת.
 
מַשְׁמָעוּת הַהַחְלָטָה הִיא הַגְדָּלַת תַּשְׁלוּם הַמַּס לָעוֹבֵד בְּגִין שֹׁוִוי הַשִּׁימּוּשׁ לָרֶכֶב הַצָּמוּד בֵּין 20 שֶׁקֶל לְחוֹדֶשׁ לְ-400 שֶׁקֶל לְחוֹדֶשׁ לְכָל הַיּוֹתֵר.
