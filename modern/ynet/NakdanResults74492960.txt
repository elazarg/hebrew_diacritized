אִישּׁוּם נֶגֶד מַדְבִּיר, שֶׁהִשְׁאִיר חוֹמֶר מְסוּכָּן בְּפִינַּת בּוּבּוֹת בְּגַן יְלָדִים
מַדְבִּיר שֶׁבִּיצֵּעַ עֲבוֹדוֹת הַדְבָּרָה בְּגַן יְלָדִים בַּשְּׁפֵלָה, הִשְׁאִיר קוּפְסַת תַּחֲלִיף חָלָב לְתִינוֹקוֹת עִם הַכִּיתּוּב "רַעַל" וּבְתוֹכָהּ חוֹמֶר הַדְבָּרָה. לְמָחֳרָת פָּעוֹטוֹת פָּתְחוּ אֶת הַקּוּפְסָה וְנָגְעוּ בַּחוֹמֶר. הוּא מוֹאֳשָׁם בְּשִׁימּוּשׁ בְּחוֹמֶר מַרְעִיל אוֹ מְסוּכָּן וְהַפְקָרָתוֹ בְּדֶרֶךְ נִמְהֶרֶת אוֹ רַשְׁלָנִית שֶׁיֵּשׁ בָּהּ כְּדֵי לְסַכֵּן חַיֵּי אָדָם

אִילָנָה קוּרִיאֵל
פּוּרְסַם: 17.02.21 , 10:17

מַדְבִּיר, שֶׁהִשְׁאִיר קוּפְסָה עַל רַעַל בְּפִינַּת הַבּוּבּוֹת בְּגַן יְלָדִים בְּאֵזוֹר הַשְּׁפֵלָה בְּתוֹךְ קוּפְסָה שֶׁל תַּחֲלִיף חָלָב לְתִינוֹקוֹת, הוֹאֳשַׁם בְּשׁוּרָה שֶׁל עֲבֵירוֹת. בְּנוֹסָף הִגִּישׁ הַמִּשְׂרָד לַהֲגָנַת הַסְּבִיבָה כְּתַב אִישּׁוּם נֶגֶד חֶבְרַת הַהַדְבָּרָה וּמְנַהֲלֶיהָ. כְּתַב הָאִישּׁוּם הוּגַּשׁ בְּתוֹם חֲקִירָה שֶׁל הַמִּשְׁטָרָה הַיְּרוּקָּה.

עַל פִּי כְּתַב הָאִישּׁוּם, בְּמַהֲלַךְ דֵּצֶמְבֶּר 2018, הַמַּדְבִּיר, שֶׁרִישְׁיוֹנוֹ מַתִּיר לוֹ לְבַצֵּעַ פְּעוּלּוֹת הַדְבָּרָה אַךְ וְרַק בְּדִירוֹת (וְלֹא בְּמִבְנִים, כְּפִי שֶׁנִּדְרָשׁ בְּבִיצּוּעַ פְּעוּלַּת הַדְבָּרָה בְּגַנֵּי יְלָדִים) הֵבִיא עִמּוֹ לְגַן יְלָדִים בְּאֵזוֹר הַשְּׁפֵלָה קוּפְסַת תַּחֲלִיף חָלָב לְתִינוֹקוֹת שֶׁעָלֶיהָ מַדְבֵּקָה עִם הַכִּיתּוּב "רַעַל" וַאֲשֶׁר הֵכִילָה מִסְפַּר קוּבִּיּוֹת שֶׁל חוֹמֶר הַדְבָּרָה בְּשֵׁם "סוֹלוֹ בְּלוֹק", הַמְּיוֹעָד לְהַדְבָּרָה תַּבְרוּאִית נֶגֶד מְכַרְסְמִים עַל-יְדֵי מַדְבִּירִים מוּרְשִׁים. 

עוֹד עוֹלֶה מִכְתַּב הָאִישּׁוּם כִּי הַחֶבְרָה וּמְנַהֲלֶיהָ נָהֲגוּ בְּאוֹתָהּ הָעֵת לְסַפֵּק לַמַּדְבִּירִים שֶׁאוֹתָם הֶעֱסִיקוּ, קוּפְסָאוֹת רֵיקוֹת שֶׁל מְזוֹן תִּינוֹקוֹת, וַעֲלֵיהֶן מַדְבֵּקָה "רַעַל", לְצוֹרֶךְ אַחְסָנַת כֵּלִים וְחוֹמְרֵי הַדְבָּרָה. לְאַחַר בִּיצּוּעַ עֲבוֹדַת הַהַדְבָּרָה, בְּתוֹךְ הַגַּן וּמְחוּצָה לוֹ, הִשְׁאִיר הַמַּדְבִּיר אֶת הָאֲרִיזָה בְּפִינַּת הַתֵּיאַטְרוֹן שֶׁבַּגַּן, לְיַד הַבּוּבּוֹת.
לְמָחֳרָת, בִּשְׁעוֹת הַבּוֹקֶר, הִבְחִינוּ כַּמָּה פָּעוֹטוֹת בַּקּוּפְסָה, וּשְׁנַיִים מֵהֶם אַף הִצְלִיחוּ לִפְתּוֹחַ אוֹתָהּ וְנָגְעוּ בְּחוֹמֶר הַהַדְבָּרָה שֶׁהָיָה בְּתוֹךְ הַקּוּפְסָה. מִשֶּׁהִבְחִינָה הַגַּנֶּנֶת בַּמִּתְרַחֵשׁ, הִיא הִרְחִיקָה אֶת הַיְּלָדִים מֵהַחוֹמֶר וּמֵהַקּוּפְסָה, וְהֵם הוּפְנוּ לִבְדִיקָה רְפוּאִית. לְמַרְבֵּה הַמַּזָּל, הִתְבָּרֵר כִּי לֹא אֵירַע נֶזֶק בְּרִיאוּתִי לִילָדִים שֶׁשִּׁיחֲקוּ בַּחוֹמֶר. 
עִם קַבָּלַת דִּיוּוּחַ עַל הַמִּקְרֶה, נִפְתְּחָה חֲקִירָה מְשׁוּתֶּפֶת עַל-יְדֵי הַמִּשְׁטָרָה הַיְּרוּקָּה שֶׁל הַמִּשְׂרָד לַהֲגָנַת הַסְּבִיבָה וּמִשְׁטֶרֶת לוֹד כְּנֶגֶד חֶבְרַת הַהַדְבָּרָה וְהַמַּדְבִּיר שֶׁבִּיצֵּעַ בְּפוֹעַל אֶת הַהַדְבָּרָה. בְּמִסְגֶּרֶת הַחֲקִירָה נֶחְקְרוּ הַגּוֹרְמִים הַמְּעוֹרָבִים בָּאֵירוּעַ וְנִלְקְחוּ דְּגִימוֹת שֶׁל הַחוֹמֶר עַל-יְדֵי הַמִּשְׁטָרָה הַיְּרוּקָּה לַמַּעְבָּדָה.
בְּנִיגּוּד לְחוֹבָה בַּחוֹק, כָּךְ נִטְעַן, הַנֶּאֱשָׁמִים לֹא דִּיוְּוחוּ לָרַשָּׁם לְעִנְיְינֵי מַדְבִּירִים עַל אוֹדוֹת הָאֵירוּעַ. 



 כְּנֶגֶד הַמַּדְבִּיר הוּגַּשׁ כְּתַב אִישּׁוּם בְּגִין הַדְבָּרָה לְלֹא רִישָּׁיוֹן מַדְבִּיר מַתְאִים; שִׁימּוּשׁ בְּתַכְשִׁיר הַדְבָּרָה שֶׁלֹּא בְּהֶתְאֵם לַהוֹרָאוֹת הַקְּבוּעוֹת בְּתָוִוית הַתַּכְשִׁיר; בִּיצּוּעַ הַדְבָּרָה שֶׁלֹּא בְּהֶתְאֵם לַהוֹרָאוֹת הַקְּבוּעוֹת בְּתָוִוית הַתַּכְשִׁיר בִּנְסִיבּוֹת מַחְמִירוֹת;  שִׁימּוּשׁ בְּחוֹמֶר מַרְעִיל אוֹ מְסוּכָּן וְהַפְקָרָתוֹ בְּדֶרֶךְ נִמְהֶרֶת אוֹ רַשְׁלָנִית שֶׁיֵּשׁ בָּהּ כְּדֵי לְסַכֵּן חַיֵּי אָדָם אוֹ לִגְרוֹם לוֹ חַבָּלָה וַהֲפָרַת חוֹבַת דִּיוּוּחַ לָרַשָּׁם וּלְמַפְעִיל עַל עִנְיָין הַקָּשׁוּר בְּהַדְבָּרָה שֶׁבּוּצְּעָה, הֶעָלוּל לִגְרוֹם לִפְגִיעָה בַּצִּיבּוּר אוֹ בַּסְּבִיבָה.
