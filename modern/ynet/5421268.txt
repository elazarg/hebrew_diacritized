רֶכֶב שֶׁחָצָה הָעֶרֶב (יוֹם חֲמִישִׁי) מִפְגַּשׁ כְּבִישׁ מְסִילָּה כַּחוֹק סָמוּךְ לְאַשְׁדּוֹד, הֶחֱלִיק - כְּכָל הַנִּרְאֶה בְּשֶׁל מֶזֶג הָאֲוִויר הַסּוֹעֵר - אִיבֵּד שְׁלִיטָה וְנִתְקַע עַל פַּסֵּי הָרַכֶּבֶת.
מַעַרְכוֹת הַהַתְרָאָה הַטֶּכְנוֹלוֹגְיוֹת שֶׁכָּלְלוּ בֵּין הַיֶּתֶר מַצְלֵמָה לְזִיהוּי תְּזוּזוֹת וְכֵן מְפַקֵּחַ הַמִּפְגָּשׁ שֶׁהוּצַּב בַּמָּקוֹם כִּבְדֶרֶךְ קֶבַע, הִתְרִיעוּ עַל הָאֵירוּעַ לְמוֹקֵד הַבְּטִיחוּת הָאַרְצִי.
הַמּוֹקֵד הִנְחָה אֶת נָהַג הָרַכֶּבֶת - שֶׁהִתְקָרְבָה לְאֵזוֹר בִּמְהִירוּת שֶׁל 140 קָמָ"שׁ - לִבְלוֹם מִיָּידִית בְּאֶמְצָעוּת עֲצִירַת חֵירוּם.
הָרַכֶּבֶת בָּלְמָה, מְפַקֵּחַ הַמִּפְגָּשׁ בַּמָּקוֹם סִייֵּעַ לַנּוֹסְעִים לְחַלֵּץ אֶת רִכְבָּם לְמָקוֹם בָּטוּחַ, וּתְנוּעַת הָרַכָּבוֹת חוּדְּשָׁה.
שִׁיר דָּנִינוֹ, אַחְרָאִית מִשְׁמֶרֶת בְּמוֹקֵד בְּטִיחוּת אַרְצִי שֶׁל רַכֶּבֶת יִשְׂרָאֵל, סִיפְּרָה לְ-ynet עַל שֶׁאֵירַע: "קִיבַּלְנוּ אֶת הַדִּיוּוּחַ מֵהַמְּפַקֵּחַ שֶׁהָיָה בַּמִּפְגָּשׁ וְהוֹצֵאנוּ דִּיוּוּחַ נוֹסָף עַל הַתַּקְרִית לְכָל הַנַּהָגִים בְּקֶשֶׁר.
הַמַּחְסוֹם נִסְגַּר כְּשֶׁרַכֶּבֶת מְהִירָה הָיְיתָה בַּדֶּרֶךְ לְשֵׁם, וְהִיא בָּלְמָה בַּעֲצִירַת חֵירוּם.
זֶה הָיָה עִנְיָין שֶׁל דַּקּוֹת עַד שֶׁהִיא הָיְיתָה מַגִּיעָה לַנְּקוּדָּה שֶׁבָּהּ נִתְקַע הָרֶכֶב".
דָּנִינוֹ, שֶׁצָּפְתָה בַּמִּתְרַחֵשׁ מֵחֲדַר הַבַּקָּרָה, הוֹסִיפָה כִּי הֶרְכֵּב שֶׁנִּתְקַע עַל הַפַּסִּים הוּרַם פִיזִית, וְכִי פִּינּוּיוֹ אָרַךְ דַּקּוֹת אֲרוּכּוֹת.
מֶרְכֶּבֶת יִשְׂרָאֵל נִמְסַר: "אָנוּ מַזְכִּירִים לִכְלַל מִשְׁתַּמְּשֵׁי הַדֶּרֶךְ לְהִיזָּהֵר בְּקִרְבַת פַּסֵּי הָרַכֶּבֶת, וּבְיִיחוּד בִּתְנָאֵי מֶזֶג הָאֲוִויר הַסּוֹעֵר".
