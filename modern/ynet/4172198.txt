אֶלְיָנִיב בַּרְדָּה וּקְבוּצָתוֹ גֵּנְק עוֹשִׂים בְּיָמִים אֵלּוּ בְּמַחֲנֵה אִימּוּנִים שָׁאנְטָלְיָה שֶׁבְּטוּרְקִיָּה, אַךְ הֵם לֹא הַבֶּלְגִּים הַיְּחִידִים שֶׁנִּמְצָאִים שָׁם.
גֵּנְק וּבַרְדָּה פָּגְשׁוּ הַיּוֹם (ו') יְרִיבָה מֵאוֹתָהּ הַלִּיגָה - בִּירְשְׁחוּט, וְהֵבִיסָה אוֹתָהּ 1:5 בִּקְלִילוּת בְּמִשְׂחַק יְדִידוּת.
 
הַשְּׁתַּיִים כְּבָר נִפְגְּשׁוּ הָעוֹנָה פַּעֲמַיִים, כְּשֶׁגַּנְק נִיצְּחָה בְּבֵיתָהּ 1:3 וְנוּצְּחָה בַּחוּץ 2:0.
הַלִּיגָה הַבֶּלְגִּית תָּשׁוּב לִפְעִילוּת בְּ-14 בְּיָנוּאָר, סוֹף הַשָּׁבוּעַ הַבָּא, כְּשֶׁגֶּנְק שֶׁבַּמָּקוֹם הַשִּׁישִּׁי תִּתְאָרֵחַ בְּמֶכְלֶן, קְבוּצָתוֹ שֶׁל לִירוֹי צְעִירִי, שֶׁתּוֹפֶסֶת אֶת הַמָּקוֹם הָעֲשִׂירִי כָּעֵת.
