הַשּׁוּתָּפוֹת בִּפְּרוֹיֶקְט יָם תֶּטִיס, קְבוּצַת דֶּלֶק וְנוֹבֵּל אֵנֶרְגִּ'י, הִקְטִינוּ חַד-צְדָדִית אֶת הֶיקֵּף הַתַּמְלוּגִים שֶׁהֵן מְשַׁלְּמוֹת לַמְּדִינָה.
הַחֲבָרוֹת, שֶׁשִּׁילְּמוּ מֵרֵאשִׁית הַשָּׁנָה רַק כִּשְׁלִישׁ מֵהַתִּמְלוּגִים שֶׁנִּדְרְשׁוּ לְשַׁלֵּם עַל יְדֵי הַמְּדִינָה, הַפְחִיתוּ אֶת הַתַּשְׁלוּמִים בִּכְ-10 מִילְיוֹן שֶׁקֶל לְחוֹדֶשׁ.

נוֹבֵּל וְדֶלֶק טוֹעֲנוֹת כִּי הַמְּדִינָה צְרִיכָה לְהַכִּיר לָהֶן בְּהוֹצָאוֹת שֶׁהָיוּ לָהֶן עַל פִּיתּוּחַ מַאַגְרֵי הַפִינִיקֶלְס הַצְּמוּדִים לַיָּם תֶּטִיס וְחִיבּוּר אַסְדַּת תָּמָר לְאַסְדַּת יָם תֶּטִיס.

מַאֲגַר יָם תֶּטִיס הָיָה עַד לִפְנֵי כַּמָּה שָׁבוּעוֹת מַאֲגַר הַגַּז הַטִּבְעִי הַיַּחַד שֶׁסִּיפֵּק גַּז טִבְעִי לְיִשְׂרָאֵל בִּקְבִיעוּת, וּכְכָזֶה גַּם הָיָה אָמוּר לִהְיוֹת מְקוֹר הַתַּמְלוּגִים הָעִיקָּרִי וְכִמְעַט הַבִּלְעָדִי לְקוּפַּת הַמְּדִינָה.
 
דּוּ"חַ וַעֲדַת שֶׁשִׁינְסְקִי הֶעֱלָה כִּי הַשִּׁיטָה הַנְּהוּגָה בְּיִשְׂרָאֵל הֵבִיאָה לִידֵי כָּךְ שֶׁבַּשָּׁנִים 2004־2009 עָמְדוּ הֶיקֵּפֵי הַתַּמְלוּגִים שֶׁשִּׁילְּמָה הַשּׁוּתָּפוֹת עַל 649 מִילְיוֹן שֶׁקֶל.
אֶלָּא שֶׁבְּשֶׁל הֲטָבוֹת מַס יִיחוּדִיּוּת (הַכָּרָה בַּפְּחָת בַּמַּאֲגָר וּבַהוֹצָאוֹת שׁוֹנוֹת‭,(‬ הִיא קִיבְּלָה הֲטָבוֹת מַס שֶׁל 710 מִילְיוֹן שֶׁקַּל – כָּךְ שֶׁבְּפוֹעַל קוּזְּזוּ הַתַּמְלוּגִים שֶׁשִּׁילְּמָה הַשּׁוּתָּפוּת.
הַוַּועֲדָה קָבְעָה כִּי יֵשׁ לְבַטֵּל אֶת הַהֲטָבוֹת בֶּעָתִיד.

לְצוֹרֶךְ חִישּׁוּבֵי הַתַּמְלוּגִים, הַמְּדִינָה מַכִּירָה לַשּׁוּתָּפוּיוֹת בַּהוֹצָאוֹת שֶׁיֵּשׁ לָהֶן עַל פִּיתּוּחַ מַאֲגָרִים חֲדָשִׁים.
 שׁוּתָּפוּת יָם תֶּטִיס פָּנְתָה לָאַחֲרוֹנָה לְמִשְׂרַד הָאֵנֶרְגִּיָּה וּבִיקְּשָׁה לְהַפְחִית אֶת הֶיקֵּף הַתַּמְלוּגִים שֶׁהִיא מְשַׁלֶּמֶת עַל רֶקַע הָעֲלוּיוֹת שֶׁהָיוּ לָהּ בְּפִיתּוּחַ מַאַגְרֵי הַפִינִיקֶלְס (כִּיסֵי גַּז קְטַנִּים שֶׁנִּקְדְּחוּ בִּתְקוּפַת הַמַּחְסוֹר בְּגַז טִבְעִי כְּדֵי לְנַסּוֹת לְהַגְדִּיל אֶת הַתְּפוּקָה) וְחִיבּוּר אַסְדַּת תָּמָר לְאַסְדַּת יָם תֶּטִיס לְצוֹרֶךְ הַזְרָמַת הַגַּז לָאָרֶץ.

עַד כֹּה, כְּשֶׁהָיוּ לְשׁוּתָּפוֹת הַשָּׂגוֹת, הַנּוֹהַג הָיָה שֶׁהֵן מְשַׁלְּמוֹת עַד שֶׁהַנּוֹשֵׂא נִבְדָּק עַל יְדֵי אַנְשֵׁי הַמִּשְׂרָד, וְאִם טַעֲנוֹתֵיהֶן מִתְקַבְּלוֹת, הַכֶּסֶף מְקוּזָּז לָהֶן מִתַּשְׁלוּמִים עֲתִידִיִּים.
 
גּוֹרְמִים בְּשׁוּתָּפוּת טוֹעֲנִים שֶׁשִּׁילְּמוּ בְּ-2012 יוֹתֵר תַּמְלוּגִים מִמָּה שֶׁמַּגִּיעַ לַמְּדִינָה לְפִי נוּסְחָה מוּסְכֶּמֶת, וְהַמְּדִינָה הִבְטִיחָה לִבְחוֹן אֶת טַעֲנוֹתֵיהֶם.
לְטַעֲנָתָם, הַמְּדִינָה לֹא גִּיבְּשָׁה אִיתָּם מַנְגָּנוֹן מְסוּדָּר לְהֶחְזֵר כְּסָפִים, וְלָכֵן לֹא יָכְלוּ לְהַגִּיעַ אִיתָּהּ לְסִיכּוּמִים.
לְטַעֲנָתָם, בַּיָּמִים הָאַחֲרוֹנִים הוֹעֳבַר לָהֶם מִתְוֶוה לְהֶסְדֵּר הֶחְזֵר כְּסָפִים, וְאִם זֶה יִכְלוֹל אֶת כָּל הַמַּרְכִּיבִים הַנִּדְרָשִׁים, בְּכַוַּונַת הַשּׁוּתָּפוּת לְשַׁלֵּם לַמְּדִינָה אֶת הַסְּכוּמִים הַמְּלֵאִים עַל פִּי חִישּׁוּבֶיהָ, וּבְמַקְבִּיל תִּימָּשֵׁךְ הַהִתְדַּיְּינוּת עַל הַהֶחְזֵר.
 
מִנּוֹבֵּל אֵנֶרְגִּ'י נִמְסַר שֶׁהִיא בּוֹחֶנֶת אֶת הַנּוֹשֵׂא מוּל מִשְׂרַד הָאֵנֶרְגִּיָּה וּתְשַׁלֵּם כָּל שֶׁקֶל שֶׁיִּידָּרֵשׁ.
בְּמִשְׂרַד הָאֵנֶרְגִּיָּה סֵירְבוּ לְהִתְיַיחֵס.

נוֹבֵּל אֵנֶרְגִּ'י הוֹדִיעָה אֶתְמוֹל כִּי בְּמַאֲגַר כָּרִישׁ יֵשׁ ‭TCF 1.8‬ גַּז טִבְעִי.
מְדוּבָּר ב-‭BCM 50‬ גַּז טִבְעִי, כַּמּוּת כְּפוּלָה מִמָּה שֶׁהָיָה בְּיָם תֶּטִיס בַּמָּקוֹר.
