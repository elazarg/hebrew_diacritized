יִיתָּכֵן מְאוֹד שֶׁעֲתִידוֹ הַמִּקְצוֹעִי שֶׁל יָאוּ מִינְג, הַסֶּנְטֶר הַסִּינִי שֶׁל יוּסְטוֹן רוֹקֵטְס, יִסְתַּיֵּים עוֹד הַקַּיִץ, לְלֹא אֶפְשָׁרוּת לַחֲזוֹר וּלְכַדְרֵר בְּמִגְרְשֵׁי הַכַּדּוּרְסַל.
מִינְג, שֶׁשִּׁיחֵק רַק חֲמִישָּׁה מִשְׂחָקִים הָעוֹנָה עֵקֶב פְּצִיעָה בַּקַּרְסוֹל, הוֹדָה בְּרֵאָיוֹן לַטֵּלֵוִויזְיָה הַסִּינִית כִּי עֲתִידוֹ לוֹט בַּעֲרָפֶל כָּבֵד.

"הַכֹּל תָּלוּי בַּקַּרְסוֹל שֶׁלִּי.
אִם אִם הוּא יִהְיֶה בְּסֵדֶר, אֶפְרוֹשׁ בְּעוֹד שְׁנָתַיִים.אִם לֹא, זֶה כַּנִּרְאֶה הַקַּיִץ הָאַחֲרוֹן שֶׁלִּי", אָמַר מִינְג בָּרֵאָיוֹן.
הַסֶּנְטֶר, שֶׁנִּבְחַר בַּמָּקוֹם הָרִאשׁוֹן בִּדְרַאפְט 2002, לֹא שִׂיחֵק כְּלָל בְּעוֹנַת 2009/10, וְסוֹבֵל מִפְּצִיעוֹת מְרוּבּוֹת בַּשָּׁנִים הָאַחֲרוֹנוֹת.

"אֲנִי יָכוֹל לָלֶכֶת וַאֲפִילּוּ לָרוּץ קְצָת, אֲבָל אֲנִי צָרִיךְ לְהַחֲזִיר לְעַצְמִי לְפָחוֹת 80 אָחוּז מֵהַכּוֹחוֹת שֶׁלִּי כְּדֵי לְשַׂחֵק.
כָּרֶגַע יֵשׁ לִי רַק כְּ-30 אָחוּז", הוֹסִיף מִינְג, שֶׁמִּתְנַשֵּׂא לְגוֹבַהּ שֶׁל 2.29 מ'.
מִי שֶׁמַּעֲנִיקָה לוֹ מוֹטִיבִיצְיָה לָשׁוּב לַמִּגְרָשִׁים הִיא בִּתּוֹ, אַיְּימִי, שֶׁטֶּרֶם רָאֲתָה אֶת אָבִיהָ מְשַׂחֵק.

"אֲנִי מְקַוֶּוה שֶׁהִיא תּוּכַל לִרְאוֹת אוֹתִי מִשְׂחָק, וְאוּלַי אֲפִילּוּ זוֹכֶה בָּאַלִּיפוּת", אָמַר מִינְג וְהוֹסִיף: "אֲנִי לֹא רוֹצָה שֶׁהִיא תִּצְפֶּה רַק בְּקַלָּטוֹת שֶׁלִּי כְּשַׂחְקָן, הִיא בְּהֶחְלֵט מַעֲנִיקָה לִי אֶקְסְטְרָה מוֹטִיבַצְיָה לְהַמְשִׁיךְ בְּהַחְלָמָה לַמְרוֹת שֶׁהָרֶגֶל שֶׁלִּי עֲדַיִין זְקוּקָה לְטִיפּוּלִים רַבִּים".

מִינְג בֶּן הַ-30 גַּם נִשְׁאַל מָה יַעֲשֶׂה בִּשְׁנַת הַפְּרִישָׁה שֶׁלּוֹ, וְלֹא כְּמוֹ כּוֹכָבִים מִזְדַּקְּנִים אֲחֵרִים, הוּא דַּוְוקָא לֹא מְעוּנְיָין לַעֲבוֹר לִקְבוּצָה שֶׁמִּתְמוֹדֶדֶת עַל הָאַלִּיפוּת.
"אַלִּיפוּת אֵינָהּ מַשֶּׁהוּ מֶרְכָּזִי אוֹ חָשׁוּב בַּתּוֹכְנִיּוֹת שֶׁלִּי", אָמַר מִינְג וְהוֹסִיף: "כָּרֶגַע אֲנִי לֹא יוֹדֵעַ אִם אֲנִי בִּכְלָל אוּכַל לָשׁוּב וּלְשַׂחֵק.
מָה שֶׁחָשׁוּב לִי בֶּעָשׂוֹר הַקָּרוֹב זֶה לְטַפֵּל בַּמִּשְׁפָּחָה וּלְהִתְעַסֵּק בַּפְּעִילוּת הַקְּהִילָּתִית שֶׁלִּי".
 
לְמִינְג, שֶׁנִּבְחַר שְׁמוֹנָה פְּעָמִים לְמִשְׂחָק הַאוֹלְסְטָאר, יֵשׁ חוֹזֶה לְעוֹנָה נוֹסֶפֶת בַּקְּבוּצָה מִטֶּקְסָס, כַּאֲשֶׁר גַּם אִיּוּם הַשְּׁבִיתָה עוֹמֵד בַּפֶּתַח.
"זֶה יִהְיֶה מְאוֹד עָצוּב עֲבוּרִי לַעֲזוֹב אֶת יוּסְטוֹן לִקְבוּצָה אַחֶרֶת", אָמַר מִינְג.
"בִּילִּיתִי שָׁם עֶשֶׂר שָׁנִים נִפְלָאוֹת מֵחַיַּי, וְזֶה יִהְיֶה קָשֶׁה מְאוֹד לַעֲזוֹב".
