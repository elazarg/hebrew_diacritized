הַנָּשִׂיא בּוּשׁ מְצַיֵּין בְּיָמִים אֵלֶּה שָׁנָה לִכְנִיסָתוֹ לַתַּפְקִיד.
בּוּשׁ אָמְנָם לֹא הִצְלִיחַ לְיַישֵּׂם רֵפוֹרְמוֹת בְּנוֹשְׂאֵי פָּנִים, וְגַם מְדִינִיּוּת הַחוּץ שֶׁלּוֹ סָפְגָה בִּיקּוֹרֶת קָשָׁה, אוּלָם עַל רֶקַע הַמִּלְחָמָה בְּאַפְגָּנִיסְטָן כִּמְעַט כָּל סִיכּוּמֵי הַשָּׁנָה הָרִאשׁוֹנָה שֶׁעָרְכוּ הָעִיתּוֹנִים וְרִשְׁתוֹת הַחֲדָשׁוֹת בְּאַרְהָ"ב מַחְמִיאִים.
גַּם אֲחוּזֵי הַתְּמִיכָה בְּבּוּשׁ בַּסְּקָרִים עֲדַיִין מֵרְקִיעֵי שְׁחָקִים.
אֶת כְּנִיסָתוֹ לַתַּפְקִיד שֶׁל בּוּשׁ לִיוּוּ אָקוֹרְדִּים צוֹרְמִים.
נִצְחוֹנוֹ בַּבְּחִירוֹת, מוּל הַמּוּעֲמָד הַדֵּמוֹקְרָטִי אֶל גּוּר הוּכְרַע רַק בְּתוֹם דִּיּוּנִים מְמוּשָּׁכִים בְּבָתֵּי הַמִּשְׁפָּט.
הֶחוֹדָשִׁים הָרִאשׁוֹנִים בַּתַּפְקִיד הִתְאַפְיְינוּ בְּבִיקּוֹרֶת קָשָׁה עַל דֵּעוֹתָיו בְּנוֹשְׂאֵי חוּץ, וּבֵינֵיהֶם הִתְחַמְּמוּת כַּדּוּר הָאָרֶץ וְתוֹכְנִית הַהֲגָנָה מִפְּנֵי טִילִים.
בַּבַּיִת הֶחָרִיף הַמִּיתּוּן הַכַּלְכָּלִי.
אוּלָם הַמִּלְחָמָה בְּאַפְגָּנִיסְטָן שִׁינְּתָה אֶת הַהִתְיַיחֲסוּת לְנָשִׂיא.
"מִנְּשִׂיא שֶׁנִּבְחַר כִּמְעַט בְּאוֹפֶן אַקְרָאִי, וְהִזְכִּיר יוֹתֵר בְּדִיחָה שֶׁל 'סָאטֶרְדֵּיי נַיְיט לַיְיב', הָפַךְ בּוּשׁ לַמְּפַקֵּד הָעֶלְיוֹן שֶׁל הַצָּבָא", אָמַר הַיּוֹעֵץ הַפּוֹלִיטִי הַדֵּמוֹקְרָטִי, בִּיל קָארִיק, לְ-CNN וְהוֹסִיף "הַמִּלְחָמָה עָזְרָה לַנָּשִׂיא לְהִתְרַחֵק מֵהַמַּחֲלוֹקוֹת שֶׁאִפְיְינוּ אֶת בְּחִירָתוֹ וְגָרְמָה לָאֶזְרָחִים לְקַבֵּל אוֹתוֹ רִגְשִׁית כִּנְשִׂיאָם".
"הַנָּשִׂיא הִשְׁתַּנָּה" אָמַר בְּרָאד פְרִימָן, אֶחָד מִיְּדִידָיו שֶׁל בּוּשׁ לָרֶשֶׁת FOX, "הוּא קוֹבֵעַ מִי יִחְיֶה וּמִי יָמוּת, מַצִּיב חַיָּילִים בִּמְקוֹמוֹת מְסוּכָּנִים.
יֵשׁ לְזֶה הַשְׁפָּעָה".
פְרִימָן מוֹסִיף שֶׁבּוּשׁ גַּם "נִרְאֶה מְבוּגָּר יוֹתֵר.
אֲנִי לֹא יוֹדֵעַ מָה בְּדִיּוּק הִשְׁתַּנָּה, אוּלַי שֶׁיעָרוֹ נַעֲשֶׂה אָפוֹר יוֹתֵר".
אֶת הַיּוֹם שֶׁשִּׁינָּה אֶת נְשִׂיאוּתוֹ, הָ-11 בְּסֶפְּטֶמְבֶּר, הֵחֵל בּוּשׁ בְּבֵית סֵפֶר יְסוֹדִי בִּפְלוֹרִידָה.
שָׁם שָׁמַע עַל מִתְקֶפֶת הַטֵּרוֹר עַל מִגְדְּלֵי הַתְּאוֹמִים בְּמֶרְכַּז הַסַּחַר הָעוֹלָמִי בִּנְיוּ יוֹרְק.
בַּשָּׁבוּעוֹת הַבָּאִים הִפְגִּין כִּישּׁוּרֵי מַנְהִיגוּת שֶׁרַבִּים בְּאָמֵרִיקָה פִּקְפְּקוּ בְּקִיּוּמָם.
עַל רֶקַע הַמִּלְחָמָה הַ"מוּצְלַחַת" בּוֹלֶטֶת הַזְנָחַת מְדִינִיּוּת הַפְּנִים שֶׁל מִמְשַׁל בּוּשׁ.
בֵּין הַיֶּתֶר אִיבֵּד בּוּשׁ אֶת הָרוֹב הָרֶפּוּבְּלִיקָנִי בַּסֵּנָאט כַּאֲשֶׁר פָּרַשׁ הַסֵּנָטוֹר גִ'ים גֵּ'פוֹרְדְס מִוֶּרְמוֹנְט לַמִּפְלָגָה הַדֵּמוֹקְרָטִית.
פְּרִישָׁה זוֹ הָיְתָה גּוֹרֵם מֶרְכָּזִי בְּכִשְׁלוֹנוֹ שֶׁל בּוּשׁ בְּהַעֲבָרַת חוּקִּים בְּנוֹשְׂאֵי הַפְּנִים, וּבֵינֵיהֶם רֵפוֹרְמוֹת בַּבִּיטּוּחַ הָרְפוּאִי וּבַבִּיטּוּחַ הַלְּאוּמִּי.
הַשָּׁנָה הָרִאשׁוֹנָה שֶׁל הַנָּשִׂיא בּוּשׁ בַּבַּיִת הַלָּבָן הִסְתַּיְּימָה בְּתַקְרִית מְבִיכָה שֶׁלֹּא הָיְתָה קְשׁוּרָה כְּלָל לְתִפְקוּדוֹ כְּנָשִׂיא וּמַדְגִּימָה יוֹתֵר מִכָּל אֶת הַשִּׁינּוּי שֶׁחָל בְּגִישָׁתָם שֶׁל הָאָמֵרִיקָנִים כְּלַפֵּי נְשִׂיאָם.
הַנָּשִׂיא צָפָה לְבַדּוֹ בְּמִשְׂחַק פוּטְבּוֹל בַּטֵּלֵוִויזְיָה כְּשֶׁנֶּחְנַק מִבֵּיְיגָּלֶה וְהִתְעַלֵּף.
תְּגוּבוֹת הָאָמֵרִיקָנִים הָיוּ אוֹהֲדוֹת וּמוּדְאָגוֹת, לְלֹא שֶׁמֶץ שֶׁל לִגְלוּג.
אִם הָאֵירוּעַ הָיָה מִתְרַחֵשׁ שָׁנָה קוֹדֶם לָכֵן, בַּיָּמִים בָּהֶם נֶחֱשָׁב הַנָּשִׂיא לְאָדָם עֲמָמִי וְלֹא מַבְרִיק בִּמְיוּחָד, אֵין סָפֵק שֶׁמְּטַר הַבְּדִיחוֹת הַמַּעֲלִיבוֹת וְהַלַּעֲגָנִיּוֹת הָיָה מְלַוֶּוה אוֹתוֹ יָמִים אֲרוּכִּים.
