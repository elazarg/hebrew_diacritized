הַמִּסְחָר בַּבּוּרְסָה בְּאַחַד הָעָם נִנְעַל הַיּוֹם (א') בַּיְּרִידוֹת שְׁעָרִים קַלּוֹת, עַל רֶקַע הָעֲלִיּוֹת בְּבוּרְסוֹת אֵירוֹפָּה וְהַיַּצִּיבוּת בְּווֹל סְטְרִיט בְּסוֹף הַשָּׁבוּעַ, וּבְצִיפִּיָּיה לְפִרְסוּם רִיבִּית בַּנְק יִשְׂרָאֵל לְנוֹבֶמְבֶּר, מָחָר.

מַדַּד תֵּל-אָבִיב 100 יָרַד בְּ-0.15% לְ-1,072.04 נְקוּדּוֹת.
מַדָּד בְּלוּטֵק יָרַד בְּ-0.49% לְ-324.21 נְקוּדּוֹת.

הַוַּועֲדָה הַמּוֹנִיטָרִית שֶׁל בַּנְק יִשְׂרָאֵל צְפוּיָה לְהוֹדִיעַ מָחָר עַל גּוֹבַהּ הָרִיבִּית לְנוֹבֶמְבֶּר.
לְפִי הַעֲרָכוֹת, הָרִיבִּית תִּישָּׁאֵר לְלֹא שִׁינּוּי, עַל 2.25%.
רָפִי גּוֹזְלָן, כַּלְכְּלָן הַמַּאקְרוֹ הָרָאשִׁי שֶׁל אִי.בִּי.אִי: "מַרְבִּית הַגּוֹרְמִים הַמְּקוֹמִיִּים תּוֹמְכִים בְּהַפְחָתַת רִיבִּית וּבֵינֵיהֶם הַיְּרִידָה בִּסְבִיבַת הָאִינְפְלַצְיָה, צֶפִי לְהֵיחָלְשׁוּת נִיכֶּרֶת בַּפְּעִילוּת כְּפִי שֶׁעוֹלֶה מִסִּקְרֵי הַצִּיפִּיּוֹת וְהַקְדָּמַת הַבְּחִירוֹת שֶׁהוֹרִידָה אֶת הַחֲשָׁשׁ מִפְּנֵי פְּרִיצָה תַּקְצִיבִית", טוֹעֵן גּוֹזְלָן.
"מִנֶּגֶד, הַשִּׁיפּוּר בַּשְּׁוָוֹקִים הַפִינַנְסִיִּים בָּחוֹדָשִׁים הָאַחֲרוֹנִים וְהַחֲשָׁשׁ מִפְּנֵי הִתְנַפְּחוּת בּוּעַת הַנַּדְלָ"ן תּוֹמְכִים בְּהוֹתָרַת הָרִיבִּית לְלֹא שִׁינּוּי", הוֹסִיף.
 
אוּרִי גְּרִינְפֵלְד, מְנַהֵל מַחְלֶקֶת הַמַּאקְרוֹ בִּפְסָגוֹת הִתְיַיחֵס בִּסְקִירָה הַשְּׁבוּעִית לְמַצָּב בִּשְׁוָוֹקַי הָעוֹלָם וְצִייֵּן כִּי "הַתִּמְחוּר הַיַּחֲסִי שֶׁל מַדְּדֵי הַמְּנָיוֹת הַמּוֹבִילִים בְּאֵירוֹפָּה נִרְאֶה כַּיּוֹם הַרְבֵּה פָּחוֹת אַטְרַקְטִיבִי מְאַשֵּׁר לִפְנֵי שָׁנָה, בְּעוֹד שֶׁבְּאַרְהָ"ב תִּמְחוּר הַמְּנָיוֹת עֲדַיִין מְעַנְיֵין אַךְ כְּדַאי לִזְכּוֹר אֶת הָעֲלִיָּיה בְּרָמַת הַסִּיכּוּן הַנּוֹבַעַת מֵחוֹסֶר הַהַסְכָּמָה בֵּין הַדֵּמוֹקְרָטִים לָרֶפּוּבְּלִיקָנִים וְהַגִּידּוּל בְּהִסְתַּבְּרוּת לְהִתְמַמְּשׁוּת תַּרְחִישׁ הַצּוּק הַפִיסְקָאלִי כְּתוֹצָאָה מִכָּךְ".
 
הַמִּסְחָר בְּבוּרְסוֹת אֵירוֹפָּה נִנְעַל בְּיוֹם שִׁישִּׁי בַּעֲלִיּוֹת בְּעִקְבוֹת הָעֲלִיָּיה הַמַּפְתִּיעָה בִּצְמִיחָה בְּאַרְהָ"ב - 2% בָּרִבְעוֹן הַשְּׁלִישִׁי - וְכֵן הָעֲלִיָּיה בְּמַדַּד אֵמוּן הַצַּרְכָנִים שֶׁל אוּנִיבֶרְסִיטַת מִישִׁיגֶן בְּאוֹקְטוֹבֶּר לְשִׂיא שֶׁל 5 שָׁנִים.
בּוּרְסוֹת ווֹל סְטְרִיט נִנְעֲלוּ בְּיַצִּיבוּת בְּשִׁישִּׁי אַךְ בְּסִיכּוּם שְׁבוּעִי אִיבְּדוּ עַד 1.7%.

אִגְרוֹת הַחוֹב שֶׁל מוֹעֲדוֹן 365 וְהַמַּשְׁבִּיר לַצַּרְכָן עָלוּ בְּ-6% וּבְ-13%, בְּהַתְאָמָה בְּעִקְבוֹת אֲסִיפַת מַחֲזִיקֵי הָאַגַ"ח שֶׁל מוֹעֲדוֹן 365 שֶׁהִתְקַיְּימָה בְּיוֹם חֲמִישִׁי וּבָהּ הֶעֱלוּ הַמַּשְׁקִיעִים אֶת חֲשָׁשׁוֹתֵיהֶם בְּיַחַס לְמַצָּבָהּ שֶׁל הַחֶבְרָה וְהִצִּיגוּ אֶת טַעֲנוֹתֵיהֶם בְּנוֹגֵעַ לְהִתְנַהֲלוּת הַהַנְהָלָה.
רָמִי שָׁבִיט, שֶׁנָּכַח בָּאֲסֵיפָה, נִיסָּה לְסַפֵּק תְּשׁוּבוֹת וְהִבִּיעַ אֵמוּן מָלֵא בַּיְּכוֹלֶת שֶׁל הַחֶבְרָה לְשָׁרֵת אֶת הַחוֹב.
בָּאֲסֵיפָה נָכְחוּ בְּעִיקָּר מַשְׁקִיעִים פְּרָטִיִּים וּמִקֶּרֶב הַמּוֹסָדִיִּים הִגִּיעוּ נָצִיגֵי הַרְאֵל, מְנוֹרָה וּמֵיטָב.

אוֹהַד מָרַאנִי, מַנְכַּ"ל הַכְשָׁרָה אֵנֶרְגִּיָּה, שֶׁחָווֹה בְּחוֹדְשַׁיִים הָאַחֲרוֹנִים שְׁנֵי קִידּוּחִים עֲקָרִים בְּרִישְׁיוֹנוֹת "שָׂרָה" וּ"מִירָה", אָמַר שֶׁתּוֹצְאוֹת הַקִּידּוּחַ הָיוּ אַכְזָבָה קָשָׁה: "הִשַּׂגְנוּ הֶישֵּׂגִים רַבִּים אֲבָל לְצַעֲרֵינוּ לָמַדְנוּ עַל בְּשָׂרֵינוּ אֶת הַקּוֹשִׁי וְהַסְּטָטִיסְטִיקָה הָעוֹלָמִית שֶׁל בִּיצּוּעַ קִידּוּחִים עַל סְמַךְ סְקָרִים סַסְמִיִּים בִּלְבַד.
לְטַעֲנָתוֹ, גִּיּוּס כְּסָפִים אֵינוֹ עַל הַפֶּרֶק אוּלָם בְּשָׁלָב זֶה הוּא לֹא נִדְרַשׁ".
הַדְּבָרִים נֶאֶמְרוּ בְּמִסְגֶּרֶת וְעִידַת הָאֵנֶרְגִּיָּה וְהָעֲסָקִים שֶׁל אֶקוֹ אֵנֶרְגִּ'י שֶׁנֶּעֱרֶכֶת בְּרָמַת גַּן.
 
מַנְכַּ"ל אַבְנֵר וְיוֹ"ר דֶּלֶק קִידּוּחִים גִּדְעוֹן תַּדְמוֹר אָמַר בַּוְּעִידָה כִּי "אִי אֶפְשָׁר לְדַבֵּר עַל כְּנִיסָה נוֹסֶפֶת שֶׁל צַנֶּרֶת גַּז בְּלִי אוֹפֶק שֶׁל יִיצּוּא גַּז אוֹ דִּיבּוּר עַל פְּרוֹיֶקְט חָדָשׁ שֶׁל גַּז, קְרִי 'לִוְויָתָן'.
לִדְבָרָיו, "הַמַּפְתֵּחַ הוּא אִישּׁוּר מַסְקְנוֹת וַעֲדַת צֶמַח, שֶׁלָּקַח לָהּ שָׁנָה לַעֲבוֹד לְרַבּוֹת קִיּוּם שֶׁל שִׁימּוּעַ צִיבּוּרִי.
אִישּׁוּר מַסְקָנוֹת אֵלּוּ יָקֵל עָלֵינוּ לְהַמְשִׁיךְ אֶת הַפִּיתּוּחַ.
הַמְּדִינָה מְקַדֶּמֶת אֶת כְּנִיסַת הַגַּז הַצְּפוֹנִית וַאֲנַחְנוּ נְבָרֵךְ כָּל הַחְלָטָה", אָמַר.
 
נִכְשַׁל נִיסָּיוֹן גִּיּוּס הָאַגַ"ח שֶׁל פְּלָאזָה סֶנְטֵרְס, חֶבְרַת הַבַּת שֶׁל אֶלְבִּיט הַדְמָיָה שֶׁבִּשְׁלִיטַת מוֹטִי זִיסֶר, כָּךְ נוֹדַע לְ"כַּלְכָּלִיסְט".
הַחֶבְרָה בִּיקְּשָׁה לְגַיֵּיס 100 מִילְיוֹן דּוֹלָר בְּהַנְפָּקַת אַגַ"ח דּוֹלָרִית.
עַל רֶקַע הֵיעָדֵר בִּיקּוּשִׁים מַסְפִּיקִים נִכְשַׁל הַגִּיּוּס אוֹתוֹ הוֹבִיל בַּנְק אוֹף אָמֵרִיקָה - מֵרִיל לִינְץ'.
הָאַגַ"ח הָיָה אָמוּר לָשֵׂאת רִיבִּית שְׁנָתִית שֶׁל 12%.

חַיִּים סַבָּן מִתְקַשֶּׁה בִּרְכִישַׁת פַּרְטְנֵר.

הָאצִ'יסוֹן מִתְנַגֶּדֶת לִפְרִיסַת הַחוֹב בְּמִסְגֶּרֶת הָעִסְקָה שֶׁבָּהּ רוֹצָה חַיִּים סַבָּן לִרְכּוֹשׁ אֶת פַּרְטְנֵר.
עַל פִּי הַמִּתְוֶוֹה שֶׁהוּצַּע לַחֶבְרָה, יוֹעֳבַר לָהּ תַּשְׁלוּם מִיָּידִי שֶׁל 30-20 מִילְיוֹן דּוֹלָר, וְיֶתֶר הַחוֹב שֶׁאָמוּר לִהְיוֹת מְשׁוּלָּם בְּפֶבְּרוּאָר 2014 יִיפָּרֵס עַל פְּנֵי אַרְבַּע שָׁנִים.
הַחוֹב לְסִינְיָם עוֹמֵד עַל 300 מִילְיוֹן דּוֹלָר.

קֶרֶן הַהַשְׁקָעוֹת רִיד אַלְסְוְוִיר (Reed Elsevier), הַמַּחֲזִיקָה בְּ-18% מִמְּנָיוֹת בָּבִּילוֹן, פָּנְתָה לְנוֹעָם לָנִיר שֶׁמַּחֲזִיק בִּשְׁלִיטָה בַּחֶבְרָה (24%) דֶּרֶךְ חנפ"ק וְלִיוֶורְמוֹר, וּבִיקְּשָׁה מִמֶּנּוּ לִשְׁקוֹל מַהֲלָךְ שֶׁל הַצָּעַת רֶכֶשׁ לִמְנָיוֹת בָּבִּילוֹן וַהֲפִיכָתָהּ לִפְרָטִית.
כָּךְ נוֹדַע לְ"כַּלְכָּלִיסְט".
עַל פִּי הַהַעֲרָכוֹת, הָרֶקַע לַמַּהֲלָךְ נָעוּץ בְּבִיצּוּעֵי מְנִיַּית בְּיָלוֹן וּבַתְּגוּבָה הַמְּאוּפֶּקֶת לִכְאוֹרָה שֶׁל הַשּׁוּק לְהֶסְכֵּם שֶׁנֶּחְתַּם בֵּין בָּבִּילוֹן לְגוּגֶל בַּשָּׁבוּעַ שֶׁעָבַר לְפִיתּוּחַ אַפְּלִיקַצְיַית הַתַּרְגּוּם שֶׁל בָּבִּילוֹן הַמְּיוֹעֶדֶת כַּיּוֹם לְמַחְשְׁבִים אִישִׁיִּים לִפְּלַטְפוֹרְמוֹת סֵלוֹלָרִיּוֹת וְנִיסָּיוֹן לְמִימּוּשׁ מוֹדֵל עִסְקִי מַתְאִים לְבָּבִּילוֹן בִּתְחוּם הֲסֵלוֹלָר.
