מֶחְקָר: הַחִיסּוּן שֶׁל פַיְיזֶר עָשׂוּי לִהְיוֹת פָּחוֹת יָעִיל בַּאֲנָשִׁים עִם הַשְׁמָנַת יֶתֶר
מֶחְקָר חָדָשׁ מָצָא שֶׁאֲנָשִׁים עִם הַשְׁמָנַת יֶתֶר פִּיתְּחוּ פָּחוֹת נוֹגְדָנִים לְאַחַר מְנַת הַחִיסּוּן הַשְּׁנִיָּיה שֶׁל פַיְיזֶר. הַהַעֲרָכָה: תִּידָּרֵשׁ מְנַת חִיסּוּן נוֹסֶפֶת אוֹ מִינּוּן גָּבוֹהַּ יוֹתֵר. מוּמְחֶה יִשְׂרְאֵלִי: "בְּעָיַית הַהַשְׁמָנָה חַיֶּיבֶת לִהְיוֹת בְּרֹאשׁ סֵדֶר הָעֲדִיפוּיוֹת שֶׁל מַעֲרֶכֶת הַבְּרִיאוּת"
ynet
פּוּרְסַם: 01.03.21 , 11:05

נְתוּנִים מַדְאִיגִים: חוֹקְרִים בְּאִיטַלְיָה מָצְאוּ שֶׁאֲנָשִׁים הַסּוֹבְלִים מֵהַשְׁמָנַת יֶתֶר מְיַיצְּרִים רַק מִמַּחֲצִית מִכַּמּוּת הַנּוֹגְדָנִים לְאַחַר שֶׁקִּיבְּלוּ אֶת הַמָּנָה הַשְּׁנִיָּיה שֶׁל הַחִיסּוּן לְקוֹרוֹנָה שֶׁל חֶבְרַת פַיְיזֶר, בְּהַשְׁוָואָה לְמִתְחַסְּנִים עִם מַדָּד BMI תַּקִּין. הַחוֹקְרִים טוֹעֲנִים שֶׁלַּמְרוֹת שֶׁמּוּקְדָּם עֲדַיִין לָדַעַת הַאִם הַחִיסּוּן אֵינוֹ יָעִיל בְּקֶרֶב אוֹתָהּ אוּכְלוּסִיָּיה, הַמִּמְצָאִים עֲשׂוּיִים לְלַמֵּד שֶׁאֲנָשִׁים הַסּוֹבְלִים מֵהַשְׁמָנַת יֶתֶר יִזְדַּקְּקוּ לִמְנַת חִיסּוּן נוֹסֶפֶת כְּדֵי לְהַבְטִיחַ הֲגָנָה מְסַפֶּקֶת מִפְּנֵי הַנְּגִיף שֶׁמְּשַׁתֵּק אֶת הָעוֹלָם בַּשָּׁנָה הָאַחֲרוֹנָה.
הַמֶּחְקָר הֶחָדָשׁ, שֶׁפּוּרְסַם אֶתְמוֹל (יוֹם רִאשׁוֹן) וְטֶרֶם עָבַר בִּיקּוֹרֶת עֲמִיתִים, בָּדַק אֶת כַּמּוּת הַנּוֹגְדָנִים שֶׁל הַמִּתְחַסְּנִים לְאַחַר קַבָּלַת שְׁתֵּי מְנוֹת חִיסּוּן שֶׁל פַיְיזֶר. הַחוֹקֵר אֵלְדוֹ וׇנוּטִי וַעֲמִיתִיו עָרְכוּ אֶת הַמֶּחְקָר בְּקֶרֶב 248 בְּנֵי אָדָם שֶׁעוֹבְדִים בְּמַעֲרֶכֶת הַבְּרִיאוּת הַצִּיבּוּרִית בְּאִיטַלְיָה, שִׁבְעָה יָמִים מִמּוֹעֵד קַבָּלַת מְנַת הַחִיסּוּן הַשְּׁנִיָּיה. הַחוֹקְרִים מָצְאוּ שֶׁ-99.5% מֵהַנֶּחְקָרִים פִּיתְּחוּ נוֹגְדָנִים בִּתְגוּבָה לַחִיסּוּן, אַךְ הַתְּגוּבָה הָיְיתָה פְּחוּתָה בְּקֶרֶב אֲנָשִׁים שֶׁסּוֹבְלִים מֵהַשְׁמָנַת יֶתֶר.

מֶחְקָרִים קוֹדְמִים שֶׁפּוּרְסְמוּ הַשָּׁנָה מָצְאוּ שֶׁהַשְׁמָנַת יֶתֶר – מַדַּד מַסַּת גּוּף (BMI) 30 וּמַעֲלָה –  מַגְדִּילָה אֶת הַסִּיכּוּן לִתְמוּתָה מִקּוֹרוֹנָה בְּכִמְעַט 50%, וְכֵן מַעֲלָה אֶת הַסִּיכּוּן לְהִתְאַשְׁפֵּז בְּבֵית חוֹלִים כְּתוֹצָאָה מִסִּיבּוּכֵי הַנְּגִיף בְּ-113%. הַסִּיבָּה לְכָךְ הִיא שֶׁאֲנָשִׁים עִם הַשְׁמָנַת יֶתֶר סוֹבְלִים בְּמַרְבִּית הַמִּקְרִים גַּם מִמַּחֲלוֹת רֶקַע נוֹסָפוֹת, כְּמוֹ מַחֲלוֹת לֵב אוֹ סוּכֶּרֶת מִסּוּג 2, הַמַּגְבִּירוֹת אֶת הַפְּגִיעָה שֶׁלָּהֶם כְּתוֹצָאָה מִמַּחֲלָה זִיהוּמִית. מִשְׁקָל עוֹדֶף עָלוּל גַּם לִגְרוֹם לְשִׁינּוּיִים מֵטַבּוֹלִיִּים, כְּגוֹן עֲמִידוּת לְאִינְסוּלִין וְדַלֶּקֶת, הַמַּקְשִׁים עַל הַגּוּף לְהִילָּחֵם בְּזִיהוּמִים.
הַשְׁמָנָה וְקוֹרוֹנָה: הַסַּכָּנוֹת וּמָה נִיתָּן לַעֲשׂוֹת כְּדֵי לִמְנוֹעַ?
"בְּעָיַית הַהַשְׁמָנָה מְחַיֶּיבֶת טִיפּוּל בַּעֲדִיפוּת עֶלְיוֹנָה" 
"מִכֵּיוָון שֶׁהַשְׁמָנַת יֶתֶר מְהַוָּוה גּוֹרֵם סִיכּוּן עִיקָּרִי לְתַחֲלוּאָה וְלִתְמוּתָה בְּקֶרֶב חוֹלֵי קוֹרוֹנָה, חוֹבָה לְהֵיעָרֵךְ עִם תּוֹכְנִית חִיסּוּנִים שׁוֹנָה וִיעִילָה יוֹתֵר עֲבוּר אוֹתָהּ אוּכְלוּסִיָּיה", אָמְרוּ הַחוֹקְרִים בְּאִיטַלְיָה, "לַמְרוֹת שֶׁיֵּשׁ צוֹרֶךְ בְּמֶחְקָרִים נוֹסָפִים, לִנְתוּנִים אֵלּוּ עֲשׂוּיוֹת לִהְיוֹת הַשְׁלָכוֹת חֲשׁוּבוֹת עַל פִּיתּוּחַ אַסְטְרָטֶגְיוֹת חִיסּוּן נֶגֶד הַקּוֹרוֹנָה, וּבִפְרָט עֲבוּר אֲנָשִׁים בַּעֲלֵי מִשְׁקָל עוֹדֶף. אִם הַנְּתוּנִים יְאוּמְּתוּ עַל יְדֵי מֶחְקָרִים גְּדוֹלִים יוֹתֵר, יִהְיֶה צוֹרֶךְ לִשְׁקוֹל אֶפְשָׁרוּת לָתֵת לַאֲנָשִׁים בַּעֲלֵי מִשְׁקָל עוֹדֶף מָנָה נוֹסֶפֶת שֶׁל הַחִיסּוּן אוֹ מִינּוּן גָּבוֹהַּ יוֹתֵר".

"מֶחְקָרִים רַבִּים הוֹכִיחוּ שֶׁהַשְׁמָנָה מְהַוָּוה גּוֹרֵם סִיכּוּן מַשְׁמָעוּתִי לְסִיבּוּכִים אוֹ תְּמוּתָה בְּמִקְרִים שֶׁל הַדְבָּקָה בַּנְּגִיף הַקּוֹרוֹנָה", אוֹמֵר לְ-ynet ד"ר רָז הַגּוֹאֵל, רוֹפֵא מוּמְחֶה וְהַמְּנַהֵל שֶׁל הַמֶּרְכָּז הָרְפוּאִי לְטִיפּוּל בְּהַשְׁמָנָה. "יַחַד עִם זֹאת, הַמֶּחְקָר הַזֶּה קוֹבֵעַ שֶׁגַּם הַחִיסּוּן עֲבוּר הָאוּכְלוּסִיָּיה הַסּוֹבֶלֶת מֵהַשְׁמָנַת יֶתֶר עָלוּל לִהְיוֹת פָּחוֹת אֵפֶקְטִיבִי וְיִיתָּכֵן שֶׁהַמְּטוּפָּלִים יִזְדַּקְּקוּ לִמְנַת חִיסּוּן נוֹסֶפֶת. בְּעָיַית הַהַשְׁמָנָה מְחַיֶּיבֶת טִיפּוּל בַּעֲדִיפוּת עֶלְיוֹנָה אֵצֶל מְקַבְּלֵי הַהַחְלָטוֹת בְּמַעֲרֶכֶת הַבְּרִיאוּת וְהַשָּׁבוּעַ יְצַיְּינוּ בְּרַחֲבֵי הָעוֹלָם אֶת יוֹם הַמּוּדָעוּת לְהַשְׁמָנָה. צָרִיךְ לִזְכּוֹר שֶׁיְּרִידָה בְּמִשְׁקָל הִיא תַּהֲלִיךְ מִתְמַשֵּׁךְ שֶׁאֵין בּוֹ פִּתְרוֹנוֹת קֶסֶם. הַתַּהֲלִיךְ חַיָּיב לְהִתְבַּצֵּעַ לְצַד לִיוּוּי, פִּיקּוּחַ וּמַעֲקָב שֶׁל צֶוֶות מְיוּמָּן וּמִקְצוֹעִי. בְּהֶיבֵּט זֶה, מַגֵּפַת הַקּוֹרוֹנָה יְכוֹלָה לְהַוּוֹת הִזְדַּמְּנוּת לְמִי שֶׁסּוֹבֵל מֵהַשְׁמָנָה - לְהַתְחִיל טִיפּוּל לִירִידָה בַּמִּשְׁקָל תּוֹךְ שִׁינּוּי אוֹרַח הַחַיִּים".
