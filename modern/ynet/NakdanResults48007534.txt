נְתַנְיָהוּ לְבָגָ"ץ: לַיּוֹעָמָ"שׁ אֵין סַמְכוּת לִקְבּוֹעַ אֶת הֶסְדֵּר נִיגּוּד הָעִנְיָינִים שֶׁלִּי
פְּרַקְלִיטֵי רה"מ טוֹעֲנִים בִּתְגוּבָה לָעֲתִירָה לְבָגָ"ץ כִּי רַק מְבַקֵּר הַמְּדִינָה מוּסְמָךְ לִקְבּוֹעַ אֶת נִיגּוּדֵי הָעִנְיָינִים שֶׁלּוֹ. "אֵין בְּסַמְכוּתוֹ שֶׁבְּדִין בִּכְדֵי לְהַכְרִיז עַל רה"מ כְּמָצוּי בְּנִיגּוּד עִנְיָינִים אוֹ לִכְבּוֹל אוֹתוֹ"
יָעֵל פְרִידְסוֹן
פּוּרְסַם: 05.11.20 , 22:56

רֹאשׁ הַמֶּמְשָׁלָה בִּנְיָמִין נְתַנְיָהוּ הִגִּישׁ לְבָגָ"ץ הָעֶרֶב (חֲמִישִׁי) אֶת תְּגוּבָתוֹ לַעֲתִירוֹת בְּעִנְיַין הֶסְדֵּר נִיגּוּד הָעִנְיָינִים שֶׁלּוֹ, בְּעִנְיָינִים הַנּוֹגְעִים לְמַעֲרֶכֶת אֲכִיפַת הַחוֹק. בִּתְגוּבָה טוֹעֵן נְתַנְיָהוּ כִּי לַיּוֹעֵץ הַמִּשְׁפָּטִי לַמֶּמְשָׁלָה אֵין סַמְכוּת לִקְבּוֹעַ אֶת הֶסְדֵּר נִיגּוּד הָעִנְיָינִים, אֶלָּא רַק לִמְבַקֵּר הַמְּדִינָה.

בִּתְגוּבָה שֶׁהוּגְּשָׁה מַדְגִּישׁ נְתַנְיָהוּ כִּי הַכְּלָל קוֹבֵעַ חוֹבַת הִיוּוֹעֲצוּת בַּיּוֹעֵץ הַמִּשְׁפָּטִי לַמֶּמְשָׁלָה לְגַבֵּי נִיגּוּד עִנְיָינִים אַךְ וְרַק בְּמַצָּב שֶׁבּוֹ הַשַּׂר עַצְמוֹ (אוֹ רה"מ) קָבַע כִּי הוּא נִמְצָא בְּנִיגּוּד עִנְיָינִים, וְאֵין מְדוּבָּר בְּמַתַּן סַמְכוּת לַיּוֹעֵץ הַמִּשְׁפָּטִי לַמֶּמְשָׁלָה לִקְבּוֹעַ שֶׁמָּא הוּא מָצוּי בְּנִיגּוּד עִנְיָינִים אוֹ לֹא.

"אֵין בְּסַמְכוּתוֹ שֶׁבְּדִין בִּכְדֵי לְהַכְרִיז עַל רֹאשׁ הַמֶּמְשָׁלָה כְּמָצוּי בְּנִיגּוּד עִנְיָינִים, אֵין בְּסַמְכוּתוֹ בִּכְדֵי לִכְבּוֹל אֶת רֹאשׁ הַמֶּמְשָׁלָה, אֵין בְּסַמְכוּתוֹ בִּכְדֵי לִמְנוֹעַ מֵרֹאשׁ הַמֶּמְשָׁלָה מִלְּבַצֵּעַ תַּפְקִידָיו שֶׁבַּדִּין הַקְּבוּעִים בַּחֲקִיקַת הַיְּסוֹד", נִכְתַּב בִּתְגוּבָה שֶׁהִגִּישׁוּ עוֹרְכֵי דִּינוֹ שֶׁל נְתַנְיָהוּ.
נְתַנְיָהוּ הוֹדִיעַ כִּי הוּא מְקַבֵּל "מִיּוֹזַמְתּוֹ, כִּי לְמַעַן הַנִּרְאוּת הַצִּיבּוּרִית, הוּא לֹא יִתְעָרֵב בְּמִינּוּי יוֹעֵץ מִשְׁפָּטִי לַמֶּמְשָׁלָה, פְּרַקְלִיט הַמְּדִינָה וּמַפְכָּ"ל הַמִּשְׁטָרָה וְהוּא מְקַבֵּל עַל עַצְמוֹ אֶת הָאָמוּר בְּחַוַּות הַדַּעַת, לְמַעֵט הַהַגְבָּלוֹת הַנּוֹגְעוֹת לַחֲקִיקָה, וְהַגְבָּלוֹת הַנּוֹגְעוֹת לְשִׁיטַת הַמִּינּוּיִים", וְכֵן אֶת הַהַגְבָּלוֹת בְּנוֹגֵעַ לְמִשְׂרַד הַתִּקְשׁוֹרֶת.
מַנְדֶּלְבְּלִיט בְּסֶפְּטֶמְבֶּר: "נְגוֹנֵן עַל הַדֵּמוֹקְרַטְיָה"

עִם זֹאת, הוּא הוֹדִיעַ כִּי "אֵין כָּל מָקוֹם לְהַגְבִּיל אֶת רֹאשׁ הַמֶּמְשָׁלָה בְּעִנְיְינֵי חֲקִיקָה בַּעֲלַת הַשְׁפָּעָה עַל הַהֲלִיךְ הַפְּלִילִי. הַהַגְבָּלוֹת שֶׁחָלוֹת עַל רֹאשׁ הַמֶּמְשָׁלָה בְּעִנְיְינֵי חֲקִיקָה הֵן הַהַגְבָּלוֹת עַל-פִּי כְּלָלֵי הָאֵתִיקָה שֶׁל חַבְרֵי הַכְּנֶסֶת. הַגְבָּלוֹת אֵלּוּ אֵינָן עִנְיָין לַהֲלִיךְ דְּנַן הָעוֹסֵק בְּהֶסְדֵּר נִיגּוּד הָעִנְיָינִים הַמִּינְהָלִי".

בְּנוֹסָף טוֹעֵן נְתַנְיָהוּ כִּי אֵין סִיבָּה לְהַטִּיל עַל רֹאשׁ הַמֶּמְשָׁלָה מִגְבָּלוֹת בְּנוֹגֵעַ לִבְחִירַת שׁוֹפְטִים חֲדָשִׁים מֵעֵבֶר לַמִּגְבָּלוֹת הַחַלּוֹת עַל שָׂרִים אֲחֵרִים.
בְּשׁוּלֵי הַתְּגוּבָה רוֹמֵז נְתַנְיָהוּ כִּי יִפְעַל לְפַצֵּל אֶת תַּפְקִיד הַיּוֹעָמָ"שׁ לְאַחַר מִינּוּי מַחֲלִיפוֹ שֶׁל מַנְדֶּלְבְּלִיט. "אֵין כָּל מְנִיעָה שֶׁרֹאשׁ הַמֶּמְשָׁלָה יוֹבִיל מַהֲלָךְ שֶׁל פִּיצּוּל תַּפְקִיד הַיּוֹעֵץ הַמִּשְׁפָּטִי לַמֶּמְשָׁלָה הַבָּא מִתַּפְקִיד רֹאשׁ הַתְּבִיעָה הַפְּלִילִית. מְדוּבָּר בְּסוּגְיָיה עֶקְרוֹנִית שֶׁאֵינָהּ קְשׁוּרָה כְּלָל וְעִיקָּר לַהֲלִיךְ הַפְּלִילִי כְּנֶגֶד רֹאשׁ הַמֶּמְשָׁלָה", נִכְתַּב.
כְּמוֹ כֵן טוֹעֲנִים עוֹרְכֵי הַדִּין שֶׁל נְתַנְיָהוּ כִּי הַהֶסְכֵּם הַקּוֹאָלִיצְיוֹנִי עִם כָּחוֹל לָבָן לֹא הִגְבִּיל אֶת מִינּוּיֵי הַבְּכִירִים לַחֲצִי שָׁנָה, אֶלָּא לִ"תְּקוּפַת חֵירוּם" שֶׁמִּשְּׁכָהּ לֹא הוּגְדַּר.
מַנְדֶּלְבְּלִיט הֶעֱבִיר בְּיוֹם שֵׁנִי אֶת הַסֵּדֶר נִיגּוּד הָעִנְיָינִים לְרֹאשׁ הַמֶּמְשָׁלָה נְתַנְיָהוּ. הַהֶסְדֵּר נוֹעַד לִמְנוֹעַ מִנְּתַנְיָהוּ שִׁימּוּשׁ בְּכוֹחוֹ כְּדֵי לְהַשִּׂיג יִתְרוֹנוֹת בַּמִּשְׁפָּט שֶׁבּוֹ הוּא מוֹאֳשָׁם בְּשׁוֹחַד, מִרְמָה וַהֲפָרַת אֱמוּנִים. בַּהֶסְדֵּר נִקְבַּע כִּי נְתַנְיָהוּ לֹא יִתְעָרֵב בְּמִינּוּי יוֹעָמָ"שׁ, פְּרַקְלִיט מְדִינָה וּמַפְכָּ"ל, וְכֵן בְּמִינּוּי שׁוֹפְטִים לַמְּחוֹזִי בִּירוּשָׁלַיִם וְלָעֶלְיוֹן, הַחְלָטוֹת הַנּוֹגְעוֹת לְעֵדִים בַּמִּשְׁפָּט, פְּעִילוּת שֶׁל מִשְׂרַד הַתִּקְשׁוֹרֶת וַחֲקִיקָה בַּעֲלַת הַשְׁפָּעָה עַל הַהֲלִיךְ הַפְּלִילִי
