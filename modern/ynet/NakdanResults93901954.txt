זִיהוּם הַזֶּפֶת: "אֵין כֶּתֶם שֶׁמֶן נוֹסָף בַּדֶּרֶךְ לְיִשְׂרָאֵל"
בְּמִסְגֶּרֶת שִׁיתּוּף פְּעוּלָּה בֵּין הַמִּשְׂרָד לַהֲגָנַת הַסְּבִיבָה לְצַהַ"ל נִסְרְקוּ כְּ-2,200 קמ"ר בַּיָּם הַתִּיכוֹן - כְּ-120 עַד 140 ק"מ מַעֲרָבִית לְיִשְׂרָאֵל. בִּסְרִיקוֹת לֹא הִתְגַּלָּה אַף כֶּתֶם שֶׁמֶן. בְּמַקְבִּיל, כְּ-1,300 אִישׁ הִמְשִׁיכוּ הַיּוֹם בְּנִיקּוּי חוֹפֵי הָאָרֶץ מִזִּיהוּם הַזֶּפֶת
אִילָנָה קוּרִיאֵל
פּוּרְסַם: 28.02.21 , 19:19

בַּמִּשְׂרָד לַהֲגָנַת הַסְּבִיבָה עִדְכְּנוּ הָעֶרֶב (יוֹם א'), לְאַחַר בְּדִיקָה נוֹסֶפֶת, כִּי אֵין אַף כֶּתֶם שֶׁמֶן נוֹסָף שֶׁעוֹשֶׂה אֶת דַּרְכּוֹ לְיִשְׂרָאֵל - זֹאת בְּנִיגּוּד לְפִרְסוּמִים אֶתְמוֹל.
עֲדַיִין נִפְלֶטֶת זֶפֶת בְּחֵיפָה

בְּהוֹדָעַת הַמִּשְׂרָד נֶאֱמַר כִּי הַיּוֹם יָצְאָה גִּיחַת סִיּוּר יַמִּי אֶל הַכֶּתֶם, שֶׁהָאִינְדִּיקַצְיָה לְמִיקּוּמוֹ הָיְיתָה 120-140 ק"מ מַעֲרָבִית לְיִשְׂרָאֵל - בַּשֶּׁטַח הַנִּמְצָא בִּתְחוּם הַמַּיִם הַכַּלְכָּלִיִּים שֶׁל מִצְרַיִם. זֹאת, בְּהֶמְשֵׁךְ לִיצִיאַת כְּלִי טַיִס מְאוּיָּשׁ מֵרָחוֹק (כטמ"מ) אֶתְמוֹל. בְּמִסְגֶּרֶת שִׁיתּוּף הַפְּעוּלָּה שֶׁל הַמִּשְׂרָד לַהֲגָנַת הַסְּבִיבָה עִם זְרוֹעַ הַיָּם שֶׁל צַהַ"ל, וְכַמָּה יְחִידוֹת בְּחֵיל הָאֲוִויר, בּוּצְּעָה סְרִיקָה וְנִבְדַּק שֶׁטַח שֶׁל 2,200 קָמָ"ר. 
שָׁחוֹר בְּחוֹף בְּאַשְׁדּוֹד

מֵהַמִּשְׂרָד לַהֲגָנַת הַסְּבִיבָה נִמְסַר כִּי בְּטִיסַת סִיּוּר זוֹ נִשְׁלְלוּ כָּל הַמִּמְצָאִים הַקּוֹדְמִים וְלֹא זוֹהֲתָה נוֹכְחוּת נֵפְט אוֹ דֶּלֶק. עוֹד נִמְסַר כִּי בְּמִסְגֶּרֶת הַטִּיסָה נִבְדַּק גַּם כֶּתֶם כֵּהֶה בְּאוֹרֶךְ שֶׁל כְּ-20 ק"מ שֶׁאוּתַּר, וְנִשְׁלְלָה הֱיוֹתוֹ סוּג שֶׁל דֶּלֶק אוֹ שֶׁמֶן. בַּמִּשְׂרָד לַהֲגָנַת הַסְּבִיבָה הִדְגִּישׁוּ כִּי מֵעֵבֶר לְכָךְ, גַּם בִּתְמוּנוֹת הַלַּוְויָין סַנְסִיטֵל 2 כָּעֵת, אֵין כָּל אִינְדִּיקַצְיָה לְכֶתֶם בְּדַרְכּוֹ לְיִשְׂרָאֵל אוֹ לְמִצְרַיִם. מִמְצָאֵי גִּיחַת הַסִּיּוּר יוּעַבְרוּ לַגּוֹרְמִים הָרֵלֵוַונְטִיִּים. הַמִּשְׂרָד לַהֲגָנַת הַסְּבִיבָה יַמְשִׁיךְ לַעֲקוֹב אַחַר הִתְפַּתְּחוּיוֹת נוֹסָפוֹת.

בַּמִּשְׂרָד לַהֲגָנַת הַסְּבִיבָה הִדְגִּישׁוּ כִּי מַרְבִּית הַזִּיהוּיִים הַמַּגִּיעִים מִתַּצְלוּמֵי לַוְויָין הֵם False-Positive, כְּלוֹמַר: הַלַּוְויָין מְאַתֵּר הַפְרָעָה, אַךְ בְּמַרְבִּית הַמִּקְרִים מְקוֹרָהּ אֵינוֹ שֶׁמֶן. הַלַּוְויָין קוֹלֵט אָנוֹמַלְיוֹת מִסּוּגִים שׁוֹנִים שֶׁיְּכוֹלִים לְהִידָּמוֹת לְכֶתֶם דֶּלֶק כְּמוֹ: הֶבְדֵּלֵי טֶמְפֵּרָטוּרָה, מֵי שִׁיפּוּלִים שֶׁל אֳונִיּוֹת הַמּוּזְרָמִים בְּהֶתְאֵם לְכָל דִּין, רַחַף יָם, עֵשֶׂב יָם הַמְּעוֹרָב עִם חֶלְקִיקִים מְרַחֲפִים בַּמַּיִם, פְּרִיחַת אַצּוֹת, הֶבְדֵּלֵי עוֹמֶק וְעוֹד.
הַשָּׂרָה לַהֲגָנַת הַסְּבִיבָה, גִּילָּה גַּמְלִיאֵל, אָמְרָה: "אֲנַחְנוּ מוֹדִים לְצַהַ"ל עַל שִׁיתּוּף הַפְּעוּלָּה הַמְּצוּיָּן, שֶׁסִּייֵּעַ לָנוּ לִשְׁלוֹל אֶת קִיּוּמוֹ שֶׁל כֶּתֶם נֵפְט אוֹ דֶּלֶק מַעֲרָבִית לְיִשְׂרָאֵל. הַכֶּתֶם כְּפִי שֶׁהֶעֱלָה הַחֲשָׁד בַּלַּוְויָינִים הָיָה רָחוֹק מְאוֹד, אַךְ חָשׁוּב הָיָה לְהָסִיר אֶת הָאִיּוּם. אָנוּ מַמְשִׁיכִים לַעֲקוֹב אַחַר הִתְפַּתְּחוּיוֹת נוֹסָפוֹת, כְּכָל שֶׁיִּהְיוּ".
בְּמַקְבִּיל, נִמְשְׁכוּ הַיּוֹם פְּעוּלּוֹת הַנִּיקָּיוֹן בְּחוֹפֵי הָאָרֶץ. בַּמִּשְׂרָד לַהֲגָנַת הַסְּבִיבָה אָמְרוּ כִּי חוֹף רֹאשׁ הַנִּקְרָה הָפַךְ לְבַעַל זִיהוּם בֵּינוֹנִי-כָּבֵד בִּגְלַל נְחִיתוֹת זֶפֶת חֲדָשׁוֹת מֵהַיּוֹם.



מִי שֶׁמִּשְׁתַּתְּפִים בִּפְעוּלּוֹת הַנִּיקָּיוֹן הֵם גַּם אַנְשֵׁי רָשׁוּת הַטֶּבַע וְהַגַּנִּים וּמִתְנַדְּבִים. כְּ-1,300 מִתְנַדְּבִים מִקֶּרֶב קְבוּצוֹת מְאוּרְגָּנוֹת וִיחִידִים סִייְּעוּ הַיּוֹם לְפַקָּחֵי וְעוֹבְדֵי רָשׁוּת הַטֶּבַע וְהַגַּנִּים בְּמַאֲמַצֵּי נִיקּוּי וְשִׁיקּוּם הַחוֹפִים מִזִּיהוּם הַזֶּפֶת בְּחוֹפֵי שְׁמוּרוֹת הַטֶּבַע וְהַגַּנִּים הַלְּאוּמִּיִּים מִצָּפוֹן וְעַד דָּרוֹם. תַּחֲנוֹת קְלִיטַת הַמִּתְנַדְּבִים נָכוֹן לְעַתָּה מְתוּכָּנוֹת לִפְעוֹל עַד יוֹם שְׁלִישִׁי, כַּאֲשֶׁר בַּיָּמִים הַגְּשׁוּמִים הַצְּפוּיִים בְּהֶמְשֵׁךְ הַשָּׁבוּעַ תֵּיעָשֶׂה הֲפוּגָה בִּפְעִילוּת הַמִּתְנַדְּבִים מִשִּׁיקּוּלֵי בְּטִיחוּת.

מֵהַעֲרָכַת מַצָּב שֶׁהִתְקַיְּימָה הַיּוֹם בִּרְשׁוּת הַטֶּבַע וְהַגַּנִּים עוֹלֶה כִּי נָכוֹן לְעַתָּה עִיקַּר הָאֶתְגָּר בְּפִינּוּי וְנִיקּוּי הַזֶּפֶת הוּא בַּמִּשְׁטָחִים הַסַּלְעִיִּים בַּחוֹפִים שֶׁבִּשְׁמוּרוֹת הַטֶּבַע וְהַגַּנִּים הַלְּאוּמִּיִּים. יַחַד עִם זֹאת עֲדַיִין נִדְרֶשֶׁת עֲבוֹדָה מְאוּמֶּצֶת וְנִרְחֶבֶת לְפִינּוּי הַזֶּפֶת מֵהַמִּשְׁטָחִים הַחוֹלִיִּים בָּהֶם הַזֶּפֶת שֶׁנּוֹתְרָה הִיא בְּעִיקָּר בְּגוֹדֶל שֶׁל גּוּשִׁים קְטַנִּים יַחֲסִית (גּוֹדֶל מַטְבֵּעַ/פְּתִיתִים). 
הָרָשׁוּת מַמְשִׁיכָה לִבְחוֹן אֶמְצָעִים טֶכְנוֹלוֹגִיִּים, מֵכָנִיִּים וְכִימִיִּים מִתְקַדְּמִים לְפִינּוּי הַזֶּפֶת מֵהַמִּשְׁטָחִים הַחוֹלִיִּים והסלעיים כָּךְ שֶׁיִּהְיוּ יְעִילִים וְלֹא יָסֵבּוּ נְזָקִים אֲחֵרִים לַסְּבִיבָה הַחוֹפִית. בְּנוֹסָף, הָרָשׁוּת פּוֹעֶלֶת לְקַדֵּם תָּכְנִית נִיטּוּר שֶׁתְּבוּצַּע בְּשִׁילּוּב כָּל הַגּוֹרְמִים הָרֵלֵוַונְטִיִּים לְמַעֲקָב אַחַר זִיהוּם הַזֶּפֶת וְהַשְׁלָכוֹתָיו.

זִיהוּם הַזֶּפֶת הִגִּיעַ בַּיָּמִים הָאַחֲרוֹנִים גַּם לְחוֹפֵי דְּרוֹם לְבָנוֹן, וְגַם שָׁם נֶעֶרְכוּ פְּעוּלּוֹת לְנִיקְיוֹן הַחוֹפִים.
