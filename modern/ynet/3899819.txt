בֵּית מִשְׁפַּט הַשָּׁלוֹם בְּתֵל אָבִיב דָּחָה תְּבִיעָה בִּסְכוּם 480 אֶלֶף שֶׁקֶל שֶׁהִגִּישׁ הַבַּמַּאי הָאוֹסְטְרִי, פָּאוּלוּס מֶנְקֶּר, נֶגֶד הַתֵּיאַטְרוֹן הַקָּאמֵרִי.
מֶנְקֶּר טָעַן בַּתְּבִיעָה, שֶׁהוּגְּשָׁה בְּ-2006, כִּי הַתֵּיאַטְרוֹן הָעִירוֹנִי שֶׁל תֵּל אָבִיב הֵפֵר אֶת תְּנָאֵי הַהֶסְכֵּם שֶׁחָתַם עִמּוֹ בְּנוֹגֵעַ לַהַצָּגָה "עֵד רְאִייָּה", שֶׁעָלְתָה בַּקָּאמֶרִי בְּ-2003 בְּכִיכּוּבוֹ שֶׁל אִיתַי טִירָאן.

בְּעוֹד שֶׁמֶּנְקֶּר טָעַן בִּתְבִיעָתוֹ כִּי הַהֲפָקָה הוֹעַלְתָה בְּנִיגּוּד לַחוֹזֶה מִחוּץ לְאוּלַם הַבַּיִת לְלֹא יְדִיעָתוֹ אוֹ הַסְכָּמָתוֹ, טָעֲנוּ בַּקָּאמֶרִי לָהָדָ"ם וְהִדְגִּישׁוּ כִּי כָּל הַגָּשַׁת הַתְּבִיעָה יְסוֹדָהּ בִּשְׁאִיפַת נָקָם מִצַּד הַבַּמַּאי.

בֵּית הַמִּשְׁפָּט קִיבֵּל בְּאוֹפֶן חֶלְקִי אֶת גִּרְסַת הַתֵּיאַטְרוֹן, וְהַשּׁוֹפֵט יַעֲקֹב שַׁיְינְמָן כָּתַב בִּפְסַק הַדִּין: "מַר מֶנְקֶּר, שֶׁעֲדַיִין כּוֹעֵס עַל הַקָּאמֵרִי, הָיָה נָכוֹן לִשְׁנוֹת גִּרְסוֹתָיו לְעִנְיַין יְדִיעַת הַעֲלָאַת הַמַּחֲזֶה מִחוּץ לָאוּלָם 4 וְהַכֹּל בְּמַטָּרָה שֶׁהֵם 'יְשַׁלְּמוּ'.
שִׁינּוּיֵי גִּרְסָאוֹת אֵלֶּה מוֹבִילִים גַּם הֵם לְמַסְקָנָה כִּי מֶנְקֶּר יָדַע עַל הַעֲלָאַת הַמַּחֲזֶה מִחוּץ לְאוּלַם הַבַּיִת, וְלֹא הִבִּיעַ כָּל מְחָאָה בִּזְמַן אֱמֶת אֶלָּא רַק בְּדִיעֲבַד.
בְּהֶקְשֵׁר זֶה עוֹלוֹת תְּמִיהוֹת לְגַבֵּי מְנִיעָיו הַנִּסְתָּרִים שֶׁל הַבַּמַּאי לְהַגָּשַׁת הַתּוּבְעָנָה הַמּוּסְבֶּרֶת רַק בְּכַעֲסוֹ עַל בִּיטּוּל מַסַּע הוֹפָעוֹת בְּוִוינָה".
