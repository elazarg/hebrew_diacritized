הַפְּעִילוּת בַּמֶּשֶׁק הִמְשִׁיכָה לִהְיוֹת נְמוּכָה גַּם בַּמַּחֲצִית הַשְּׁנִיָּיה שֶׁל 2003, לַמְרוֹת שֶׁהִסְתַּמְּנָה בְּמַחֲצִית הִתְאוֹשְׁשׁוּת בִּפְעִילוּת הַמִּגְזָר הָעִסְקִי, שֶׁמְּקוֹרָהּ בְּגִידּוּל הַיִּיצּוּא וְהַצְּרִיכָה הַפְּרָטִית.
כָּךְ עוֹלֶה מֵהַסְּקִירָה הַתְּקוּפָתִית שֶׁל הַהִתְפַּתְּחוּיוֹת הַכַּלְכָּלִיּוֹת שֶׁפִּרְסֵם הַיּוֹם (א') בַּנְק יִשְׂרָאֵל.

עַל פִּי הַסְּקִירָה, הַגִּידּוּל בַּבִּיקּוּשִׁים נָבַע מִשְּׁנֵי מְקוֹרוֹת: עֲלִיָּיה בְּתוֹצָר הַמְּדִינוֹת הַמְּפוּתָּחוֹת, לְאַחַר תְּקוּפָה אֲרוּכָּה שֶׁל שֵׁפֶל, שֶׁהֵבִיא לְהִתְרַחֲבוּת הַסַּחַר הָעוֹלָמִי וּבְמַקְבִּיל

לְגִידּוּל הַיִּיצּוּא הַיִּשְׂרְאֵלִי בְּעִיקָּר בִּתְחוּם הַהַיי-טֵק.
וְכֵן הַהֲקָלָה הַמְּסוּיֶּימֶת בַּמַּצָּב הַבִּיטְחוֹנִי וְהִתְמַתְּנוּת בְּאִי הַוַּודָּאוּת הַכַּלְכָּלִית.

"עַל אַף סִימָנֵי הַתַּפְנִית", אוֹמְרִים בְּבַנְק יִשְׂרָאֵל, "מְעִידָה הִתְנַהֲגוּת הַיַּצְרָנִים עַל אִי וַדָּאוּת לְגַבֵּי עוֹצְמַת הַהִתְאוֹשְׁשׁוּת וְיַצִּיבוּתָהּ.
גַּם קֶצֶב גִּידּוּלוֹ שֶׁל הַתּוֹצָר הָעוֹמֵד עַל 1.2% בִּשְׁנַת 2003, מִתּוֹכָם 0.8% בַּמַּחֲצִית הַשְּׁנִיָּיה - הָיָה נָמוּךְ מִמָּה שֶׁמְּאַפְשֵׁר מִצְבּוֹר הַהוֹן וְכוֹחַ הָאָדָם בַּמְּדִינָה וּלְפִיכָךְ לֹא מִיצִּינוּ אֶת הַפּוֹטֶנְצְיָאל שֶׁל הַמֶּשֶׁק.
כְּמוֹ כֵן, הִצְטַמְצֵם הַתּוֹצָר לְנֶפֶשׁ זוֹ הַשָּׁנָה הַשְּׁלִישִׁית בִּרְצִיפוּת".

בְּנוֹסָף מַדְגִּישִׁים בְּבַנְק יִשְׂרָאֵל, כִּי "בַּמַּחֲצִית הַשְּׁנִיָּיה שֶׁל שְׁנַת 2003 הָיְתָה רָמַת הַפְּעִילוּת נְמוּכָה וְשִׁיעוּר הָאַבְטָלָה נִשְׁאַר גָּבוֹהַּ - 10.7% -  גַּם בָּרִבְעוֹן הַשְּׁלִישִׁי שֶׁל שְׁנַת 2003".

נִמְשַׁךְ הַמַּשְׁבֵּר בַּעֲנַף הַבְּנִיָּיה.

בְּנִיגּוּד לַהִתְאוֹשְׁשׁוּת בְּעַנְפֵי הַתַּעֲשִׂיָּיה, הַמִּסְחָר, הַשֵּׁירוּתִים וְהַתַּיָּירוֹת נִמְשַׁךְ הַשֵּׁפֶל בַּעֲנַף הַבְּנִיָּיה וְאַף הֶחָרִיף.
עֲנַף הַבְּנִיָּיה, שֶׁמְּהַוֶּוה בְּדֶרֶךְ כְּלָל אִינְדִּיקָטוֹר לְמִיתּוּן, סֵבֶל בִּשְׁנַת 2003 מִיְּרִידָה נִיכֶּרֶת בְּמִסְפַּר הַתְחָלוֹת הַבְּנִיָּיה.

בִּסְקִירָה נֶאֱמַר, כִּי לְאַחַר הִתְאוֹשְׁשׁוּת זְמַנִּית בִּמְכִירוֹת בַּעֲנַף הַבְּנִיָּיה בָּרִבְעוֹן הַשֵּׁנִי, כַּנִּרְאֶה עֵקֶב יִיסּוּף הַשֶּׁקֶל שֶׁהֵבִיא לִירִידַת הַמְּחִירִים הַדּוֹלָרִיִּים שֶׁל הַדִּירוֹת, יָרַד מִסְפַּר הַדִּירוֹת הַחֲדָשׁוֹת שֶׁנִּמְכְּרוּ בְּמַהֲלַךְ הַחוֹדָשִׁים יוּלִי-אוֹקְטוֹבֶּר בְּ-13% לְעוּמַּת הַתְּקוּפָה הַמַּקְבִּילָה בְּ-2002.
גַּם הֶיצֵּעַ הַדִּירוֹת הָעוֹמְדוֹת לִמְכִירָה הִמְשִׁיךְ לָרֶדֶת בַּתְּקוּפָה הָאֲמוּרָה וְהִגִּיעַ לְשֵׁפֶל לְלֹא תַּקְדִּים.

הַהַעֲרָכוֹת בְּבַנְק יִשְׂרָאֵל הֵן, כִּי הַיְּרִידָה הַגְּדוֹלָה בִּמְלַאי הַדִּירוֹת עֲשׂוּיָה לִהְיוֹת בֶּעָתִיד הַגּוֹרֵם הַמֶּרְכָּזִי לְגִידּוּל הַפְּעִילוּת בַּמֶּשֶׁק.
