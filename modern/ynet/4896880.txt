עִם תּוֹם הַסִּיבּוּב הָרִאשׁוֹן אֵין בִּכְלָל עוֹרְרִין מִי הִיא הַקְּבוּצָה הַטּוֹבָה בַּלִּיגָה.
הָאַלּוּפָה, מַכַּבִּי תֵּל  אָבִיב, מַמְשִׁיכָה אֶת הַמּוֹמֶנְטוּם מֵהָעוֹנָה שֶׁעָבְרָה וְאַחֲרֵי 11 מַחְזוֹרִים הִיא לְלֹא הֶפְסֵד, מְדוֹרֶגֶת  בְּפִסְגַּת הַטַּבְלָה בְּפַעַר שֶׁל 6 נְקוּדּוֹת מִבְנֵי הֶרְצְלִיָּה ורמה"ש בִּמְקוֹמוֹת 2-3.
יְתֵרָה מִכָּךְ, אֶפְשָׁר  לְהַגִּיד שֶׁרַק קָטַסְטְרוֹפָה מִקְצוֹעִית תִּמְנַע מִמֶּנָּה אֶת יִתְרוֹן הַבֵּיתִיּוֹת לְאוֹרֶךְ כָּל הַפְּלֶיְיאוֹף.
מָחָר  (ו', 16:00) פּוֹגְשִׁים הַצְּהוּבִּים אֶת א.ס רָמַת הַשָּׁרוֹן בְּרֶבַע גְּמַר גְּבִיעַ הַמְּדִינָה, וְהֵם רוֹצִים  לְהַמְשִׁיךְ בְּאוֹתָהּ הַדֶּרֶךְ.
מִצַּד שֵׁנִי, קְבוּצָתוֹ שֶׁל רְאוּבֵן יוֹסֶפוֹלְסְקִי הִיא כְּכָל הַנִּרְאֶה הַיְּרִיבָה הַקָּשָׁה  בְּיוֹתֵר שֶׁמַּכַּבִּי ת"א יְכוֹלָה הָיְיתָה לִפְגּוֹשׁ בַּשָּׁלָב הַזֶּה.
אַחֲרֵי פְּתִיחַת עוֹנָה לֹא טוֹבָה, הִצְלִיחַ  יוֹסֶפוֹלְסְקִי לְיַישֵּׁר אֶת הַשּׁוּרוֹת, וְאֶת הַסִּיבּוּב הָרִאשׁוֹן סָגְרָה רמה"ש עִם חֲמִישָּׁה נִיצְחוֹנוֹת  רְצוּפִים.
הַשְּׁאֵלָה הַגְּדוֹלָה הִיא הָאֵם עוֹד קְבוּצָה שֶׁנִּרְאֵית מְצוּיָּן הָעוֹנָה הִיא בְּנֵי הֶרְצְלִיָּה.
הַחֲבוּרָה  שֶׁל בְּנֵי אִינְגֵּל מַפְתִּיעָה אֲפִילּוּ אֶת עַצְמָהּ וְעִם סִדְרַת נִיצְחוֹנוֹת מַרְשִׁימָה הִתְמַקְּמָה לָהּ כָּאָמוּר בַּמָּקוֹם  הַשֵּׁנִי.
בְּנוֹסָף, הֶרְצְלִיָּה בִּיצְּעָה חִיזּוּק מַשְׁמָעוּתִי לִקְרַאת סִיּוּם הַסִּיבּוּב כְּשֶׁצֵּירְפָה אֶת דּוֹר גֶּבַע  וְנָדָב רוּבִּין.
הֶרְצְלִיָּה תִּתְאָרֵחַ בְּבֵית מַכַּבִּי אֵצֶל סְגָנִית הָאַלּוּפָה, מַכַּבִּי רִאשׁוֹן לְצִיּוֹן (16:00).
 הַקְּבוּצָה שֶׁל סַעַר פְרֶנְקֶל סוֹף סוֹף מַתְחִילָה לְשַׂחֵק אֶת הַכַּדּוּרְיָד שֶׁהַמְּאַמֵּן כ"כ רָצָה לִרְאוֹת.
צֵירוּפוֹ  שֶׁל אִיבָן קְרָאצִ'יץ' לְצַד חֲזָרָתוֹ לִפְעִילוּת שֶׁל רוֹבֵּרְט מַרְקוּטִיץ', הָפְכוּ אֶת רַאשְׁלָ"צ לְאַחַת הַחֲזָקוֹת  בַּלִּיגָה.
לִרְאָיָיה הִיא עִם חֲמִישָּׁה נִיצְחוֹנוֹת רְצוּפִים וְעָלְתָה מֵהַתַּחְתִּית כְּבָר לַמָּקוֹם הָרְבִיעִי.
הַהֶפְסֵד  הַטֶּכְנִי בַּמִּפְגָּשׁ מוּל אָסָ"א ת"א שִׁיבֵּשׁ אֶת תּוֹכְנִיּוֹת הַמִּיקּוּם שֶׁל רַאשְׁלָ"צ, אֲבָל אֵין סָפֵק שֶׁבַּסִּיבּוּב  הַשֵּׁנִי הִיא תִּתְקַדֵּם בְּמַעֲלֵה הַטַּבְלָה.
יֶתֶר מִשְׂחֲקֵי רֶבַע הַגְּמָר: עִירוֹנִי רְחוֹבוֹת - הַפּוֹעֵל רַאשְׁלָ"צ  (15:00), אָסָ"א ת"א - א.כ נֵס צִיּוֹנָה (16:00).
