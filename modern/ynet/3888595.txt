מַדּוּעַ הָרַמַטְכָּ"ל לְשֶׁעָבַר, דָּן חָלוּץ, מְקַשֵּׁר בְּעַמּוּד הַפֶיְיסְבּוּק שֶׁלּוֹ לִידִיעָה עַל תַּעֲרוּכָה שֶׁמַּאֲשִׁימָה אֶת  יִשְׂרָאֵל בִּגְנֵיבַת מַיִם, וּמֵגִיב "יִשְׂרָאֵל (הַפֶּלֶג הַפָאשִׁיסְטִי שֶׁל הֶעָם הַיְּהוּדִי) אֵינָהּ יְכוֹלָה לִגְנוֹב מַיִם, הִיא פָּשׁוּט לוֹקַחַת אוֹתָם, כְּפִי שֶׁאֱלוֹהִים הוֹרָה לָהּ בַּחֲלוֹם"? לֹא מְדוּבָּר בֶּחָלוּץ עַצְמוֹ שֶׁמִּתְחַבֵּר לְפֶיְיסְבּוּק וּ וּמְהַגֵּג.
זֶהוּ פְּרוֹפִיל מְזוּיָּיף שֶׁמּוּפְעָל עַל יְדֵי מִי שֶׁמְּזַהֶה עַצְמוֹ כְּחַיָּיל לְשֶׁעָבַר, וְטוֹעֵן שֶׁזּוֹ סָאטִירָה, וּמַאֲשִׁים אֶת חֲלוּץ בְּפִשְׁעֵי מִלְחָמָה.
פֶיְיסְבּוּק, הָרֶשֶׁת הַחֶבְרָתִית הַגְּדוֹלָה  בָּעוֹלָם, מְאַפְשֶׁרֶת לְכָל אֶחָד לִפְתּוֹחַ פְּרוֹפִיל בְּכָל שָׁם וּלְהַעֲלוֹת אֵלָיו תְּמוּנוֹת.
מֵאַחַר שֶׁלֹּא נִיתָּן לְאַמֵּת אֶת זֶהוּתוֹ שֶׁל מִי שֶׁפּוֹתֵחַ פְּרוֹפִיל, תּוֹפְעוֹת הִתְחַזּוּת הִינָּן שְׁכִיחוּת בָּאִינְטֶרְנֵט מֵאָז וּמֵעוֹלָם,  וּבִפְרָט בָּרְשָׁתוֹת הַחֶבְרָתִיּוֹת מֵאָז שֶׁפָּרְחוּ בַּשָּׁנִים הָאַחֲרוֹנוֹת.
יְדוּעָנִים זוֹכִים לִפְעָמִים לִפְּרוֹפִילִים  מְזוּיָּיפִים לְשֵׁם לַעַג וְסָאטִירָה, כַּאֲשֶׁר בְּחֵלֶק מֵהַמִּקְרִים מוּבְהָר כִּי מָדוֹר בִּפְּרוֹפִיל מְזוּיָּיף.
בְּעַמּוּד  הַפֶיְיסְבּוּק שֶׁל חָלוּץ נִמְצֵאת תְּמוּנָתוֹ כְּשֶׁהוּא יוֹשֵׁב מְחוּיָּיךְ עִם קַסְדָּה.
בִּכְתוֹבֶת מְגוּרָיו מְצוּיָּין הֶלְסִינְקִי, פִינְלַנְד.
בְּמַבָּט רִאשׁוֹן אֶפְשָׁר לִטְעוֹת וְלַחֲשׁוֹב כִּי אָכֵן מְדוּבָּר בְּחָלוּץ, אַךְ בִּקְרִיאָה  מַעֲמִיקָה יוֹתֵר נִיתָּן לִרְאוֹת כִּי נִמְתַּחַת בִּיקּוֹרֶת נוֹקֶבֶת וְקִיצוֹנִית עַל הָרַמַטְכָּ"ל לְשֶׁעָבַר, לִכְאוֹרָה  בִּשְׁמוֹ.
בֵּין תְּחוּמִי הָעִנְיָין שֶׁל "דָּן חָלוּץ" בַּפְּרוֹפִיל מוֹפִיעִים "הַפְּצָצָה אֲרוּכַּת טְווֹח", וּ"תִּיקֵי הַשְׁקָעוֹת".
בִּסְעִיף הַדֵּעוֹת הַפּוֹלִיטִיּוֹת נִכְתַּב "לַהֲרוֹג אֶת כּוּלָּם" וּבִסְעִיף הַהִשְׁתַּיְּיכוּת הַדָּתִית -  "הָאֱלוֹהִים שֶׁלְּךָ מַתִּיר לַהֲרוֹג אֲנָשִׁים? אֲנִי בָּעִנְיָין".
הַמּוּזִיקָה הָאֲהוּבָה עָלָיו הִיא "רַעַשׁ שֶׁל  נָפָּאלְם בַּבּוֹקֶר" - פָּרַפְרָזָה עַל הַצִּיטּוּט הַמְּפוּרְסָם מִסֶּרֶט הַמִּלְחָמָה "אַפּוֹקָלִיפְּסָה עַכְשָׁיו".
מַפְעִיל  הַפְּרוֹפִיל מְפַרְסֵם תְּמוּנוֹת מֵהַפְגָּנוֹת מִבִּילְעִין וּמְקוֹמוֹת נוֹסָפִים וְכֵן מֵגִיב בְּתוֹר "חָלוּץ" עַל  כַּתָּבוֹת וּפִרְסוּמִים נוֹסָפִים אֲשֶׁר עוֹלִים בָּרֶשֶׁת, בִּלְשׁוֹן מַלְעִיזָה וּמַתְרִיסָה.
עַל כַּתָּבָה בְּנוֹשֵׂא פָּרָשַׁת הָרִיגּוּל הָאַחֲרוֹנָה שֶׁנֶּחְשְׂפָה הַשָּׁבוּעַ כָּתַב "אִם הֶעְלַמְתִּי אֶזְרָח וְאַף אֶחָד לֹא יוֹדֵעַ, הַאִם הוּא בֶּאֱמֶת נֶעֱלָם?".
מַפְעִיל הַפְּרוֹפִיל, שֶׁלֹּא מוּכָן לִמְסוֹר אֶת שְׁמוֹ, כּוֹתֵב לְ-ynet בִּתְשׁוּבָה כִּי הוּא "אֶזְרָח  יִשְׂרְאֵלִי, בֶּן לְאֵם יְהוּדִיָּיה, שֵׁירַתְתִּי בַּצָּבָא וּבְאוֹפֶן מְסוּיָּים תַּחַת פִּיקּוּדוֹ שֶׁל רא"ל חָלוּץ", הוּא  אוֹמֵר שֶׁהוּא בֵּן 30 מֵרַעֲנָנָה בַּמָּקוֹר, "שֶׁשּׁוֹכֶנֶת בְּחֶלְקָה עַל אַדְמַת הַכְּפָר טָבּסוּר", לִדְבָרָיו.
אוֹתוֹ  מַפְעִיל מִסְתּוֹרִי אוֹמֵר כִּי הוּא נִפְגַּע וּמִתְרַעֵם עַל הַגְדָּרַת הַפְּרוֹפִיל כְּ'מְזוּיָּיף'.
"הַפְּרוֹפִיל הַזֶּה  אֵינֶנּוּ מְזוּיָּיף אֶלָּא סָאטִירִי" הוּא כּוֹתֵב, "זֶה הֶבְדֵּל חָשׁוּב, מַבְחִינָה אוֹמָּנוּתִית, וְגַם חוּקִּית".
לִדְבָרָיו אֲנָשִׁים מְבִינִים אֶת הָאִירוֹנְיָה בִּדְבָרָיו וּמוּבְהָר כִּי הוּא לֹא מְנַסֶּה לְהִתְחַזּוֹת אֶלָּא לְהַצִּיג  דְּמוּת בְּאוֹפֶן סָאטִירִי.
בְּפֶיְיסְבּוּק חֲבֵרִים, עַל פִּי הַחֶבְרָה, יוֹתֵר מִשְּׁנֵי מִילְיוֹן גּוֹלְשִׁים יִשְׂרְאֵלִים.
לִדְמוּתוֹ שֶׁל חֲלוּץ בְּפֶיְיסְבּוּק יֵשׁ כָּעֵת 260 חֲבֵרִים בִּלְבַד וְהוּא פָּעִיל בָּרֶשֶׁת, וּמֵגִיב גַּם לִפְעִילוּת אֵצֶל חֲבֵרִים אֲחֵרִים בְּנוֹסָף לְעִדְכּוּנִים שֶׁהוּא מְפַרְסֵם בָּעַמּוּד שֶׁלּוֹ.
מַדּוּעַ הֶחֱלִיט לָצֵאת בִּמְחָאָה נֶגֶד  חֲלוּץ בְּפֶיְיסְבּוּק? "הַהַחְלָטָה נוֹבַעַת מִכַּמָּה סִיבּוֹת", הוּא כּוֹתֵב, "בֵּינֵיהֶן תִּיעוּב עָמוֹק שֶׁאֲנִי חָשׁ כְּלַפֵּי רַב אַלּוּף חָלוּץ מִתּוֹקֶף הֶיכֵּרוּתֵנוּ, וּמִמַּעֲשָׂיו וּדְרָכָיו לְהִתְבַּטֵּא.
פְּלַטְפוֹרְמַת הַפֶיְיסְבּוּק מְשַׁמֶּשֶׁת מְקוֹם מִפְגָּשׁ חֶבְרָתִי לַאֲנָשִׁים שֶׁלְּעוֹלָם לֹא הָיוּ נִפְגָּשִׁים אוֹ מְשׂוֹחֲחִים, וְכֵן בְּמָה לְהַבָּעַת עֲמָדוֹת  בְּצוּרָה חוֹפְשִׁית".
"דָּן חָלוּץ הוּא פּוֹשֵׁעַ מִלְחָמָה מְתוֹעָב לֹא פָּחוֹת מִשְּׁאָר הַגֵּנֵרָלִים וְהַפּוֹלִיטִיקָאִים  בְּיִשְׂרָאֵל", הוּא מַאֲשִׁים, "יֵשׁ לִי עִנְיָין אִישִׁי בּוֹ מִכֵּיוָון שֶׁהוּא וַאֲנִי שׁוּתָּפִים לַפֶּשַׁע.
שׁוּב,  מְדוּבָּר בְּסָאטִירָה וְלֹא בְּזִיּוּף מִכֵּיוָון שֶׁאֵינֶנִּי מִתְיַימֵּר לִהְיוֹת דָּן חָלוּץ, אֶלָּא רַק לְהַצִּיג אוֹתוֹ  וְאֶת אִישִׁיּוּתוֹ בְּצוּרָה מְגוֹחֶכֶת וְסָאטִירִית, וְזֶה עַל מְנַת שֶׁאֲנָשִׁים יְנַסּוּ לְהָעֵיז וּלְהַבִּיט בְּמִפְלֶצֶת  בְּלִי לִבְרוֹחַ בְּפַחַד.
אֶפְשָׁר אוּלַי גַּם לִקְרֹא לְזֶה הוּמוֹר שָׁחוֹר".
מִלִּשְׁכָּתוֹ שֶׁל חָלוּץ נִמְסַר "אָנוּ  דּוֹחִים אֶת הַנּוֹשֵׂא בִּשְׁאָט נֶפֶשׁ".
