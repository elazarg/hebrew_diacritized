בֶּהָלָה בְּסִין: וִירוּס מֵעַיִים אַלִּים שֶׁהִתְפַּשֵּׁט בַּמְּדִינָה הָרַג עַד כֹּה 22 יְלָדִים וְהִדְבִּיק אֲלָפִים.
מֶמְשֶׁלֶת סִין הֵפִיצָה אַזְהָרָה לְרַחֲבֵי הָעוֹלָם בְּנִסָּיוֹן לִמְנוֹעַ הִתְפַּשְּׁטוּת שֶׁל הַוִּוירוּס הַקַּטְלָנִי מֵעֵבֶר לַמְּדִינָה.
 
בְּהוֹדָעָה שֶׁהוֹצִיא מִשְׂרַד הַבְּרִיאוּת הַסִּינִי, נוֹקֶטֶת הַמְּדִינָה בְּאֶמְצָעִים לִמְנִיעַת הִתְפַּשְּׁטוּת זִיהוּם הַוִּוירוּס הַקָּרוּי אֶנְטְרוֹ-וִירוּס 71 (EV71).
זִיהוּמִים בְּוִוירוּסֵי מִשְׁפַּחַת ה"אֶנְטְרוֹ" שְׁכִיחִים מְאוֹד בְּרַחֲבֵי הָעוֹלָם, וְהֵם אַחְרָאִים עַל כְּמַחֲצִית מִכָּל מִקְרֵי מַחֲלוֹת הַחוֹם בִּילָדִים.
הַוִּוירוּסִים הַלָּלוּ תּוֹקְפִים אַךְ וְרַק בְּנֵי אָדָם.
הֵם מִתְפַּשְּׁטִים מֵאָדָם לָאָדָם דֶּרֶךְ מַגָּע (בְּעִיקָּר עִם הַפְרָשׁוֹת צוֹאָה), דֶּרֶךְ מַעֲרֶכֶת הַנְּשִׁימָה, וְגַם דֶּרֶךְ הָאֵם לָעוּבָּר.
 
אֶנְטְרוֹ-וִירוּס 71 מוּכָּר בַּשָּׁנִים הָאַחֲרוֹנוֹת בְּמִסְפַּר הִתְפָּרְצוּיוֹת שֶׁדּוּוְּחוּ בְּמָלֶזְיָה, יַפָּן, טַיְוָואן וְאוֹסְטְרַלְיָה.
הַמְּדוּבָּר בְּוִוירוּס מֵעַיִים מְדַבֵּק בְּיוֹתֵר הַגּוֹרֵם לְחוֹם, שַׁלְפּוּחִיּוֹת בַּפֶּה וּפְרִיחָה עַל הַיָּדַיִים וְהָרַגְלַיִים.
בְּמִקְרִים קִיצוֹנִיִּים גּוֹרֵם הַוִּוירוּס לְנֶזֶק כְּלַל מַעֲרַכְתִּי לְאֵיבָרִים שׁוֹנִים בַּגּוּף וּבָהֶם מוֹחַ, לֵב וְרֵיאוֹת.
 
הַהִתְפָּרְצוּת הַנּוֹכְחִית הֵחֵלָּה לְאַחַר שֶׁפָּעוֹט בֶּן שָׁנָה וָחֵצִי מֵת בִּדְרוֹם מְחוֹז גְּוַואנְגְּדּוֹנְג.
מֵאָז דּוַּוח עַל יוֹתֵר מִ-3,000 יְלָדִים נוֹסָפִים נְגוּעִים בַּנְּגִיף בִּמְחוֹזוֹת אַנֶּהֱוֵי וְזָגִ'יאַנְג.
הַהִתְפָּרְצוּת הֵחֵלָּה לְהִתְפַּשֵּׁט בָּעִיר פוֹיַאנְג בְּמֵרְץ הָאַחֲרוֹן, אוּלָם דּוְּוחָה רַק בַּשָּׁבוּעַ שֶׁעָבַר.
מוּמְחִים הִבִּיעוּ חֲשָׁשׁ שֶׁהַוִּוירוּס הִתְפַּשֵּׁט מִדְּרוֹם הַמְּדִינָה לִצְפוּנָה.
רָשׁוּיוֹת בְּרִיאוּת בְּרַחֲבֵי סִין מְדַוְּוחִים לְלֹא הֶרֶף עַל כָּל חוֹלָה חָדָשׁ שֶׁנִּמְצָא נָגוּעַ בְּוִוירוּס.
 
הָעִיכּוּב בַּדִּיוּוּחַ הוֹבִיל לַהַאֲשָׁמוֹת לְפִיהֶן הַמֶּמְשָׁלָה מַסְתִּירָה אֶת מִקְרֵי הַהַדְבָּקָה בַּנְּגִיף הָאַלִּים.
שַׂר הַבְּרִיאוּת הַסִּינִי דָּחָה אֶת הַהַאֲשָׁמוֹת לְפִיהֶן כָּשְׁלָה סִין בְּטִיפּוּל בַּמַּצָּב, וְאָמַר כִּי עַד אָז לֹא הָיָה בָּרוּר מָה טִיבָהּ שֶׁל הַמַּחֲלָה.
 
מִסְפַּר הַיְּלָדִים שֶׁנִּדְבְּקוּ בַּנְּגִיף EV71 מַמְשִׁיךְ עֲדַיִין לַעֲלוֹת.
מוּמְחֵי אִרְגּוּן הַבְּרִיאוּת הָעוֹלָמִי סְבוּרִים כִּי הַהִתְפָּרְצוּת תַּמְשִׁיךְ גַּם בָּחוֹדָשִׁים הַקְּרוֹבִים, הִתְפָּרְצוּת שֶׁעֲלוּלָה לִגְבּוֹת קוֹרְבָּנוֹת נוֹסָפִים, מַרְבִּיתָם – יְלָדִים מִתַּחַת לְגִיל 6 שָׁנִים.
 
כְּ-1,000 יְלָדִים מְאוּשְׁפָּזִים בְּיָמִים אֵלֶּה בְּבָתֵּי הַחוֹלִים בְּסִין לְאַחַר שֶׁאוּבְחֲנוּ כִּנְגוּעִים בִּנְגִיף, מַצָּבָם שֶׁל 58 מֵהֶם הוּגְדַּר קָשֶׁה עַד אָנוּשׁ.
בֵּינְתַיִים הֵחֵלּוּ בְּסִין לִנְקוֹט בְּאֶמְצָעִים לִמְנִיעַת הִתְפַּשְּׁטוּת הַהַדְבָּקָה.
