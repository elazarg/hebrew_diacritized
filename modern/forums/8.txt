עָבַרְתִּי דִּירָה.
בָּאוּ אֵלַיי אַבָּא שֶׁלִּי, מוֹיְשִׁי וְיִשְׂרָאֵל עִם מְכוֹנִית עֲנָקִית מֵהַמַּחְסָן שֶׁל אַבָּא שֶׁלִּי וְעָזְרוּ לִי לְהַעֲבִיר.
הַמְּכוֹנִית הָיְיתָה עֲנָקִית אֲבָל לֹא מַסְפִּיק, וְכָכָה עָמַדְנוּ שָׁם אַרְבַּעְתֵּנוּ וְנִיסִּינוּ לְפַצֵּחַ אֶת הַטֶּטְרִיס הָרַב מְמַדִּי שֶׁנִּקְרָא הַחֲפָצִים שֶׁלִּי, שֶׁכְּמוֹ הַחַיִּים לִפְעָמִים יֵשׁ מֵהֶם פָּשׁוּט יוֹתֵר מִדַּי.
זֶה הָיָה נִרְאֶה חֲסַר סִיכּוּי וַאֲנִי כְּבָר הִתְחַלְתִּי לְהַפְרִיד בֵּין קוּפְסָאוֹת וַחֲפָצִים הֶכְרֵחִיִּים [מַרְאָה, עַמּוּד תְּלִיָּיה לִבְגָדִים לֹא מַסְפִּיק נְקִיִּים בִּשְׁבִיל אָרוֹן אֲבָל לֹא מַסְפִּיק מְלוּכְלָכִים בִּשְׁבִיל כְּבִיסָה] לַחֲפָצִים בַּעֲדִיפוּת שְׁנִיָּיה שֶׁאוּלַי נַשְׁאִיר מֵאָחוֹר [קוּפְסָא שֶׁמְּכִילָה קוּפְסָאוֹת שֶׁמְּכִילוֹת קוּפְסָאוֹת שֶׁמְּכִילוֹת סִיכּוֹת בִּיטָּחוֹן, כִּסֵּא שִׁבְעָה שֶׁל כֶּתֶר] -- אֲבָל אָז יִשְׂרָאֵל אָמַר שֶׁשְּׁמוּאֵל מֵהַמַּחְסָן לִימֵּד אוֹתוֹ פַּעַם שֶׁכָּל מָה שֶׁשָּׂמִים לִפְנֵי הָרֶכֶב וְאוֹמְרִים שֶׁחַיָּיב לְהִיכָּנֵס בַּסּוֹף נִכְנַס, וְכָל מָה שֶׁשָּׂמִים בַּעֲדִיפוּת שְׁנִיָּיה וְאוֹמְרִים שֶׁיִּיכָּנֵס אִם יִישָּׁאֵר מָקוֹם לֹא נִכְנָס.
אָז הֶחְלַטְנוּ שֶׁהַכֹּל יִיכָּנֵס וְהַכֹּל בֶּאֱמֶת נִכְנָס.
חוּץ מִקּוּפְסָה שֶׁל מָחִית קָארִי שֶׁשַּׁמְתִּי בַּתִּיק שֶׁל הַקּוֹרְקִינֵט וְלָקַחְתִּי אִיתִּי כִּי כְּשֶׁאָמַרְנוּ שֶׁהַכֹּל חַיָּיב לְהִיכָּנֵס הִיא הָיְיתָה בַּמְּקָרֵר וְלֹא נָכְחָה בְּמַעֲמָד הַמְּחַיֵּיב.
פַּעֲמַיִים הִיא נָפְלָה לִי בַּדֶּרֶךְ בִּגְלַל בָּמְפֶּרִים אֲבָל עַכְשָׁיו אַחֲרֵי נִיקָּיוֹן יְסוֹדִי וְהַתְקָנַת נוּרוֹת בַּמְּקָרֵר גַּם לָהּ יֵשׁ מָקוֹם.
הַבּוֹקֶר יָצָאתִי לַעֲבוֹדָה וּפָגַשְׁתִּי אֶת רוֹחִי מִוַּעַד בַּיִת, שֶׁבִּזְמַן שֶׁהֶעֱבַרְנוּ אֶת הַחֲפָצִים חָלְפָה עַל פָּנֵינוּ וּבִיקְּשָׁה שָׁלוֹשׁ בַּקָּשׁוֹת שׁוֹנוֹת שֶׁקְּשׁוּרוֹת לַחֲדַר הַמַּדְרֵגוֹת.
הַבּוֹקֶר הִיא חָלְפָה שׁוּב וְשׁוּב בִּיקְּשָׁה, שְׁתַּיִים, גַּם כֵּן קְשׁוּרוֹת.
