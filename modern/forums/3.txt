שָׁלוֹם,
 
אוּלַי כָּתְבוּ כְּבָר הוֹדָעָה בַּנּוֹשֵׂא הַזֶּה.
יֵשׁ לִי רַעֲיוֹן לְמוּצָר שֶׁהוּא לֹא מוּצָר טֶכְנוֹלוֹגִי, יוֹתֵר בְּכִיוּוּן שֶׁל קוֹסְמֶטִיקָה - לַמְרוֹת שֶׁלֹּא מְדוּבָּר בְּמִשְׂחֶה אוֹ קְרֵם וכ"ו.
אֲנִי לֹא מְעַצֶּבֶת תַּעֲשִׂיָּיתִית וְאֵין לִי מוּשָּׂג אֵיךְ יוֹצְרִים אֶת הַמּוּצָר הַזֶּה וּבֶטַח אֵיךְ מְיַיצְּרִים אוֹתוֹ.
חָשַׁבְתִּי לְחַפֵּשׂ מִפְעָלִים בָּאָרֶץ שֶׁמְּיַיצְּרִים מוּצָרִים פְּלַסְטִיִּים שֶׁהֵם אֵלֶּה שֶׁיִּשְׁתַּמְּשׁוּ בַּמְּעַצֵּב הַתַּעֲשִׂיָּיתִי שֶׁלָּהֶם כְּדֵי לְעַצֵּב אֶת הַמּוּצָר וְהֵם אֵלֶּה שֶׁיִּיצְּרוּ, יְשַׁוְּוקוּ וְיִמְכְּרוּ אוֹתוֹ.
אֲנִי רַק אֶתֵּן לָהֶם אֶת הָרַעְיוֹן (אִם הֵם יִרְצוּ).
 
בִּשְׁבִיל זֶה חָשַׁבְתִּי לְהַחְתִּים אוֹתָם קוֹדֶם עַל הֶסְכֵּם סוֹדִיּוּת כֵּיוָון שֶׁכָּל מָה שֶׁיֵּשׁ לִי הוּא הָרַעְיוֹן.
בָּרֶגַע שֶׁאֲגַלֶּה לָהֶם אוֹתוֹ הֵם לֹא צְרִיכִים אוֹתִי.
 
רָאִיתִי בָּאִינְטֶרְנֵט הֶסְכֵּם סוֹדִיּוּת חִינָּמִי וְגַם דִּיבַּרְתִּי בֶּעָבַר עִם עו"ד שֶׁמִּתְמַחֶה בִּיצִירַת הֶסְכֵּמֵי סוֹדִיּוּת לַהֲגָנָה עַל קִנְיָין רוּחָנִי.
מָה שֶׁאֲנִי לֹא מְבִינָה לְגַבֵּי הַהֶסְכֵּמִים הַלָּלוּ זֶה - כְּדֵי שֶׁבַּעַל מִפְעָל יַסְכִּים לַחְתּוֹם עַל הֶסְכֵּם כָּזֶה, צָרִיךְ שֶׁהָרַעְיוֹן יִהְיֶה כְּבָר כָּתוּב בַּהֶסְכֵּם - כִּי הוּא הֲרֵי לֹא יִרְצֶה לְהִתְחַיֵּיב לֹא לְגַלּוֹת מַשֶּׁהוּ שֶׁהוּא אֲפִילּוּ לֹא יוֹדֵעַ מָה וְלֹא יוֹדֵעַ אִם זֶה רַעֲיוֹן שֶׁאוּלַי הוּא אוֹ מִישֶׁהוּ חָשְׁבוּ עָלָיו כְּבָר קוֹדֶם.
מִצַּד שֵׁנִי, בָּרֶגַע שֶׁאֲפָרֵט מָה הָרַעְיוֹן בַּהֶסְכֵּם כְּבָר אֵין טַעַם בַּהֶסְכֵּם.
גִּילִּיתִי אֶת הַקְּלָפִים שֶׁלִּי וּמִישֶׁהוּ לֹא יָשָׁר יָכוֹל בְּשָׁלָב זֶה לְהַגִּיד לִי תּוֹדָה רַבָּה, אֲנִי לֹא מְעוּנְיָין לַחְתּוֹם עַל שׁוּם דָּבָר וְיֵלֵךְ וִיפַתֵּחַ אֶת הַמּוּצָר בְּעַצְמוֹ.
אֵיךְ מִתְגַּבְּרִים עַל הַקּוֹשִׁי הַזֶּה? - שֶׁבַּעַל הַמִּפְעָל אוֹ מִי שֶׁאֲנִי רוֹצֶה לְשַׁתֵּף אִיתּוֹ פְּעוּלָּה יַחְתּוֹם עַל הַהֶסְכֵּם בְּלִי לְפָרֵט בַּהֶסְכֵּם מָה הָרַעְיוֹן שֶׁלִּי.
 
שְׁאֵלָה שְׁנִיָּה בַּנּוֹשֵׂא - אֲנִי חָשַׁבְתִּי לִמְכּוֹר אֶת הָרַעְיוֹן שֶׁלִּי תְּמוּרַת אָחוּז בִּמְכִירוֹת.
הַאִם זֶה מְקוּבָּל אוֹ שבדר"כ מוֹכְרִים רַעֲיוֹן עֲבוּר סְכוּם כֶּסֶף קָבוּעַ.
אִם מְדוּבָּר בְּאָחוּז מִמְּכִירוֹת, אֵיךְ אֲנִי יְכוֹלָה לָדַעַת בְּכַמָּה כֶּסֶף הַמּוּצָר נִמְכַּר? אֲנִי צְרִיכָה בִּשְׁבִיל זֶה לְהַאֲמִין לְמָה שֶׁאוֹמְרִים לִי, הֲרֵי לֹא יִתְּנוּ לִי גִּישָׁה לְסִפְרֵי הַחֶשְׁבּוֹנוֹת שֶׁלָּהֶם (?).
אִם מְדוּבָּר בִּסְכוּם נָקוּב - אֵיךְ אֶפְשָׁר לְהַעֲרִיךְ סְכוּם כָּזֶה בַּהֲנָחָה שֶׁהַמּוּצָר אִם יַצְלִיחַ יָכוֹל לְהִימָּכֵר בְּמֶשֶׁךְ שָׁנִים בְּהַרְבֵּה כֶּסֶף.
 
וּשְׁאֵלָה שֶׁקְּשׁוּרָה לְזֶה - רָאִיתִי בְּהֶסְכֵּם סוֹדִיּוּת חִינָּמִי בָּאִינְטֶרְנֵט וְשָׁם יֵשׁ שׁוּרָה שֶׁאוֹמֶרֶת שֶׁאִם מִי שֶׁחָתַם עַל הַהֶסְכֵּם הֵפֵר אוֹתוֹ הוּא מְחוּיָּיב לְפַצּוֹת אוֹתִי בִּסְכוּם מְסוּיָּים - אֵיךְ אֶפְשָׁר לְהַעֲרִיךְ מֵרֹאשׁ אֶת הַסְּכוּם?
