וַאֲנִי נִזְכָּר שֶׁכְּשֶׁבִּיקַּשְׁתִּי בֶּעָבָר לְקַבֵּל מִבְצָעִים שֶׁקַּיָּימִים אֵצֶל הַמִּתְחָרִים נֶאֱמַר לִי שֶׁהֵם זֶה הֵם וַאֲנַחְנוּ זֶה אֲנַחְנוּ וְלֹא מַעְתִּיקִים מֵאֶחָד לַשֵּׁנִי, אֲבָל כַּנִּרְאֶה שֶׁבְּיִיעוּל זֶה מַשֶּׁהוּ אַחֵר.
אָז אֲנִי מְנַסֶּה לִשְׁאוֹל הֲרֵי כָּל חוֹדֶשׁ אֲנִי מְשַׁלֵּם הַרְבֵּה כֶּסֶף עַל הַתּוֹסָפוֹת שֶׁל הַמַּכְשִׁיר וְאַתֶּם בַּחֶבְרָה לֹא מַפְסִיקִים לְפַרְסֵם שֶׁיֵּשׁ הַכֹּל בְּמַכְשִׁיר וְאַתֶּם יוֹדְעִים כַּמָּה זֶה חָשׁוּב
אָז מָה אַתְּ פִּתְאוֹם זֶה נִהְיָה לֹא נוֹרָא? וּבִכְלָל אֲנִי מְשַׁלֵּם עַל חֲבִילָה בֵּין אִם נִיצַּלְתִּי וּבֵין אִם לֹא, מָה קוֹרֶה בַּיּוֹמַיִים הָאֵלּוּ? תַּכְלֶס לֹא קִיבַּלְתִּי תְּשׁוּבָה מְסַפֶּקֶת.
בְּקִיצּוּר מְדוּבָּר בְּיִיעוּל אֲבָל רַק אֲנִי כַּנִּרְאֶה לֹא רוֹאֶה אֶת הַיְּיעִילוּת.

אֲנִי מַגְזִים בַּתְּגוּבָה? אוּלַי, אֲבָל אֲנִי עֲדַיִין חוֹשֵׁב שֶׁלֹּא מְדוּבָּר בְּיִיעוּל לְטוֹבַת הַצַּרְכָן, אֶלָּא יִיעוּל לְטוֹבַת הַחֶבְרָה לַמְרוֹת מְחָאַת הַצַּרְחָן.
אָז בְּבַקָּשָׁה תּוֹאִילוּ בְּטוּבְכֶם לְהַבִּיעַ אֶת דַּעְתְּכֶם לָעִנְיָין בַּסֶּקֶר שֶׁאֲנִי מְצָרֵף, וְנִרְאֶה מָה דַּעַת הַגּוֹלְשִׁים.
אִם אָכֵן דַּעַת רוֹב הַגּוֹלְשִׁים שֶׁלֹּא מְדוּבָּר בְּיִיעוּל אֶלָּא בְּמִטְרָד, אָנְסָה לְהִתְלוֹנֵן "בַּדְּרָגִים הַגְּבוֹהִים" לְמַעַן שִׁינּוּי הַשֵּׁירוּת.
