שָׁאוּל צָ'אפָּאתוּל בַּמִּשְׁפָּחָה אַחַת הָיוּ שְׁלֹשָּׂה יְלָדִים: שְׁתֵּי אֲחָיוֹת :דּוֹרִית הַבּוֹגֶרֶת, תָּמִיד מְסֻדֶּרֶת, עִם עֵינַיִם שְׁחֹרוֹת שֶׁרוֹאוֹת כָּל דָּבָר שֶׁאָהֲבָה לְהִתְבּוֹנֵן בְּכָל פִּנָּה וּמָקוֹם וּלְנַחֵשׁ: אֵיזֶה מֶזֶג אֲוִיר יִהְיֶה הַיּוֹם?
דָּלִיָּה, חַיְכָנִית עִם לֵב טוֹב, עִם שֵׂעָר זָהוֹב שֶׁאָהֲבָה לַעֲזֹר בַּמִּטְבָּח בַּאֲפִיָּה, וּבִישוּלִים, אֲבָל לֹא לְהַדִּיחַ כֵּלִים! גַּם הִיא שָּׂמָה, כֹּל דָּבָר בְּמָקוֹם יָדוּעַ וּמְסֻיָּם, וְתָמִיד יָדְעָה וְלֹא שָׁכְחָה, הֵיכָן! לִשְׁתֵּי הָאֲחָיוֹת הָיָה גַּם אָח צָעִיר: שָׁאוּל, יֶלֶד גִ'ינְגִ'י אוֹהֵב תַּעֲלוּל, שְׁכִּינוֹּ אוֹתוֹ: שָׁאוּל צָ'אפָּאצ'וּל! וְלָמָּה צָ'אפָּאצ'וּל? 
הוּא הָיָה יֶלֶד פִּקֵּחַ וְשׁוֹבָב שֶׁהגִּיעַ בְּלִי בְּעָיוֹת לְכָל מָקוֹם בַּמֶּרְחָב, וּבְכָל מָקוֹם שֶׁאֵלָיו התְקָרֵב הוּא מָשַׁךְ הַרְבֵּה תְּשׂוּמֶת לֵב, אֲבָל שֶׁלֹּא כְּמוֹ שְׁתֵּי הָאֲחָיוֹת הוּא הָיָה קְצָת מְפֻזָּר וְלֹא זָכַר תָּמִיד הֵיכָן, הֵנִיחַ כָּל דָּבָר-אֲפִילּוּ אֶת הַנַּעֲלַיִם שֶׁנָּעַל אֶתְמוֹל בַּצָּהֳרַיִם! וּפַעַם, כְּשֶׁחָלַץ אוֹתָן מֵהָרַגְלַיִם וְהִסְתּוֹבֵב קְצָת יָחֵף, בְּלִי לָשִׂים לֵב, הִרְגִּיש מִיָּד בָּרֶגֶל כְּאֵב, וְהֶחְלִיק עַל הָרִצְפָּה, (לְמַזָּלוֹ, לְיַד הַסַּפָּה).

אֵיזֶה מַזָּל שֶׁלֹּא קָרָה שׁוּם דָּבָר וְעַד הָעֶרֶב, הַכֹּל עָבָר! וּמֵאָז קָרְאוּ לוֹ: צָ'אפָּאצ'וּל.

הָיָה לוֹ רֹאשׁ מָלֵא תַּלְתַּלִים אֲבָל הוּא לֹא אָהַב לְהִסְתָּרֵק וְאֶת הַכִּפָּה הָיָה בְּסִכָּה מְהַדֵּק אֲבָל כְּשֶׁהָיָה עִם חֲבֵרָיו מְשַׂחֵק הַכִּפָּה הָיְתָה לִפְעָמִים נוֹפֶלֶת הַרְחֵק! וּכְשֶׁחָזַר הַבַּיְתָה גַּם אֶת הַגָּרְבַּיִם לֹא תָּמִיד לָבַשׁ בְּסֵדֶר, כִּי לֹא פַּעַם הֵן הָיוּ בִּמְקוֹמוֹת שׁוֹנִים בַּחֶדֶר.

אִם בָאָרוֹן הָיוּ מֻנָּחוֹת כְּזוּג מַתְאִים בְּאוֹתָם הַצְּבָעִים ,עַד שֶׁלָּבַשׁ אוֹתָן צָ'אפָּאצ'וּל: לִפְעָמִים גֶּרֶב עִם עִגּוּלִים ברֶגֶל יָמִין וְעַל רֶגֶל שְׂמֹאל- גֶּרֶב עִם פַּסִים אֲדֻמִּים.

לִנְעֹל נַעֲלַיִם- הוּא לֹא אָהַב בִּכְלָל, לָרֹב הֶעֱדִיף לָלֶכֶת יָחֵף אוֹ לְפָחוֹת בַּסַּנְדָּל וְכָךְ הִסְתּוֹבֵב סָבִיב בַּסְּתָו, בַּקַּיִץ וּבָאָבִיב, וְרַק בַּחֹרֶף כְּשֶׁגֶּשֶׁם יָרַד, נָעַל אוֹ מַגָּפַיִם חַמּוֹת בִּמְיֻחָד.

יוֹם אֶחָד נִתְקַל בַּחֲתַלְתּוּל, עַל הָעֵץ שֶׁבֶּחָצֵר, חָתוּל גִ'נְגִ'י עִם שָׂפָם חָמוּד מְאֹד וְעֵינַיִם יְרֻקּוֹת, שֶׁלַּמְרוֹת הַכֶּלֶב שֶׁנָּבַח אֵלָיו, לֹּא הִשְׁמִיעַ יְלָלָה רְגִילָה:"מִיָאוּ- מִיָאוּ "כְּמוֹ שֶׁנּוֹהֵג כָּל חָתוּל ,אֶלָּא הוֹצִיא לָשׁוֹן, לִקֵּק אֶת הַשָּׂפָם- וְרַק אָז יִלֵל: "צִי-אָהּ - צִ'-אָה"- שָּׁאוּל- צָ'אפָּאצ'וּל, כָּל כָּךְ הִתְלַהֵב, עַד שֶׁבְּזֶה הַחֲתַלְתּוּל, מִיָּד הִתְאַהֵב.

הִבְרִיחַ אֶת הַכֶּלֶב, וְהַחֲתַלְתּוּל קָפַץ מִיָּד אֵלָיו עַל שְׁתֵּי רַגְלָיו, הוֹצִיא שׁוּב לָּשׁוֹן וְלִקֵּק אֶת כַּף יָדוֹ שֶׁל צָ'אפָּאצ'וּל, עַד שֶׁלָּקַח אֶת הַחֲתַלְתּוּל מֵהַגִּנָּה, וּמָקוֹם בַּמִּרְפֶּסֶת לוֹ פִּנָּה.
עַל מַגֶּבֶת כְּחֻלָּה, הֵכִין לוֹ קַעֲרִית עֲגֻלָה וּמִלֵא אוֹתָהּ בְּחָלָב, שֶׁהַחֲתַלְתּוּל לִקֵּק בְּעֹנֶג רָב, וּכְשֶׁסִּיֵּם לְלַקֵּק בַּשָּׂפָם וּבַלָּשׁוֹן, יִלֵל צִ'י-אָה צִ'י–אָה בְּשָׂשׂוֹן.

שָׁאוּל צָ'אפָּאצ'וּל כַּמּוּבָן, מְאֹד שָׂמַח וְרַק אָז אֶת הֶחָתוּל לַחֶדֶר שֶׁלּוֹ לָקַח.
הַיַּלְקוּט הָיָה מֻטָּל בְּתוֹךְ הַסַּל וְהַקַּלְמָר הֶחָדָשׁ הָיָה נִרְאָה מְשֻׁמָּשׁ עִם קִשְׁקוּשִׁים, בְּצִבְעֵי הַ"טּוּשִׁים".

הֶחָתוּל נֶעֱצַר לְרֶגַע, בְּגַב מְקֻמָּר, בַּחֶדֶר הִסְתּוֹבֵב בְּמֶשֶׁךְ כַּמָּה דַּקּוֹת, רִחְרַח אֶת כָּל רִחְרַח אֶת כָּל הַ"מְּצִיאוֹת" יִלֵל: צִ'י -אָה צִ'י -אָה צִ'י –אָה ואַחַר כָּךְ נִכְנָס מִתַּחַת לַמִּטָּה, וּכְשֶׁמָּצָא אֶת הַגֶּרֶב הַחוּמָה תָּפַשׂ אוֹתָהּ בְּפִיו וּמִהֵר לְהַנִּיחַ אוֹתָהּ לְיַד הַגֶּרֶב הַשְּׁנִיָּה וְכָךְ תּוֹךְ דַּקָּה אוֹ שְׁתַּיִם, הֶחֱזִיר לִמְקוֹמָן אֶת כָּל הַגַּרְבַּיִם.

אֶת הַכַּדּוּר הִקְפִּיץ בְּאַפּוֹ עַד שֶׁנָּפַל עַל הַסַּפָּה, אֶת הַשְּׂרוֹכִים הִכְנִיס לַנַּעֲלַיִםוְגַם אֶת הַכִּפָּה שֶׁנָּפְלָה (כְּמוֹ תָּמִיד)הֵרִים בְּצִפָּרְנָיו וְלְשָׁאוּל אוֹתָהּ הוֹשִׁיט וְיִלֵל: צִ'י-אָה צִ'י–אָה כְּאִלּוּ אָמַר: חֲבוֹשׁ אוֹתָהּ בְּלִי בּוּשָׁה! שָׁאוּל מִיָּד הִתְכּוֹפֵף אֵלָיו וּבְשִׂמְחָה הֵרִים אוֹתוֹ עַל יָדָיו, אֶת הַחֲתַלְתּוּל, חִבֵּק וְלִטֵּף וְהִרְגִּישׁ מַמָּשׁ שִׂמְּחָה בַּלֵּב! כְּשֶׁדּוֹרִית וְדָלִיָּה שָׁמְעוּ אֶת הַיְּלָלוֹת, נִגְּשׁוּ לִבְדֹּק אִם אֵין תַּקָּלוֹת, כְּשֶׁפָּגְשׁוּ אֶת שְׁנֵי הַגִּ'ינְגִ'ים וְרָאוּ אֶת הָרִצְפָּה המְסֻדֶּרֶת הֵבִינוּ שֶׁלְ צָ'אפָּאצ'וּל יֵשׁ מַמָּשׁ חָבֵר חָדָשׁ נֶהֱדָר שֶׁבְּעֶזְרַת הַשֵּׁם, יִשְׁמֹר עָלָיו לְבַל יהְיֶה יוֹתֵר מְפֻזָּר! מֵאוֹתוֹ רֶגַע הָפְכוּ שָׁאוּל וְהַחֲתַלְתּוּל, לְצֶמֶד חֶמֶד שֶׁהִסְתּוֹבֵב יַחַד בְּכָל פִּנָּה וְזָכָה לַשֵּׁם: צָ'אפָּאתוּל
