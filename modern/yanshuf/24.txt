הַשְּׁמֵנִים נֶגֶד כָּל הָעוֹלָם
"אֲנַחְנוּ רוֹצִים שֶׁרִשְׁתוֹת הַמָּזוֹן יַקְטִינוּ אֶת הַמָּנוֹת, יַגְדִּילוּ אֶת הֶחֳמָרִים הַבְּרִיאוּתִיִּים וְיַפְסִיקוּ לְאַרְגֵּן מִבְצָעִים", אוֹמְרִים אִרְגּוּנֵי צַרְכָנִים בְּאַרְצוֹת הַבְּרִית. הֵם טוֹעֲנִים נֶגֶד חֶבְרוֹת הַמָּזוֹן הַמָּהִיר כְּמוֹ מֵק-דוֹנָלְד וַאֲחֵרוֹת. זֶהוּ רַק קְצֵה הַקַּרְחוֹן שֶׁל מַאֲבָק קָשֶׁה שֶׁמִּתְפַּתֵּחַ בַּיָּמִים אֵלֶּה בְּאָמֶרִיקָה וְיַעֲבֹר אוּלַי גַּם לְיִשְׂרָאֵל.
מִסְפָּר הַשְּׁמֵנִים בְּאָמֶרִיקָה עוֹלֶה מִיּוֹם לְיוֹם. רֹב הָאָמֶרִיקָאִים נֶחְשָׁבִים לִשְׁמֵנִים. גַּם הַיְּלָדִים סוֹבְלִים מִמִּשְׁקַל עֹדֶף. אֶת הַסִּבּוֹת לְהַשְּׁמָנַת הַיֶּתֶר תּוֹלִים הַשְּׁמֵנִים דַּוְקָא בְּחֶבְרוֹת הַמָּזוֹן הַמָּהִיר. "הֵם מְפַתִּים אוֹתָנוּ לִקְנוֹת אֶצְלָם. הֵם עוֹשִׂים מִבְצָעִים, הָאֹכֶל שֶׁלָּהֶם לֹא בָּרִיא וְהַמָּנוֹת שֶׁלָּהֶם גְּדוֹלוֹת", הֵם מִתְלוֹנְנִים.
לְדַעֲתָם, חֶבְרוֹת הַמָּזוֹן הַמָּהִיר עוֹשׂוֹת הַכֹּל כְּדֵי שֶׁהֵם יִקְנוּ אֶצְלָם וְלָכֵן הֵם מַשְׁמִינִים. בְּיָמִים אֵלֶּה, מֻגָּשׁוֹת תְּבִיעוֹת רַבּוֹת לְבָתֵּי הַמִּשְׁפָּט, בָּהֵן הֵם דּוֹרְשִׁים מֵחֶבְרוֹת הַמָּזוֹן לְפַצּוֹת אוֹתָם בְּמִילְיוֹנֵי דּוֹלָרִים. חֶבְרוֹת הַמָּזוֹן לֹא נִשְׁאָרוֹת חַיָּבוֹת. "הַבְּעָיָה הִיא לֹא בָּנוּ. מְדֻבָּר בַּאֲנָשִׁים שֶׁאוֹכְלִים בְּצוּרָה לֹא מְאֻזֶּנֶת וְלֹא עוֹשִׂים סְפּוֹרְט. אֲנַחְנוּ מְפַרְסְמִים כָּל הַזְּמַן אֶת הַצֹּרֶךְ בִּסְפּוֹרְט וּבִבְרִיאוּת. אֲנַחְנוּ בְּטוּחִים, שֶׁאַף אֶחָד לֹא יַצְלִיחַ לִרְזוֹת דֶּרֶךְ בָּתֵּי הַמִּשְׁפָּט", אוֹמֵר גִּ'ין גְרָאבּוֹבְסְקִי, דּוֹבֵר אִרְגּוּן יַצְרָנֵי הַמָּזוֹן בְּאָמֶרִיקָה.
"חַיָּבִים לִדְאֹג לְשִׁנּוּי דֶּרֶךְ מַעֲרֶכֶת הַחִנּוּךְ, עוֹד בִּזְמַן שֶׁהַיְּלָדִים קְטַנִּים. כֻּלָּנוּ חַיָּבִים לְהָבִין שֶׁלַּסְּפּוֹרְט יֵשׁ חֲשִׁיבוּת", קוֹבֵעַ הַדּוֹבֵר.
בְּאַרְצוֹת הַבְּרִית וּבִמְדִינוֹת אֲחֵרוֹת חוֹשְׁשִׁים, שֶׁבְּעוֹד כַּמָּה שָׁנִים תַּהֲפֹךְ הַהַשְּׁמָנָה לְדָבָר מְסֻכָּן מְאֹד. כַּיּוֹם, הָעִשּׁוּן הוּא הַגּוֹרֵם מִסְפָּר אֶחָד לְמָוֶת. אִם אֲנָשִׁים יַמְשִׁיכוּ לֶאֱכֹל בְּכַמֻּיּוֹת וְלֹא יַעֲשׂוּ סְפּוֹרְט, תַּהֲפֹךְ הַהַשְּׁמָנָה לַגּוֹרֵם הָעִקָּרִי לְמָוֶת.
אִרְגּוּנִי הַשְּׁמֵנִים לֹא מִתְרַשְּׁמִים מֵהַהֶסְבֵּרִים שֶׁל יַצְרָנִיּוֹת הַמָּזוֹן. הֵם דּוֹרְשִׁים לְהָטִיל קְנָס עַל חֲבָרוֹת שֶׁמְּיַצְּרוֹת מָזוֹן לֹא בָּרִיא. הֵם קוֹרְאִים לַמֶּמְשָׁלָה לְחוֹקֵק חֻקִּים נֶגֶד אוֹתָן הַחֲבָרוֹת. מִצַּד שֵׁנִי, הֵם קוֹרְאִים לְמִשְׂרַד הַחִנּוּךְ לִקְבֹּעַ לַיְּלָדִים מַה לֶּאֱכֹל בַּצָּהֳרַיִם.
פְּקִידֵי מֶמְשָׁלָה אָמֶרִיקָאִיִּים לֹא חוֹשְׁבִים שֶׁזֶּה מִתַּפְקִידָם לְהִכָּנֵס לַצַּלָּחוֹת שֶׁל הָאֶזְרָחִים. "הַמְּבֻגָּרִים יְכוֹלִים לְהַחְלִיט בְּעַצְמָם מַה לֶּאֱכֹל", אֹמֶר אָרִיק בּוּסְט, סְגַן הַשַּׂר הָאַחֲרָאִי עַל הַמָּזוֹן בארה"ב. "הֵם גַּם צְרִיכִים לִהְיוֹת אַחֲרָאִיִּים עַל הָאוֹכֵל שֶׁהַיְּלָדִים שֶׁלָּהֶם אוֹכְלִים. אִם מִישֶׁהוּ סוֹבֵל מֵעֹדֶף מִשְׁקָל, זֶה בִּגְלַל שֶׁהוּא לָקַח אֹכֶל מַשְׁמִין וְלֹא בָּרִיא", הוּא אוֹמֵר.
הַהוֹרִים בְּאַרְצוֹת הַבְּרִית, הָרוֹאִים אֶת הַוִּכּוּחַ בֵּין הַצְּדָדִים, לֹא מוּכָנִים לָקַחַת עַל עַצְמָם אֶת הָאַחֲרָיוּת לְהַשְּׁמָנַת הַיְּלָדִים וּלְהַשְּׁמָנָה שֶׁלָּהֶם עַצְמָם. הֵם מַפְנִים אֶצְבַּע מַאֲשִׁימָה לְעֵבֶר הַפִּרְסוֹמוֹת הַמְּפַתּוֹת וְעַל כָּךְ שֶׁהַיְּלָדִים אוֹהֲבִים דַּוְקָא אֹכֶל מָהִיר כְּמוֹ הַמְּבּוּרְגֶרִים. 
בְּדִיקָה קְצָרָה מַעֲלָה, שֶׁהַתַּקְצִיב לְפִרְסוֹמוֹת בִּתְחוּם הָאֹכֶל הוּא גָּדוֹל מְאֹד - כ-30 מִילְיַארְד דּוֹלָר. רֹב הַכֶּסֶף מֻפְנֶה לְכִוּוּן הַיְּלָדִים וּבְנֵי הַנֹּעַר.
כָּךְ קוֹרֶה, שֶׁהַהַאֲשָׁמוֹת עוֹבְרוֹת מֵאֶחָד לַשֵּׁנִי, כַּאֲשֶׁר אַף אֶחָד לֹא מוּכָן לָקַחַת אַחֲרָיוּת עַל הַהַשְּׁמָנָה שֶׁלּוֹ עַצְמוֹ וְשֶׁל הַיְּלָדִים שֶׁלּוֹ. הַנִּסָּיוֹן לְבַקֵּשׁ אֶת הַחְלָטַת בֵּית הַמִּשְׁפָּט בַּנּוֹשֵׂא הַזֶּה הוּא מְעַנְיֵן, אַךְ אֵין בּוֹ כְּדֵי לִפְתֹּר אֶת הַבְּעָיָה. בְּכָל מִקְרֶה, אֲנָשִׁים רַבִּים בָּעוֹלָם מַשְׁמִינִים מִיּוֹם לְיוֹם, וּבְרֹאשָׁם הָאָמֶרִיקָאִים.
