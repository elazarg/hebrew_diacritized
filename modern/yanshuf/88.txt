בִּמְדִינַת טוּרקמֶנִיסטָאן בִּטְלוּ אֶת שְׁמוֹת הֶחֳדָשִׁים וְהַיָּמִים
בְּאַסְיָה יֶשְׁנָהּ מְדִינָה בְּשֵׁם "טוּרקמֶנִיסטָאן". מְדִינָה זוֹ הָיְתָה פָּעַם חֵלֶק מִבְּרִית הַמּוֹעָצוֹת, וְכַיּוֹם הִיא מְדִינָה עַצְמָאִית.
בְּרֹאשׁ הַמְּדִינָה עוֹמֵד הַנָּשִׂיא מוּרָאד נִיאַזוֹב. הוּא לֹא נָשִׂיא רָגִיל. לְדַעֲתוֹ, הַמְּדִינָה צְרִיכָה לִפְעֹל לְפִי מַה שֶּׁהוּא חוֹשֵׁב. כָּךְ הוּא הֶחְלִיט לְהַחְלִיף אֶת שְׁמוֹת הֶחֳדָשִׁים בְּלוּחַ הַשָּׁנָה. לְמָשָׁל, בִּמְקוֹם חֹדֶשׁ יָנוּאָר הִתְקַבֵּל הַשֵּׁם הֶחָדָשׁ: "טוּרְקֶמֵנְבָּאשִׁי". שֵׁם זֶה הוּא הַכִּנּוּי שֶׁל הַנָּשִׂיא בִּכְבוֹדוֹ וּבְעַצְמוֹ.
כְּדֵי שֶׁלָּאֶזְרָחִים יִהְיֶה קַל יוֹתֵר לְכַבֵּד אֶת בְּנֵי הַמִּשְׁפָּחָה שֶׁלּוֹ, הוּא קָרָא לְחֹדֶשׁ אַפְּרִיל עַל שֵׁם הָאִמָּא שֶׁלּוֹ. לָכֵן, אִם אַתֶּם רוֹצִים לִקְבֹּעַ פְּגִישָׁה עִם מִישֶׁהוּ מִטּוּרְקְמֶנִיסְטָאן בְּחֹדֶשׁ אַפְּרִיל, כְּדַאי לָכֶם לֹא לְהִתְבַּלְבֵּל.
הַנָּשִׂיא הֶחָרוּץ לֹא שָׁכַח לְשַׁנּוֹת גַּם אֶת שְׁמוֹת הַיָּמִים בַּשָּׁבוּעַ. מֵעַכְשָׁו, אֵין עוֹד יְמֵי רִאשׁוֹן, שֵׁנִי וּשְׁלִישִׁי, אֶלָּא "יוֹם הַמְּנוּחָה", שֶׁזֶּה יוֹם רִאשׁוֹן, "יוֹם עִקָּרִי", "יוֹם צָעִיר", "יוֹם מַתְאִים", "יוֹם מְבֹרָךְ", "יוֹם שִׁשִּׁי" ו"יוֹם הָרוּחָנִיּוּת הַגְּבוֹהָה".
כְּדֵי שֶׁלֹּא יִקְרֶה מַצָּב שֶׁלְּתוֹשָׁבֵי הַמְּדִינָה הַחֲדָשָׁה יִמָּאֵס מֵהַנָּשִׂיא הַמּוּזָר שֶׁלָּהֶם, הוּא קָבַע שֶׁהוּא יִהְיֶה נָשִׂיא לְכָל יְמֵי חַיָּיו. בְּהַחְלָטָה זוֹ תָּמְכוּ כָּל חַבְרֵי הַפַּרְלָמֶנְט שֶׁל טוּרְקְמֶנִיסְטָאן.
