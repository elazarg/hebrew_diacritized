הַמַּדְפֶּסֶת שֶׁמְּשַׁגַעַת אֶת הָעוֹלָם
מְסֻגֶלֶת לְהַדְּפִּיס צַלָחוֹת וְכוֹסוֹת אֲמִתִיִים * חֲבָרוֹת גְּדוֹלוֹת בָּעוֹלָם קָנוּ אֶת הַמַּדְפֶּסֶת כְּדֵי לַחְסוֹךְ בְּהוֹצָאוֹת יִצוּר * מַדוּעַ חֶבְרַת צִיוּד הַסְּפּוֹרְט "נַיְקִי" וְחַבְרוֹת הָרֶכֶב זְקוּקוֹת לְמַדְפֶּסֶת הַמְּיֻצֶרֶת בְּיִשְׂרָאֵל? *
תָאֲרוּ לְעַצְמְכֶם, הִזְמַנְתֶם אוֹרְחִים לַאֲרוּחַת צָהֳרַיִם. אַתֶם עוֹרְכִים אֶת הַצְּלָחוֹת עַל הַשֻּׁלְחָן, אֲבָל הַתִּינוֹק שֶׁבַּבַּיִת מָשַׁךְ אֶת הַמַּפָּה וְשָׁבַר אֶת כָּל הַצַּלָחוֹת – אַף צַלַחַת נוֹרְמָלִית לֹא נִשְׁאֶרֶת בַּבַּיִת. הַפִּתְרוֹן – לָרוּץ לִקְנוֹת צַלָחוֹת חֲדָשׁוֹת, אַךְ זֶה לֹא תָמִיד אֶפְשָׁרִי, אוֹ לְהִתְקַשֵׁר לָאוֹרְחִים וּלְהוֹדִיעַ לָהֶם לְהָבִיא צַלָחוֹת מֵהַבַּיִת. קָשֶׁה לְהַאֲמִין, שֶׁיִהְיֶה מִי שֶׁיַעֲשֶׂה אֶת זֶה.
בְּקָרוֹב, תִּהְיֶה לָכֶם אֶפְשָׁרוּת נוֹסֶפֶת – לְהַדְּפִּיס אֶת הַצַּלָחוֹת. הַכַּוָנָה לְהַדְּפִּיס צַלָחוֹת אֲמִתִיוֹת, וְלֹא עַל הַנְּיָר. חֶבְרָה יִשְׂרְאֵלִית הִמְצִיאָה אֶת הַמַּדְפֶּסֶת הַמְּשֻׁכְלֶלֶת, שֶׁיְכוֹלָה לְהַדְּפִּיס קֻפְסָאוֹת, נַעֲלַיִם, מִבְרְשׁוֹת שִׁנַיִם וְכִמְעַט כָּל דָּבָר הָעוֹלֶה עַל רוּחֲכֶם.
אִם כֵּן, כָּל מַה שֶׁנִּשְׁאַר לָנוּ לַעֲשׂוֹת, הוּא לְהַזְּמִין מַדְפָּסוֹת כְּאֵלֶה וּלְהַתְּחִיל לַחְגוֹג.
זֶה לֹא כָּל כָּךְ פָּשׁוּט - מַדְפֶּסֶת כָּזוֹ עוֹלָה כַּחֲצִי מִילְיוֹן שְׁקָלִים.
אִם כֵּן, מִי צָרִיךְ מַדְפֶּסֶת כָּל כָּךְ יְקָרָה שֶׁמְּסֻגֶלֶת לְהַדְּפִּיס צַלָחוֹת וְכוֹסוֹת?
מִתְבָּרֵר, שֶׁחֲבָרוֹת עוֹלָמִיוֹת מְאֹד אוֹהֲבוֹת אֶת הַמַּדְפֶּסֶת הַיִּשְׂרְאֵלִית. חֶבְרַת "אָדִידָס", לְמָשָׁל, מְיַצֶרֶת אֶת דִגְמֵי הַנַּעֲלַיִם הַחֲדָשִׁים שֶׁלָּהֶם בְּאֶמְצָעוּת הַמַּדְפֶּסֶת. זֹאת אוֹמֶרֶת, שֶׁהַמְּעַצֵב שֶׁל הַנַּעַל הַחֲדָשָׁה מְעַצֵב אוֹתָה עַל הַמַּחְשֵׁב וּלְאַחַר מִכֶּן הוּא מַדְפִּיס אוֹתָה בַּמַּדְפֶּסֶת הַתְּלַת מְמַדִית. כַּאֲשֶׁר בּוֹדְקִים אֶת הַמִּדוֹת שֶׁל הַנַּעַל, מַתְחִילִים בְּיִצוּר הֲמוֹנִי.
"הַמַּדְפֶּסֶת הַחֲדָשָׁה שִׁנְתָה לָנוּ אֶת כָּל שִׁיטַת הָעֲבוֹדָה", אוֹמְרִים ב"אָדִידָס". "תַהֲלִיכֵי הַיִּצוּר מְהִירִים וּמְדֻיָקִים יוֹתֵר".
גַם חֶבְרוֹת הַמְּכוֹנִיוֹת מִשְׁתַּמְשׁוֹת בַּמַּדְפֶּסֶת, כְּדֵי לִבְנוֹת דִגְמֵי מְכוֹנִיוֹת קְטַנִים וּלְפַתֵחַ חֲלָקִים מְיֻחָדִים שֶׁל תַעֲשִׂיַת הָרֶכֶב.
וּבְכָל זֹאת, אִם בָּא לָכֶם לִקְנוֹת מַדְפֶּסֶת כָּזוֹ, שֶׁהַגֹּדֶל שֶׁלָה כְּגֹדֶל מְכוֹנַת כְּבִיסָה, אַתֶם מֻזְמָנִים לְהוֹצִיא חֲצִי מִילְיוֹן שֶׁקֶל. אֲבָל אִם אֵין לָכֶם אֶת הַכֶּסֶף הַזֶּה, כְּדַאי לְחַכּוֹת. בְּעוֹד כַּמָה שָׁנִים, הַמַּדְפֶּסֶת הַמַּדְלִיקָה הַזּוֹ תִּהְיֶה זוֹלָה יוֹתֵר. אֲבָל, שִׂימוּ לֵב, הַ"דְיוֹ" שֶׁל הַמַּדְפֶּסֶת, שֶׁהוּא בְּעֶצֶם פְּלַסְטִיק חַם, יִהְיֶה יָקָר מְאֹד. כָּךְ, שֶׁגַּם אָז לֹא בָּטוּחַ שֶׁכְּדַאי יִהְיֶה לִקְנוֹת אֶת ה"צַעֲצוּעַ" הַמְּיֻחָד וְהַמְּעַנְיֵן הַזֶּה.
