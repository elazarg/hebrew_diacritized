קַבְּלוּ אֶת מִשְׁפַּחַת הַצִּ'יפְּסוֹנִים הָרִאשׁוֹנָה בָּעוֹלָם
מִשְׁפַּחַת גֵ'ייקוֹבְס הָאָמֶרִיקָאִית זָכְתָה לַכִּנּוּי "מִשְׁפַּחַת הַצִּ'יפְּסוֹנִים ". כָּל זֹאת בִּגְלַל שְׁבָב אֶלֶקְטְרוֹנִי קָטָן שֶׁהֻשְׁתַּל מִתַּחַת לָעוֹר שֶׁל כָּל אֶחָד מִבְנֵי הַמִּשְׁפָּחָה. הַשְּׁבָב הָאֶלֶקְטְרוֹנִי (בָּאַנְגְּלִית - צִ'יפְּס) מֵכִיל מֵידָע חָשׁוּב מְאֹד עַל בְּנֵי הַמִּשְׁפָּחָה. הוּא מְיֹעָד, בְּעִקָּר, לָתֵת מֵידָע בְּנוֹשְׂאִים רְפוּאִיִּים עַל דֵרֶק, בֵּן ה-14 וְהַהוֹרִים לֵזְלִי וְגֵ'פְרִי.
לְמַעֲשֶׂה, דרק וְאִמּוֹ, לזלי, אֵינָם חוֹלִים כְּלָל. הֵם מִשְׁתַּתְּפִים בְּנִסּוּי שֶׁאוּלַי יָבִיא לְכָךְ, שֶׁכָּל אָדָם בָּעוֹלָם יִשָּׂא מִתַּחַת לָעוֹר שֶׁלּוֹ שְׁבָב מְיֻחָד עִם כָּל הַפְּרָטִים הָרְפוּאִיִּים שֶׁלּוֹ. אֹמֶר דר' ווּלְקָן, מִי שֶׁמְּבַצֵּעַ אֶת הַשְּׁתָּלוֹת הַשְּׁבָבִים: "לִפְעָמִים, מַגִּיעִים לְבֵית הַחוֹלִים אֲנָשִׁים שֶׁלֹּא יְכוֹלִים לְדַבֵּר. אֲנַחְנוּ לֹא יוֹדְעִים עֲלֵיהֶם דָּבָר וְקָשֶׁה לָנוּ לְטַפֵּל בָּהֶם. בַּשְּׁבָב הָאֶלֶקְטְרוֹנִי קַיָּמִים הַפְּרָטִים הָאִישִׁיִּים שֶׁל הָאָדָם, כּוֹלֵל מִסְפְּרֵי הַטֶּלֶפוֹן שֶׁלּוֹ. אִם יֵשׁ צֹרֶךְ, אֲנַחְנוּ מִתְקַשְּׁרִים לַבַּיִת שֶׁל הָאִישׁ הַמַּגִּיעַ לְבֵית הַחוֹלִים וּמַזְמִינִים אֶת בְּנֵי הַמִּשְׁפָּחָה שֶׁלּוֹ.
"כַּמּוּבָן, שֶׁהַטִּפּוּל בְּאִישׁ שֶׁכָּזֶה יִהְיֶה מְאֹד יָעִיל וּבְמִקְרִים מְסֻיָּמִים יוּכַל הַשְּׁבָב לְהַצִּיל חַיִּים".
כַּמָּה זְמַן לוֹקֵחַ לְהַשְּׁתִּיל שְׁבָב אֶלֶקְטְרוֹנִי שֶׁכָּזֶה? רַק חָמֵשׁ דַּקּוֹת.
לֹא כֻּלָּם מְקַבְּלִים אֶת הָרַעְיוֹן בְּשִׂמְחָה. אִרְגּוּנֵי זְכֻיּוֹת הָאָדָם חוֹשְׁשִׁים מֵהַיּוֹם שֶׁבּוֹ יֻשְׁתְּלוּ שְׁבָבִים כְּאֵלֶּה בִּבְנֵי אָדָם כְּדֵי שֶׁיּוּכְלוּ לַעֲקֹב אַחֲרֵיהֶם. כְּבָר עַכְשָׁו יָדוּעַ שֶׁנִּתַּן לַעֲקֹב אַחֲרֵי בְּנֵי אָדָם הַנּוֹשְׂאִים שְׁבָבִים כְּאֵלֶּה בְּאֶמְצָעוּת לַוְיָנִים.
"לָאֲנָשִׁים תֹּאבַד הפְּרָטִיּוּת. הֵם יִהְיוּ חֲשׂוּפִים בכָל מָקוֹם אֵלָיו יַגִּיעוּ", אֹמֶר מַרְק רוטנברג, מְנַהֵל הַמֶּרְכָּז לִפְרָטִיּוּת הַמֵּידָע הָאֶלֶקְטְרוֹנִי. "הוֹרִים יַחְדִּירוּ שְׁבָבִים כָּאֵלֶּה בַּיְּלָדִים שֶׁלָּהֶם כְּדֵי לָדַעַת הֵיכָן הֵם נִמְצָאִים וְכַדּוֹמֶה".
הַשִּׁלְטוֹנוֹת הָאָמֶרִיקָאִיִּים דּוֹחִים אֶת הַטְּעָנוֹת הָאֵלֶּה וְאוֹמְרִים: "הַשִּׁמּוּשׁ בַּשְּׁבָבִים נַעֲשָׂה מִתּוֹךְ בְּחִירָה חָפְשִׁית. אִם אָדָם רוֹצֶה לְהַחְדִּיר שְׁבָב אֶלֶקְטְרוֹנִי לַגּוּף שֶׁלּוֹ, הוּא יוּכַל לַעֲשׂוֹת זֹאת מִתּוֹךְ שִׁקּוּלֵי בְּרִיאוּת וְשִׁקּוּלִים אֲחֵרִים. אִם הוּא לֹא רוֹצֶה לַעֲשׂוֹת זֹאת, אַף אֶחָד לֹא יוּכַל לְהַכְרִיחַ אוֹתוֹ".
