הַאִם אֲנַחְנוּ הוֹפְכִים לִהְיוֹת אָמֶרִיקָאִים
הָאָמֶרִיקָאִים כּוֹבְשִׁים אֶת הָעוֹלָם. בְּכָל מָקוֹם שׁוֹתִים קוקה קוֹלָה, אוֹכְלִים מֵק-דוֹנָלְד וְרוֹאִים סִרְטֵי קוֹלְנוֹעַ אָמֶרִיקָנִיִים. בְּיִשְׂרָאֵל כְּבָר דֻּבַּר נֶגֶד הַתּוֹפָעָה בָּהּ בְּנֵי הַנֹּעַר מִתְלַבְּשִׁים כְּמוֹ אָמֶרִיקָאִיִים וּמִזְדַּהִים עִם הַתַּרְבּוּת שֶׁל הַמַּעֲצָמָה הַגְּדוֹלָה בְּיוֹתֵר בָּעוֹלָם. אַל דְּאָגָה, אֲנַחְנוּ לֹא לְבַד. גַם הַצָּרְפָתִים וְהָאַנְגְּלִים מְנַסִּים לַעֲצוֹר אֶת הַתּוֹפָעָה.
לִפְנֵי מִסְפַּר שָׁנִים, הִתְלוֹנֵן עֵזֶר וייצמן, נְשִׂיא מְדִינַת יִשְׂרָאֵל בְּאוֹתָם יָמִים, שֶׁהַנֹּעַר הַיִּשְׂרְאֵלִי הוֹלֵךְ אַחֲרֵי הַתַּרְבּוּת הָאָמֶרִיקָאִית. הוּא יָצָא נֶגֶד תַרְבּוּת הַמֵּק-דוֹנָלְד, הוּא דִבֵּר עַל כָּךְ שֶׁאֲנַחְנוּ מִתְלַבְּשִׁים בִּבְגָדִים "אָמֶרִיקָאִיִים", עִם סְמָלִים אָמֶרִיקָאִיִים, מַעֲדִיפִים שִׁירִים בְּלוֹעֲזִית וְעוֹד.
הַנָּשִׂיא לְשֶׁעָבַר, דִבֵּר עַל הַצּוֹרֵךְ לִשְׁמוֹר עַל הַזֶּהוּת הַיִּשְׂרְאֵלִית, וְעַל כָּךְ, שֶׁלֹּא כָּל מַה שֶׁבָּא מֵאָמֶרִיקָה הוּא טוֹב.
מִתְבָּרֵר, שֶׁהַדִּיוּן הַזֶּה, נֶגֶד הָאָמֶרִיקַנִיזַצְיָה שֶׁל הַחֶבְרָה לֹא קַיָם רַק בְּיִשְׂרָאֵל. בְּצָרְפַת, לְמָשָׁל, הָיְתָה הִתְנַגְדוּת גְּדוֹלָה לִלְמוֹד אַנְגְּלִית. הַצָּרְפָתִים הִרְגִישׁוּ, שֶׁהַשָּׂפָה שֶׁלָּהֶם חֲשׁוּבָה יוֹתֵר. תַיַר הַמַּגִיעַ לְצָרְפַת יִתְקַשֶׁה לִמְצוֹא אֶזְרָחִים דּוֹבְרֵי אַנְגְּלִית.
אוּלַם הַהַפְּתָעָה הַגְּדוֹלָה בְּיוֹתֵר בָּאָה מִכִּווּן הָאַנְגְּלִים. לַמְרוֹת שֶׁהַשָּׂפָה הָאַנְגְּלִית נְהוּגָה בִּשְׁנֵי הַמְּקוֹמוֹת, הָאַנְגְּלִים לֹא אוֹהֲבִים אֶת הִשְׁתַּלְטוּת הַתַּרְבּוּת הָאָמֶרִיקָאִית עַל הָאָרֶץ שֶׁלָּהֶם. הֵם מִתְנַגְדִים לְהַעֲסָקַת שַׂחְקָנֵי קוֹלְנוֹעַ אָמֶרִיקָאִיִים בִּסְרָטִים תוֹצֶרֶת אַנְגְּלִיָה, הֵם לֹא אוֹהֲבִים אֶת הַכְּנִיסָה שֶׁל חֲבָרוֹת כְּמוֹ קוקה קוֹלָה ומֵק-דוֹנָלְד לְאַנְגְּלִיָה. הֵם רוֹאִים אֵיךְ הַפִּרְסוֹמוֹת הָאָמֶרִיקָאִיוֹת מִשְׁתַּלְטוֹת עַל הַטֶּלֶוִיזְיָה שֶׁלָּהֶם וְלֹא מְרֻצִים.
לַמְרוֹת הַהִתְנַגְדוּת לְכָךְ שֶׁהַתַּרְבּוּת הָאָמֶרִיקָאִית מִשְׁתַּלֶּטֶת עַל הַחֶבְרָה הָעוֹלָמִית, יֵשׁ גַם תוֹמְכִים. לְדַעֲתָם, כָּל חֶבְרָה צְרִיכָה לִשְׁמוֹר עַל הַיִּחוּדִיוּת שֶׁלָה. עִם זֹאת, הִיא צְרִיכָה לְהַכִּיר בְּכָךְ שֶׁהָעוֹלָם הוֹפֵךְ לְמָקוֹם אֶחָד גָדוֹל וּמְשֻׁתָף. "יֵשׁ לְזֶה יִתְרוֹנוֹת גְּדוֹלִים וּכְדַאי מְאֹד לְנַצֵל זֹאת", אוֹמְרִים הַתּוֹמְכִים.
הֵם מַצְבִּיעִים עַל פְּתִיחַת הַמִּסְעָדוֹת שֶׁל מֵק-דוֹנָלְד בְּרוּסְיָה כְּעַל אֶחָד הַסִּימָנִים הַחֲשׁוּבִים בְּיוֹתֵר לְכָךְ שֶׁהַמִּלְחָמָה שֶׁלָה עִם אַרְצוֹת הַבְּרִית הִסְתַּיְמָה. הֵם מַצְבִּיעִים עַל הָעֻבְדָה הַמְּעַנְיֶנֶת, שֶׁמֵעוֹלָם לֹא נֶעֶרְכָה מִלְחָמָה בֵּין שְׁתֵי מְדִינוֹת שֶׁיֵשׁ בָּהֵן מִסְעָדוֹת שֶׁל מֵק-דוֹנָלְד. כַּמוּבָן, שֶׁזֶה לֹא בִּגְלַל הַמִּסְעָדוֹת, אֶלָא בִּגְלַל הָרוּחַ הַדֶּמוֹקְרָטִית שֶׁיֵשׁ בְּאוֹתָן מְדִינוֹת. כַּאֲשֶׁר שַׁלִיטֵי הַמְּדִינָה מַסְכִּימִים לְקַבֵּל אֶת הַחֲבָרוֹת הָאָמֶרִיקָאִיוֹת, הֵם כְּאִלּוּ מַכְרִיזִים עַל כָּךְ שֶׁהֵם מוּכָנִים לִפְעוֹל עַל פִּי הַחֻקִים הַבֵּין לְאֻמִיִים, אוֹתָם מוֹבִילָה אַרְצוֹת הַבְּרִית.
יֵשׁ כְּאֵלֶה שֶׁאֲפִלוּ קוֹרְאִים לִלְמוֹד חֲלָקִים מֵהַתַּרְבּוּת הָאָמֶרִיקָאִית כְּמוֹ: זְכֻיוֹת אָדָם, דֶמוֹקְרַטְיָה וְעוֹד.
נִרְאָה, שֶׁסִימָנֵי הַתַּרְבּוּת הָאָמֶרִיקָאִית יַגִיעוּ כִּמְעַט לְכָל מָקוֹם בָּעוֹלָם. לַמִתְנַגְדִים יִהְיֶה קָשֶׁה לַעֲצוֹר אֶת הָרָצוֹן שֶׁל רַבִּים מֵאִתָנוּ לֶאֱכוֹל הַמְּבּוּרְגֶרִים שֶׁל מֵק-דוֹנָלְד, לִשְׁתּוֹת קוקה קוֹלָה וְלֶאֱהוֹב אֶת הַכַּדוּרְסַל שֶׁל ה-NBA. "מַה שֶׁכָּל אֶחָד חַיָּב לִזְכּוֹר, שֶׁהוּא לֹא צָרִיךְ לְוַתֵר עַל הַזֶּהוּת הָאִישִׁית שֶׁלוֹ, גַם כַּאֲשֶׁר הוּא רוֹצֶה לִהְיוֹת חֵלֶק מֵהָעוֹלָם הַגָּדוֹל", אוֹמְרִים הָאֲנָשִׁים שֶׁרוֹצִים לֵהָנוֹת מֵהַתַּרְבּוּת הָאָמֶרִיקָאִית בְּלִי לְוַתֵר עַל הַתַּרְבּוּת שֶׁלָּהֶם.
