הַמְּפֻרְסָמִים מַצִּיגִים אֶת הַחַיִּים הַפְּרָטִיִּים שֶׁלָּהֶם
תָּכְנִיּוֹת טֶלֶוִיזְיָה חַדְשׁוֹת מַצִּיגוֹת שַׂחְקָנֵי קוֹלְנוֹעַ וְזַמָּרִים בַּחַיִּים הַפְּרָטִיִּים שֶׁלָּהֶם. הַקָּהָל רוֹאֶה אוֹתָם מְבַשְּׁלִים, יְשֵׁנִים, מְנַקִּים אֶת הַבַּיִת וּלְבוּשִׁים בְּפִּיגָ'מָה. לְדַעַתְכֶם זֶה מְשַׁעֲמֵם? הָאָמֶרִיקָאִים חוֹשְׁבִים שֶׁזֶּה מְעַנְיֵן מְאֹד וְצוֹפִים בַּטֶּלֶוִיזְיָה בְּמֶשֶׁךְ שָׁעוֹת. הַאִם יֵשׁ לְפָנֵינוּ פִּתּוּחַ נוֹסָף שֶׁל תָּכְנִיּוֹת הַטֶּלֶוִיזְיָה?
אוֹזִי אוֹסבּוֹרְן, זַמַּר רֹק אָמֶרִיקָאִי מְפֻרְסָם, הִרְשָׁה לְמַצְלֵמוֹת הַטֶּלֶוִיזְיָה לְצַלֵּם אוֹתוֹ בַּמֶּשֶׁךְ 24 שָׁעוֹת. לֹא רַק הַזַּמָּר צֻלַּם, אֶלָּא גַּם אִשְׁתּוֹ וְהַיְּלָדִים. צוֹפֵי הַטֶּלֶוִיזְיָה רָאוּ אֵיךְ אוֹזִי רָב עִם אִשְׁתּוֹ, מִתְוַכֵּחַ עִם הַיְּלָדִים, מֵכִין לָהֶם אֹכֶל, שׁוֹלֵחַ אוֹתָם לְבֵית הַסֵּפֶר וּמִתְאַמֵּן לִקְרַאת הַהוֹפָעָה הַחֲדָשָׁה שֶׁלּוֹ.
עֲרוּץ הַטֶּלֶוִיזְיָה MTV. שֶׁשִּׁדֵּר אֶת הצִלּוּמִים מֵחַיָּיו הַפְּרָטִיִּים שֶׁל אוֹסבּוֹרְן, הוֹדִיעַ עַל הִתְעַנְיְנוּת רַבָּה מְאֹד בַּתָּכְנִית. מִילְיוֹנֵי צוֹפִים בְּרַחֲבֵי אַרְצוֹת הַבְּרִית וְהעוֹלָם הִתְיַשְּׁבוּ כָּל יוֹם מוּל הַמִּרְקָע וְצָפוּ בְּמִשְׁפַּחַת אוֹסבּוֹרְן.
הַכֶּסֶף לֹא מְשַׁנֶּה אֶת הָאָדָם
"זוֹ מִשְׁפָּחָה רְגִילָה לְגַמְרֵי", אָמְרָה קֶרֶן ג'וֹנְס, שֶׁצָּפְתָה בַּתָּכְנִית. זֶה הָיָה מְעַנְיֵן אוֹתִי לִרְאוֹת אֵיךְ מִשְׁפָּחָה שֶׁל עֲשִׁירִים וּמְפֻרְסָמִים מִתְנַהֶגֶת. זֶה הָיָה מְעַנְיֵן לְגַלּוֹת, שֶׁהַבְּעָיוֹת שֶׁלָּהֶם דּוֹמוֹת לַבְּעָיוֹת שֶׁל כָּל אֶחָד. הַכֶּסֶף לֹא מְשַׁנֶּה אֶת הָאָדָם, גַּם אִם הוּא גָּר בְּבַיִת עֲנָק".
הַצְּלָחַת הַתָּכְנִית הֵבִיאָה לְשִׁנוּי אֵצֶל הַמְּפֻרְסָמִים. בְּעוֹד שֶׁבֶּעָבָר הֵם לֹא הִסְכִּימוּ שֶׁמַּצְלֵמוֹת הַטֶּלֶוִיזְיָה תִּפְעַלְנָה בַּבַּיִת שֶׁלָּהֶם, הַיּוֹם הֵם שִׁנּוּ אֶת דַּעְתָּם. "אָנוּ מְקַבְּלִים פְּנִיּוֹת רַבּוֹת מִשַּׂחְקָנֵי קוֹלְנוֹעַ וּמְפֻרְסָמִים אֲחֵרִים, הַמְּעֻנְיָנִים שֶׁנַּחְשֹׂף אוֹתָם לַצִּבּוּר. הַיּוֹם הַסֶּלֶבְּרִיטָאִים פְּתוּחִים יוֹתֵר לַצַּלָּמִים וְלַהַצְּלָחָה שֶׁל מִשְׁפַּחַת אוֹסבּוֹרְן יֵשׁ חֵלֶק בַּזֶּה", אוֹמְרִים ב-MTV. 
הַחֲשִׂיפָה שֶׁל מִשְׁפַּחַת אוֹסבּוֹרְן לֹא נִשְׁאֲרָה רַק בַּעֲרוּץ הַמּוּסִיקָה. הַמִּשְׁפָּחָה הֻזְמְנָה לְכָל תָּכְנִיּוֹת הָאֵרוּחַ הַחֲשׁוּבוֹת וַאֲפִלּוּ הַנָּשִׂיא בּוּשׁ אֵרֵחַ אֶת בְּנֵי הַמִּשְׁפָּחָה, שֶׁהָפְכוּ לִהְיוֹת פּוֹפּוּלָרִיִּים יוֹתֵר וְיוֹתֵר. בְּנֵי הַנֹּעַר בְּאָמֶרִיקָה בָּחֲרוּ אֶת יַלְדֵי מִשְׁפַּחַת אוֹסבּוֹרְן לַיְלָדִים שֶׁל הַשָּׁנָה.
נְצִיגַת חוֹלֵי הַסַּרְטָן
בַּחֹדֶשׁ שֶׁעָבַר, סִפְּרָה הָאֵם, שָׁרוֹן אוֹסבּוֹרְן, שֶׁנִּתְגַּלְּתָה אֶצְלָהּ מַחֲלַת הַסַּרְטָן. בְּתוֹךְ זְמַן קָצָר, הָפְכָה שָׁרוֹן לִנְצִיגַת חוֹלֵי הַסַּרְטָן. הִיא סִפְּרָה עַל הַבְּעָיוֹת בַּמַּחֲלָה שֶׁלָּהּ, עַל הַצּוּרָה בָּהּ זֶה נִתְגַּלָּה. הִיא גַּם הֵחֵלָּה לָתֵת עֵצוֹת לְנָשִׁים אֵיךְ לְהִתְגַּבֵּר עַל מַחֲלַת הַסַּרְטָן.
הַכְּנִיסָה לַחַיִּים הַפְּרָטִיִּים שֶׁל הַמִּשְׁפָּחָה הַמְּפֻרְסֶמֶת הִכְנִיסָה לָהּ סְכוּם שֶׁל מִילְיוֹנֵי דּוֹלָרִים. הֵם סִכְּמוּ כְּבָר עַל הֶמְשֵׁךְ הַצִּלּוּמִים, עֲבוּרָם יְקַבְּלוּ סְכוּם נָאֶה שֶׁל 7 מִילְיוֹן דּוֹלָר.
אוּלָם, לַמְרוֹת הַהִתְעַנְיְנוּת הָרַבָּה וְהִתְלַהֲבוּת הַצּוֹפִים, יֵשׁ כְּאֵלֶּה שֶׁאֵינָם מְרֻצִּים מֵהַתּוֹפָעָה. כָּזֶה הוּא הַשַּׂחְקָן בִּיל קוֹסְבִּי, שֶׁהָיָה כּוֹכַב תָּכְנִית טֶלֶוִיזְיָה שֶׁעָסְקָה בְּמִשְׁפַּחַת שְׁחֹרִים חֲבִיבָה. "אָסוּר לָנוּ לְהִתְעַסֵּק בְּתָכְנִיּוֹת כְּאֵלֶּה", אוֹמֵר בִּיל קוסבי. "מַה שֶּׁרָאִינוּ לְפָנֵינוּ זוֹ מִשְׁפָּחָה עֲצוּבָה, עִם מְרִיבוֹת וּמְתָחִים. זֶה לֹא בִּדּוּר וְלֹא תָּכְנִית מְתֻחְכֶּמֶת. אֲנִי לֹא הָיִיתִי רוֹצֶה לָקַחַת חֵלֶק בְּכָךְ".
הַצִּבּוּר יַתְחִיל לְהִשְׁתַּעְמֵם
מְבַקְּרֵי הַטֶּלֶוִיזְיָה לֹא מַאֲמִינִים שֶׁתָּכְנִיּוֹת מִסּוּג כָּזֶה יַמְשִׁיכוּ לִהְיוֹת הַצְּלָחָה. "אֵין לְתָכְנִיּוֹת כְּאֵלָה עֲתִיד. אִי אֶפְשָׁר לְחַדֵּשׁ בָּהֶן. מַה יִּהְיֶה הַשָּׁלָב הַבָּא?", שׁוֹאֵל מְבַקֵּר הַטֶּלֶוִיזְיָה שֶׁל ה"ניו-יוֹרֵק טיימס", "הַאִם זֶה יִהְיֶה שִׁיטוֹת הַנִּקָּיוֹן שֶׁל הָעֲשִׁירִים וְהַמְּפֻרְסָמִים?". הַמֻּמְחִים חוֹשְׁבִים שֶׁבְּתוֹךְ שָׁנָה לֹא נִרְאֶה תָּכְנִיּוֹת כְּאֵלֶּה. "תִּרְאוּ מַה קָּרָה עִם הַתָּכְנִית הַפּוֹפּוּלָרִית בְּיוֹתֵר בַּשָּׁנָה שֶׁעָבְרָה, "מִי רוֹצֶה לִהְיוֹת מִילְיוֹנֶר". הַצִּבּוּר הֵבִין אֶת הָרַעְיוֹן וְהִפְסִיק לִצְפּוֹת", הֵם קוֹבְעִים.
בְּיִשְׂרָאֵל קָשֶׁה לִרְאוֹת זַמָּרִים וְשַׂחְקָנִים שֶׁיַּסְכִּימוּ שֶׁיְּצַלְּמוּ אוֹתָם בַּחֲדַר הַשֵּׁנָה שֶׁלָּהֶם, אוֹ בִּמְרִיבָה עִם הַיְּלָדִים שֶׁלָּהֶם. קָשֶׁה עוֹד יוֹתֵר לְהַאֲמִין, שֶׁתִּהְיֶה תַּחֲנַת טֶלֶוִיזְיָה שֶׁתִּהְיֶה מוּכָנָה לְשַׁדֵּר תָּכְנִית מִסּוּג שֶׁכָּזֶה כָּל יוֹם. עוֹד יוֹתֵר קָשֶׁה לְהַאֲמִין, שֶׁיִּהְיוּ מַסְפִּיק צוֹפִים לְסוּג זֶה שֶׁל בִּדּוּר. בְּקִצּוּר, אֵין סִכּוּי שֶׁתָּכְנִית כָּזוֹ תַּגִּיעַ לְיִשְׂרָאֵל.
