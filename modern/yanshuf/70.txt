סֶקֶר: הַיַּלְקוּט כָּבֵד מִדַּי
אִם הַיַּלְקוּט שֶׁלָּכֶם שׁוֹקֵל יוֹתֵר מֵעֲשָׂרָה אָחוּז מִמִּשְׁקַל הַגּוּף שֶׁלָּכֶם, אַתֶּם בִּבְעָיָה וְאַתֶּם לֹא לְבַד.
סֶקֶר שֶׁנֶּעֱרַךְ בְּקֶרֶב תַּלְמִידֵי בָּתֵּי הַסֵּפֶר מְגַלֶּה – הַיַּלְקוּטִים שֶׁל הַתַּלְמִידִים כְּבֵדִים מִדַּי. 
עוֹרֶכֶת הַסֶּקֶר, סוּזָן לִימוֹן, מִמִּשְׂרַד הַבְּרִיאוּת, בָּדְקָה לְמַעְלָה מ-10,000 תַּלְמִידִים וְגִלְּתָה כִּי רַבִּים מֵהֶם סוֹחֲבִים עַל הַגַּב שֶׁלָּהֶם מִשְׁקָל הַגָּבוֹהַּ ב-50% מֵהָרָצוּי.
"הַמִּשְׁקָל הָעוֹדֵף הַזֶּה עָלוּל לִפְגֹּעַ בַּגַּב שֶׁל הַתַּלְמִידִים. יֵשׁ לַעֲשׂוֹת הַכֹּל, כְּדֵי שֶׁמִּשְׁקַל הַיַּלְקוּט לֹא יַעֲלֶה עַל 10% מִמִּשְׁקַל הַגּוּף שֶׁל הַיֶּלֶד", הִיא אוֹמֶרֶת. "יֶלֶד שֶׁסּוֹבֵל מִכְּאֵבֵי גַּב בְּגִיל צָעִיר, יִסְבֹּל מֵהַגַּב שֶׁלּוֹ הַרְבֵּה יוֹתֵר כַּאֲשֶׁר יִהְיֶה מְבֻגָּר".
בְּעָיָה אַחֶרֶת הָעֲלוּלָה לִפְגֹּעַ בַּגַּב שֶׁל הַתַּלְמִידִים הִיא, הַיְּשִׁיבָה הָאֲרֻכָּה בַּשִּׁעוּרִים. מֻמְחִים אָמֶרִיקָנִיִּים מַמְלִיצִים שֶׁלֹּא לָשֶׁבֶת יוֹתֵר מֵחֲצִי שָׁעָה בִּרְצִיפוּת. כְּיָדוּעַ, הַשִּׁעוּרִים בְּאֶרֶץ נִמְשָׁכִים שְׁלֹשֶׁת רִבְעֵי הַשָּׁעָה.
