הַנְּמָלָה חֲכָמָה יוֹתֵר מֵהָאָדָם
מָה הַקֶּשֶׁר בֵּין הַנְּמָלִים וְהַפְּקָקִים בָּרְחוֹבוֹת? אֵיךְ נוּכַל לְטַפֵּל בַּזֶּבֶל שֶׁכָּל אֶחָד מֵאִתָנוּ מְיַצֵר בַּצוּרָה הַיְּעִילָה בְּיוֹתֵר? עַל הַקֶּשֶׁר שֶׁבֵּין שְׁלֹמֹה הַמֶּלֶךְ, בְּנִיָּה, נְמָלִים וּפְּרוֹפֶסוֹרִים בָּאוּנִיבֶרְסִיטָה *
שְׁלֹמֹה הַמֶּלֶךְ, הֶחָכָם מִכָּל אָדָם, מַמְלִיץ לְכָל אֶחָד מֵאִתָנוּ לָלֶכֶת אֶל הַנְּמָלָה, לִלְמוֹד מִמֶנָה וְלִהְיוֹת חֲכָמִים יוֹתֵר.
נִרְאֶה, שֶׁבְּנֵי הָאָדָם לֹא קִבְּלוּ אֶת הַהַמְּלָצוֹת שֶׁל הֶחָכָם מִכָּל הָאָדָם בְּמֶשֶׁךְ אַלְפֵי שָׁנִים. רַק בָּאַחֲרוֹנָה, גִּלוּ מַדְעָנִים אַנְגְּלִים, שֶׁהַמֶּלֶךְ מֵהַתָּנָ"ךְ צָדַק.
הַחוֹקְרִים יָצְאוּ לִיעָרוֹת הַגֶּשֶׁם שֶׁבְּפָנָמָה. הֵם עָקְבוּ אַחֲרֵי הִתְנַהֲגוּת נְמָלִים. בְּעֶזְרַת תָכְנַת מַחְשֵׁב מְדֻיֶקֶת, הֵם בָּדְקוּ אֶת הַתְּנוּעָה שֶׁל קְבוּצוֹת עֲנָקִיוֹת שֶׁל נְמָלִים. הֵם גִּלוּ, שֶׁלַמְרוֹת שֶׁהַנְּמָלָה עִוֶּרֶת וּקְטַנָה מְאֹד, הִיא מַצְלִיחָה לְנַהֵל אֶת הַתַּעֲבוּרָה שֶׁל כָּל פְּרָט וּפְרָט. הַחוֹקְרִים גַם בָּדְקוּ מָה עוֹשׂוֹת הַנְּמָלִים כְּשֶׁהֵן צְרִיכוֹת לַעֲבוֹר מִמָּקוֹם לְמָקוֹם בִּמְהִירוּת. "אֵצֶל בְּנֵי הָאָדָם, כַּאֲשֶׁר יֵשׁ צֹרֶךְ שֶׁל קְבוּצוֹת גְּדוֹלוֹת שֶׁל אֲנָשִׁים לַעֲבוֹר מִמָּקוֹם לְמָקוֹם, נוֹצָרִים פְּקָקִים. אֵצֶל הַנְּמָלָה אֵין מֻשָּׂג כָּזֶה "פְּקָקִים". הִיא מַצְלִיחָה לִפְתוֹר אֶת הַבְּעָיוֹת שֶׁלָה עַל יְדֵי הַגְּדָלַת שֶׁטַח הַתְּנוּעָה", אוֹמֵר פרופ' נָייגֶ'ל פְרַאנְקס, רֹאשׁ צֶוֶת הַמֶּחְקָר. "הֵן מִתְנַהֲגוֹת בְּצוּרָה קְבוּצָתִית וּפוֹתְרוֹת אֶת הַבְּעָיָה בְּצוּרָה יְעִילָה בְּיוֹתֵר".
לְאַחַר שְׁנוֹת מֶחְקָר אֲרֻכּוֹת, הִגִיעוּ הַחוֹקְרִים לַמַּסְקָנָה, שֶׁבְּנֵי הָאָדָם יְכוֹלִים לִלְמוֹד מֵהַנְּמָלִים עוֹד דְּבָרִים רַבִּים אֲחֵרִים. "גַם לַנְּמָלִים יֵשׁ בְּעָיוֹת שֶׁל סִלוּק זֶבֶל וְשֶׁל בְּנִיָּה. לֶעָרִים שֶׁל הַנְּמָלִים יֵשׁ בְּעָיוֹת הַדּוֹמוֹת לְאֵלֶה שֶׁיֵשׁ בֶּעָרִים שֶׁלָּנוּ" אוֹמֵר פרופ' נָייגֶ'ל. "הַטֶּבַע כְּבָר מָצָא אֶת הַפִּתְרוֹנוֹת לְאוֹתָן בְּעָיוֹת. אִם נִלְמַד מֵהַנְּמָלִים וּמֵחַיוֹת אֲחֵרוֹת, אֵיךְ לִפְתוֹר אֶת אוֹתָן הַבְּעָיוֹת, נַרְוִיחַ מִכָּךְ הַרְבֵּה".
אִם הַפְּרוֹפֶסוֹר הַמְּכֻבָּד הָיָה קוֹרֵא אֶת הַדְּבָרִים שֶׁל שְׁלֹמֹה הַמֶּלֶךְ, אוּלַי הוּא הָיָה יָכֹל לַחְסוֹךְ לְעַצְמוֹ אֶת הַמַּאֲמָץ שֶׁל הַמֶּחְקָר הַקָּשֶׁה שֶׁעָרַךְ.
