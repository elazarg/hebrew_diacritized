פִּנוֹקְיוֹ
יֶלֶד הָעֵץ הַמֻּכָּר בְּיוֹתֵר בָּעוֹלָם, חוֹזֵר שׁוּב לְמָסַכֵּי הַקּוֹלְנוֹעַ. לְאַחַר שֶׁכִּכֵב בְּסֶרֶט מְצֻיָר, זוֹכֶה פִּנוֹקְיוֹ לְטִפּוּלוֹ שֶׁל הַקּוֹמִיקַאי הָאִיטַלְקִי רוֹבֵּרְטוֹ בֵּנִינִי, שֶׁגַם מְשַׂחֵק אֶת הַבֻּבָּה שֶׁיָצַר סַבָּא גֵ'פֵּטוֹ * "זֶהוּ סֶרֶט שֶׁמַצִיג אֶת הָרָצוֹן שֶׁל הַזָּר לְהִשְׁתַלֵב בַּחֶבְרָה", אוֹמְרִים יוֹצְרֵי הַסֶּרֶט * הַבֻּבָּה עַל הַחוּטִים מִמְשִׁיכָה לְהַלְּהִיב אֶת הַצּוֹפִים *
אַרְבַּע שָׁנִים לְאַחַר הַהַצְּלָחָה הַגְּדוֹלָה שֶׁל רוֹבֵּרְטוֹ בֵּנִינִי, בַּסֶרֶט "הַחַיִים הַיָּפִים", חוֹזֵר אֵלֵינוּ הַשַּׂחְקָן הָאִיטַלְקִי בְּסֶרֶט חָדָשׁ – פִּנוֹקְיוֹ. 
הַסֶּרֶט מְבֻסָס עַל הַסִּפּוּר שֶׁל קַרְלוֹ קוֹלוֹדִי, אוֹדוֹת הַנַּגָר סַבָּא גֵ'פֵּטוֹ, שֶׁיָצַר בֻּבַּת עֵץ בַּשֵׁם "פִּנוֹקְיוֹ". לְגֵ'פֵּטוֹ אֵין יְלָדִים וְהוּא חוֹלֵם עַל יֶלֶד מִשֶׁלוֹ. וְהִנֵה, הַבֻּבָּה שֶׁיָצַר הָפְכָה לְיֶלֶד. אוּלָם, זֶהוּ לֹא יֶלֶד רָגִיל. הַפֵּיָה הַטּוֹבָה וגֵ'פֵּטוֹ  מְנַסִים לַעֲזוֹר לפִּנוֹקְיוֹ, אַךְ הוּא רוֹצֶה לִהְיוֹת יֶלֶד כְּכָל הַיְּלָדִים.
כְּדֵי לְהַשִּׂיג אֶת הַמַּטָרָה, יוֹצֵא פִּנוֹקְיוֹ לְמַסָע אָרֹךְ, בּוֹ הוּא מִסְתַבֵּךְ בְּצָרוֹת. הוּא מָאֳשָׁם בְּפֶשַׁע שֶׁלֹא עָשָׂה וּמִתְאַכְזֵב מֵהָעוֹלָם שֶׁכָּךְ נוֹהֵג. הַדְּמֻיוֹת הַקְּרוֹבוֹת אֵלָיו - סַבָּא גֵ'פֵּטוֹ, הַפֵּיָה הַטּוֹבָה וְהַצְּרָצַר הַמְּדַבֵּר, מְנַסִים לְשַׁכְנֵעַ אוֹתוֹ לְקַבֵּל אֶת הַחֻקִים וְהָעֲרָכִים שֶׁל הַחַיִים בַּחֶבְרָה הָאֱנוֹשִׁית, לַמְרוֹת שֶׁהֵם לֹא תָמִיד צוֹדְקִים.
הַסֶּרֶט שׁוֹנֶה מֵהַסִּפּוּר הַמְּקוֹרִי, אַךְ שׁוֹמֵר עַל הָרַעְיוֹנוֹת הַמֶּרְכָּזִיִים שֶׁלוֹ. רוֹבֵּרְטוֹ בֵּנִינִי מְשַׂחֵק אֶת פִּנוֹקְיוֹ כְּבֻבָּה הַמֻּפְעֶלֶת בְּחוּטִים. פִּנוֹקְיוֹ מַמְשִׁיךְ לִהְיוֹת אוֹתָה הַדְּמוּת הַתְּמִימָה, שֶׁאוֹהֶבֶת מְאֹד אֶת הַחַיִים. הוּא מְנַסֶה לְהִשְׁתַלֵב בָּעוֹלָם הָרָגִיל וְרוֹצֶה לְהָבִין אוֹתוֹ.
"יֵשׁ כָּאן רָצוֹן לְהַרְאוֹת אֶת הַמַּאֲבָק שֶׁל הָאָדָם הַזָּר, שֶׁרוֹצֶה לְהִשְׁתַלֵב בַּחֶבְרָה" אוֹמְרִים יוֹצְרֵי הַסֶּרֶט.
שׁוֹטְרִים לְלֹא הַפְּסָקָה
זָוִית צִלוּם לֹא נְכוֹנָה, דְבוֹרָה מִזְדַמֶנֶת וְשׁוֹטֵר שַׁחְצָן מְשַׁנִים אֶת סֵדֶר הַחַיִים שֶׁל הַדְּמוּיוֹת הָרָאשִׁיוֹת, הַמּוֹצְאוֹת אֶת עַצְמָן בְּסַכָּנוֹת לֹא צְפוּיוֹת * סֶרֶט פְּעֻלָה מַלְהִיב וּמַצְחִיק, שֶׁאֵינוֹ דוֹמֶה לְסִרְטֵי הַפְּעֻלָה שֶׁהִתְרַגַלְנוּ אֲלֵיהֶם * 
דְבוֹרָה קְטַנָה מְשַׁנָה אֶת הַחַיִים שֶׁל שְׁנֵי אֲנָשִׁים, הַמּוֹצְאִים אֶת עַצְמָם בְּתוֹךְ סַכָּנוֹת וְהַרְפַּתְקָאוֹת מְשַׁעַשְׁעוֹת.
מַרְטִין לוֹרֵנְס, הוּא כּוֹכַב סֶרֶט הַפְּעֻלָה "שׁוֹטְרִים לְלֹא הַפְּסָקָה". הוּא רוֹצֶה לִהְיוֹת שׁוֹטֵר, אֲבָל בְּבֵית הַסֵּפֶר לְשׁוֹטְרִים לֹא אוֹהֲבִים אוֹתוֹ כָּל כָּךְ. הוּא שַׁחְצָן וְלֹא מַצְלִיחַ לִשְׁלוֹט עַל הָעֲצַבִּים שֶׁלוֹ. הַמַּדְרִיכִים מַחְלִיטִים לְסַלֵק אוֹתוֹ מִבֵּית הַסֵּפֶר לְשׁוֹטְרִים.
מֵחֹסֶר בְּרֵרָה, הוּא הוֹפֵךְ לִהְיוֹת שׁוֹמֵר בְּמִשְׁמָר הַלְּאֻמִי הָאָמֶרִיקָאִי. 
קֲצִין מִשְׁטָרָה בְּשֵׁם הָאנְק רַפֵּרְטִי, הַלֹּא הוּא סְטִיב זָ'אן, עוֹצֵר אֶת מַרְטִין לוֹרֵנְס, בַּחֲשָׁד שֶׁהוּא פָּרַץ לָרֶכֶב שֶׁלוֹ. לוֹרֵנְס, שְׁחוּם הָעוֹר, מְנַסֶה לְשַׁכְנֵעַ אֶת קֲצִין הַמִּשְׁטָרָה הָעַצְבָּנִי, שֶׁהוּא טוֹעֶה. וְכָאן נִכְנֶסֶת הַדְּבוֹרָה, שֶׁמְשַׁנָה לַשְׁנַיִם אֶת הַחַיִים. הָאנְק, שֶׁרוֹצֶה לְהִפָּטֵר מֵהַדְּבוֹרָה, מְנוֹפֵף בְּיָדָיו כְּדֵי לְסַלֵק אוֹתָה. אוּלָם, מֵרָחוֹק זֶה נִרְאָה כְּאִלוּ הוּא מַכֶּה אֶת הָאִישׁ הַשָּׁחֹר שֶׁלְיָדוֹ. הַתְּנוּעוֹת שֶׁלוֹ מְצֻלָמוֹת עַל יְדֵי אֶזְרָח שֶׁעָמַד בְּמֶרְחָק מֵהַשְּׁנַיִם. הוּא מַעֲבִיר אֶת הַצִּלוּמִים לְרֶשֶׁת טֶלֶוִיזְיָה, שֶׁמַצִיגָה אֶת הַשּׁוֹטֵר כְּמִי שֶׁמַכֶּה וּמִתְעַלֵל בְּאֶזְרָחִים. הוּא מָאֳשָׁם בְּהַכָּאַת אָדָם שָׁחֹר, מֻכְנָס לְכֶלֶא וּמְסֻלָק מֵהַמִּשְׁטָרָה.
מֵחֹסֶר בְּרֵרָה, גַם הוּא מוֹצֵא אֶת עַצְמוֹ מִתְגַיֵס לַמִשְׁמָר הַלְּאֻמִי, שָׁם הוּא פּוֹגֵשׁ שׁוּב אֶת מַרְטִין לוֹרֵנְס. כַּמוּבָן, שֶׁאֵין בְּנֵיהֶם אַהֲבָה גְדוֹלָה, אַךְ סַכָּנוֹת אֲחֵרוֹת גוֹרְמוֹת לָהֶם לְשַׁתֵף פְּעֻלָה.
"שׁוֹטְרִים לְלֹא הַפְּסָקָה" הוּא סֶרֶט פְּעֻלָה מִצַד אֶחָד וּמַצְחִיק מִצַד שֵׁנִי. הַצּוֹפִים נֶהֱנִים מִסִפּוּר עֲלִילָה מוֹתֵחַ וּמִשְׁנֵי שַׂחְקָנִים מַצְחִיקִים.
