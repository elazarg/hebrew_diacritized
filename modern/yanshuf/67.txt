250 מִלְיוֹן יְלָדִים בְּרַחֲבֵי הָעוֹלָם מְנֻצָּלִים כְּכֹחַ עֲבוֹדָה
הַיְּלָדִים, שֶׁאֵינָם יְכוֹלִים לְהָגֵן עַל עַצְמָם, נִלְקָחִים לַעֲבוֹדוֹת קָשׁוֹת. הַתַּשְׁלוּם שֶׁהֵם מְקַבְּלִים קָטָן בְּיוֹתֵר וּפְעָמִים רַבּוֹת הֵם אֲפִלּוּ לֹא מְקַבְּלִים תַּשְׁלוּם כְּלָל. חֵלֶק מֵהֶם נִפְגַּע בַּעֲבוֹדוֹת וְנִשְׁאַר נָכֶה לְכָל חַיָּיו. הַסִּכּוּיִים שֶׁל יְלָדִים אֵלֶּה, שֶׁלֹּא לוֹמְדִים, לְהַצְלִיחַ בְּחַיִּים, קָטָן מְאֹד. הָאוּ"ם מְנַסֶּה לְהָגֵן עַל אוֹתָם יְלָדִים, שֶׁרוּבָּם נִמְצָאִים בְּמִזְרַח אַסְיָה וּבְאַפְרִיקָה.
אִרְגּוּן יוּנִיצֶ"ף שֶׁל הָאוּ"ם פִּרְסֵם הֶחָדָשׁ נְתוּנִים קָשִׁים עַל זְכִיּוֹת הַיְּלָדִים בָּעוֹלָם. עַל פִּי פִּרְסוּמִי הָאִרְגּוּן, לְמַעְלָה מִ-250 מִלְיוֹן יְלָדִים בְּרַחֲבֵי הָעוֹלָם עוֹבְדִים בַּעֲבוֹדוֹת שׁוֹנוֹת. הַמַּעֲסִיקִים שֶׁלָּהֶם מַכְרִיחִים אוֹתָם לַעֲבוֹד, לַמְרוֹת שֶׁהֵם לֹא רוֹצִים.
כְּתוֹצָאָה מִכָּךְ הַיְּלָדִים נֶחֱשָׂפִים לַתְּאוּנוֹת וְלָמוּת, הַבְּרִיאוּת שֶׁלָּהֶם נִפְגַּעַת וְהֵם לֹא הוֹלְכִים לִלְמוֹד. כַּאֲשֶׁר הַיְּלָדִים הָאֵלֶּה גְּדֵלִים, הֵם נִשְׁאָרִים בְּלִי הַשְׂכָּלָה וְעִם הַרְבֵּה בְּעָיוֹת רְפוּאִיּוֹת. הֶעָתִיד שֶׁלָּהֶם אֵינוֹ טוֹב וְהַסִּכּוּיִים שֶׁלָּהֶם לְהַצְלִיחַ וּלְהִתְקַדֵּם קְטַנִּים.
רַב הַיְּלָדִים הָעוֹבְדִים נֶגֶד רְצוֹנָם, נִמְצָאִים בְּמִזְרַח אַסְיָה. יֶשְׁנָן מְדִינוֹת בָּהֶן הַיְּלָדִים עוֹבְדִים בְּתַעֲשִׂיַּת הַשְּׁטִיחִים. הַיָּדַיִם הַקְּטַנּוֹת שֶׁלָּהֶם מַתְאִימוֹת לַהֲכָנַת הַשְּׁטִיחִים. כַּאֲשֶׁר אוֹתָם יְלָדִים גְּדֵלִים, הֵם נִזְרָקִים מִמְּקוֹמוֹת הָעֲבוֹדָה.
יֶשְׁנָן מְדִינוֹת הַמַּעֲסִיקוֹת יְלָדִים בִּתְנָאִים גְּרוּעִים מְאֹד. הַמַּעֲסִיקִים שֶׁלָּהֶם מְשַׁלְּמִים לָהֶם שָׂכָר נָמוּךְ וּמְנַצְּלִים אוֹתָם. פֻּרְסַם, שֶׁנַּעֲלֵי חֶבְרַת נָיְקִי מִיצְרִים בְּמִפְעָלִים הַמְּנַצְּלִים יְלָדִים. רַבִּים בְּאָמֵרִיקָה הִצִּיעוּ לְהַחֲרִים אֶת מוּצְרֵי הַסְּפּוֹרְט שֶׁל הַחֶבְרָה כְּדֵי לְהַכְרִיחַ אוֹתָהּ לַעֲזוֹר לַיְּלָדִים וְלֹא לִגְרוֹם לַנִּצּוּל שֶׁלָּהֶם.
גַּם בַּחֲלָקִים אֲחֵרִים בָּעוֹלָם יֶשְׁנָם יְלָדִים רַבִּים הַנֶּאֱלָצִים לַעֲבוֹד, מִפְּנֵי שֶׁהַמְּבֻגָּרִים מַכְרִיחִים אוֹתָם.
בְּיִשְׂרָאֵל יֶשְׁנוֹ חֹק הָאוֹסֵר עַל הַעֲסָקַת יְלָדִים, וּבֶאֱמֶת מִסְפַּר הַיְּלָדִים הָעוֹבְדִים בְּאֶרֶץ קָטָן בְּיוֹתֵר. 
אִרְגּוּן יוּנִיצֶ"ף שַׁיָּךְ לָאוּ"ם וְתַפְקִידוֹ לְטַפֵּל בִּזְכִיּוֹת הַיֶּלֶד בָּעוֹלָם. בָּאַחֲרוֹנָה, יָצָא הָאִרְגּוּן בְּסִדְרַת מוֹדָעוֹת, לְטוֹבַת הַיְּלָדִים.
מֵעַל הַתְּמוּנָה שֶׁלִּפְנֵיכֶם, שֶׁפֻּרְסְמָה עַל יְדֵי יוּנִיצֶ"ף, נִכְתַּב: "מֵרָחוֹק זֶה נִרְאֶה יָפֶה". הַתְּמוּנָה מַצִּיגָה יְלָדִים מִשְׂחָקִים וְנֶהֱנִים. אוּלָם, כַּאֲשֶׁר מִסְתַּכְּלִים טוֹב יוֹתֵר בַּפְּרָטִים, אֶפְשָׁר לִרְאוֹת שֶׁהַיְּלָדִים עוֹבְדִים, שֶׁיֵּשׁ שׁוֹמְרִים שֶׁמּוֹנְעִים מֵהֶם לִבְרוֹחַ, וְשֶׁבְּעֶצֶם הַחַיִּים שֶׁלָּהֶם לֹא כָּל כָּךְ יָפִים.
