מִיקִי מָאוּס וְדוֹנָלְד דַּאק בִּבְעָיוֹת
תַּיָּרִים הִפְסִיקוּ לָבוֹא לְפַּארְקֵי הַשַּׁעֲשׁוּעִים וְהַהֶפְסֵדִים הַכַּסְפִּיִּים גָּדְלוּ * נִסָּיוֹן לְהוֹצִיא סֶרֶט קֻפָּתִי נִכְשַׁל, ו-180 מִילְיוֹן דּוֹלָר הָלְכוּ לְאִבּוּד * מְבַקְּרֵי הַקּוֹלְנוֹעַ: אֲנַחְנוּ מְחַכִּים לְסֶרֶט מַצְלִיחַ בְּסִגְנוֹן "מֶלֶךְ הָאֲרָיוֹת", אוֹ "בַּת הַיָּם הַקְּטַנָּה" * "אֲנַחְנוּ בִּתְקוּפָה לֹא טוֹבָה", מוֹדִים הַבְּעָלִים שֶׁל מִיקִי מָאוּס וְדוֹנָלְד דַּאק *
אַחַת הַבְּעָיוֹת הַגְּדוֹלוֹת שֶׁל פַּארְקֵי הַשַּׁעֲשׁוּעִים, הֵם הַתּוֹרִים הָאֲרֻכִּים לִפְנֵי הַמִּתְקָנִים. לִפְעָמִים, צָרִיךְ לַעֲמֹד בַּתוֹר לְמַעְלָה מִשָּׁעָה כְּדֵי לְהִכָּנֵס לְמִתְקָן מֻצְלָח. לֹא עוֹד. כַּיּוֹם, מִי שֶׁלֹּא אוֹהֵב אֶת הַתּוֹרִים הָאֲרֻכִּים הָאֵלֶּה, יָכֹל לָבוֹא לְאֶחָד מִפַּארְקֵי הַשַּׁעֲשׁוּעִים שֶׁל ווֹלְט דִּיסְנִי וְכִמְעַט לֹא לַעֲמֹד בַּתּוֹר. הַסִּבָּה – מְעַט מְאֹד מְבַקְּרִים.
"אֲנַחְנוּ בִּתְקוּפָה לֹא טוֹבָה", אוֹמְרִים הַמְּנַהֲלִים שֶׁל דִּיסְנִי ווֹרְלְד בְּאַרְצוֹת הַבְּרִית. "בִּגְלַל הַתְּקָפוֹת הַטֶּרוֹר, פָּחוֹת תַּיָּרִים בָּאִים לְכָאן וַאֲנַחְנוּ מַרְגִּישִׁים זֹאת הֵיטֵב. בְּנוֹסָף לְכָךְ, הַמַּצָּב הַכַּלְכָּלִי בָּעוֹלָם וּבְאָמֶרִיקָה, גּוֹרֵם לְרַבִּים לַחְסֹךְ כֶּסֶף וְלֹא לָבוֹא אֵלֵינוּ".
בִּגְלַל הַקְּשָׁיִים, הוֹדִיעוּ בְּחֶבְרַת דִּיסְנִי ווֹרְלְד, שֶׁאוֹרְחִים הַמִּתְגוֹרְרִים בַּמְּלוֹנוֹת שֶׁלָּהֶם לֹא יוּכְלוּ לְקַבֵּל הֲנָחוֹת בְּפַּארְק הַשַּׁעֲשׁוּעִים כְּמוֹ בֶּעָבָר. אֶלָּא שֶׁבִּמְקוֹם לְהַרְוִיחַ מֵהַקִּצּוּץ, אוֹרְחֵי הַמְּלוֹנוֹת כָּעֲסוּ וְהִפְסִיקוּ לָבוֹא.
הַבְּעָיוֹת שֶׁל דִּיסְנִי לֹא מִסְתַּיְּמוֹת רַק בַּפַּארְקִים. בָּאַחֲרוֹנָה, הֵם הֵפִיקוּ כַּמָּה סְרָטִים, מִתּוֹךְ תִּקְוָה שֶׁהָרְוָחִים יַגִּיעוּ מִשָּׁם. אֶלָּא שֶׁגַּם כָּאן הֵם נִכְשְׁלוּ. הַסֶּרֶט "אִי הַמַּטְמוֹן", עָלָיו עָבְדוּ לְמַעְלָה מִשְּׁנָתַיִם, לֹא הִצְלִיחַ וְהַשְּׁקָעָה שֶׁל 180 מִילְיוֹן דּוֹלָר הָלְכָה לְאִבּוּד. מְבַקְּרֵי הַקּוֹלְנוֹעַ קָטְלוּ אֶת הַסֶּרֶט וַאֲנָשִׁים הִפְסִיקוּ לָבוֹא לְבָתֵּי הַקּוֹלְנוֹעַ.
"אֲנַחְנוּ עֲדַיִן מְחַכִּים לִסְרָטִים טוֹבִים כְּמוֹ "הַיָּפָה וְהַחַיָּה", "בַּת הַיָּם הַקְּטָנָה" וְ"אָלָאדִין", אוֹמְרִים מְבַקְּרֵי הַקּוֹלְנוֹעַ. "הַסְּרָטִים הַחֲדָשִׁים שֶׁל דִּיסְנִי אֵינָם טוֹבִים וַחֲבָל לְבַזְבֵּז עֲלֵיהֶם אֶת הַכֶּסֶף".
בְּאֻלְפְּנֵי דִּיסְנִי מְנַסִּים לִמְצֹא אֶת הַנֻּסְחָה שֶׁתַּחְזִיר אֶת הַמִּשְׁפָּחוֹת שֶׁמִּלְּאוּ אֶת בָתֵּי הַקּוֹלְנוֹעַ, כְּמוֹ שֶׁעָשׂוּ בֶּעָבָר "מֶלֶךְ הָאֲרָיוֹת", "פַנְטַזְיָה" וַאֲחֵרִים.
