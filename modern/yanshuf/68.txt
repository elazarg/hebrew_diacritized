אִרְגוּן יוניצ"ף שֶׁל האו"ם פִּרְסֵם הַחֹדֶשׁ נְתוּנִים קַשִׁים עַל זְכֻיוֹת הַיְּלָדִים בָּעוֹלָם. עַל פִּי פִּרְסוּמֵי הָאִרְגוּן, לְמַעְלָה מ-250 מִלִיוֹן יְלָדִים בְּרַחֲבֵי הָעוֹלָם עוֹבְדִים בַּעֲבוֹדוֹת שׁוֹנוֹת. הַמַּעֲסִיקִים שֶׁלָהֶם מַכְרִיחִים אוֹתָם לַעֲבוֹד, לַמְרוֹת שֶׁהֵם לֹא רוֹצִים.
כְּתוֹצָאָה מִכָּךְ הַיְּלָדִים נֶחְשָׂפִים לִתְאוּנוֹת וּלְמָוֶת, הַבְּרִיאוּת שֶׁלָהֶם נִפְגַעַת וְהֵם לֹא הוֹלְכִים לִלְמוֹד. כַּאֲשֶׁר הַיְּלָדִים הָאֵלֶה גְדֵלִים, הֵם נִשְׁאָרִים בְּלִי הַשְּׂכָּלָה וְעִם הַרְבֵּה בְּעָיוֹת רְפוּאִיוֹת. הֶעָתִיד שֶׁלָהֶם אֵינוֹ טוֹב וְהַסִּכּוּיִים שֶׁלָהֶם לְהַצְּלִיחַ וּלְהִתְקַדֵם קְטַנִים.
רֹב הַיְּלָדִים הָעוֹבְדִים נֶגֶד רְצוֹנָם, נִמְצָאִים בְּמִזְרַח אַסְיָה. יֶשְׁנָן מְדִינוֹת בָּהֵן הַיְּלָדִים עוֹבְדִים בְּתַעֲשִׂיַת הַשְּׁטִיחִים. הַיָּדַיִם הַקְּטַנוֹת שֶׁלָהֶם מַתְאִימוֹת לַהֲכָנַת הַשְּׁטִיחִים. כַּאֲשֶׁר אוֹתָם יְלָדִים גְדֵלִים, הֵם נִזְרָקִים מִמְקוֹמוֹת הָעֲבוֹדָה.
יֶשְׁנָן מְדִינוֹת הַמַּעֲסִיקוֹת יְלָדִים בִּתְנָאִים גְרוּעִים מְאֹד. הַמַּעֲסִיקִים שֶׁלָהֶם מְשַׁלְמִים לָהֶם שָׂכָר נָמוּךְ וּמְנַצְלִים אוֹתָם. פֻּרְסַם, שֶׁנַעֲלֵי חֶבְרַת נַיְקִי מְיֻצָרִים בְּמִפְעָלִים הַמְּנַצְלִים יְלָדִים. רַבִּים בְּאָמֶרִיקָה הִצִיעוּ לְהַחְרִים אֶת מוּצְרֵי הַסְּפּוֹרְט שֶׁל הַחֶבְרָה כְּדֵי לְהַכְּרִיחַ אוֹתָה לַעֲזוֹר לִילָדִים וְלֹא לִגְרוֹם לְנִצוּל שֶׁלָהֶם.
גַם בַּחֲלָקִים אֲחֵרִים בָּעוֹלָם יֶשְׁנָם יְלָדִים רַבִּים הַנֶּאֱלָצִים לַעֲבוֹד, מִפְּנֵי שֶׁהַמְּבֻגָרִים מַכְרִיחִים אוֹתָם.
בְּיִשְׂרָאֵל יֶשְׁנוֹ חֹק הָאוֹסֵר עַל הַעֲסָקַת יְלָדִים, וּבֶאֱמֶת מִסְפַּר הַיְּלָדִים הָעוֹבְדִים בָּאָרֶץ קָטָן בְּיוֹתֵר. 
אִרְגוּן יוניצ"ף שַׁיָךְ לאו"ם וְתַפְקִידוֹ לְטַפֵּל בִּזְכֻיוֹת הַיֶּלֶד בָּעוֹלָם. בָּאַחֲרוֹנָה, יָצָא הָאִרְגוּן בְּסִדְרַת מוֹדָעוֹת, לְטוֹבַת הַיְּלָדִים.
מֵעַל הַתְּמוּנָה שֶׁלִפְנֵיכֶם, שֶׁפֻּרְסְמָה עַל יְדֵי יוניצ"ף, נִכְתַב: "מֵרָחוֹק זֶה נִרְאֶה יָפֶה". הַתְּמוּנָה מַצִיגָה יְלָדִים מְשַׂחֲקִים וְנֶהֱנִים. אוּלָם, כַּאֲשֶׁר מִסְתַכְּלִים טוֹב יוֹתֵר בַּפְּרָטִים, אֶפְשָׁר לִרְאוֹת שֶׁהַיְּלָדִים עוֹבְדִים, שֶׁיֵשׁ שׁוֹמְרִים שֶׁמוֹנְעִים מֵהֶם לִבְרוֹחַ, וְשֶׁבְּעֶצֶם הַחַיִים שֶׁלָהֶם לֹא כָּל כָּךְ יָפִים.
