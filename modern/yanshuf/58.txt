עוֹלַם הַמַּדָע וְהַתִּקְשֹׁרֶת יוֹצֵא לֹא פַּעַם בְּכוֹתְרוֹת עֲנָק עַל תַגְלִית מַדָעִית חֲשׁוּבָה אוֹ עַל תוֹפַעַת טֶבַע יוֹצֵאת מֵהַכְּלָל. מַדְעָנִים מִכָּל רַחֲבֵי הָעוֹלָם נִפְגָשִׁים כְּדֵי לָדוּן בְּתַגְלִיוֹת הַמְּעַנְיְנוֹת, מְנַסִּים לִמְצוֹא לָהֶם תְּשׁוּבָה וּמְבַזְבְּזִים שָׁנִים בְּמֶחְקָרִים. אוּלָם, לֹא פַּעַם מִתְבָּרֶרֶת הָאֱמֶת – אוֹתָן תוֹפָעוֹת מְעַנְיְנוֹת לֹא קָרוּ בַּמְּצִיאוּת, אֶלָא הָיוּ תַרְמִית שֶׁל בְּנֵי אָדָם.
אַחַת הַתּוֹפָעוֹת הַמְּפֻרְסָמוֹת הִיא מַעְגְלֵי הַחִטָה בְּאַנְגְלִיָה. בְּאֶחָד הַבְּקָרִים, גִלוּ תוֹשָׁבִים שֶׁבִּשְׂדֵה הַחִטָה שֶׁלְיַד הַכְּפָר שֶׁלָּהֶם, נוֹצְרוּ מַעְגָלִים עֲנָקִיִים. אִי אֶפְשָׁר הָיָה לָתֵת סִבָּה הֶגְיוֹנִית לַמַּעְגָלִים הָאֵלֶה.
מַדְעָנִים מִכָּל רַחֲבֵי הָעוֹלָם הִגִיעוּ לַמָקוֹם בְּנִסָיוֹן לִמְצוֹא תְּשׁוּבָה לַתוֹפָעָה הַמּוּזָרָה. הָיוּ שֶׁחָשְׁבוּ, שֶׁאוֹתָם מַעְגָלִים נוֹצְרוּ כְּתוֹצָאָה מִמְעַרְבֹּלֶת שֶׁל רוּחוֹת, אַךְ הַדִּיוּק הָרַב שֶׁלָּהֶם, פָּסַל אֶת הָאֶפְשָׁרוּת הַזּוֹ.
לְאַחַר מַחְשָׁבוֹת וּמְדִידוֹת, הִגִיעוּ הַחוֹקְרִים לַמַּסְקָנָה, שֶׁמְּדֻבָּר בְּחַיְיזָרִים שֶׁבָּאוּ לְבִקוּר בְּכַדוּר הָאָרֶץ וְהַחֲלָלִית שֶׁלָּהֶם גָרְמָה לַסִימָנִים הָאֵלֶה. הַתַּעֲלוּמָה גָבְרָה עוֹד יוֹתֵר, כַּאֲשֶׁר בִּשְׂדוֹת חִטָה אֲחֵרִים נִתְגַלוּ עוֹד עִגוּלִים מִסְתוֹרִיִים שֶׁכְּאֵלֶה. הַמְּעַנְיֵן הָיָה שֶׁכָּל הַמַּעְגָלִים נוֹצְרוּ בַּלַּיְלָה. הַדָּבָר הֵבִיא לְכָךְ, שֶׁחוֹקְרִים רַבִּים וְחוֹבְבֵי חַיְיזָרִים, הִגִיעוּ לְאוֹתוֹ אֵזוֹר חַקְלַאי בְּתִקְוָה לִמְצוֹא אֶת אוֹתָן חֲלָלִיוֹת. הֵם הָיוּ בְּטוּחִים, שֶׁהַחַיזָרִים מַגִיעִים רַק בַּלֵּילוֹת, כְּדֵי לֹא לִפְגוֹשׁ בִּבְנֵי הָאָדָם.
כָּךְ בְּמֶשֶׁךְ עַשְׂרוֹת שָׁנִים, נִסוּ הַחוֹקְרִים לְהָבִין מָה הַמַּשְׁמָעוּת שֶׁל הַקַּוִים הָאֵלֶה, וְהַאִם הֵם מְסַמְּלִים מַשֶׁהוּ מֵהָעוֹלָם הַחִיצוֹן.
רַק בִּשְׁנַת 1991 הוֹדוּ שְׁנֵי גְּבָרִים מְבֻגָרִים, שֶׁהֵם אַחֲרָאִים לַצִיוּרִים הַמִּסְתוֹרִיִים. הֵם גִלוּ אֵיךְ הֵם צִיְרוּ אֶת הַמַּעְגָלִים הָאֵלֶה. "עָשִׂינוּ אֶת זֶה בִּשְׁבִיל הכייף", סִפְּרוּ דָאַג בָּאוּאֵר וְדֵייב צ'וֹרְטְלִי. "לֹא הֶאֱמַנוּ שֶׁהָעוֹלָם כֻּלוֹ יִתְעַנְיֵן בַּצִיוּרִים שֶׁלָּנוּ. אֲנַחְנוּ שְׂמֵחִים שֶׁיָצַרְנוּ עִנְיָן, שֶׁלֹּא הִזִיק לָאַף אֶחָד".
הַמִּקְרֶה שֶׁל מַעְגְלֵי הַחִטָה הָיָה בְּדִיחָה נֶחְמָדָה, שֶׁלֹּא הִזִיקָה לָאַף אֶחָד. אוּלָם, יֶשְׁנָם שְׁקָרִים שֶׁבְּהֶחְלֵט הִזִיקוּ לָאֲנָשִׁים, אוֹ גָרְמוּ לַמַּמְצִיאִים שֶׁלָּהֶם לְהַרְוִיחַ הַרְבֵּה כֶּסֶף.
אֶחָד שֶׁכָּזֶה הוּא הַמִּקְרֶה שֶׁל שְׁתֵי יְלָדוֹת אַנְגְלִיוֹת, שֶׁסִפְּרוּ שֶׁהֵן פָּגְשׁוּ פֵיוֹת. הָיָה זֶה בִּשְׁנַת 1917 אֲנָשִׁים בָעוֹלָם הִתְרַגְשׁוּ מְאֹד. סוֹף, סוֹף יֵשׁ אֱמֶת בְּסִפּוּרֵי הַיַּלְדוּת שֶׁלָּהֶם וְהַפֵּיוֹת הַטּוֹבוֹת אָכֵן קַיָמוֹת, כָּךְ חָשְׁבוּ. הַיְּלָדוֹת הָאַנְגְלִיוֹת צִלְמוּ תְּמוּנוֹת בָּהֵן הֵן נִרְאוֹת עִם הַפֵּיוֹת. הַתְּמוּנוֹת נִרְאוּ אֲמִתִיוֹת וְהַפֵּיוֹת נִרְאוּ בְּבֵרוּר. הֵן פָּשׁוּט לָקְחוּ בְּרִיסְטוֹלִים גְּדוֹלִים וְצִיְרוּ עֲלֵיהֶם פֵיוֹת וְהַכֹּל נִרְאָה אֲמִתִי בְּיוֹתֵר.
הַהִתְרַגְשׁוּת תָפְסָה סוֹפְרִים וְעִתוֹנָאִים חֲשׁוּבִים. 
לְאַחַר כְּשִׁשִׁים שָׁנָה, הוֹדוּ אֵלְסִי רָיְיט וּפְרָנְסִיס גְּרִיפִית, שֶׁהַכֹּל הָיָה שֶׁקֶר. "רָצִינוּ לִמְתוֹחַ אֶת הַמִּשְׁפָּחָה וְהַחֲבֵרִים", סִפְּרוּ הַשְּׁתַיִם. "לֹא תֵּאַרְנוּ לְעַצְמֵנוּ שֶׁהָעִנְיָן יְגִיעַ לִמְמַדִּים כָּאֵלֶּה. לֹא יָדַעְנוּ אֵיךְ לְהַפְּסִיק אֶת זֶה".
בְּעוֹלַם הַקּוֹלְנוֹעַ אָהֲבוּ אֶת הָרַעְיוֹן שֶׁל הַפֵּיוֹת וְאַף עָשׂוּ סֶרֶט עַל שְׁתֵי הַיְּלָדוֹת הָאַנְגְלִיוֹת. יוֹצְרֵי הַסֶּרֶט הִתְעַלְמוּ מֵהָעֻבְדָה שֶׁהַכֹּל הָיָה שֶׁקֶר אֶחָד גָדוֹל.
אֶחָד הַמִּקְרִים הַבּוֹלְטִים הוּא הַסִּפּוּר עַל הַצַּלַחַת הַמְּעוֹפֶפֶת באַרְצוֹת הַבְּרִית. בָּלוֹן אֲוִיר רֵיק נָפַל בְּשָׂדֶה חַקְלַאי. הָיָה זֶה בִּשְׁנַת 1947. זֶה הָיָה דָבָר כִּמְעַט רָגִיל בְּאוֹתָן הַשָּׁנִים. אֶלָא שֶׁבְּצוּרָה לֹא מוּבֶנֶת, הִשְׁתַּנָה הַסִּפּוּר. עִתוֹנִים דִוְחוּ עַל צַלַחַת מְעוֹפֶפֶת שֶׁנָּפְלָה מֵהֶחָלָל הַחִיצוֹן וְהָיוּ בָּה חַיזָרִים. לְטַעֲנַת מַמְצִיאֵי הַסִּפּוּרִים, הַחַיזָרִים נִתְפְּסוּ עַל יְדֵי הַצָּבָא הָאָמֶרִיקָנִי וְהֵם מֻחְזָקִים בְּכֶלֶא מְיֻחָד. אֶחָד מֵהֶם שֶׁנִּפְצַע, אַף עָבַר נִתּוּחַ שֶׁצֻלַם בְּמַצְלֵמוֹת טֶלֶוִיזְיָה. כַּמוּבָן שֶׁהַכֹּל הָיָה שֶׁקֶר. מֵהַצִּלוּמִים אֶפְשָׁר לִרְאוֹת בְּבֵרוּר שֶׁמְּדֻבָּר בְּבֻבַּת גוּמִי. אוּלַם הָאָמֶרִיקָאִים לֹא מוּכָנִים לְהַאֲמִין לַסִפּוּר. לְדַעֲתָם, הַמֶּמְשָׁלָה וְהַנָּשִׂיא הֶחְלִיטוּ לֹא לְפַרְסֵם אֶת הַדָּבָר כְּדֵי לֹא לִיצוֹר בֶּהָלָה בְּקֶרֶב הַצִּבּוּר.
גַם לְאַחַר שֶׁהַתַּרְמִית הִתְגַלְתָה וְכַיוֹם בָּרוּר שֶׁמְּדֻבָּר בְּשֶׁקֶר, רַבִּים מֵאֶזְרְחֵי אַרְצוֹת הַבְּרִית עֲדַיִן מַאֲמִינִים בַּסִפּוּר. סֶקֶר שֶׁנֶּעֱרַךְ לִפְנֵי 12 שָׁנָה מְגַלָה כִּי 31% מַאֲמִינִים לְסִפּוּרֵי הַצַּלַחַת הַמְּעוֹפֶפֶת.
הַסִּפּוּר אַף מָשַׁךְ אֶת כּוֹתְבֵי הַסְּרָטִים שֶׁל הוֹלִיווּד, שֶׁכָּתְבוּ סְרָטִים רַבִּים עַל הַנּוֹשֵׂא, בינהם "הַיּוֹם השלישי" ו"גְבָרִים בְּשָׁחֹר". גַם סִדְרַת הַטֶּלֶוִיזְיָה הַמְּפֻרְסֶמֶת "תִּיקִים בָּאֲפֵלָה" הֻשְׁפְּעָה מִסִּפּוּרֵי הַחַיזָרִים.
הַאִם, בָּעִדָן שֶׁל הַיּוֹם, יֵשׁ סִכּוּי שֶׁנִּפֹּל בַּפַּח עִם סִפּוּרִים מוּזָרִים וְלֹא אֲמִתִיִים? יֵשׁ לְהַאֲמִין שֶׁכֵּן. הַצִּבּוּר רוֹצֶה לִשְׁמוֹעַ סִפּוּרִים מוּזָרִים וּמְעַנְיְנִים וְלִפְעָמִים זֶה לֹא מְעַנְיֵן אוֹתוֹ, שֶׁהֵם לֹא אֲמִתִיִים. מְעַנְיֵן לִרְאוֹת, אִם כֵּן, מַה תִהְיֶה הַמְּתִיחָה הַבָּאָה.
