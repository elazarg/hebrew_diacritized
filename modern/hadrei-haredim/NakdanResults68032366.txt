עַמּוּד הַבַּיִת 
בְּרִיאוּת
 חַדְשׁוֹת בְּרִיאוּת
הִתְנַזְּרוּת מְסוּכֶּנֶת: צִמְחוֹנִיִּים סוֹבְלִים יוֹתֵר מַשְׁבְּרִים
מֶחְקָר שֶׁנֶּעֱרַךְ בְּמֶשֶׁךְ שָׁנִים קוֹבֵעַ: טִבְעוֹנִיִּים וְצִמְחוֹנִיִּים סוֹבְלִים יוֹתֵר מִשְּׁבָרִים | עוֹרְכֵי מֶחְקָר עָקְבוּ מֵאָז 1993 וְעַד 2016 אַחֲרֵי אַרְבַּע קְבוּצוֹת שׁוֹנוֹת וְהִגִּיעוּ לַמַּסְקָנוֹת הַלָּלוּ


יוֹסֵף קְלַייְן, ז' כִּסְלֵו תשפ"א 23/11/2020 20:39


אֵין בָּשָׂר, אֵין עֲצָמוֹת
הַצִּמְחוֹנִיִּים לֹא יֹאהֲבוּ לִקְרוֹא אֶת הַשּׁוּרוֹת הַבָּאוֹת, הַטִּבְעוֹנִים יוּכּוּ שׁוֹק עַל יָרֵךְ: מֶחְקָר שֶׁנֶּעֱרַךְ בְּמֶשֶׁךְ 23 שָׁנִים קוֹבֵעַ כִּי מִי שֶׁמְּאַמֵּץ אוֹרַח חַיִּים טִבְעוֹנִי אוֹ צִמְחוֹנִי מִסְתַּכֵּן בִּשְׁבָרִים בְּיַחַס לְמִי שֶׁלֹּא מִתְנַזֵּר מִמְּזוֹנוֹת כָּאֵלּוּ אוֹ אֲחֵרִים.

מַאֲמָר שֶׁפּוּרְסַם בִּכְתַב עֵת רְפוּאִי נֶחֱשָׁב מְדַוֵּוחַ עַל מֶחְקָר שֶׁכָּלַל מַעֲקָב אֲרוֹךְ טוַּוח, שֶׁהֵחֵל בִּשְׁנַת 1993. בַּמֶּחְקָר עָקְבוּ מֵאָז וְעַד 2016 אַחֲרֵי אַרְבַּע קְבוּצוֹת: כְּ-30,000 אוֹכְלֵי בָּשָׂר, 8,000 אוֹכְלֵי דָּגִים, 15,000 צִמְחוֹנִים וְ-2,000 טִבְעוֹנִים.

בַּמֶּחְקָר עָקְבוּ הַחוֹקְרִים אַחֲרֵי דִּיוּוּחִים שֶׁל הַמִּשְׁתַּתְּפִים עַל שְׁבָרִים בְּאוֹפֶן כְּלָלִי, וּבְאוֹפֶן מְקוֹמִי (יָרֵךְ, רֶגֶל, צֶלַע, יָד, פֶּרֶק יָד, וְעוֹד וָעוֹד).

לְאַחַר נִטְרוּל מִשְׁתַּנֵּי רֶקַע נִמְצָא שֶׁלְּטִבְעוֹנִיָם סִיכּוּי יוֹתֵר מִכָּפוּל לִסְבּוֹל מִשְּׁבָרִים (פִּי 2.3), וּלְצִמְחוֹנִים וּלְאוֹכְלֵי דָּגִים בִּלְבַד, מְקַדֵּם הַסִּיכּוּן הָיָה 1.25.

מֵהַמֶּחְקָר עוֹלֶה כִּי הַגּוּף דּוֹרֵשׁ אֶת הָעֲרָכִים הַתְּזוּנָתִיִּים הַמִּתְקַבְּלִים מִמַּאֲכַל בָּשָׂר, עוֹפוֹת וְדָגִים וְחֶסְרוֹנָם עָלוּל לְהַשְׁפִּיעַ  לְרָעָה עַל חוֹסֶן הָעֲצָמוֹת, כָּךְ שֶׁנְּפִילוֹת רַבּוֹת יוֹתֵר יִסְתַּיְּימוּ בְּשֶׁבֶר אוֹתוֹ יֵשׁ לְאַחוֹת.

כְּפִי שֶׁנִּיתָּן לִרְאוֹת, טִבְעוֹנִיִּים חֲשׂוּפִים לְסַכָּנָה גְּדוֹלָה יוֹתֵר מֵאֲשֶׁר צִמְחוֹנִיִּים. הַסִּיבָּה לְכָךְ הִיא כְּכָל הַנִּרְאֶה הִתְנַזְּרוּת שֶׁל הַטִּבְעוֹנִיִּים מִכָּל מוּצָר מִן הַחַי, כּוֹלֵל גְּבִינָה, חָלָב, בֵּיצִים וַאֲפִילּוּ דְּבַשׁ. הַצִּמְחוֹנִיִּים לְעוּמָּתָם מִתְנַזְּרִים מִבָּשָׂר, עוֹפוֹת וְדָגִים, אַךְ נֶהֱנִים מִגְּבִינָה וּבֵיצִים וְעוֹד.
