הַסִּיפּוּר הַמּוּזָר שֶׁלָּנוּ לְבוֹקֶר זֶה, שֶׁדּוּוַּח בְּסֵפֶר הַזִּיכְרוֹנוֹת מְרוּבֵּה הַכְּרָכִים שֶׁל סְטָנִיסְלָב צַ'רְנוֹבְסְקִי, מִתְרַחֵשׁ בְּבִנְיַין פִּינָּתִי בְּוַורְשָׁה שֶׁל 1874, בְּמִפְגַּשׁ הָרְחוֹבוֹת וְיֶילְקָה, שְׁלִיסְקָה וּסְיֶינָה. לְמַרְבֵּה הַצַּעַר, אֵין תְּמוּנָה שֶׁל הַבִּנְיָין וַאֲפִילּוּ הַתָּכְנִית שֶׁבַּתְּמוּנָה גֵּנֵרִית, בְּעִיקָּר כִּי הָרוֹב הוּשְׁמַד במלה"ע הַשְּׁנִיָּיה (שֶׁבְּמַהֲלָכָהּ הָיוּ הָרְחוֹבוֹת הָאֵלּוּ בְּתוֹךְ הַגֵּטוֹ). טְרִיגֵר אַלִּימוּת מִינִית, נִרְאֶה לִי.
כְּמוֹ רַבִּים מֵרְחוֹבוֹת וַרְשָׁה, גַּם רְחוֹבוֹת אֵלּוּ הָיוּ בְּנוּיִים מִבָּתֵּי מְגוּרִים שֶׁהָפְכוּ, עִם הַזְּמַן, לִצְפוּפִים וּמוּזְנָחִים בְּמִידָּה. אַחֲרֵי הַמֶּרֶד שֶׁל 1863, הַשִּׁלְטוֹנוֹת הָרוּסִיִּים לֹא הִתְעַנְיְינוּ בִּמְיוּחָד בִּגְדִילָתָהּ שֶׁל הָעִיר וּבְתִכְנוּנָהּ, כָּךְ שֶׁבִּנְיָינִים אֵלּוּ עַל הַדֶּשֶׁא הַנָּסוֹג בָּהֶם נִדְמוּ לַאֲנָשִׁים כְּמוֹ יֶזִ'י וּוּלְדּוֹרְף, אַתֶּם לֹא אֲמוּרִים לְהַכִּיר אוֹתוֹ, כְּכַוָּורוֹת הַבְּנוּיוֹת מִ"תָּאִים אֲפוֹרִים, עֲמוּקִּים וְרֵיקִים מִדְּבַשׁ". 
בְּאֶמְצַע הַמִּבְנֶה הַמְּרוּבָּע הָיְיתָה חָצֵר קְטַנָּה שֶׁמּוּפְרֶדֶת מִן הָרְחוֹב בַּשַּׁעַר, שֶׁעָלָיו מוּפְקָד שׁוֹטֵר בַּדַּרְגָּה הַנְּמוּכָה בְּיוֹתֵר בְּכוֹחַ הַמִּשְׁטָרָה שֶׁל הָעִיר, לָרוֹב מְהַגֵּר טָרִי. הוּא הָיָה אָמוּר לְתַחְזֵק אֶת הֶחָצֵר וְאֶת גֶּרֶם הַמַּדְרֵגוֹת וּלְפַקֵּחַ עַל הַנִּכְנָסִים וְהַיּוֹצְאִים מִן הַבּוֹקֶר וְעַד 23:00 בַּלַּיְלָה.
הַבְּנִיָּיה, שֶׁלֹּא תָּמִיד הָיְיתָה אֵיכוּתִית, אִפְשְׁרָה מְצִיצָנוּת. לְכָל הַדִּירוֹת הָיוּ חַלּוֹנוֹת הַמַּשְׁקִיפִים לֶחָצֵר הַפְּנִימִית וְהַקִּירוֹת הָיוּ דַּקִּים מַסְפִּיק כְּדֵי שֶׁנִּיתָּן הָיָה לְהַאֲזִין לַמִּתְרַחֵשׁ בַּדִּירוֹת שֶׁמִּסָּבִיב. בְּמַקְבִּיל, הַבִּנְיָינִים חִיבְּרוּ בֵּין מַעֲמָדוֹת. הַקּוֹמָה הָרִאשׁוֹנָה וְהַשְּׁנִיָּיה נִשְׁמְרָה לַעֲשִׁירִים וּכְכָל שֶׁעָלִיתָ בַּגּוֹבַהּ כָּךְ יָרַד מַצָּבָם הַחֶבְרָתִי וְהַכַּלְכָּלִי שֶׁל הַתּוֹשָׁבִים, לְפָחוֹת עַד שֶׁנִּבְנוּ מַעֲלִיּוֹת. בַּקּוֹמוֹת הָעֶלְיוֹנוֹת בְּיוֹתֵר הָיוּ הַדִּירוֹת הַקְּטַנּוֹת בְּיוֹתֵר, לָרוֹב לְחַסְרֵי מִשְׁפָּחוֹת, בְּעוֹד שֶׁהָעֲנִיִּים מִכּוֹל שָׁכְנוּ דַּוְוקָא בַּמַּרְתֵּף.
הַחַיִּים בַּבָּתִּים הָאֵלּוּ לֹא הָיוּ מַלְהִיבִים בְּהֶכְרֵחַ. אֶחָד הָעִיתּוֹנִים, לְמָשָׁל, שָׁאַל בִּשְׁנַת 1868 "מָה לַעֲשׂוֹת אִם הַדַּיָּיר לְיַדְכֶם מִתְאַמֵּן עַל נְגִינָה בַּחֲצוֹצְרָה כָּל הַיּוֹם?". "מִשְׁפָּטִית", הוּא אָמַר, "לֹא נִיתָּן לְהַאֲשִׁים אוֹתוֹ בִּכְלוּם". הוּא אַף צִיטֵּט אֶת הַפִּתְגָּם הַפּוֹלָנִי הֶחָבִיב, "Wolnoć Tomku w swoim domk" שֶׁלָּקוּחַ מֵאַגָּדַת עַם וְאוֹמֶרֶת בְּגָדוֹל כָּל אֶחָד חוֹפְשִׁי בְּבֵיתוֹ. "אֲבָל מָה יַעֲשֶׂה הַשָּׁכֵן הָאוּמְלָל?"
אֲבָל לִגְבֶרֶת ב', "שֶׁשָּׂכְרָה דִּירָה נֶחְמָדָה עִם אֲחוֹתָהּ מָשְׁיָה וְעִם בִּתָּהּ הַקְּטַנָּה קוֹצִי", הָיְיתָה בְּעָיָה חֲמוּרָה מִכָּךְ. "לְצַעֲרָהּ, הַדִּירָה הַנֶּגְדִּית הָיְיתָה לָרוֹב זִירַת הִתְרַחֲשׁוּתָן שֶׁל אוֹרְגִּיּוֹת שֶׁלֹּא נִשְׁמַע כְּמוֹתָן".
הִתְגּוֹרֵר בָּהּ הַבְּעָלִים שֶׁל הַבִּנְיָין בְּעַצְמוֹ, "אִישׁ עָשִׁיר, פְרִיק שֶׁהָיְיתָה לוֹ אִישָּׁה צְעִירָה וְיָפָה. בַּלַּיְלָה, כְּשֶׁמֶּזֶג הָאֲוִויר חַם, נִיתָּן הָיָה לִרְאוֹת מַרְאוֹת יוֹצְאֵי דּוֹפֶן מִבַּעַד לְחַלּוֹנוֹתָיו הַפְּתוּחִים שֶׁל דִּירָתוֹ שֶׁל בַּעַל הַבִּנְיָין. לָרוֹב אַחֲרֵי הַצָּהֳרַיִים, כִּמְעַט בְּכָל יוֹם, נִיתָּן הָיָה לִשְׁמוֹעַ שָׁם צְעָקוֹת מוּזָרוֹת. מַר' ר' רָדַף אַחֲרֵי אִשְׁתּוֹ הַצְּעִירָה וְהַחֲצִי-עֵירוּמָה בְּרַחֲבֵי הַדִּירָה עִם שׁוֹט עֵץ לִבְנֶה, כְּשֶׁהִיא רַק בַּחוּלְצָה דַּקָּה עִם קַו מָתְנַיִים נָמוּךְ בִּמְיוּחָד, חֲתוּכָה מִיָּמִין וּמִשְּׂמֹאל. אַחֲרֵי שֶׁתָּפַס אוֹתָהּ, הוּא הָיָה תּוֹפֵס אֶת צַמּוֹתֶיהָ הָאֲרוּכּוֹת כְּמוֹ מוֹשְׁכוֹת וְרוֹדֵף אַחֲרֵי הָאִישָּׁה הַמִּסְכֵּנָה אֲפִילּוּ עוֹד יוֹתֵר. אַחֲרֵי הַמִּרְדָּף וְהַבֶּכִי שָׂרְרָה דְּמָמָה.
כִּרְכָּרָה פְּתוּחָה נִכְנְסָה בַּשַּׁעַר הָרָאשִׁי. הגב' ר' הִתְלַבְּשָׁה לִיצִיאָה בְּמֶשִׁי וּמַלְמָלָה, נִכְנְסָה פְּנִימָה עִם בַּעֲלָהּ שֶׁגָּעַר בָּהּ לַעֲטוֹת הַבָּעַת פָּנִים נְעִימָה וּלְחַיֵּיךְ, אַחֶרֶת תִּהְיֶה הַצְלָפָה חֲדָשָׁה לְאַחַר שֶׁתָּשׁוּב מִבִּיקּוּר בַּשְּׂדֵרָה הַבַּלְוָודָרִית הָאָפְנָתִית. דַּיָּירֵי בִּנְיַין הַדִּירוֹת הָעֲנָק הַזֶּה, דֶּרֶךְ כָּל הַחַלּוֹנוֹת מִן הֶחָצֵר כְּאִילּוּ הָיוּ אֵלּוּ תָּאֵי תֵּיאַטְרוֹן, תָּמִיד צָפוּ בְּאוֹרְגִּיּוֹת הַסָּאדִיסְטִיּוֹת הָאֵלּוּ, הֵעִירוּ אֶת הֶעָרוֹתֵיהֶם וְסִיפְּרוּ פְּרָטִים מְצַמְרְרִים שׁוֹנִים שֶׁרָאוּ בֶּעָבָר. הַשּׁוֹמֵר בַּכְּנִיסָה הָרָאשִׁית צוּוָה לְהַחֲזִיק שׁוֹט מוּכָן לְפִי בַּקָּשָׁה, שֶׁאוֹתוֹ הוּא לָרוֹב הֵכִין בֶּחָצֵר וְהִשְׁרָה בִּדְלִי שֶׁל מַיִם.
פַּעַם אַחַת, בְּמַהֲלַךְ סוּפַת רְעָמִים, בָּרָק כָּבֵד שָׁבַר כִּמְעַט אֶת כָּל הַחַלּוֹנוֹת בַּבִּנְיָין הַזֶּה. בַּעַל הַבִּנְיָין הָיָה צָרִיךְ לְהַחֲלִיף אֶת כּוּלָּם עַל חֶשְׁבּוֹנוֹ. כְּשֶׁזֶּה קָרָה, הַנָּשִׁים שֶׁנָּטְרוּ לוֹ טִינָה נָזְפוּ בּוֹ וְטָעֲנוּ שֶׁהָיָה זֶה עוֹנֶשׁ מִשָּׁמַיִם עַל הִתְעַלְּלוּת אַכְזָרִית בְּאִישְׁתּוֹ הַתַּמָּה."
