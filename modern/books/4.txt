מֵאָז הֱיוֹתוֹ גֶּבֶר צָעִיר, גִּילָּה תָּמִיד עִנְיָין מְיוּחָד בִּבְכוֹר אַחְיָינָיו - בָּנָה-יְחִידָה שֶׁל אֲחוֹתוֹ הַיְּחִידָה. חַיִּים אוּמְלָלִים הָיוּ לְדוֹדָתִי, שֶׁבְּמַהֲלָכָם עָבְרָה כַּמָּה תְּקוּפוֹת נִישּׂוּאִים קָשׁוֹת, וּבְנָהּ שִׁילֵּם אֶת הַמְּחִיר: שׁוּלַּח לִפְנִימִיּוֹת צְבָאִיּוֹת, מֵעוֹלָם לֹא זָכָה לַבַּיִת שֶׁל מַמָּשׁ. מְנִיעָיו שֶׁל אָבִי לֹא הָיוּ, לְדַעְתִּי, אֶלָּא טוּב-לֵב וּתְחוּשַׁת אַחְרָיוּת, וְהוּא פָּרַשׁ חָסוּתוֹ עַל הַנַּעַר. הוּא תָּמַךְ בּוֹ בְּעִידוּד בִּלְתִּי פּוֹסֵק, לִימֵּד אוֹתוֹ אֵיךְ לְהִסְתַּדֵּר בָּעוֹלָם. אַחַר-כָּךְ סִייֵּעַ לוֹ בַּעֲסָקִים, וּכְשֶׁהִתְעוֹרְרוּ בְּעָיוֹת תָּמִיד הָיָה מוּכָן לְהַאֲזִין וּלְיַיעֵץ. גַּם לְאַחַר שֶׁדּוֹדָנִי הִתְחַתֵּן וְהֵקִים מִשְׁפָּחָה מִשֶּׁלּוֹ, הוֹסִיף אָבִי לְגַלּוֹת עִנְיָין פָּעִיל, שִׁיכֵּן אוֹתָם בְּבֵיתוֹ בְּשָׁלָב מְסוּיָּים בְּמֶשֶׁךְ יוֹתֵר מִשָּׁנָה, הִתְמִיד בַּאֲדִיקוּת לַעֲנִיק מַתָּנוֹת יוֹם-הוּלֶּדֶת לְאַרְבַּעַת יְלָדָיו שֶׁל הָאַחְיָין, וּתְכוּפוֹת הִתְאָרֵחַ אֶצְלָם לַאֲרוּחַת-עֶרֶב.
דּוֹדָנִי זֶה זוּעְזַע מִמּוֹתוֹ שֶׁל אָבִי יוֹתֵר מִכָּל שְׁאָר קְרוֹבִי. בַּהִתְכַּנְּסוּת הַמִּשְׁפַּחְתִּית לְאַחַר הַלְּוָויָה נִיגַּשׁ אֵלַי שָׁלוֹשׁ אוֹ אַרְבַּע פְּעָמִים וְאָמַר, "נִתְקַלְתִּי בּוֹ בְּמִיקְרֶה רַק לִפְנֵי כַּמָּה יָמִים. הָיִינוּ אֲמוּרִים לֶאֱכוֹל יַחַד אֲרוּחַת-עֶרֶב בְּיוֹם שִׁשִּׁי."
הַמִּלִּים שֶׁבָּהֶן הִשְׁתַּמֵּשׁ הָיוּ זֵהוֹת בְּדִיּוּק בְּכָל פַּעַם, כְּאִילּוּ אֵינוֹ יוֹדֵעַ עוֹד מָה הוּא שָׂח.
הִרְגַּשְׁתִּי שֶׁאֵיכְשֶׁהוּ הִתְחַלַּפְנוּ בְּתַפְקִידִים, שֶׁהוּא הַבֵּן הָאָבֵל וַאֲנִי הָאַחְיָין הַמְּנַחֵם. רָצִיתִי לְחַבֵּק אֶת כְּתֵפוֹ וְלוֹמַר לוֹ אֵיזֶה אִישׁ טוֹב הָיָה אָבִיו. אַחֲרֵי כִּכְלוֹת הַכֹּל, הוּא הָיָה הַבֵּן הָאֲמִיתִּי, הַבֵּן שֶׁלֹּא יָכוֹלְתִּי לְהָבִיא אֶת עַצְמִי לִהְיוֹת.
בַּשְּׁבוּעַיִים הָאַחֲרוֹנִים הִידְהֲדוּ בְּרָאשֵׁי שׁוּרוֹת אֵלּוּ מִשֶּׁל מוֹרִיס בּלָאנְשׁוֹ: "דָּבָר אֶחָד יֵשׁ לְהָבִין: לֹא אָמַרְתִּי שׁוּם דָּבָר יוֹצֵא-דּוֹפֶן אוֹ אֲפִילּוּ מַפְתִּיעַ. הַיּוֹצֵא-דּוֹפֶן מַתְחִיל בָּרֶגַע שֶׁאֲנִי מִשְׁתַּתֵּק. אֲבָל עַל כָּךְ שׁוּב אֵינֶנִּי יָכוֹל לְדַבֵּר."
לִפְתּוֹחַ בְּמָוֶות. לְפַלֵּס מִשָּׁם דֶּרֶךְ חֲזָרָה אֶל הַחַיִּים, וְאָז, לְבַסּוֹף, לַחֲזוֹר אֶל הַמָּוֶות.
וְאוּלַי: חוֹסֶר-הַשַּׁחַר שֶׁבַּנִּיסָּיוֹן לוֹמַר מַשֶּׁהוּ עַל מִישֶׁהוּ.
