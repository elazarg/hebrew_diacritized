חִינּוּךְ
גַּנֵּי הַחִינּוּךְ הַמְּיוּחָד מַמְשִׁיכִים כָּרָגִיל? הַאִם זֹאת הַחְלָטָה סוֹפִית?
הַמֶּמְשָׁלָה הֶחֱרִיגָה בִּתְקוּפַת הַסֶּגֶר אֶת הַפְּעִילוּת הַחִינּוּכִית שֶׁל תַּלְמִידֵי הַחִינּוּךְ הַמְּיוּחָד וְאֶת הַפְּעִילוּת שֶׁל יְלָדִים וְנוֹעַר בְּסִיכּוּן - כָּךְ שֶׁהֵם אֲמוּרִים לִפְעוֹל. עִם זֹאת, הִסְתַּדְּרוּת הַמּוֹרִים מִתְנַגֶּדֶת לְכָךְ וְטוֹעֶנֶת כִּי הַדָּבָר מְסַכֵּן אֶת עוֹבְדֵי הַהוֹרָאָה.  
הַאִם תִּתְקַיֵּים לְמִידָה מֵרָחוֹק?
כֵּן. עִם כְּנִיסַת הַסֶּגֶר, בְּיָמִים שֶׁל לְמִידָה (לֹא כּוֹלֵל יְמֵי חַג) תִּתְקַיֵּים לְמִידָה מֵרָחוֹק בְּכָל שִׁכְבוֹת הַגִּיל. עִם זֹאת, יֵשׁ לְצַיֵּין כִּי הַנְהָגַת הַהוֹרִים מִתְנַגֶּדֶת לְכָךְ נֶחְרָצוֹת וּמְאַיֶּימֶת לֹא לְשַׁתֵּף פְּעוּלָּה כְּדֵי שֶׁיְּמֵי הַלִּימּוּד יוּחְזְרוּ בִּמְלוֹאָם לְאַחַר הַסֶּגֶר.

 
אֵיךְ תֵּירָאֶה הַלְּמִידָה מֵרָחוֹק בַּשְּׁכָבוֹת הַנְּמוּכוֹת?
בְּמִשְׂרַד הַחִינּוּךְ עֲדַיִין לֹא פִּרְסְמוּ אֶת הַמִּתְווֹה הַמָּלֵא.
הַאִם הֵיעָדְרוּת מִשִּׁיעוּר בְּזוּם נֶחְשֶׁבֶת הֵיעָדְרוּת "רְגִילָה"?
כֵּן. מִשְׂרַד הַחִינּוּךְ קָבַע כִּי הַשָּׁנָה הוּא יֶאֱכוֹף אֶת חוֹק חִינּוּךְ חוֹבָה כְּדֵי לְהַבְטִיחַ שֶׁהַתַּלְמִיד לוֹקֵחַ חֵלֶק פָּעִיל בְּתוֹךְ תַּהֲלִיךְ הַלְּמִידָה. בָּתֵּי הַסֵּפֶר יִבְדְּקוּ נוֹכְחוּת וְיִתְיַיחֲסוּ לַנּוֹשֵׂא בְּהֶתְאֵם. בְּמִשְׂרַד הַחִינּוּךְ מְצַיְּינִים כִּי כְּכָל שֶׁלְּתַלְמִיד יֵשׁ סִיבָּה מוּצְדֶּקֶת לְהֵיעָדְרוּת, הֲרֵי שֶׁהוּא יַסְדִּיר אֶת הָעִנְיָין עִם בֵּית הַסֵּפֶר. 
מָה לְגַבֵּי תַּשְׁלוּמֵי הוֹרִים לַצָּהֳרוֹנִים? 
בְּמִשְׂרַד הַחִינּוּךְ עֲדַיִין אֵין תְּשׁוּבוֹת בְּרוּרוֹת. עִם זֹאת, בַּגַּל הָרִאשׁוֹן הוּא הוֹדִיעַ כִּי הַהוֹרִים יְקַבְּלוּ הֶחְזֵר עֲבוּר שֵׁירוּתִים שֶׁלֹּא קִיבְּלוּ.
הַאִם מְעוֹנוֹת הַיּוֹם לְגִילֵי לֵידָה עַד 3 יַעַבְדוּ בִּתְקוּפַת הַסֶּגֶר?
לֹא. הֵם חֵלֶק מִמַּעֲרֶכֶת הַחִינּוּךְ וְלָכֵן יִישָּׁאֲרוּ סְגוּרִים.
תְּפִילּוֹת
מָה עוֹשִׂים עִם תַּשְׁלִיךְ לְאוֹר מִגְבְּלוֹת הַהִתְקַהֲלוּת הַחֲדָשׁוֹת?
הָרַב הָרָאשִׁי דָּוִד לָאוּ הוֹרָה שֶׁלֹּא לְבַצֵּעַ אֶת מִנְהַג הַתַּשְׁלִיךְ בְּסָמוּךְ לִמְקוֹרוֹת מַיִם בְּשֶׁל הַחֲשָׁשׁ מֵהִתְקַהֲלוּת. נִיתָּן לוֹמַר גַּם בְּרִיחוּק מָקוֹם". 
אֲנִי בְּבִידוּד, מָה עוֹשִׂים עִם תְּקִיעוֹת שׁוֹפָר?
הָרַב יִצְחָק נֵרִיָּה, רֹאשׁ יְשִׁיבַת תּוֹרָה בְּצִיּוֹן, מַסְבִּיר כִּי אָדָם אֵינוֹ מְסוּגָּל לָצֵאת יְדֵי חוֹבַת הַמִּצְווֹה בִּתְקִיעַת שׁוֹפָר עַצְמָאִית. "תַּקָּנָה זוֹ הִיא רַק בַּצִּיבּוּר. אֶת תְּקִיעוֹת הַשּׁוֹפָר תִּשְׁמְעוּ כַּאֲשֶׁר יָבוֹאוּ לִתְקוֹעַ לָכֶם מֵהַחַלּוֹן אוֹ חֲדַר הַמַּדְרֵגוֹת". 
חַם בַּחוּץ וּבִלְתִּי אֶפְשָׁרִי לְהִתְפַּלֵּל בְּמֶשֶׁךְ שָׁעוֹת אֲרוּכּוֹת, אֶפְשָׁר לְקַצֵּר אֶת הַתְּפִילָּה?
הַתְּשׁוּבָה הִיא חַד־מַשְׁמָעִית כֵּן. הָרַב הָרָאשִׁי יִצְחָק יוֹסֵף מַדְגִּישׁ כִּי "יֵשׁ לְקַצֵּר בְּמַנְגִּינוֹת וּבְפִיּוּטִים וְלֹא לְהַאֲרִיךְ בָּהֶם". כְּמוֹ כֵן, יֵשׁ לְקַצֵּר עַד כַּמָּה שֶׁאֶפְשָׁר בַּדְּרָשָׁה שֶׁל הָרַב. הָרַב יוּבָל שַׁרְלוֹ מְבַקֵּשׁ לִזְכּוֹר גַּם אֶת בַּעֲלֵי הַמּוּגְבָּלוּיוֹת וְהַצְּרָכִים הַמְּיוּחָדִים וּלְהַתְאִים לָהֶם אֶת הַתְּפִילָּה בְּמִנְיַין הֶחָצֵר. "הַפִּתְרוֹנוֹת הֶכְרֵחִיִּים: תְּשׂוּמֶת לֵב, הַצְלָלָה, קִיצּוּר הַתְּפִילָּה בִּתְחוּם הַפִּיּוּטִים וּמְקוֹמוֹת נוֹסָפִים שֶׁבְּשָׁנָה כְּזוֹ יֵשׁ לְדַלֵּג עֲלֵיהֶם".
כְּשֶׁמְּבַצְּעִים אֶת הַתְּקִיעוֹת בְּשׁוֹפָר יֵשׁ חֲשָׁשׁ מֵהַתָּזַת רוֹק. יֵשׁ פִּתְרוֹן?

 
הָרַב יוֹסֵף צְבִי רִימּוֹן, רַב בֵּית הַמִּדְרָשׁ בַּמֶּרְכָּז הָאָקָדֵמִי לֵב, פּוֹסֵק כִּי יֵשׁ לְכַסּוֹת אֶת הַשּׁוֹפָר בְּמַסֵּיכָה (בַּצַּד הַגָּדוֹל) כְּדֵי לִמְנוֹעַ הַדְבָּקָה. "יֵשׁ לִשְׁמוֹר עַל מֶרְחָק רַב יוֹתֵר בֵּין הַתּוֹקֵעַ לַשּׁוֹמְעִים. אִם חוֹשְׁשִׁים מֵהָרוֹק הַיּוֹצֵא מֵהַשּׁוֹפֵר מוּתָּר לַהֲפוֹךְ אוֹתוֹ כְּלַפֵּי מַטָּה".
תַּחְבּוּרָה
הַאִם הַתַּחְבּוּרָה הַצִּיבּוּרִית תִּפְעַל בִּזְמַן הַסֶּגֶר? 
כֵּן, אַךְ בְּצוּרָה מְצוּמְצֶמֶת. עַל פִּי הַחְלָטַת שָׂרַת הַתַּחְבּוּרָה בְּתוֹם יְשִׁיבַת הַעֲרָכַת מַצָּב, הַתַּחְבּוּרָה תִּפְעַל בִּתְפוּסָה שֶׁל עַד 50 אֲחוּזִים וְלֹא בְּמֶשֶׁךְ כָּל הַיּוֹם.
מָה תִּהְיֶינָה שְׁעוֹת הַפְּעִילוּת? 
אוֹטוֹבּוּסִים בֵּין־עִירוֹנִיִּים יִפְעֲלוּ בְּיָמִים רִאשׁוֹן עַד חֲמִישִׁי בִּלְבַד. הַקַּוִּוים לֹא יִפְעֲלוּ בִּימֵי שִׁישִּׁי, בְּמוֹצָאֵי שַׁבָּת וּבְעַרְבֵי וּמוֹצְאֵי הַחַגִּים. אוֹטוֹבּוּסִים עִירוֹנִיִּים יִפְעֲלוּ בְּיָמִים רִאשׁוֹן עַד חֲמִישִׁי בֵּין הַשָּׁעוֹת 05:00 לְ־22:00. 
אִילּוּ קַוִּוים יַעַבְדוּ? 
אֵלֶּה הַמַּגִּיעִים לְמוֹקְדֵי הַתַּעֲסוּקָה וְלַצְּרָכִים חִיּוּנִיִּים בִּלְבַד. קַוִּוים אֲחֵרִים, וּבָהֶם קַוֵּוי לַיְלָה, קַוֵּוי תַּלְמִידִים, קַוֵּוי אָקָדֶמְיָה וְנוֹסָפִים יְבוּטְּלוּ.
מָה לְגַבֵּי הָרַכֶּבֶת? 
הָרַכֶּבֶת תַּמְשִׁיךְ לִפְעוֹל בְּהֶתְאֵם לְמוֹדֵל הַוּוֹאוּצֵ'ר וְהַזְמָנַת הַנְּסִיעָה בְּיָמִים רִאשׁוֹן עַד חֲמִישִׁי בֵּין הַשָּׁעוֹת 05:00 לְ־22:00. הָרַכֶּבֶת לֹא תִּפְעַל בִּימֵי שִׁישִּׁי, בְּמוֹצָאֵי שַׁבָּת, בְּעַרְבֵי חַג וּבְמוֹצָאֵי חַג.
מָה לְגַבֵּי שֵׁירוּתֵי מוֹנִיּוֹת וּמוֹנִיּוֹת שֵׁירוּת? 
שֵׁירוּתִי "בָּאבֶּל" וּ"טִיקְטָק" יִפְעֲלוּ בֶּעָרִים הָרֵלֵוַונְטִיּוֹת בְּיָמִים רִאשׁוֹן עַד חֲמִישִׁי בַּשָּׁעוֹת הַסְּדִירוֹת. מוֹנִיּוֹת שֵׁירוּת עִירוֹנִיּוֹת יִפְעֲלוּ לְפִי מַתְוֶוה הָאוֹטוֹבּוּסִים הָעִירוֹנִיִּים, וּמוֹנִיּוֹת שֵׁירוּת בֵּין־עִירוֹנִיּוֹת לְפִי מַתְוֶוה הָאוֹטוֹבּוּסִים הַבֵּין־עִירוֹנִיִּים.
הַאִם יִהְיֶה אֶפְשָׁר לִנְסוֹעַ לְאֵילַת? 
שֵׁירוּת הָאוֹטוֹבּוּסִים לְאֵילַת יוּפְעַל בְּאוֹפֶן מְצוּמְצָם בְּהַזְמָנַת מָקוֹם מֵרֹאשׁ עַל מְנַת לְשָׁרֵת אֶת תּוֹשָׁבֵי אֵילַת לִנְסִיעוֹת חִיּוּנִיּוֹת.

 
הַאִם תִּהְיֶינָה מִגְבָּלוֹת בַּנְּסִיעָה? 
כְּמוֹ שֶׁהָיָה עַד עַכְשָׁיו. נוֹסָף עַל כָּךְ, לֹא תִּתְאַפְשֵׁר יְשִׁיבָה בְּאוֹטוֹבּוּסִים מֵאֲחוֹרֵי הַנַּהָג, לְמַעֵט בְּאוֹטוֹבּוּסִים שֶׁכְּבָר הוּתְקְנוּ בָּהֶם מְחִיצוֹת מָגֵן לְנֶהָגִים. 
הַאִם תִּהְיֶה אֲכִיפָה? 
אֲכִיפָה תִּתְבַּצֵּעַ עַל יְדֵי מִשְׂרַד הַתַּחְבּוּרָה לִמְנִיעַת נִיצּוּל שֶׁל מַעֲרַךְ הַתַּחְבּוּרָה הַצִּיבּוּרִית לִצְרָכִים שֶׁאֵינָם חִיּוּנִיִּים וְתוֹךְ הֲפָרַת הַסֶּגֶר.
הֵיכָן נִיתָּן יִהְיֶה לְקַבֵּל מֵידָע? 
כָּל הַמֵּידָע זָמִין בְּמוֹקֵד הַטֵּלֵפוֹנִי 8787*, בָּאַפְּלִיקַצְיוֹת הַיִּיעוּדִיּוֹת, בַּאֲתָרֵי הַמַּפְעִילִים וּבַאֲתַר הָרַכֶּבֶת.
בַּהֲכָנַת הַיְּדִיעָה הִשְׁתַּתְּפוּ: מֵיטַל יַסְעוּר בֵּית-אוֹר, נוֹעַם (דְּבוּל) דְּבִיר, חָנָן גְּרִינְווֹד וּשְׁלוֹמִי דִּיאֵז
