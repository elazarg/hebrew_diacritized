3 תּוֹפָעוֹת חֲדָשׁוֹת
גַּם הֲדַס יוֹסֵף, אֵם לְשָׁלוֹשׁ, וּבָהֶן יַלְדָּה שֶׁעוֹלָה לְכִיתָּה ג', בָּחֲרָה לִרְכּוֹשׁ צִיּוּד לְבֵית הַסֵּפֶר בִּמְחִיר שָׁפוּי. "הֶחְלַטְתִּי לֹא לְהִיסָּחֵף לְמוּתָגִים בִּמְחִירֵי עָתֵק, זֶה גַּם חִינּוּךְ בְּסִיסִי בְּעֵינַיי. הֶחְלַטְתִּי לְנַהֵל עִם הַבָּנוֹת שֶׁלִּי שִׁגְרָה רְגִילָה כְּכָל הָאֶפְשָׁר. לָקַחְתִּי אֶת הַבַּת שֶׁלִּי, רוֹנִּי, אִיתִּי לַחֲנוּת וְנָתַתִּי לָהּ לְהִתְרַגֵּשׁ וְלִבְחוֹר לְעַצְמָהּ קַלְמָר, יוֹמָן וָתִיק חֲדָשִׁים, וְכַמּוּבָן אֶת כָּל הַצִּיּוּד הַנִּדְרָשׁ לְבֵית הַסֵּפֶר. בְּמַקְבִּיל רָכַשְׁתִּי מַסֵּיכוֹת צִבְעוֹנִיּוֹת רַב־פַּעֲמִיּוֹת, אַלְכּוֹגֶ'ל מְחוּבָּר לַתִּיק שֶׁנִּיתָּן לְמִילּוּי, יוֹתֵר מַגָּבוֹת מֵהָרָגִיל כְּדֵי שֶׁהָאוֹכֶל לֹא יֵישֵׁב עַל שׁוּלְחַן הַכִּיתָּה וּכְפָפוֹת לַשֵּׁירוּתִים
בֶּנִי גְּרִינָאפֶל, מַנְכַּ"ל רִשְׁתוֹת עִידָּן 2000 וּכְפַר הַשַּׁעֲשׁוּעִים, מַצְבִּיעַ עַל מְגַמּוֹת חֲדָשׁוֹת בַּצְּרִיכָה לִקְרַאת שְׁנַת הַלִּימּוּדִים הַקְּרוֹבָה. "אָנוּ מְזַהִים שָׁלוֹשׁ תּוֹפָעוֹת חֲדָשׁוֹת הָעוֹנָה, לְהַבְדִּיל מִשָּׁנִים קוֹדְמוֹת. הָרִאשׁוֹנָה הִיא הֶאָטָה בְּהִצְטַיְּידוּת. תּוֹפָעָה שְׁנִיָּיה הִיא שֶׁאֲנָשִׁים קוֹנִים מַהֵר וּמְבַלִּים זְמַן קָצָר בַּחֲנוּת, כְּבָר אֵין אֶת הַסִּיּוּר הַמְּקוּבָּל וְהַחֲגִיגִי סְבִיב רְכִישַׁת הַצִּיּוּד לְבֵית הַסֵּפֶר שֶׁבּוֹלֵט בִּמְיוּחָד בַּכִּיתּוֹת הַנְּמוּכוֹת. הַתּוֹפָעָה הַשְּׁלִישִׁית שֶׁאֲנַחְנוּ מְזַהִים הִיא שֶׁלְּצַד הָרְכִישָׁה שֶׁל הַצִּיּוּד לְבֵית הַסֵּפֶר, אֲנָשִׁים רוֹכְשִׁים גַּם מִשְׂחֲקֵי יְצִירָה, מִשְׂחֲקֵי קוּפְסָה וּבְעִיקָּר מִשְׂחָקִים חִינּוּכִיִּים. מְדוּבָּר בְּתוֹפָעָה שֶׁלֹּא הָיְתָה בַּנֶּפַח הַזֶּה בַּתְּקוּפָה הַמַּקְבִּילָה אֶשְׁתָּקַד. כְּמוֹ כֵן, סָבָא וְסָבְתָא כְּבָר לֹא מִצְטָרְפִים לִקְנִיּוֹת עִם הַנְּכָדִים".
גַּם עָמִית שַׁבְּתַאי, מְנַהֶלֶת סַחַר וִיבוּא בִּקְרָבִיץ, מְעִידָה עַל שִׁינּוּיִים. "הַסַּל הַמְּמוּצָּע בַּכִּיתּוֹת הַגְּבוֹהוֹת הִשְׁתַּנָּה כִּי מְשׁוּלָּבִים בּוֹ מוּצָרִים טֶכְנוֹלוֹגִיִּים לִלְמִידָה מֵהַבַּיִת. כְּמוֹ כֵן, לָקוֹחוֹת מַגִּיעִים הַרְבֵּה יוֹתֵר מוּכָנִים וּמְמוּקָּדִים לִרְכִישָׁה לְאַחַר שֶׁעָשׂוּ בְּדִיקוֹת בָּאִינְטֶרְנֵט. קַיָּים אִיזּוּן בֵּין הַמּוּצָרִים הַמְּמוּתָּגִים לְאֵלּוּ שֶׁאֵינָם מְמוּתָּגִים, וְלָקוֹחוֹת מַעֲדִיפִים לִקְנוֹת מוּצָרִים אֵיכוּתִיִּים וּמְעַטִּים לְעוּמַּת מוּצָרִים רַבִּים וְזוֹלִים".

לִדְבָרֶיהָ, הַטְּרֶנְדִּים בְּמוּצְרֵי הַחֲזָרָה לְבֵית הַסֵּפֶר תּוֹאֲמִים אֶת הַמְּגַמָּה הָעוֹלָמִית שֶׁל חֲזָרָה לַפַּשְׁטוּת. "אוֹפְנַת אָמָּנוּת הָרְחוֹב (סְטְרִיט אַרְט) פּוֹפּוּלָרִית בְּקֶרֶב יְלָדִים. נִיתָּן לִרְאוֹת שֶׁיֵּשׁ הַעֲדָפָה לְמוּצָרִים גֵּנֵרִיִּים עַל פְּנֵי מוּצָרִים מְמוּתָּגִים, דִּיסְנִי לְמָשָׁל, וְכֵן הַעֲדָפָה וּצְמִיחָה דְּרָמָטִית בְּכָל הַקָּשׁוּר לְמִיחְזוּר, גַּם בְּעוֹלַם הַתִּיקִים יֵשׁ עֲלִיָּיה בְּמוּתְגֵי הַתִּיקִים הַיְּדִידוּתִיִּים לַסְּבִיבָה, וְכֵן הַחֲשִׁיבוּת לְפוּנְקְצְיוֹנָלִיּוּת שֶׁל הַתִּיק גְּבוֹהָה יוֹתֵר הַשָּׁנָה".
נִיל בֶּן דְּרוֹר, מְנַהֵל מוּתָג JanSport בְּיִשְׂרָאֵל, מְאַשֵּׁר אֶת מָה שֶׁמַּצִּיגָה שַׁבְּתַאי. "רוֹב הַדְּגָמִים בַּקּוֹלֶקְצִיָּה הָעוֹנָה כּוֹלְלִים תָּא יִיעוּדִי לְמַחְשֵׁב נַיָּיד, וְיֵשׁ דְּגָמִים הַכּוֹלְלִים גַּם תָּא לְטָאבְּלֶט. אֲנַחְנוּ רוֹאִים עֲלִיָּיה בְּהִתְעַנְיְינוּת בַּתְּחוּם הַסְּבִיבָתִי וּבַחֲשִׁיבָה הַיְּרוּקָּה, מִחְזַרְנוּ יוֹתֵר מִ־13 מִילְיוֹן בַּקְבּוּקֵי פְּלַסְטִיק שֶׁשּׁוּלְּבוּ בַּתִּיקִים שֶׁלָּנוּ הָעוֹנָה וְכֵן אֲנַחְנוּ מַצִּיעִים דֶּגֶם שֶׁכּוּלּוֹ מוּרְכָּב מֵחֳומָרִים מְמוּחְזָרִים".
קוֹנִים פָּחוֹת צִיּוּד רָגִיל
אוֹפֶן הַלְּמִידָה הֶחָדָשׁ מֵהַבַּיִת מְחַיֵּיב הִתְאַרְגְּנוּת מְיוּחֶדֶת וַעֲזָרִים שֶׁיְּסַיְּיעוּ לַיְּלָדִים לְהִתְרַגֵּל לַמְּצִיאוּת הַחֲדָשָׁה. מְנַהֲלֵי "תִּיכוֹנִיסְטִים מְצַיְּיצִים", קְהִילַּת תַּלְמִידֵי הַתִּיכוֹן וְהַחֲטִיבָה הַגְּדוֹלָה בְּיִשְׂרָאֵל, מְדַבְּרִים עַל סְגִירַת אַרְנָקִים. "תְּקוּפַת הַקּוֹרוֹנָה לִימְּדָה אֲנָשִׁים לְהִתְמוֹדֵד לָרִאשׁוֹנָה עִם רְכִישׁוֹת בְּאוֹנְלַיְין, וְאָנוּ רוֹאִים שֶׁחֵלֶק מֵהַצַּרְכָנִים כְּבָר אִימְּצוּ אֶת זֶה כְּהֶרְגֵּל לְאוֹרֶךְ זְמַן. סַל הַקְּנִיּוֹת הַמְּמוּצָּע בָּרֶשֶׁת הוּא 30-20 פְּרִיטִים, מִמַּכְשִׁירֵי כְּתִיבָה וּמוּצָרִים בְּסִיסִיִּים לְלִימּוּדִים וְעַד קַלְמָרִים וְתִיקִים. גַּם אִם לֹא תִּהְיֶה חֲזָרָה רְגִילָה לְשִׁגְרַת הַלִּימּוּדִים".
מִיכַל רָבִיב, סמנכ"לית שִׁיוּוּק בַּאֲתַר הַסַּחַר טֶרְמִינָל X, מוֹסִיפָה כִּי "לְאוֹרֶךְ הֶחוֹדְשַׁיִים הָאַחֲרוֹנִים קַיָּים בִּיקּוּשׁ רַב לְמוּצָרִים בְּקָטֵגוֹרְיַית 'חוֹזְרִים לְבֵית הַסֵּפֶר', שֶׁצּוֹבֵר תְּאוּצָה כְּכָל שֶׁמִּתְקָרְבִים לִתְחִילַּת שְׁנַת הַלִּימּוּדִים הַחֲדָשָׁה. קַיֶּימֶת קְפִיצָה שֶׁל מֵאוֹת אֲחוּזִים בָּרְכִישׁוֹת בְּיַחַס לַשָּׁנָה שֶׁעָבְרָה". לִדְבָרֶיהָ, "הַמּוּצָרִים הַנִּמְכָּרִים בְּיוֹתֵר הֵם חוּלְצוֹת בֵּית סֵפֶר עִם סֵמֶל, תִּיקֵי גַּב, צִיּוּד כְּגוֹן קַלְמָרִים וְנַעֲלֵי הִתְעַמְּלוּת".
עִם זֹאת, בְּרִשְׁתוֹת הָאוֹפְנָה זֶבְּרָה וְסֶלֶקְט נִשְׁמַע קוֹל אַחֵר. לְטַעֲנַת אַלּוֹן דּוֹיְטְשׁ, מַנְכַּ"ל הָרְשָׁתוֹת הָאֵלּוּ, "הַקְּנִיָּיה בַּחֲנוּיוֹת הַפִיזִיּוֹת זֵהָה לְשָׁנִים קוֹדְמוֹת עַד לְרֶגַע זֶה. הָרוֹב הַמּוּחְלָט מַעֲדִיף לְהַגִּיעַ לַחֲנוּיוֹת הַפִיזִיּוֹת, לִבְחוֹר אֶת הַחוּלְצוֹת, לִמְדּוֹד אוֹתָן וּלְאַחַר מִכֵּן לְהַדְפִּיס אֶת סֵמֶל בֵּית הַסֵּפֶר. בָּאִינְטֶרְנֵט זֶה קְצָת מְסוּרְבָּל וְגַם אִיטִּי בְּיַחַס לַחֲנוּיוֹת הַפִיזִיּוֹת. אֲנָשִׁים אוֹהֲבִים לְהַרְגִּישׁ אֶת חֲוָויַית הַקְּנִיָּיה הַפִיזִית".
