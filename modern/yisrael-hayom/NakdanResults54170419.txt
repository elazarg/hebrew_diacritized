דֵּיְיט בַּבַּיִת
הַקּוֹרוֹנָה הֵבִיאָה אִיתָּהּ אֶתְגָּרִים לֹא פְּשׁוּטִים, וְאֶחָד מֵהֶם לְלֹא סָפֵק הוּא מְצוּקַת הַלּוֹקֵיְישֶׁנִים לְדֶיְיטִים • הַמִּסְעָדוֹת וְהַקַּנְיוֹנִים נִסְגְּרוּ, וְלָשֶׁבֶת בַּחוּץ קַר • הַחוֹרֶף כָּאן, וְיֵשׁ לַמּוֹחַ הַיְּהוּדִי פִּתְרוֹנוֹת יְצִירָתִיִּים בִּמְיוּחָד

עִנְבָּל אֶלְמָלִיחַ

תְּהִילָּה פֵּרְל

לִירוֹן וַאֲבִיחַי, שְׁנֵי רַוָּוקִים שֶׁמְּחַפְּשִׂים זוּגִיּוּת, הֵבִינוּ שֶׁבִּתְקוּפַת הַקּוֹרוֹנָה הָאוֹפְּצִיָּה הַטּוֹבָה בְּיוֹתֵר עֲבוּרָם הִיא לְהִיפָּגֵשׁ בַּחוּץ. הַבְּעָיָה הִיא שֶׁהֵם נֶאֶלְצוּ לָאַחֲרוֹנָה לִדְחוֹת אֶת הַדֵּיְיט שֶׁלָּהֶם שָׁלוֹשׁ פְּעָמִים בִּגְלַל גֶּשֶׁם שׁוֹטֵף. "הַסְּגָרִים עִיכְּבוּ אֶת הֲדֶיְיטִים. הָיוּ אוֹפְּצִיּוֹת כְּמוֹ פְּגִישׁוֹת בַּזּוּם, אֲבָל לִי זֶה לֹא הִתְאִים. חָשׁוּב לִי לִרְאוֹת פִיזִית אֶת הָאָדָם שֶׁאֲנִי יוֹצֵאת אִיתּוֹ, הָיָה לִי קָשֶׁה עִם הַפָּן הַטֶּכְנוֹלוֹגִי. לְאַחַר שֶׁאוּשְּׁרוּ הַהֲקַלּוֹת, הֲדֶיְיטִים הִתְחַדְּשׁוּ וְעָבְרוּ לְפַארְקִים צִיבּוּרִיִּים אוֹ לִמְקוֹמוֹת שֶׁאֶפְשָׁר לָשֶׁבֶת בָּהֶם בִּפְרָטִיּוּת בָּאֲוִויר הַפָּתוּחַ. הִגִּיעַ הַגֶּשֶׁם, וְאִיתּוֹ נִקְטְעוּ הַסִּיכּוּיִים לְדֶיְיטִים בַּחוּץ", מְסַפֶּרֶת לִירוֹן, "אִי אֶפְשָׁר לָשֶׁבֶת בַּבַּיִת, כִּי אֲנִי לֹא אַזְמִין אֵלִי אוֹ לֹא אֶרְצֶה לָלֶכֶת לְדֵיְיט רִאשׁוֹן לְגֶבֶר שֶׁעֲדַיִין לֹא פָּגַשְׁתִּי, הַמִּסְעָדוֹת וְהַקַּנְיוֹנִים סְגוּרִים, כָּךְ שֶׁנּוֹתַרְנוּ לְלֹא קוֹרַת גַּג וּלְלֹא אֶפְשָׁרוּת לְהִיפָּגֵשׁ בְּלִי לִקְפּוֹא מָקוֹר".

עִם הַדְּבָרִים הַלָּלוּ יִזְדַּהוּ כָּל מִי שֶׁנִּמְצָאִים בִּסְצֵנַת הֲדֶיְיטִים בִּתְקוּפַת הַקּוֹרוֹנָה. "זוֹ תְּקוּפָה לֹא פְּשׁוּטָה לָצֵאת לְדֶיְיטִים. הָרַוָּוקִים צְרִיכִים חֶסֶד שֶׁל אֲחֵרִים כְּדֵי לְקַיֵּים פְּגִישׁוֹת", אוֹמֶרֶת תְּהִילָּה פֵּרְל מִיְּרוּשָׁלַיִם, שֶׁלִּפְנֵי הַקּוֹרוֹנָה אֵירְחָה בְּבֵיתָהּ אַחַת לְשָׁבוּעַ רַוָּוקִים וְרַוָּוקוֹת לְמִפְגְּשֵׁי הֶיכֵּרוּיוֹת. פֵּרְל הֵבִינָה שֶׁנּוֹצְרָה בְּעָיָה גְּדוֹלָה בְּקֶרֶב רַוָּוקִים שֶׁרוֹצִים לָצֵאת, אֲבָל לֹא מוֹצְאִים הֵיכָן לָשֶׁבֶת כָּעֵת.

"הַכֹּל הִתְחִיל כְּשֶׁדִּיבַּרְתִּי עִם אַחְיָינִית שֶׁלִּי, סְטוּדֶנְטִית צְעִירָה, וְתוֹךְ כְּדֵי שִׂיחָה שָׁאַלְתִּי אוֹתָהּ אֵיךְ הִיא בַּתְּקוּפָה הַזּוֹ וְאֵיפָה הַקּוֹרוֹנָה תָּפְסָה אוֹתָהּ. הִתְגַּלְגַּלְנוּ בַּשִּׂיחָה וְהֵבַנְתִּי שֶׁמְּצוּקַת הֲדֶיְיטִים, כְּמוֹ כָּל דָּבָר בַּקּוֹרוֹנָה, רַק הִקְצִינָה. תְּקוּפוֹת הַסֶּגֶר, מְנִיעַת הַמִּפְגָּשִׁים, וּכְשֶׁסּוֹף-סוֹף אֶפְשָׁר לִנְסוֹעַ וּלְהִיפָּגֵשׁ - אֵין אֵיפֹה. בִּשְׁנִיּוֹת נִדְלְקוּ לִי כָּל כַּפְתּוֹרֵי אַזְעָקַת הָאֱמֶת שֶׁלִּי, וְעוֹד בְּאוֹתוֹ הָעֶרֶב כָּתַבְתִּי לְכָל הָאַחְיָינִים שֶׁלִּי שֶׁבַּגִּילִים הָרֵלֵוַונְטִיִּים הוֹדָעָה מַזְמִינָה לְדֵיְיט אֶצְלֵנוּ בַּסָּלוֹן, עִם סוֹדִיּוּת מוּבְטַחַת. הַהוֹדָעוֹת הִתְקַבְּלוּ עַל לְבָבוֹת שֶׁמְּשַׁוְּועִים לַמָּקוֹם, לְאַרְבָּעָה קִירוֹת וּשְׁתִיָּיה חַמָּה, לְמַשֶּׁהוּ שֶׁיְּאַפְשֵׁר לָהֶם בְּתוֹךְ הַכָּאוֹס הַזֶּה שֶׁכּוּלָּנוּ נִמְצָאִים בּוֹ לִבְנוֹת אֶת הַפִּינָּה שֶׁלָּהֶם בַּחַיִּים שֶׁכָּל כָּךְ מְחַכָּה לְיַחֵד. הֵבַנְתִּי שֶׁיֵּשׁ הֲמוֹן אֲנָשִׁים שֶׁרַק מְחַכִּים שֶׁמִּישֶׁהוּ יְאַפְשֵׁר לָהֶם מָקוֹם, וְהִצַּעְתִּי בְּפֶיְיסְבּוּק שֶׁלִּי לִכְתּוֹב בִּתְגוּבוֹת מִי מוּכָן לְאָרֵחַ".

לְאוֹר הַתְּגוּבוֹת הָרַבּוֹת לַפּוֹסְט, הִיא הֵקִימָה קְבוּצַת פֶיְיסְבּוּק בַּשֵּׁם "דֶּיְיטִים בֵּיתִיִּים בַּקּוֹרוֹנָה". "לֹא הֶאֱמַנְתִּי שֶׁזֶּה יַגִּיעַ לְסֵדֶר גּוֹדֶל שֶׁכְּזֶה. מִשְׁפָּחוֹת הִצְטָרְפוּ לַקְּבוּצָה וְהִתְחִילוּ לְהַצִּיעַ אֶת בָּתֵּיהֶן מִכָּל הַלֵּב. אֲנָשִׁים גַּם הִצִּיעוּ לְרַוָּוקִים הַמִּתְאָרְחִים אֶצְלָם אֲרוּחוֹת, מְרָקִים, קָפֶה וְעוּגָה. הָיוּ כָּאֵלֶּה שֶׁכְּבָר אֵירְחוּ בְּאוֹתוֹ עֶרֶב שֶׁפָּתַחְתִּי אֶת הַקְּבוּצָה וְזֶה מְרַגֵּשׁ מְאוֹד", אוֹמֶרֶת פֵּרְל. קְבוּצַת הֲפֶיְיסְבּוּק מוֹנָה כְּאֶלֶף אִישׁ, וּבָהּ פּוֹסְטִים רַבִּים שֶׁל אֲנָשִׁים הַמַּזְמִינִים פְּנוּיִים וּפְנוּיוֹת לָשֶׁבֶת בְּבֵיתָם לְדֵיְיט.

אֵין חָשַׁשׁ לְהַכְנִיס זָרִים לַבַּיִת בִּגְלַל הַקּוֹרוֹנָה?

"אֲנִי חֲרַדָּתִית. כְּשֶׁהַזּוּג הָרִאשׁוֹן הִגִּיעַ, בִּיקַּשְׁתִּי שֶׁיִּהְיוּ בַּבַּיִת עִם מַסֵּיכָה וְאַחַר כָּךְ כְּשֶׁהֵם הָלְכוּ שָׁטַפְתִּי אֶת הַכֵּלִים שֶׁהֵם אָכְלוּ בָּהֶם בְּסַבּוֹן. בַּקּוֹרוֹנָה חוֹשְׁשִׁים גַּם מֵאֲנָשִׁים לֹא זָרִים כִּי אִי אֶפְשָׁר לָדַעַת מִי חוֹלֶה אִם אֵין תַּסְמִינִים, אֲנִי נִזְהֶרֶת עִם כּוּלָּם. בְּנוֹגֵעַ לִבְטִיחוּת, נָשִׁים צְעִירוֹת שֶׁהַבַּעַל שֶׁלָּהֶן בְּקֶבַע כָּתְבוּ שֶׁהַבַּיִת רֵיק כָּל הַשָּׁבוּעַ וְאֶפְשָׁר לְהַגִּיעַ לְדֵיְיט. בִּיקַּשְׁתִּי מֵהֶן לְהוֹרִיד אֶת הַפּוֹסְט כִּי זוֹ פִּרְצָה שֶׁלְּעִיתִּים קוֹרֵאת לְגַנָּב. מְדוּבָּר בְּכָל זֹאת בִּקְבוּצָה בָּרֶשֶׁת, וְשָׁם לֹא כּוּלָּם צַדִּיקִים וְטוֹבִים. תָּמִיד מַשֶּׁהוּ רַע יָכוֹל לִקְרוֹת אֲבָל לְדַעְתִּי הַתּוֹעֶלֶת עוֹלָה עַל הַחֶסֶר, וַאֲנִי מְקַוָּוה שֶׁאַף אֶחָד לֹא יִינָּזֵק".

רָחֵלִי בֶּן שִׁימוֹל מִפֶּתַח תִּקְוָוה, חֲבֵרָה בַּקְּבוּצָה, נְשׂוּאָה וְאֵם לַחֲמִישָּׁה, מְסַפֶּרֶת שֶׁלֹּא הָיָה לָהּ חָשַׁשׁ לְהַזְמִין זוּג שֶׁהִיא לֹא מַכִּירָה. "אַחֲרֵי שֶׁפִּרְסַמְתִּי בְּפֶיְיסְבּוּק שֶׁיֵּשׁ לָנוּ מִרְפֶּסֶת פְּנוּיָה בְּפֶתַח תִּקְוָוה עִם נוֹף יָפֶה, בָּחוּר אֶחָד יָצַר אִיתִּי קֶשֶׁר וְשָׁאַל אִם יוּכַל לְהַגִּיעַ לְדֵיְיט בְּמוֹצָאֵי הַשַּׁבָּת, וּמוּבָן שֶׁהִסְכַּמְתִּי. הִתְחַבַּרְתִּי מְאוֹד לַמֵּיזָם כִּי נוֹשֵׂא הַהֶיכֵּרוּיוֹת מְאוֹד קָרוֹב לְלִיבִּי. אֲנִי מְנַסָּה לְשַׁדֵּךְ הַרְבֵּה וְגַם פְּעִילָה בִּקְבוּצַת וָּוֶטְסָאפּ שֶׁל שַׁדְכָנִיּוֹת. הַמִּפְגָּשׁ עִם הַזּוּג הָיָה נֶחְמָד, וְזוֹ אֲוִוירָה נְעִימָה יוֹתֵר כְּשֶׁהַדֵּיְיט בַּבַּיִת שֶׁל מִישֶׁהוּ. הַמְּאָרֵחַ יוֹצֵר עִם הַזּוּג מִין סְמוֹל טוֹק וְזֶה שׁוֹבֵר אֶת הַקֶּרַח, בִּמְיוּחָד בְּדֵיְיט רִאשׁוֹן. אֲנִי רוֹאֶה מֵהַתְּגוּבוֹת בְּפֶיְיסְבּוּק שֶׁנִּרְאֶה שֶׁהַפְּנוּיִים וְהַפְּנוּיוֹת שְׂמֵחִים עַל הַמֵּיזָם וְעַל כָּךְ שֶׁאֲנָשִׁים עֵרִים לִמְצוּקָתָם".

הַפוֹרְמָט הַחֲרֵדִי

דֵּיְיט בַּבַּיִת לֹא זָר לַמִּגְזָר הַחֲרֵדִי. עֲבוּרָם, מְקוּבָּל שֶׁהַדֵּיְיט מִתְקַיֵּים בְּבַיִת שֶׁל אֶחָד הַמְּשׁוּדָּכִים אוֹ בְּלוֹבִּי שֶׁל בֵּית מָלוֹן. דָּוִד, גָּרוּשׁ מֵאַשְׁדּוֹד, מְסַפֵּר כִּי אֵירֵחַ דֵּיְיט בְּגַג בֵּיתוֹ: "הַפוֹרְמָט שֶׁל דֵּיְיט בַּבַּיִת מוּכָּר לַחֲרֵדִים הַרְבֵּה לִפְנֵי הַקּוֹרוֹנָה, אַךְ מְדוּבָּר בִּנְקוּדָּה רְגִישָׁה עֲבוּר גְּרוּשִׁים חֲרֵדִים מוֹדֶרְנִיִּים שֶׁמְּחַפְּשִׂים פֶּרֶק ב'. הֵם לֹא הֶאֱמִינוּ שֶׁיַּחְזְרוּ לְהִיפָּגֵשׁ בַּבַּיִת לְשִׁידּוּךְ כְּפִי שֶׁעָשׂוּ כְּשֶׁהָיוּ צְעִירִים, וְיֵשׁ כָּאֵלֶּה שֶׁאַף הִבְטִיחוּ לְעַצְמָם לֹא לַעֲשׂוֹת זֹאת. הִזְמַנְתִּי חָבֵר לַדֵּיְיט אֶצְלִי בַּמִּרְפֶּסֶת, הִדְלַקְתִּי נֵרוֹת וְהֵבֵאתִי לָהֶם בַּקְבּוּק יַיִן, וְהֵם סִיפְּרוּ שֶׁזּוֹ הָיְתָה חֲוָויָה נְעִימָה וְשׁוֹנָה מֵהַדֵייטִים שֶׁהִכִּירוּ בֶּעָבָר".

לִירוֹן וַאֲבִיחַי בָּחֲרוּ בְּסוֹפוֹ שֶׁל דָּבָר לָצֵאת לְדֵיְיט אֵצֶל חֲבֵרָה שֶׁל לִירוֹן בַּסָּלוֹן, וַעֲבוּרָם הַחֲוָויָה הָיְתָה מְעַנְיֶינֶת. "הַיְּלָדִים שֶׁלָּהּ הָיוּ בַּבַּיִת וְלֹא הִתְכּוֹנַנּוּ מֵרֹאשׁ לִשְׁאֵלוֹת מְבִיכוֹת שֶׁלָּהֶם", הִיא מְסַפֶּרֶת,"עִם זֹאת, מְדוּבָּר בְּחֶבֶל הַצָּלָה שֶׁל מַמָּשׁ. אֲנִי מְקַוָּוה שֶׁאֲנָשִׁים נוֹסָפִים יַצִּיעוּ אֶת הַבָּתִּים שֶׁלָּהֶם וִיאָרְחוּ רַוָּוקִים וְרַוָּוקוֹת שֶׁרוֹצִים לִמְצוֹא אֶת הָאֶחָד אוֹ אֶת הָאַחַת שֶׁלָּהֶם וּלְהִתְחַתֵּן, גַּם בַּקּוֹרוֹנָה".
