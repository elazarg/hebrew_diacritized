
טִיסוֹת, צוֹלְלוֹת - וְחִילּוּפֵי הַאֲשָׁמוֹת
הִשְׁתַּלְשְׁלוּת הָאֵירוּעִים בַּכְּנֶסֶת רַק חִידְּדָה אֶת הֲלַךְ הָרוּחַ בַּמַּעֲרֶכֶת הַפּוֹלִיטִית: הַמֶּמְשָׁלָה עֲדַיִין  קַיֶּימֶת - אֲבָל הָאַחְדוּת כְּבָר מִזְּמַן לֹא • גַּנְץ הִזְהִיר: "לֹא חוֹשֵׁשׁ מִבְּחִירוֹת, נָבוֹא בְּשָׁלוֹם  לְקִידּוּם תַּקְצִיב - אַךְ נֵדַע לְהִשְׁתַּמֵּשׁ גַּם בְּחֶרֶב" • נְתַנְיָהוּ הֶאֱשִׁים: "גַּנְץ מִשְׁתַּמֵּשׁ בְּצַהַ"ל  כִּכְלִי נִיגּוּחַ פּוֹלִיטִי" • וְדֶרְעִי בָּחַר צַד: "גַּנְץ חָצָה קַו אָדוֹם בִּגְלַל שִׁיקּוּלִים פּוֹלִיטִיִּים"
יְהוּדָה שְׁלֵזִינְגֶּר, דָּנִיֵּאל רוֹט-אַבְנֵרִי וְגִדְעוֹן אַלּוֹן



הִתְרַחֲשׁוּיוֹת בְּמִשְׁכַּן הַכְּנֶסֶת

אֶתְמוֹל יִיזָּכְרוּ כְּעוֹד יוֹם שֶׁקֵּירֵב אֶת הִתְפָּרְקוּת הַמֶּמְשָׁלָה, בַּדֶּרֶךְ לְמַעֲרֶכֶת בְּחִירוֹת נוֹסֶפֶת.


זֶה הִתְחִיל בְּהִתְבַּטְּאוּיוֹת אַגְרֵסִיבִיּוֹת שֶׁל שַׂר הַבִּיטָּחוֹן גַּנְץ כְּלַפֵּי  רה"מ נְתַנְיָהוּ, הִמְשִׁיךְ בְּהַסְתָּרַת הַנְּסִיעָה הַמְּדִינִית שֶׁל נְתַנְיָהוּ מִגַּנְץ וּמִשַּׂר הַחוּץ אַשְׁכְּנַזִּי, מִשָּׁם הִידַּרְדְּרָה הָרַכֶּבֶת הַפּוֹלִיטִית לְהַאֲשָׁמָה מִצַּד גַּנְץ בִּפְגִיעָה בְּבִיטְחוֹן הַמְּדִינָה - וְנִגְמַר  בְּהִתְנַעֲרוּת פּוּמְבִּית שֶׁל יוֹ"ר שָׁ"ס אַרְיֵה דֶּרְעִי מֵהִתְחַיְּיבוּתוֹ כְּלַפֵּי גַּנְץ.

הַסְּעָרָה הַתּוֹרָנִית הֵחֵלָּה כְּשֶׁיּוֹ"ר כָּחוֹל לָבָן הוֹדִיעַ עַל וַעֲדַת בְּדִיקָה בְּעִנְיַין פָּרָשַׁת הַצּוֹלְלוֹת, וְאָמַר בָּרֵאָיוֹן  שֶׁפּוּרְסַם אֶתְמוֹל: "הַמְּדִינָה עַל אֵשׁ קְטַנָּה כִּי נְתַנְיָהוּ חוֹשֵׁב קוֹדֶם עַל הַמִּשְׁפָּט. יַאלְלָה פָאוּדָה". בִּשְׁעוֹת הַבּוֹקֶר הַמּוּקְדָּמוֹת תָּקְפוּ אַנְשֵׁי הַלִּיכּוּד אֶת גַּנְץ לְנוֹכַח הִתְבַּטְּאוּיוֹתָיו וּבְשֶׁל  הֲקָמַת וַעֲדַת הַבְּדִיקָה. "לֹא הוּפְתַּעְנוּ מֵהַהַחְלָטָה, בְּכָל פַּעַם שֶׁגַּנְץ יוֹרֵד  בַּסְּקָרִים הוּא מְיַיצֵּר סְפִּין", אָמַר שַׂר  הַמּוֹדִיעִין אֵלִי כֹּהֵן, "גַּנְץ אָמַר מֵאָז כְּנִיסָתוֹ לַמֶּמְשָׁלָה שֶׁהַדָּבָר הַיָּחִיד שֶׁצָּרִיךְ לַעֲסוֹק בּוֹ זֶה מַשְׁבֵּר הֲקוֹרוֹנָה. הַיָּחִיד שֶׁמְּיַישֵּׂם אֶת זֶה הוּא רֹאשׁ הַמֶּמְשָׁלָה".

שָׁעָה קַלָּה לְאַחַר מִכֵּן פּוּרְסַם כִּי רה"מ נְתַנְיָהוּ בִּיקֵּר בְּסָעוּדְיָה שִׁלְשׁוֹם לִפְגִישָׁה עִם יוֹרֵשׁ הָעֶצֶר מוּחַמַּד בֶּן סַלְמָאן - וְכִי גַּנְץ וְאַשְׁכְּנַזִּי לֹא הָיוּ מְעוֹרָבִים אוֹ מְעוּדְכָּנִים בְּפִרְטֵי הַפְּגִישָׁה, וּבְהַחְלָטָה לְהַתִּיר אֶת פִּרְסוּמָהּ.

גַּנְץ: "לֹא רוֹצֶה בְּחִירוֹת"

"הַדְלָפַת הַטִּיסָה הַחֲשָׁאִית שֶׁל רֹאשׁ הַמֶּמְשָׁלָה הִיא צַעַד חֲסַר אַחְרָיוּת, לֹא  כָּךְ אֲנִי נוֹהֵג", אָמַר גַּנְץ בִּשְׁעוֹת הַצָּהֳרַיִים בִּישִׁיבַת סִיעָתוֹ. "מֵעוֹלָם לֹא נָהַגְתִּי כָּךְ וּמֵעוֹלָם לֹא אֶנְהַג כָּךְ, וַאֲנִי חוֹשֵׁב שֶׁבַּהֶקְשֵׁר הַזֶּה אֶזְרְחֵי יִשְׂרָאֵל  צְרִיכִים לִהְיוֹת מוּטְרָדִים". הוּא הוֹסִיף כִּי "נָבוֹא בְּשָׁלוֹם לְקִידּוּמוֹ שֶׁל תַּקְצִיב וּמֶמְשָׁלָה מְתַפְקֶדֶת, אַךְ נֵדַע לְהִשְׁתַּמֵּשׁ גַּם בְּחֶרֶב. אִם לֹא נַחֲזוֹר וּמִיָּיד לְמַסְלוּל שֶׁל עֲשִׂיָּיה לְמַעַן כְּלַל אֶזְרְחֵי יִשְׂרָאֵל, לֹא תִּהְיֶה בְּרֵירָה וְנֵצֵא לַמַּעֲרָכָה מֵחָדָשׁ בִּשְׂדֵה הַבְּחִירוֹת.

"מְדִינַת יִשְׂרָאֵל צְרִיכָה מֶמְשָׁלָה מְתַפְקֶדֶת מִגּוֹבַהּ בַּתַּקְצִיב, שֶׁמִּתְבַּסֶּסֶת עַל שִׁיתּוּף פְּעוּלָּה. אֲנִי לֹא חוֹשֵׁשׁ מִבְּחִירוֹת - אֲנִי פָּשׁוּט לֹא חוֹשֵׁב שֶׁהֵן הַדָּבָר הַנִּדְרָשׁ לִמְדִינַת יִשְׂרָאֵל, וְאַמְשִׁיךְ לִפְעוֹל כְּכָל שֶׁאוּכַל לְמַעַן  הַמֶּמְשָׁלָה הַזֹּאת".

בְּהִתְיַיחֵס לְפָרָשַׁת הַצּוֹלְלוֹת אָמַר  גַּנְץ: "לְאַחַר שֶׁבָּחַנְתִּי מֵחָדָשׁ אֶת נוֹשֵׂא הַצּוֹלְלוֹת וְעִדְכַּנְתִּי אֶת רֹאשׁ הַמֶּמְשָׁלָה, הֶחְלַטְתִּי לְהָקִים וַעֲדַת בְּדִיקָה מֶמְשַׁלְתִּית לְרֶכֶשׁ סְפִינוֹת הַמָּגֵן  וְהַצּוֹלְלוֹת. לִכְאוֹרָה, רֹאשׁ הַמֶּמְשָׁלָה צָרִיךְ הָיָה לִהְיוֹת מְעוּנְיָין יוֹתֵר מִכָּל  אָדָם אַחֵר בְּוַועֲדַת הַבְּדִיקָה הַזּוֹ. אִם, כְּפִי שֶׁטּוֹעֲנִים הוּא וַאֲנָשָׁיו - הַנּוֹשֵׂא נִסְרַק בְּמַסְרְקוֹת בַּרְזֶל וְאֵין בּוֹ רְבָב - הֲרֵי כָּל אָדָם אַחֵר הַמְּשׁוּכְנָע בְּחַפּוּתוֹ הָיָה שָׂמֵחַ עַל הַהִזְדַּמְּנוּת לְהוֹכִיחַ לְעֵינֵי  כָּל אֶת צִדְקָתוֹ".

נְתַנְיָהוּ: "וַעֲדָה תְּפוּרָה"

רֹאשׁ הַמֶּמְשָׁלָה נְתַנְיָהוּ לֹא נוֹתַר חַיָּיב וְהֵשִׁיב לְגַנְץ בִּישִׁיבַת סִיעַת הַלִּיכּוּד.  "בּוּשָׁה שֶׁגַּנְץ מִשְׁתַּמֵּשׁ בְּצַהַ"ל כִּכְלִי  נִיגּוּחַ פּוֹלִיטִי", תָּקַף רה"מ, "אַחֲרֵי שֶׁכָּל גּוֹרְמֵי הָאֲכִיפָה קָבְעוּ פַּעַם אַחַר פַּעַם שֶׁלְּרֹאשׁ הַמֶּמְשָׁלָה נְתַנְיָהוּ אֵין שׁוּם קֶשֶׁר  לְפָרָשַׁת הַצּוֹלְלוֹת, גַּנְץ מִינָה 'וַעֲדָה תְּפוּרָה' בְּרָאשׁוּת מִי שֶׁאָמַר לִפְנֵי שָׁנָה שֶׁ'רֹאשׁ הַמֶּמְשָׁלָה צָרִיךְ לְהִתְפַּטֵּר'. מְעַנְיֵין לְחַכּוֹת וְלִרְאוֹת מָה תִּהְיֶה מַסְקְנָתוֹ. אִי אֶפְשָׁר לְשַׁעֵר. הַוַּועֲדָה הַתְּפוּרָה הַזֹּאת נוֹעֲדָה לְהָבִיא מַסְקָנָה תְּפוּרָה וּלְקוֹשֵׁשׁ קוֹלוֹת. זֹאת בּוּשָׁה גְּדוֹלָה. אַף שַׂר בִּיטָּחוֹן לֹא עָשָׂה שִׁימּוּשׁ פּוֹלִיטִי כָּל  כָּךְ בּוֹטֶה בְּצַהַ"ל כְּפִי שֶׁעָשָׂה גַּנְץ. אֲנִי מְקַוֶּוה שֶׁגַּנְץ יִתְעַשֵּׁת מֵהַפׇאוּדָה הַזֹּאת וְיַחֲזוֹר לַעֲבוֹד בְּיַחַד, זֶה מָה שֶׁאֶזְרְחֵי  יִשְׂרָאֵל מְצַפִּים מֵאִיתָּנוּ".

גַּם יוֹ"ר הַכְּנֶסֶת יָרִיב לֵוִין תָּקַף אֶת  גַּנְץ וְאָמַר: "גַּנְץ הֵקִים וַעֲדָה פּוֹלִיטִית  לַחֲלוּטִין, זֶה אֵירוּעַ חָמוּר וַחֲסַר תַּקְדִּים.  שִׂיא שֶׁל צִינִיּוּת".

כָּחוֹל לָבָן: "דֶּרְעִי יְקַיֵּים"

בְּמַקְבִּיל לִדְבָרָיו שֶׁל גַּנְץ, יָצָא יוֹ"ר  שָׁ"ס אַרְיֵה דֶּרְעִי בִּישִׁיבַת סִיעַת שָׁ"ס בַּכְּנֶסֶת בְּהַצְהָרָה חֲרִיגָה שֶׁבָּהּ אָמַר כִּי הוּא בְּסָפֵק אִם עֲדַיִין יֵשׁ הַצְדָּקָה לְהֶמְשֵׁךְ הַשּׁוּתָּפוּת בְּמֶמְשֶׁלֶת הָאַחְדוּת. "אֲנִי פּוֹנֶה לִנְתַנְיָהוּ וְגַנְץ, הִגַּעְנוּ לִשְׁעַת הָאֱמֶת, לִשְׁעַת הַנְּעִילָה שֶׁל  הַמֶּמְשָׁלָה", אָמַר דֶּרְעִי, "צָרִיךְ לָשִׂים הַכֹּל בַּצַּד וּלְהִתְמַקֵּד בִּבְרִיאוּת וּבַפַּרְנָסָה שֶׁל עַם יִשְׂרָאֵל; גַּנְץ חָצָה קַו אָדוֹם בִּגְלַל שִׁיקּוּלִים פּוֹלִיטִיִּים - קוֹרֵא לוֹ  לַחֲזוֹר בּוֹ". הוּא הִדְגִּישׁ כִּי "חָשַׁבְתִּי שֶׁאַחְדוּת הִיא הַדָּבָר הַטּוֹב בְּיוֹתֵר לְעַם  יִשְׂרָאֵל, אֲבָל אֲנִי כְּבָר לֹא בָּטוּחַ".

בִּתְגוּבָה עַל דִּבְרֵי דֶּרְעִי וְאַנְשֵׁי הַלִּיכּוּד, מָסְרוּ בְּכָחוֹל לָבָן: "הֲקָמַת הַוַּועֲדָה לֹא נוֹבַעַת מִשִּׁיקּוּלִים פּוֹלִיטִיִּים אֶלָּא מִתּוֹךְ אַחְרָיוּת. מֵחוֹבָתָהּ שֶׁל מַעֲרֶכֶת הַבִּיטָּחוֹן לִבְדּוֹק אֶת תַּהֲלִיכֵי הָרֶכֶשׁ וּלְהָפִיק לְקָחִים. לֹא תִּהְיֶה פָּרָשַׁת צוֹלְלוֹת ‭ .2‬ כָּחוֹל לָבָן לֹא מְטִילָה סָפֵק כִּי בְּרֶגַע הָאֱמֶת הַשַּׂר דֶּרְעִי יְקַיֵּים אֶת הִתְחַיְּיבוּתוֹ וְאֶת הָעַרְבוּת שֶׁנָּתַן לְיִישּׂוּם מָלֵא שֶׁל הַהֶסְכֵּם שֶׁנֶּחְתַּם עִם  הֲקָמַת הַמֶּמְשָׁלָה".

לַפִּיד: "הִגִּיעַ זְמַן בְּחִירוֹת"

בְּתוֹךְ כָּךְ, יוֹ"ר הָאוֹפּוֹזִיצְיָה חַ"כּ  יָאִיר לַפִּיד (יֵשׁ עָתִיד) הוֹדִיעַ אֶתְמוֹל  כִּי בְּיוֹם רְבִיעִי הַבָּא, 2 בְּדֵצֶמְבֶּר, בְּכַוּוֹנָתוֹ לְהָבִיא לְאִישּׁוּר מְלִיאַת הַכְּנֶסֶת בִּקְרִיאָה טְרוֹמִית אֶת הַצָּעַת הַחוֹק  לְפִיזּוּר הַכְּנֶסֶת. לַפִּיד, שֶׁבַּשָּׁבוּעוֹת הָאַחֲרוֹנִים מָשַׁךְ פַּעֲמַיִים אֶת הַצָּעַת הַחוֹק בָּרֶגַע הָאַחֲרוֹן, הִבְטִיחַ בִּישִׁיבַת הַסִּיעָה: "לֹא נִמְשׁוֹךְ אוֹתוֹ בָּרֶגַע הָאַחֲרוֹן וְלֹא נְנַהֵל עַל זֶה מוּ"מ. הַחוֹק  יַעֲלֶה לַהַצְבָּעָה. הִגִּיעַ הַזְּמַן לִבְחִירוֹת. מֵהַמֶּמְשָׁלָה הַזּוֹ לֹא יֵיצֵא כְּלוּם. מְדוּבָּר בְּאוֹסֶף פּוֹלִיטִיקָאִים שֶׁעֲסוּקִים רַק בְּעַצְמָם, בַּגּ'וֹבִּים שֶׁלָּהֶם וּבַמְּרִיבוֹת הָעֲלוּבוֹת שֶׁלָּהֶם. הֵם לֹא עוֹבְדִים בִּשְׁבִיל  אֶזְרְחֵי יִשְׂרָאֵל, אֶלָּא רַק לְמַעַן עַצְמָם. אֲנִי קוֹרֵא לְכָל סִיעוֹת הַבַּיִת - זֶה רֶגַע  הַמִּבְחָן שֶׁלָּכֶם".

יְצוּיַּן כִּי עַל מְנַת שֶׁהַצָּעַת הַחוֹק  תַּעֲבוֹר, יֵשׁ צוֹרֶךְ בְּרוֹב שֶׁל ‭ 61‬ ח"כים. גַּם אִם יוּשַּׂג רוֹב כָּזֶה, עִנְיָין אֲשֶׁר מוּתְנֶה בִּתְמִיכַת כָּחוֹל לָבָן, הַצָּעַת הַחוֹק תַּגִּיעַ לְוַועֲדַת הַכְּנֶסֶת שֶׁתִּצְטָרֵךְ לְהָכִין אֶת הַצָּעַת הַחוֹק לִקְרִיאָה רִאשׁוֹנָה - וְשׁוּב הַדָּבָר יְחַיֵּיב אֶת תְּמִיכָתָהּ, מִשּׁוּם  שֶׁיּוֹ"ר וַעֲדַת הַכְּנֶסֶת הוּא חַ"כּ אֵיתָן גִּינְזְבּוּרְג מִכָּחוֹל לָבָן שֶׁקּוֹבֵעַ אֶת סֵדֶר  הַיּוֹם שֶׁל דִּיּוּנֵי הַוַּועֲדָה. לְאַחַר מִכֵּן, הַצָּעַת הַחוֹק תִּצְטָרֵךְ לַעֲבוֹר גַּם אֶת הַקְּרִיאָה הָרִאשׁוֹנָה וּלְאַחֲרֶיהָ הַקְּרִיאוֹת הַשְּׁנִיָּיה וְהַשְּׁלִישִׁית.

בְּכָל מִקְרֶה, אִם עַד ‭ 23‬ בְּדֵצֶמְבֶּר לֹא יְאוּשַּׁר תַּקְצִיב הַמְּדִינָה, הַכְּנֶסֶת תִּתְפַּזֵּר אוֹטוֹמָטִית - גַּם לְלֹא אִישּׁוּר חוֹק לְהִתְפַּזְּרוּתָהּ.

"יוֹ"ר כָּחוֹל לָבָן וְשַׂר הַבִּיטָּחוֹן  בְּנֵי גַּנְץ: "הַהַדְלָפָה בִּדְבַר הַטִּיסָה הַחֲשָׁאִית שֶׁל רֹאשׁ הַמֶּמְשָׁלָה הִיא צַעַד חֲסַר אַחְרָיוּת. לֹא כָּךְ אֲנִי נוֹהֵג. מֵעוֹלָם לֹא נָהַגְתִּי כָּךְ וּמֵעוֹלָם לֹא אֶנְהַג כָּךְ, וַאֲנִי חוֹשֵׁב שֶׁבַּהֶקְשֵׁר הַזֶּה אֶזְרְחֵי יִשְׂרָאֵל צְרִיכִים לִהְיוֹת מוּטְרָדִים" "רה"מ נְתַנְיָהוּ: "אַחֲרֵי שֶׁכָּל גּוֹרְמֵי הָאֲכִיפָה קָבְעוּ פַּעַם אַחַר פַּעַם שֶׁלְּרֹאשׁ הַמֶּמְשָׁלָה נְתַנְיָהוּ אֵין  שׁוּם קֶשֶׁר לְפָרָשַׁת הַצּוֹלְלוֹת, גַּנְץ מִינָה 'וַעֲדָה  תְּפוּרָה' בְּרָאשׁוּת מִי שֶׁאָמַר לִפְנֵי שָׁנָה שֶׁ'רֹאשׁ  הַמֶּמְשָׁלָה צָרִיךְ לְהִתְפַּטֵּר'. מְעַנְיֵין לִרְאוֹת מָה  תִּהְיֶה מַסְקְנָתוֹ. אִי אֶפְשָׁר לְשַׁעֵר" "יוֹ"ר שָׁ"ס הַשַּׂר אַרְיֵה דֶּרְעִי:  "אֲנִי פּוֹנֶה לִנְתַנְיָהוּ וְגַנְץ, הִגַּעְנוּ לִשְׁעַת הָאֱמֶת, לִשְׁעַת הַנְּעִילָה שֶׁל הַמֶּמְשָׁלָה. צָרִיךְ לָשִׂים הַכֹּל בַּצַּד וּלְהִתְמַקֵּד בִּבְרִיאוּת וּבַפַּרְנָסָה שֶׁל עַם יִשְׂרָאֵל. חָשַׁבְתִּי שֶׁאַחְדוּת הִיא הַדָּבָר הַטּוֹב בְּיוֹתֵר  לְעַם יִשְׂרָאֵל, אֲבָל אֲנִי כְּבָר לֹא בָּטוּחַ"
