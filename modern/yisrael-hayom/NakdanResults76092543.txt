מִי נִבְחַר לִהְיוֹת עָשָׁיו אוֹ יַעֲקֹב?

אָבִי דַּבּוּשׁ

"עַל חַרְבְּךָ תִּחְיֶה". הַנְּבוּאָה-קְלָלָה הַזּוֹ מִפָּרָשַׁת תּוֹלָדוֹת מְכַכֶּבֶת בִּנְאוּמִים שֶׁל מַנְהִיגִים וּבְכִירֵי מַעֲרֶכֶת  הַבִּיטָּחוֹן. זוֹ גְּזֵירַת הַגּוֹרָל שֶׁלָּנוּ, הֵם אוֹמְרִים. הֵם שׁוֹכְחִים  דָּבָר אֶחָד: הַמִּילִּים הַלָּלוּ נֶאֶמְרוּ בַּפָּרָשָׁה דַּוְוקָא לְעֵשָׂיו,  וְלֹא לְיַעֲקֹב אָבִינוּ...

‭20‬ שָׁנָה אַחֲרֵי הַדְּבָרִים, יָדַע יַעֲקֹב לִפְתּוֹר קוֹנְפְלִיקְטִים בְּדֶרֶךְ אַחֶרֶת לְגַמְרֵי. לַמִּפְגָּשׁ הַטָּעוּן וְהַמְּסוּכָּן עִם אָחִיו  הַפָּגוּעַ הֵכִין יַעֲקֹב שְׁלוֹשָׁה דְּבָרִים: דּוֹרוֹן, תְּפִילָּה וּמִלְחָמָה. בַּסֵּדֶר הַזֶּה. יַעֲקֹב אָבִינוּ יָדַע שֶׁהַחֶרֶב אֵינָהּ הַדֶּרֶךְ הַיְּחִידָה  לִפְתּוֹר קוֹנְפְלִיקְטִים. לְמַעֲשֶׂה, הִיא הַדֶּרֶךְ הַגְּרוּעָה בְּיוֹתֵר,  דֶּרֶךְ שֶׁל חוֹסֶר בְּרֵירָה. דֶּרֶךְ הַפִּיּוּס וְהַתִּקְוָוה קוֹדְמוֹת לָהּ, וְטוֹבוֹת מִמֶּנָּה לְאֵין שִׁיעוּר.

הַסִּכְסוּךְ הַיִּשְׂרְאֵלִי-פָלַשְׂתִּינִי נִקְלַע לְשָׁם. מִבְּחִינָתֵנוּ,  כְּיִשְׂרְאֵלִים, דֶּרֶךְ הַפִּיּוּס וְהַתִּקְוָוה נִכְזְבוּ. לִפְנֵי הָאִינְתִיפַאדַה הָרִאשׁוֹנָה, כ-%‭17‬ מֵהַיִּשְׂרְאֵלִים צִידְּדוּ בְּפִתְרוֹן  שְׁתֵּי הַמְּדִינוֹת. בְּסוֹפָהּ הַמִּסְפָּר עָלָה ל-%‭.57‬ הָאִינְתִיפַאדַה  הַשְּׁנִיָּיה, שֶׁפָּרְצָה בְּאוֹקְטוֹבֶּר ‭ ,2000‬ הָפְכָה אֶת הַמְּגַמָּה.

אֲנַחְנוּ חוֹשְׁשִׁים, וּבְצֶדֶק. נִיסְּיוֹנוֹת הַפִּיּוּס וְהַהַסְכָּמָה  הַכּוֹשְׁלִים הוּחְלְפוּ בִּצְעָדִים חַד-צְדָדִיִּים שֶׁהֵבִיאוּ, כְּמוֹ בַּהִתְנַתְּקוּת, לְפִתְרוֹנוֹת צְבָאִיִּים וּלְהַקְטָנַת מִסְפַּר הָאֲבֵידוֹת שֶׁלָּנוּ, אֲבָל לֹא הוֹבִילוּ לְפִתְרוֹן. הַכְּהוּנָּה שֶׁל טְרַאמְפּ הִדְגִּימָה אֶת חוֹסֶר הַתּוֹחֶלֶת הַזֶּה הֵיטֵב. בִּיטּוּל הַהֶסְכֵּם עִם  אִירָאן רָשׁוּם עַל שְׁמָם שֶׁל נְתַנְיָהוּ וּטְרַאמְפּ. לְמַעֲשֶׂה, הוּא הֵבִיא אוֹתָנוּ לַחֲזָרָה לַמֵּרוֹץ הַחִימּוּשׁ הַגַּרְעִינִי הָאִירָאנִי.  גַּם תּוֹכְנִית הַסִּיפּוּחַ, שֶׁנִּיסְּתָה לְדַלֵּג מֵעַל הַפָלַשְׂתִּינִים,  נִתְקְלָה בְּקִיר יַרְדֵּנִי-אֵירוֹפִּי. הַיְּרִידָה מֵהָעֵץ הַמְּסוּכָּן נוּצְּלָה, בְּאוֹפֶן חִיּוּבִי, לְהֶסְכֵּמִים עִם מְדִינוֹת הַמִּפְרָץ.

בְּחִירָתוֹ שֶׁל בַּיְידֶּן עֲשׂוּיָה לְהָבִיא לְרוּחַ אַחֶרֶת. הַשְּׁאֵלָה הַגְּדוֹלָה הִיא אִם נִרְצֶה לְנַצֵּל אֶת הַהִזְדַּמְּנוּת. קִיר הַבַּרְזֶל  שֶׁל זַ'בּוֹטִינְסְקִי פָּעַל אֶת פְּעוּלָּתוֹ. בַּתְּפִיסָה שֶׁל הַמְּדִינַאי, מְיַיסֵּד בֵּיתָ"ר (שֶׁהוֹבִילָה לַהֲקָמַת חֵרוּת וְהַלִּיכּוּד), זֶה בְּדִיּוּק הַזְּמַן לְמַמֵּשׁ אֶת הַנִּיצָּחוֹן לְטוֹבַת חַיֵּי שָׁלוֹם וְסִיּוּם  הַשְּׁלִיטָה הַצְּבָאִית בָּעָם שֶׁלְּצִידֵּנוּ. זֶהוּ שִׂיא הַדִּילֵמָה: הַאִם אֲנַחְנוּ רוֹצִים לִהְיוֹת עֵשָׂיו אוֹ יַעֲקֹב?

הַדְּבָרִים תְּלוּיִים בָּנוּ. הָעֲמִידָה שֶׁל נְתַנְיָהוּ מוּל מִמְשַׁל אוֹבָּאמָה הוֹכִיחָה אֶת זֶה. לַפָלַשְׂתִּינִים יֵשׁ הָאַחְרָיוּת הַחֶלְקִית שֶׁלָּהֶם לַמַּצָּב הַנּוֹכְחִי, אַף שֶׁהֵם הַצַּד הַחַלָּשׁ בְּמִשְׁוְואַת הַסִּכְסוּךְ. הַשְּׁאֵלָה הִיא אִם אֲנַחְנוּ רוֹצִים לְנַצֵּל אֶת הַחוֹזְקָה שֶׁלָּנוּ כְּדֵי לְהָבִיא לְפִתְרוֹנוֹת.

בְּאוֹגוּסְט ‭ 2014‬ הִסְתַּיְּימָה מִלְחֶמֶת צוּק אֵיתָן. חֲבֵרַיי  וַאֲנִי, מִשְּׂדֵרוֹת וּמֵעוֹטֵף עַזָּה, חָזַרְנוּ לְבָתֵּינוּ וְהִתְחַלְנוּ לְשַׁקֵּם אֶת חַיֵּינוּ הַמְּטוּלְטָלִים. נְתַנְיָהוּ יָשַׁב לְצַד גַּנְץ  הָרַמַטְכָּ"ל וְיַעֲלוֹן שַׂר הַבִּיטָּחוֹן וְדִיבֵּר עַל "חַלּוֹן הִזְדַּמְּנוּיוֹת מְדִינִי", שֶׁנִּפְתַּח. לְמַרְבֵּה הַצַּעַר, הוּא לֹא נִיצֵּל אוֹתוֹ.

בְּכִירֵי מַעַרְכוֹת הַבִּיטָּחוֹן מַסְבִּירִים שׁוּב וָשׁוּב שֶׁכּוֹחַ מוּגְבָּל. שֶׁבַּסּוֹף כָּל נִיצָּחוֹן צְבָאִי זְמַנִּי חַיָּיב לָבוֹא מַהֲלָךְ מְדִינִי. שֶׁהַדֶּרֶךְ לְמַמֵּשׁ אֶת חֲזוֹן מְגִילַּת הָעַצְמָאוּת עוֹבֶרֶת  בְּפִיּוּס וּבְתִקְוָוה, כְּפִי שֶׁחֲתוּמִים שָׁם שְׂמֹאל וְיָמִין, דָּתִיִּים  וְחִילּוֹנִים: "אָנוּ מוֹשִׁיטִים יָד לְשָׁלוֹם וְלִשְׁכֵנוּת טוֹבָה לְכָל  הַמְּדִינוֹת הַשְּׁכֵנוֹת וְעַמֵּיהֶן, וְקוֹרְאִים לָהֶם לְשִׁתּוּף פְּעוּלָּה  וְעֶזְרָה הֲדָדִית... מְדִינַת יִשְׂרָאֵל מוּכָנָה לִתְרוֹם חֶלְקָהּ בְּמַאֲמָץ מְשׁוּתָּף לְקַדֵּם אֶת  הַמִּזְרָח הַתִּיכוֹן ".

לְחַיִּים עַל הַחֶרֶב יֵשׁ מְחִיר. הֵם מְבִיאִים אֶת הַיְּלָדִים שֶׁלָּנוּ לַקַּסְבּוֹת שֶׁל שְׁכֶם וְחֶבְרוֹן. הֵם הוֹפְכִים אֶת הַפִּתְרוֹנוֹת שֶׁלָּנוּ לַחֶבְרָה הַיִּשְׂרְאֵלִית פְּנִימָה לְ"אֲנַחְנוּ אוֹ הֵם". לְכוֹחָנוּת וּלְאַלִּימוּת. הַחֶרֶב צְרִיכָה לִהְיוֹת הַפִּתְרוֹן הָאַחֲרוֹן  בְּאַרְגַּז הַכֵּלִים שֶׁלָּנוּ, וְלֹא פִּתְרוֹן לַחַיִּים.

בְּחִירָתוֹ שֶׁל בַּיְידֶּן עֲשׂוּיָה לְהָבִיא רוּחַ אַחֶרֶת.  קִיר הַבַּרְזֶל שֶׁל זַ'בּוֹטִינְסְקִי פָּעַל אֶת פְּעוּלָּתוֹ. זֶה הַזְּמַן לְמַמֵּשׁ אֶת הַנִּיצָּחוֹן לְטוֹבַת חַיֵּי שָׁלוֹם וְסִיּוּם הַשְּׁלִיטָה הַצְּבָאִית בְּעַם שֶׁלְּצִידֵּנוּ
