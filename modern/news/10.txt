בְּ-2017 אֵין כִּמְעַט רַוָּוקים וְרַוָּוקוֹת שֶׁלֹּא נִמְצָאִים בַּאֲתָרֵי הַהֶיכֵּרוּיוֹת הַשּׁוֹנִים.
חֵלֶק מֵהָאֲנָשִׁים בֶּאֱמֶת מְנַסִּים לִמְצוֹא אַהֲבָה, אֲחֵרִים נִמְצָאִים שָׁם בְּעִיקָּר כְּדֵי לִמְצוֹא אַהֲבָה לְלַיְלָה אֶחָד.
הַבְּעָיוֹת מַתְחִילוֹת כְּשֶׁאֵין תֵּיאוּם צִיפִּיּוֹת בֵּין הָרְצוֹנוֹת הַשּׁוֹנִים, וּכְשֶׁגְּבָרִים - וְגַם נָשִׁים - מַרְשִׁים לְעַצְמָם לְהִתְנַהֵג בְּצוּרָה מְרוּשַּׁעַת לַאֲחֵרִים.
חוּץ מִזֶּה, אִי אֶפְשָׁר בֶּאֱמֶת לָדַעַת מַשֶּׁהוּ עַל הָאָדָם שֶׁאִיתּוֹ אֲנַחְנוּ יוֹצְאִים רַק עַל סְמַךְ מִסְפַּר הִתְכַּתְּבוּיוֹת סְפוּרוֹת בָּאֲתָר, וְכִמְעַט תָּמִיד מְדוּבָּר בְּדֵיְיט שֶׁהוּא לַחֲלוּטִין "חָתוּל בַּשַּׂק" - אוֹ שֶׁנְּהַנֶּה מִמֶּנּוּ, אוֹ שֶׁנִּמְצָא אֶת עַצְמֵנוּ מִתְפַּלְּצִים נוֹכַח הַיְּצוּר שֶׁחוֹלֵק אִיתָּנוּ שׁוּלְחָן.

הַמִּתְחַזֶּה

הַסִּיפּוּר שֶׁל יָעֵל (27) יְעַצְבֵּן וִיתַסְכֵּל רַוָּוקוֹת רַבּוֹת שֶׁלְּצַעֲרִי הָרַב חוֹוּ מַשֶּׁהוּ דּוֹמֶה עַל בְּשָׂרָן.
זֶה הִתְחִיל כְּשֶׁיָּעֵל נִתְקְלָה בְּבָחוּר שֶׁמָּצָא חֵן בְּעֵינֶיהָ בַּפֶיְיסְבּוּק, וְהִצִּיעָה לוֹ חֲבֵרוֹת.
"רַק אַחֲרֵי שֶׁהוּא אִישֵּׁר אוֹתִי הֵבַנְתִּי שֶׁיֵּשׁ בְּעָיָה קְטַנָּה - הוּא לֹא גָּר בְּיִשְׂרָאֵל.
לַמְרוֹת זֹאת, הִתְחַלְנוּ לְהִתְכַּתֵּב כִּמְעַט כָּל יוֹם, וּבְמֶשֶׁךְ חוֹדְשַׁיִים הוּא הִשְׁמִיעַ לִי הַבְטָחוֹת מִכָּאן וְעַד קָנָדָה שֶׁהוּא אוֹטוֹטוֹ מַגִּיעַ לָאָרֶץ, וְשֶׁסּוֹף-סוֹף נִפְגַּשׁ.
בְּסוֹפוֹ שֶׁל דָּבָר, הוּא בֶּאֱמֶת הִגִּיעַ, אֲבָל הוּא לֹא יָצַר קֶשֶׁר.
לְאַחַר זְמַן מָה, חָקַרְתִּי אֶת הָעִנְיָין וְגִילִּיתִי שֶׁכָּל הַזְּמַן הַזֶּה שֶׁדִּיבַּרְנוּ הָיְיתָה לוֹ וַעֲדַיִין יֵשׁ לוֹ חֶבְרָה, הוּא פָּשׁוּט דּוֹאֵג לְהַסְתִּיר אוֹתָהּ הֵיטֵב בָּרְשָׁתוֹת הַחֶבְרָתִיּוֹת.
שָׁלַחְתִּי לָהּ הוֹדָעָה מִפְּרוֹפִיל אַחֵר כְּדֵי לְסַפֵּר לָהּ עִם מָה הִיא חָיָה, וְכוּלִּי תִּקְוָוה שֶׁהִיא רָאֲתָה אֶת הַהוֹדָעָה שֶׁלִּי וְהֶחֱלִיטָה לַעֲזוֹב אוֹתוֹ.
אַגַּב, בַּזְּמַן שֶׁהוּא הָיָה בָּאָרֶץ, גַּם הוּא וְגַם אָחִיו הַתְּאוֹם (שֶׁהוּא נָשׂוּי פְּלוּס יַלְדָּה, וְעוֹד אֶחָד בַּדֶּרֶךְ) קָפְצוּ לִי בַּטִּינְדֶּר".

אֵין סֶקְס? אָז אֵין סֶרֶט

גַּם לְנוֹעָה (30) לֹא חֲסֵרוֹת חֲוָויוֹת שְׁלִילִיּוֹת מֵאֲתָרֵי הַהֶיכֵּרוּיוֹת לְמִינֵיהֶם: "פַּעַם נִפְגַּשְׁתִּי עִם מִישֶׁהוּ שֶׁהִכַּרְתִּי בְּאוּקְיֵי-קִיּוֹפִיד.
מְאוֹד נִמְשַׁכְתִּי אֵלָיו וּבַדֵּיְיט הָרִאשׁוֹן כְּבָר שָׁכַבְנוּ, אֶלָּא שֶׁאַחֲרֵי זֶה לֹא שָׁמַעְתִּי מִמֶּנּוּ יוֹתֵר.
עָבְרָה תְּקוּפָה שֶׁל חוֹדְשַׁיִים וּקְצָת, וּלְפֶתַע הוּא הִתְקַשֵּׁר וּבִיקֵּשׁ מִמֶּנִּי הִזְדַּמְּנוּת שְׁנִיָּיה.
לְאַחַר שֶׁהִבְהַרְתִּי לוֹ שֶׁאֲנִי לֹא מְחַפֶּשֶׂת סְטוּץ אֶלָּא זוּגִיּוּת, קָבַעְנוּ שׁוּב לְהִפָּגֵשׁ.
הוּא אָסַף אוֹתִי וְנָסַעְנוּ לִרְאוֹת סֶרֶט בְּרִאשׁוֹן לְצִיּוֹן.
כָּל הַדֶּרֶךְ הוּא הִתְנַצֵּל עַל הַפַּעַם הַהִיא שֶׁנֶּעֱלַם, וּמִצַּד שְׁנִי הוּא גַּם אָמַר לִי 'אֲבָל אַל תַּעֲשִׂי לִי דַּוְוקָא עַכְשָׁיו וּתְיַיבְּשִׁי אוֹתִי בְּסוֹף הָעֶרֶב'.

הֵשַׁבְתִּי לוֹ שֶׁאֲנִי לֹא עוֹשָׂה דַּוְוקָא, אֲבָל אֲנִי בְּמַּחְזוֹר אָז גַּם כָּכָה לֹא יִקְרֶה כְּלוּם.
בָּרֶגַע שֶׁהוּא שָׁמַע אֶת זֶה, הוּא הִתְאַכְזֵב וְהִתְחִיל לְהִתְנַהֵג בְּעַצְבָּנוּת.
כְּשֶׁהִגַּעְנוּ הוּא יָצָא רִאשׁוֹן וּבַקּוֹשִׁי הִסְתַּכֵּל עָלַיי.
שָׁאַלְתִּי אִם הוּא רוֹצֶה שֶׁנְּוַותֵּר עַל הַסֶּרֶט וְשֶׁהוּא יַחֲזִיר אוֹתִי הַבַּיְתָה, וְהוּא מִיָּד הִסְכִּים.
חָזַרְנוּ לָאוֹטוֹ וְכָל הַנְּסִיעָה שָׁתַקְתִּי בְּכַעַס.
הִרְגַּשְׁתִּי כְּמוֹ כְּלִי שֶׁאֶפְשָׁר לִזְרוֹק לַפַּח בְּתוֹם הַשִּׁימּוּשׁ.
הָעוּבְדָּה שֶׁהוּא לֹא רָצָה לְצַפּוֹת אִיתִּי בַּסֶּרֶט אִם לֹא מְצַפֶּה לוֹ סֶקְס בְּסוֹף הָעֶרֶב הָיְתָה מַשְׁפִּילָה.
חָזַרְתִּי הַבַּיְתָה בּוֹכָה וּכְאוּבָה".
 
"בָּחוּר אַחֵר שֶׁהִכַּרְתִּי בָּאִינְטֶרְנֵט נָטַשׁ אוֹתִי בְּבֵית הַקָּפֶה מִיָּד כְּשֶׁהִגַּעְתִּי.
הוֹשַׁטְתִּי לוֹ אֶת הַיָּד כְּדֵי שֶׁנַּכִּיר, אֲבָל הוּא יָשָׁר אָמַר: 'זֶה לֹא זֶה', סוֹבֵב אֶת הַגַּב וְהָלַךְ.
חָזַרְתִּי הַבַּיְתָה מְתוּסְכֶּלֶת וּמְיוֹאֶשֶׁת, וַאֲנִי חוֹשֶׁבֶת שֶׁאֶת הֶמְשֵׁךְ הָעֶרֶב הֶעֱבַרְתִּי בְּלִבְכּוֹת וְלִשְׂנוֹא אֶת כָּל הַגְּבָרִים".

לְהַתְחִיל עִם הַמֶּלְצָרִית מוּל הָעֵינַיִים שֶׁלָּךְ

סִיוָון (27) יָצְאָה לַמִּסְעָדָה עִם בָּחוּר שֶׁהִכִּירָה בְּאוֹתוֹ הָעֶרֶב מַמָּשׁ בַּאֲתַר הֶיכֵּרוּיוֹת.
הוּא הוֹתִיר עָלֶיהָ רוֹשֶׁם טוֹב, עַד שֶׁהִגִּיעָה הַמֶּלְצָרִית: "לֹא הֶאֱמַנְתִּי שֶׁזֶּה קוֹרֶה לִי", הִיא מְסַפֶּרֶת.
"הַמֶּלְצָרִית שֶׁשֵּׁירְתָה אוֹתָנוּ נִיסְּתָה לְהַתְחִיל עִם הַדֵּיְיט שֶׁלִּי, וְהוּא זָרַם אִיתָּהּ וּפְלִרְטֵט אִיתָּהּ מוּל הָעֵינַיִים שֶׁלִּי, וַאֲפִילּוּ לִיטֵּף אוֹתָהּ בַּיַּשְׁבָן! בָּרוּר שֶׁלֹּא נִפְגַּשְׁנוּ יוֹתֵר".

הָאִישָּׁה שֶׁגָּרְמָה לַמֶּלְצַר לִבְכּוֹת

הַסִּיפּוּרִים הַקָּשִׁים הַלָּלוּ אֵינָם נַחֲלָתָם שֶׁל נָשִׁים בִּלְבַד, וְגַם גְּבָרִים חוֹוִים יִיבּוּשִׁים וְהַעֲלָבוֹת מְהֻדְיָיטִים שֶׁלָּהֶם.
אוֹהַד (28) הִכִּיר מִישֶׁהִי בְּאֶחָד הָאֲתָרִים וְהֵם קָבְעוּ לֶאֱכוֹל בְּיַחַד אֲרוּחַת עֶרֶב.
"כְּבָר כְּשֶׁנִּכְנַסְנוּ לַמִּסְעָדָה הִיא הִתְחִילָה לְהִתְעַלֵּל בְּצֶוֶות הַמֶּלְצָרִים וּפָשׁוּט הִשְׁפִּילָה אוֹתָם, עַד שֶׁהַמֶּלְצַר הָיָה עַל סַף דְּמָעוֹת.
אֵיךְ שֶׁסִּייַּמְנוּ לֶאֱכוֹל לִיוִּויתִי אוֹתָהּ הַבַּיְתָה וְחָזַרְתִּי לַמִּסְעָדָה כְּדֵי לְהִתְנַצֵּל.
זֶה הָיָה מַלְחִיץ וְהָזוּי".

מִי זֹאת הַשִּׁיכּוֹרָה הַזֹּאת?

גַּם אִיתַּי (26) יָצָא לְדֵיְיט עִם בַּחוּרָה שֶׁהִכִּיר בְּאֶחָד הָאֲתָרִים, אֲבָל מַמָּשׁ לֹא צִיפָּה לְסוֹף כָּל כָּךְ הָזוּי: "יָצָאתִי אִיתָּהּ לְפָּאבּ בְּנַחֲלַת בִּנְיָמִין וְאַחֲרֵי שָׁלוֹשׁ כּוֹסוֹת יַיִן הִיא הִשְׁתַּכְּרָה לְגַמְרֵי, בַּדֵּיְיט הָרִאשׁוֹן! לָקַחְתִּי לָהּ אֶת הַמַּפְתְּחוֹת שֶׁל הָאוֹטוֹ וְנָהַגְתִּי אֵלַיי, כִּי לֹא הָיָה מַצָּב שֶׁהִיא תַּגִּיעַ הַבַּיְתָה בְּשָׁלוֹם.
תּוֹךְ כְּדֵי הַנְּסִיעָה הִיא הֵחֵלָּה לְהִתְפַּשֵּׁט, וְאַתְּ יוֹדַעַת מָה זֶה תְּנוּעָה בַּחֲמִישִׁי בָּעֶרֶב בְּתֵל אָבִיב! בְּמֶשֶׁךְ חֲצִי שָׁעָה שֶׁל רַמְזוֹרִים אֲדוּמִּים כָּל גֶּבֶר וְלֹא מְעַט נָשִׁים עָבְרוּ לָגוּר בְּתוֹךְ חַלּוֹן הָרֶכֶב שֶׁלָּנוּ! לְקִינּוּחַ, עַד שֶׁהֶעֱלִיתִי אוֹתָהּ הַבַּיְתָה, הִיא דָּאֲגָה לְרֶצֶף לִי אֶת כָּל הַשֵּׁירוּתִים בָּקִיא בְּגָוֶון בּוּרְגּוּנְדִּי, עַד שֶׁלְּבַסּוֹף נִרְדְּמָה! סִיּוּט!".

סִיּוּט בְּאוּלַם הַקּוֹלְנוֹעַ

חֵן (34) לֹא חָשַׁב שֶׁלָּצֵאת לַסֶּרֶט בַּקּוֹלְנוֹעַ בַּדֵּיְיט הָרִאשׁוֹן זֶה מַשֶּׁהוּ אִידֵיאָלִי, אֲבָל הוּא הִסְכִּים לִזְרוֹם עִם בַּקָּשָׁתָהּ שֶׁל בַּחוּרָה שֶׁהִכִּיר בְּפֶיְיסְבּוּק, וְלַחֲרוֹג מֵהֶרְגֵּלוֹ.
"זֶה הִתְחִיל בַּזֶּה שֶׁהִיא יִיבְּשָׁה אוֹתִי מִתַּחַת לַבַּיִת שֶׁלָּהּ אֵיזֶה חֲצִי שָׁעָה עִם הוֹדָעוֹת שֶׁל 'אֲנִי תֶּכֶף מַגִּיעָה' כָּל חָמֵשׁ דַּקּוֹת, אֲבָל זֶה לֹא נִגְמַר שָׁם.
כְּשֶׁהִגַּעְנוּ לַקּוֹלְנוֹעַ הָיָה תּוֹר בַּקּוּפּוֹת כִּי כְּבָר הָיִינוּ בְּאִיחוּר, וּבְלִי לְמַצְמֵץ הִיא תָּפְסָה אוֹתִי בַּיָּד וְעָקְפָה אֶת כָּל הַתּוֹר דֶּרֶךְ קָרוּסֶלָת הַיְּצִיאָה.
כּוּלָּם הִסְתַּכְּלוּ עָלֵינוּ הֲמוּמִים וְגַם אֲנִי הָיִיתִי בַּשּׁוּק, אֲבָל לֹא הָיָה לִי נָעִים לוֹמַר מַשֶּׁהוּ, אָז שָׁתַקְתִּי.

מִשָּׁם הָלַכְנוּ לַמִּזְנוֹן, לָקַחְנוּ פּוֹפְּקוֹרְן וְשׁוֹקוֹלָד בְּגוֹדֶל בֵּינוֹנִי, כָּזֶה שֶׁמִּשְׁפָּחוֹת פּוֹתְחוֹת בְּטִיּוּלִים.
הֵעַפְתִּי מַבָּט בַּשּׁוֹקוֹלָד כִּי עִנְיֵין אוֹתִי בְּאֵיזֶה טַעַם הוּא, וְתוֹךְ חָמֵשׁ שְׁנִיּוֹת הוּא כְּבָר מָצָא עַצְמוֹ בַּתִּיק שֶׁלָּהּ, וְכַמּוּבָן שֶׁהִיא לֹא הִצִּיעָה לִי אַף קוּבִּיָּה.
עַד שֶׁהִתְיַישַּׁבְנוּ הִיא גַּם הִסְפִּיקָה לְחַסֵּל כִּמְעַט אֶת כָּל הַפּוֹפְּקוֹרְן, וְעוֹד רָבָה עִם אֶחָד מֵעוֹבְדֵי הַמָּקוֹם.
לְבַסּוֹף הִתְיַישַּׁבְנוּ וּלְיָדֵינוּ יָשְׁבוּ יְלָדִים שֶׁדִּיבְּרוּ בֵּינֵיהֶם בִּלְחִישָׁה.
הִיא כַּמּוּבָן הִתְעַצְבְּנָה עֲלֵיהֶם בְּטַעֲנָה שֶׁהֵם מַפְרִיעִים לָהּ, וַאֲפִילּוּ אִייְּמָה לִקְרוֹא לְסַדְרָן אוֹ לְאִימָּא שֶׁלָּהֶם.
וְאִם כָּל זֶה לֹא מַסְפִּיק, לְמָחֳרָת הִיא סִימְּסָה לִי מַשֶּׁהוּ בְּסִגְנוֹן 'נֶהֱנֵיתִי מְאוֹד, אֲבָל לֹא מַתְאִים'.
לֹא יָכוֹלְתִּי יוֹתֵר וְסִימַּסְתִּי לָהּ בַּחֲזָרָה ג'וֹרָה שֶׁל קְלָלוֹת שֶׁלֹּא הָיוּ מְבַיְּישׁוֹת נַהָג מַשָּׂאִית שִׁיכּוֹר.
דַּי, כַּמָּה אֶפְשָׁר".

הַחוֹלָה הַמְּדוּמָּה

גַּם לִיאוֹר חָוָוה דֵּיְיט רִאשׁוֹן מֵהַגֵּיהִינּוֹם בְּעִקְבוֹת בַּחוּרָה מְעַט לֹא יַצִּיבָה.
"הָיְיתָה מִישֶׁהִי שֶׁיָּצָאתִי אִיתָּהּ לְדֵיְיט רִאשׁוֹן אַחֲרֵי שֶׁהִכַּרְנוּ בְּטִינְדֶּר, וּמִסְתַּבֵּר שֶׁהִיא לֹא אוֹהֶבֶת בְּצֵל.
הִיא הֶחְלִיטָה לְהַזְמִין סָלָט, וְאָמְרָה לַמֶּלְצָרִית שֶׁהִיא אָלֶרְגִּית לַבָּצָל - לַמְרוֹת שֶׁהִיא לֹא בֶּאֱמֶת אָלֶרְגִּית, רַק כִּי הִיא חָשְׁבָה שֶׁכָּכָה בָּטוּחַ לֹא יִהְיֶה זֵכֶר לְבָצָל בַּסָּלָט שֶׁלָּהּ.
בַּסּוֹף הָיָה שָׁם מַמָּשׁ קְצָת בָּצָל, אֲבָל הִיא הִרְגִּישָׁה מְחוּיֶּבֶת לַעֲשׂוֹת סְצֵנָה, לַמְרוֹת שֶׁשּׁוּב - הִיא לֹא בֶּאֱמֶת אָלֶרְגִּית לְזֶה! הִיא הִתְחִילָה לִצְעוֹק שָׁם שֶׁהֵם יָכְלוּ לַהֲרוֹג אוֹתָהּ, וַאֲנִי בֵּינְתַיִים לֹא יָדַעְתִּי אֵיפֹה לִקְבּוֹר אֶת עַצְמִי".

לְסִיּוּם, שְׁנֵי סִיפּוּרִים חִיּוּבִיִּים:

מֵיטַל (30) מָצְאָה אֶת הָאַהֲבָה שֶׁלָּהּ בְּטִינְדֶּר, אֲבָל סִיפּוּר הַהֶיכֵּרוּת שֶׁלָּהּ יַפְתִּיעַ אֲפִילּוּ אֶתְכֶם: "לִפְנֵי שְׁנָתַיִים הוֹרַדְתִּי טִינְדֶּר לָרִאשׁוֹנָה בְּחַיַּי.
יָצָא לִי לְשׂוֹחֵחַ עִם מִסְפַּר בַּחוּרִים שֶׁלֹּא זִיהִיתִי בָּהֶם פּוֹטֶנְצְיָאל, אֲבָל בַּיּוֹם הַשֵּׁנִי שֶׁבּוֹ שָׁהִיתִי בָּאַפְּלִיקַצְיָה, אֲנִי וַחֲבֵרָה טוֹבָה שֶׁלִּי יָשַׁבְנוּ בְּיַחַד, וְתוֹךְ כְּדֵי שִׂיחָה גִּילִּינוּ שֶׁשָּׁתִינוּ מִתְכַּתְּבוֹת עִם אוֹתוֹ הַבָּחוּר.
הֶחְלַטְנוּ לְסַפֵּר לוֹ שֶׁשָּׁתִינוּ חֲבֵרוֹת וְהִצַּעְנוּ לוֹ לְהִיפָּגֵשׁ שְׁלוֹשְׁתֵּנוּ.
הוּא הִגִּיעַ וְהַלַּיְלָה הָיָה נֶהֱדָר.
יָשַׁבְנוּ בְּגִינָהּ וְלֹא הִצְלַחְנוּ לְהַפְסִיק לִצְחוֹק.
לְמָחֳרָת הִמְשַׁכְנוּ לְשׂוֹחֵחַ שְׁלוֹשְׁתֵּנוּ בַּקְּבוּצָה שֶׁפָּתַחְנוּ בִּמְיוּחָד בְּוַוטְסָאפּ, עַד שֶׁהֶחְלַטְתִּי שֶׁאֲנִי רוֹצָה לְהַכִּיר אוֹתוֹ קְצָת יוֹתֵר.
חֲבֶרְתִּי פִרְגְּנָה לִשְׁנֵינוּ, וּמֵאָז וְעַד הַיּוֹם אֲנַחְנוּ בְּנֵי זוּג, וּמְאוֹד מוּפְתָּעִים מֵהַדֶּרֶךְ שֶׁבָּהּ הִכַּרְנוּ".
 
גַּם לְיָעֵל (33) יֵשׁ סִיפּוּר חִיּוּבִי לְשֵׁם שִׁינּוּי, שֶׁיַּזְכִּיר לְכוּלָּנוּ שֶׁלֹּא הַכּוֹל רַע בָּאֲתָרִים הַלָּלוּ: "יוֹם אֶחָד הִשְׁאַרְתִּי אֶת הַאַיְיפוֹן שֶׁלִּי בְּטָעוּת לְלֹא הַשְׁגָּחָה, וְעָשׂוּ לִי פִּיגּוּעַ טִינְדֶּר - חֲבֵרִים שֶׁלִּי נִכְנְסוּ לְאַפְּלִיקַצְיָה וְהֶחֱלִיטוּ שֶׁהֵם עוֹשִׂים לְיַייֵּק וְלֵב לְכָל הַגְּבָרִים שֶׁהָיוּ שָׁם! אֶחָד מֵהֶם הָיָה בָּחוּר בְּשֵׁם יוֹאָב שֶׁגַּם הִתְחִיל לְדַבֵּר אִיתִּי בְּצָ'אט.
בַּסּוֹף יָצָאנוּ, אֲבָל הַדֵּיְיט הָיָה נוֹרָאִי.
הוּא אָמַר לְמָשָׁל שֶׁהוּא בַּחַיִּים לֹא יִתְחַתֵּן כְּהַצְהָרָה.
לֹא חָשַׁבְתִּי שֶׁאֶתֵּן לוֹ צַ'אנְס נוֹסָף בִּכְלָל, אֲבָל הַשּׁוּתָּפוֹת שֶׁלִּי לַדִּירָה שִׁכְנְעוּ אוֹתִי לָצֵאת אִיתּוֹ לְדֵיְיט שֵׁנִי, וּמֵאָז אֲנַחְנוּ בְּיַחַד, כְּבָר שָׁלוֹשׁ שָׁנִים".
