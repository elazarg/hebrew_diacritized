מִשְׁלַחַת בִּיטְחוֹנִית שֶׁל חַמַאס הִגִּיעָה הַיּוֹם (שַׁבָּת) לְקָהִיר, וּצְפוּיָה לִפְגּוֹשׁ בְּכִירִים מִצְרַיִם וְלָדוּן בְּנוֹשְׂאִים בִּיטְחוֹנִיִּים.

הַמִּשְׁלַחַת, שֶׁהִגִּיעָה לְמִצְרַיִם מֵרְצוּעַת עַזָּה דֶּרֶךְ מַעֲבַר רָפִיחַ, תִּפְגּוֹשׁ אֶת הַבְּכִירִים בַּהֶמְשֵׁךְ לִפְגִישׁוֹת שֶׁנֶּעֶרְכוּ עִם מְשַׁלַּחַת בְּרָאשׁוּת סְגַן מַנְהִיג הָאִרְגּוּן אִיסְמַאעִיל הָנִיָּיה לִפְנֵי שָׁבוּעַ וָחֵצִי.
גּוֹרְמִים בְּמִצְרַיִם אָמְרוּ לָעִיתּוֹן "אֶל-עָרָבִּי אֶל-גְּ'דִיד" כִּי הַצַּד הַמִּצְרִי בִּיקֵּשׁ מֵהַמִּשְׁלַחַת לִפְתּוֹחַ אֶת סוּגְיַית הַנֶּעֱדָרִים הַיִּשְׂרְאֵלִים הַמּוּחְזָקִים בִּרְצוּעַת עַזָּה, וְלָדוּן בַּנּוֹשֵׂא.
הַגּוֹרְמִים הֶעֱרִיכוּ כִּי יִיתָּכֵן שֶׁהַבַּקָּשָׁה הַמִּצְרִית נֶעֶשְׂתָה בְּעִקְבוֹת פְּנִיָּיה מִיִּשְׂרָאֵל.

בְּיוּלִי 2015 הוּתַּר לְפִרְסוּם כִּי אֶבְרָה מַנְגִּיסְטוּ, בֵּן 29 מֵאַשְׁקְלוֹן, מוּחְזָק בָּרְצוּעָה לְאַחַר שֶׁחָצָה אֶת הַגְּבוּל בְּסֶפְּטֶמְבֶּר 2014.
שְׁלוֹשָׁה חוֹדָשִׁים לְאַחַר מַנְגִּיסְטוּ, חָצָה אֶת הַגְּבוּל לָרְצוּעָה הִישָּׂאם א-סַיֵּיד, בֵּדוּאִי תּוֹשַׁב הַדָּרוֹם הַמּוּכָּר כְּלוֹקֶה בְּנַפְשׁוֹ.

בְּיוּלִי הָאַחֲרוֹן חָצָה אֶת הַגְּבוּל לְכִיוּוּן הָרְצוּעָה אֶזְרָח יִשְׂרְאֵלִי נוֹסָף, גּ'וּמְעָה אִבְּרָאהִים אַבּוּ עָנַיְמֶה הַמִּתְגּוֹרֵר בִּכְפָר חָאשֶׁם זָאנָה שֶׁבַּפְּזוּרָה הַבֵּדוּאִית.
בְּנוֹסָף מוּחְזָקוֹת בָּרְצוּעָה גּוּפוֹתֵיהֶם שֶׁל הַחַיָּילִים אוֹרוֹן שָׁאוּל וְהָדָר גּוֹלְדִּין, שֶׁנֶּהֶרְגוּ בְּמִבְצַע "צוּק אֵיתָן".

עַל פִּי דִּיוּוּחִים בַּמִּשְׁלַחַת נִמְצָא גַּם סְגַן רֹאשׁ הַזְּרוֹעַ הַצְּבָאִית שֶׁל חַמַאס, מָרוּאָן עִיסַא.
הַצְּדָדִים צְפוּיִים לְנַהֵל שִׂיחוֹת עַל נִיסְּיוֹנוֹתֶיהָ שֶׁל מִצְרַיִם לְהַעֲצִים אֶת הַלַּחַץ עַל שְׁלוּחַת דְּאַעָשׁ בְּסִינַי שֶׁפּוֹעֶלֶת בֵּין הַיֶּתֶר סָמוּךְ לִגְבוּל רְצוּעַת עַזָּה, וְנֶהֱנֵית בַּשָּׁנִים הָאַחֲרוֹנוֹת מִתְּמִיכַת הַזְּרוֹעַ הַצְּבָאִית שֶׁל חַמַאס.
הַצְּדָדִים צְפוּיִים לָדוּן גַּם בְּחִיזּוּק הַפְּרִיסָה שֶׁל חַמַאס בְּאֵזוֹר הַגְּבוּל עִם מִצְרַיִם.

בְּקָהִיר כְּבָר הֵחֵלּוּ לְהָקֵל עַל הַלַּחַץ בָּרְצוּעָה בְּאֶמְצָעוּת פְּתִיחַת מַעֲבַר רָפִיחַ בְּאוֹפֶן סָדִיר יוֹתֵר.
מִצְרַיִם מְאַפְשֶׁרֶת לָאַחֲרוֹנָה מֵעֵבֶר שֶׁל מִסְפָּר גָּדוֹל יוֹתֵר שֶׁל בְּנֵי אָדָם בֵּין עַזָּה לְמִצְרַיִם, וְכֵן הַכְּנֶסֶת סְחוֹרוֹת, חוֹמְרֵי בְּנִיָּיה וּמְכוֹנִיּוֹת.
