גַּם אִם אֲנַחְנוּ מַמָּשׁ מַקְפִּידִים עַל תַּפְרִיט בָּרִיא כָּל הַשָּׁנָה – לְכוּלָּנוּ יֵשׁ נְפִילוֹת קְטַנּוֹת מִדֵּי פַּעַם.
אֵצֶל חֵלֶק זוֹ אֲרוּחַת שִׁישִּׁי, אֵצֶל אֲחֵרִים זֶה חֲגִיגוֹת הַסִּילְבֶסְטֵר וְאֵצֶל חֶלְקֵינוּ זֶה בְּעֵרֶךְ כָּל חַג.
אֲנַחְנוּ כְּבָר יוֹדְעִים אֵילּוּ מָנוֹת עֲדִיפוּת לָנוּ וּמָה הַכַּמּוּת שֶׁאֲנַחְנוּ צְרִיכִים לֶאֱכוֹל, אֲבָל מָה לַעֲשׂוֹת שֶׁלִּפְעָמִים יֵשׁ לָנוּ מַצַּב רוּחַ לַחֲגִיגָה (גַּם אִם אֵין סִיבָּה מְסוּיֶּמֶת) וַאֲנַחְנוּ מִתְפַּנְּקִים קְצָת יֶתֶר עַל הַמִּידָּה.
אָז לִפְנֵי שֶׁאַתֶּם מִתְעַצְבְּנִים עַל עַצְמְכֶם עַל הֶרֶס הַדִּיאֵטָה, הִנֵּה כַּמָּה טִיפִּים שֶׁיַּעַזְרוּ לָכֶם לְהִתְאוֹשֵׁשׁ בִּמְהִירוּת וְלַחֲזוֹר לַדִּיאֵטָה בְּקַלּוֹת.

1. לְהִיפָּטֵר מֵהָאַשְׁמָה
הַדָּבָר הָרִאשׁוֹן שֶׁאַתֶּם צְרִיכִים לַעֲשׂוֹת אַחֲרֵי שֶׁשְּׁבַרְתֶּם אֶת הַדִּיאֵטָה וְנִשְׁנַשְׁתֶּם קְצָת יוֹתֵר מִדַּי מַמְתַּקִּים – זֶה לֹא לַעֲשׂוֹת לְעַצְמְכֶם חַיִּים קָשִׁים עַל זֶה.
זָכְרוּ שֶׁמָּה שֶׁבֶּאֱמֶת מַשְׁפִּיעַ עַל הַבְּרִיאוּת שֶׁלָּכֶם בַּטְּוָוח הָרָחוֹק, זֶה מָה שֶׁאַתֶּם אוֹכְלִים מִדֵּי יוֹם וְלֹא שַׂקִּית אֶקְסְטְרָה שֶׁל צִ'יפְּס שֶׁאֲכַלְתֶּם פַּעַם אַחַת מוּל הַטֶּלֶוִויזְיָה.
אִם קָרָה שֶׁאֲכַלְתֶּם קְצָת יוֹתֵר מִדַּי בְּמֶשֶׁךְ כַּמָּה יָמִים סְפוּרִים, אֵין סִיבָּה שֶׁתַּרְגִּישׁוּ רַע לְגַבֵּי זֶה.
לְהֵיפֶךְ, תֵּיהָנוּ מִזֶּה לְלֹא רִגְשׁוֹת אָשָׁם.

2. לֹא לְדַלֵּג עַל אֲרוּחוֹת
"אָכַלְתִּי עַכְשָׁיו עוּגָה, אָז פָּשׁוּט אֲדַלֵּג עַל אֲרוּחַת הָעֶרֶב" – מַכִּירִים אֶת הַמִּשְׁפָּט הַזֶּה? הָרֶגֶל כְּזֶה רַק יִגְרוֹם לָכֶם לִהְיוֹת רְעֵבִים יוֹתֵר וְאוּמְלָלִים יוֹתֵר, מָה שֶׁיִּגְרוֹם לָכֶם לֶאֱכוֹל יוֹתֵר בְּסוֹפוֹ שֶׁל דָּבָר.
בִּמְקוֹם, אִכְלוּ אֲרוּחוֹת קְטַנּוֹת וּקְלִילוֹת, עִם הַרְבֵּה פֵּירוֹת, יְרָקוֹת וְחֶלְבּוֹן רָזֶה, כְּמוֹ דָּג, עוֹף, שְׁעוּעִית אוֹ טוֹפוּ.

3. לֹא לַעֲלוֹת עַל הַמִּשְׁקָל
אִם אֲכַלְתֶּם לָאַחֲרוֹנָה קְצָת יוֹתֵר מִדַּי, כְּכָל הַנִּרְאֶה שֶׁגַּם הַמִּסְפָּר עַל הַמִּשְׁקָל יַעֲלֶה.
אֲבָל זֶה לֹא בְּהֶכְרֵחַ בִּגְלַל שֶׁבֶּאֱמֶת עֲלִיתֶם בַּמִּשְׁקָל, אֶלָּא כֵּיוָון שֶׁהַגּוּף שֶׁלָּכֶם אָגַר נוֹזְלִים מֵהַמֶּלַח הָעוֹדֶף שֶׁצְּרַכְתֶּם.
הִתְרַחֲקוּ מִמִּשְׁקָלִים וּבִמְקוֹם זֹאת הִשְׁקִיעוּ אֶת הָאֵנֶרְגִּיָּה שֶׁלָּכֶם בְּמָה שֶׁעֲלֵיכֶם לַעֲשׂוֹת כְּדֵי לַחֲזוֹר לֶאֱכוֹל בָּרִיא.

4. לְאַזֵּן בַּחֲזָרָה אֶת הַמֵּעַיִים
כַּמּוּת מוּגְזֶמֶת שֶׁל סוּכָּר וְאַלְכּוֹהוֹל עֲלוּלָה לִהְיוֹת חֲדָשׁוֹת רָעוֹת לְמַעֲרֶכֶת הָעִיכּוּל שֶׁלָּכֶם.
מָזוֹן זֶה מְשָׁרֵת אֶת הַ"חַיְידַּקִּים הָרָעִים" שֶׁבְּקֵיבָה שֶׁלָּכֶם וְלָכֵן יָכוֹל לִגְרוֹם לִנְפִיחוּת וְלִבְעָיוֹת בַּבֶּטֶן.
נְטִילַת פְּרוֹבִּיוֹטִיקָה יְכוֹלָה לְסַיֵּיעַ לְהַחֲזִיר אֶת הָאִיזּוּן בְּמַעֲרֶכֶת הָעִיכּוּל, עַל יְדֵי כָּךְ שֶׁהִיא מְסַפֶּקֶת מָזוֹן לְ"חַיְידַּקִּים הַטּוֹבִים" בַּקֵּיבָה שֶׁלָּכֶם.
גַּם צְרִיכַת מְזוֹנוֹת פָּרָה-בִּיוֹטִיִּים כְּמוֹ בָּצָל, שׁוּם, אַרְטִישׁוֹק וּבָנָנָה.

5. לֶאֱהוֹב אֶת הַכָּבֵד שֶׁלָּכֶם
הַכָּבֵד אַחְרַאי לְעִיכּוּל שׁוּמָּנִים וְנִיקּוּי הַגּוּף מֵרְעָלִים וְלָכֵן בִּתְקוּפוֹת שֶׁבָּהֶם אַתֶּם אוֹכְלִים קְצָת יוֹתֵר – יִיתָּכֵן שֶׁגַּם הַכָּבֵד שֶׁלָּכֶם יִהְיֶה מוּתָּשׁ מֵהָרָגִיל.
תְּמִיכָה בְּתִפְקוּד הַכָּבֵד, יְכוֹלָה לְסַיֵּיעַ מַשְׁמָעוּתִית לַגּוּף שֶׁלָּכֶם לְהִתְנַקּוֹת לְאַחַר אֲכִילַת יֶתֶר.
אַתֶּם יְכוֹלִים לַעֲשׂוֹת אֶת זֶה בְּאֶמְצָעוּת מַאֲכָלִים הַתּוֹרְמִים לִבְרִיאוּת הַכָּבֵד כְּמוֹ יְרָקוֹת מִמִּשְׁפַּחַת הַמַּצְלִיבִים (בְּרוֹקוֹלִי, כְּרוּבִית, כְּרוֹב וְקֵיְיל), לִימוֹן, סֶלֶק, בֵּיצִים וְכוּרְכּוּם.

6. לִשְׁתּוֹת הַרְבֵּה מַיִם
אִם יָצָא לָכֶם לָאַחֲרוֹנָה לַחְגּוֹג עִם הֲמוֹן שְׁתִיָּיה וְאוֹכֵל, בֶּטַח שַׂמְתֶּם לֵב שֶׁאַתֶּם נוֹטִים פָּחוֹת לְהַקְפִּיד עַל שְׁתִיַּית מַיִם.
הִתְיַיבְּשׁוּת יְכוֹלָה לְהַשְׁפִּיעַ עַל הַגּוּף שֶׁלָּכֶם בְּמִגְוַון דְּרָכִים, בֵּינֵיהֶן מַצַּב רוּחַ יָרוּד, רָמוֹת אֵנֶרְגִּיָּה נְמוּכוֹת וּמַעֲרֶכֶת עִיכּוּל חַסְרַת מֶרֶץ.
חָשׁוּב גַּם לִשְׁתּוֹת מַיִם כְּדֵי לְנַקּוֹת אֶת הַגּוּף מֵרְעָלִים וְלִתְמוֹךְ בַּכָּבֵד.
