צַהַ"ל תָּקַף הַלַּיְלָה (בֵּין רִאשׁוֹן לְשֵׁנִי) חוּלְיָה שֶׁל אַרְבָּעָה מְחַבְּלִים שֶׁהִנִּיחוּ מִטְעֲנֵי חַבָּלָה עַל קַו הַגְּבוּל עִם סוּרְיָה. הַחוּלְיָה מָנְתָה אַרְבָּעָה מְחַבְּלִים, לְפָחוֹת חָמוּשׁ אֶחָד וְסוּכְּלָה בְּעֵת שֶׁנִּיסָּה לְהַנִּיחַ אֶת הַמִּטְעָן. הָאַרְבָּעָה כְּכָל הַנִּרְאֶה נֶהֶרְגוּ. בְּצַהַ"ל צִייְּנוּ כִּי לֹא הָיוּ נִפְגָּעִים לְכוֹחוֹתֵינוּ. הַתַּקְרִית אֵירְעָה בְּמֶרְחָב הַמּוּכָּר לָאוֹיֵב, מֵאַחַר וּבֶעָבָר פָּעַל בּוֹ בֵּית חוֹלִים שָׂדֶה בִּתְקוּפַת "שְׁכֵנוֹת טוֹבָה".
מִדּוֹבֵר צַהַ"ל נִמְסַר כִּי "כּוֹחַ מְיוּחָד שֶׁבִּיצֵּעַ מַאֲרַב בְּאֵזוֹר דְּרוֹם רָמַת הַגּוֹלָן, סָמוּךְ לְמוּצָּב צְבָאִי, זִיהָה חוּלְיָה שֶׁל מִסְפַּר מְחַבְּלִים מְנִיחִים מִטְעָנִים בְּקִרְבַת גֶּדֶר הַגְּבוּל. הַכּוֹחַ וּכְלֵי טַיִס שֶׁל צַהַ"ל בִּיצְּעוּ יְרִי בְּמַקְבִּיל עַל הַחוּלְיָה שֶׁמָּנְתָה אַרְבָּעָה מְחַבְּלִים וּפָגְעוּ בָּהֶם". 
בְּצַהַ"ל הִדְגִּישׁוּ: "כּוֹחוֹת פִּיקּוּד הַצָּפוֹן נִמְצָאִים בִּדְרִיכוֹת גְּבוֹהָה לְכָל תַּרְחִישׁ, צַהַ"ל רוֹאֶה בַּמִּשְׁטַר הַסּוּרִי אַחְרַאי לְכָל פְּעוּלָּה הַנַּעֲשֵׂית בְּשִׁטְחוֹ וְלֹא יְאַפְשֵׁר כָּל פְּגִיעָה בָּרִיבּוֹנוּת שֶׁל מְדִינַת יִשְׂרָאֵל".
הַכּוֹחַ הַמְּיוּחָד שֶׁאָרַב לִמְחַבְּלִים, זִיהָה אוֹתָם, יָרָה וּפָגַע בָּהֶם הוּא מִיְּחִידַת מַגְלָּן מֵחֲטִיבַת הַקּוֹמַנְדּוֹ, שֶׁפְּרוּסָה מִזֶּה כַּעֲשָׂרָה יָמִים בַּמַּאֲרַבִים עַל כְּלַל אֶמְצַעֶיהָ הַיִּיעוּדִיִּים לְמִלְחָמָה, בְּכָל גְּזָרוֹת גְּבוּל הַצָּפוֹן. הַתַּקְרִית אֵירְעָה לְיַד נַחַל רוֹקָאד, אֵזוֹר בּוֹ יֵשׁ שְׁטָחִים מֵתִים רַבִּים שֶׁמְּנּוּצְּלִים כִּנְתִיבֵי חֲדִירָה לְצַד הַיִּשְׂרְאֵלִי. בָּאֵזוֹר הוּקְמָה בַּשְּׁנָתַיִים הָאַחֲרוֹנוֹת תַּשְׁתִּית שֶׁל חִיזְבַּאלְלָה.
הַזִּיהוּי הָרִאשׁוֹנִי הָיָה עַל יְדֵי תַּצְפִּיתָנִיּוֹת מִגְּדּוּד עַיִט 595 שֶׁל אוּגְדַּת הַגּוֹלָן, שֶׁהִבְחִינוּ בְּהִתְקָרְבוּיוֹת חֲשׁוּדוֹת לְמֶרְחַב הַגְּבוּל. צַהַ"ל מִיקֵּם בְּאֵזוֹר כּוֹחוֹת רַבִּים לְאַחַר שֶׁזּוֹהוּ בּוֹ בַּשָּׁבוּעַ הָאַחֲרוֹן הִתְקָרְבוּיוֹת חֲשׁוּדוֹת וְאִיסּוּף מוּגְבָּר שֶׁל מוֹדִיעִין עַל יְדֵי "רוֹעֵי צֹאן" סוּרִים.

הַמְּתִיחוּת בִּגְּבוּל הַצָּפוֹן הֵחֵלָּה לְאַחַר הַתְּקִיפָה הַמְּיוּחֶסֶת לְיִשְׂרָאֵל בְּדַמֶּשֶׂק לִפְנֵי כִּשְׁבוּעַיִים, שֶׁבְּמַהֲלָכָהּ נֶהֱרַג פָּעִיל חִיזְבַּאלְלָה. בָּאִרְגּוּן הִבְטִיחוּ לִנְקוֹם וּמֵאָז הֶעֱלָה צַהַ"ל אֶת רָמַת הַכּוֹנְנוּת בִּגְבוּל לְבָנוֹן. כַּמָּה יָמִים לְאַחַר הַתְּקִיפָה בְּדַמֶּשֶׂק, חוּלְיַית חִיזְבַּאלְלָה חֲדְרָה לְשֶׁטַח יִשְׂרָאֵל בְּמֶרְחַב הַר דֹּב שֶׁבְּגְּבוּל לְבָנוֹן, וְנִמְלְטָה מֵהַמָּקוֹם לְאַחַר שֶׁצַּהַ"ל פָּתַח בִּירִי לְעֵבֶר הַכּוֹחַ. חִיזְבַּאלְלָה הִכְחִישׁ שֶׁהָיָה מְעוֹרָב בַּתַּקְרִית וְטָעַן כִּי "יִשְׂרָאֵל מְנַסָּה לְיַיצֵּר לְעַצְמָהּ הֶישֵּׂגִים".

דּוֹבֵר צַהַ"ל תָּאָ"ל הָדִי זִילְבֶּרְמָן הִבְהִיר לְאַחַר הַתַּקְרִית כִּי "יָמִים מְתוּחִים לְפָנֵינוּ" וְאָמַר: "כּוֹחוֹת צַהַ"ל בְּמוּכָנוּת וַעֲרוּכִים לְהֶמְשֵׁךְ כְּכָל שֶׁיִּידָּרֵשׁ. לֹא נְאַפְשֵׁר פְּגִיעָה בְּבִיטָּחוֹן תּוֹשָׁבֵי מְדִינַת יִשְׂרָאֵל וּבְרִיבּוֹנוּתָהּ".
