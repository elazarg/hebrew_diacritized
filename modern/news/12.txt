יָנוּאָר עַד אַפְּרִיל: כְּבָר בִּתְחִילַּת הַשָּׁנָה צְפוּיִים רְגָעִים רוֹמַנְטִיִּים בִּסְבִיבָה שׁוֹפַעַת בְּאֵנֶרְגִּיּוֹת מִיְּסוֹד הַמַּיִם.
אֵלֶּה יָמִים חֲלוֹמִיִּים עֲבוּר אֵלֶּה מִכֶּם הַמְּבַקְּשִׁים לְקַיֵּים עִסְקִי נַדְלָ"ן, כְּמוֹ גַּם לְמִי שֶׁעוֹסְקִים בִּתְחוּמִים הַקְּשׁוּרִים בָּעִיצּוּב וּבְפִיסּוּל.

זוֹהִי תְּקוּפָה אוּלְטִימָטִיבִית לִנְסִיעוֹת, מַסָּעוֹת וְטִיּוּלִים.
חוֹדֶשׁ מַרְס מְבַשֵּׂר עַל תְּחִילָּתוֹ שֶׁל מַזַּל טָלֶה יַחַד עִם תְּחִילָּתָהּ שֶׁל שָׁנָה אַסְטְרוֹלוֹגִית חֲדָשָׁה, וְיַחַד עִם חֲגִיגוֹת הָאָבִיב שֶׁתָּמִיד מֵיטִיבוֹת עִם חַיֵּי הָאַהֲבָה שֶׁלָּכֶם.

בְּמַקְבִּיל, יֵשׁ לְהִישָּׁמֵר מִפְּנֵי נְטִיּוֹת כְּפִיָּיתִיּוֹת כְּלַפֵּי בְּנֵי הַזּוּג.
הֵיבֶּטִים הַרְמוֹנִיִּים מֵהַכּוֹכָבִים וֶנוּס וְנֶפְּטוּן רוֹמְזִים עַל קְשָׁרִים יוֹצְאֵי דּוֹפֶן עִם יְלִידֵי הַמַּזָּלוֹת מֹאזְנַיִים וְדָגִים.

מַאי עַד אוֹגוּסְט: זֶה הַזְּמַן לְהַשִּׂיג אֶת הָרָצוּי, וְלֹא לְהִתְפַּשֵּׁר עִם הַמָּצוּי.
צְפוּיִים אֲהַבְהַבִּים רוֹמַנְטִיִּים וּפִיתּוּיִים שֶׁבֶּעָבָר הִשְׁתַּדַּלְתֶּם לְהִתְרַחֵק מֵהֶם.
אַתֶּם מְגַלִּים פְּתִיחוּת לְהַרְפַּתְקָנוּת, וְיִיתָּכְנוּ מִפְגָּשִׁים עִסְקִיִּים עִם הַמּוֹבִילִים שֶׁבַּחֲבוּרָה.

מַעֲמַדְכֶם הַקַּרְיֵירִיסְטִי יְדַרְבֵּן אֶתְכֶם לְהַרְשִׁים אֶת הָאֲחֵרִים בְּכָל הַנּוֹגֵעַ לִיכוֹלוֹתֵיכֶם וּלְכִישְׁרוֹנוֹתֵיכֶם.
בִּתְקוּפָה זוֹ הָרָקִיעַ מַבְטִיחַ נִיסִּים וְנִפְלָאוֹת לְאֵלֶּה מִכֶּם שֶׁמְּבַקְּשִׁים לְמַסֵּד אֶת אַהֲבָתָם אוֹ בְּפַשְׁטוּת, לְהִתְחַתֵּן! הַנּוּמֵרוֹלוֹגְיָה מְבַשֶּׂרֶת עַל בְּנֵי זוּג אֲשֶׁר צֵירוּף אוֹתִיּוֹת שְׁמָם אוֹ סְפָרוֹת תַּאֲרִיךְ לֵידָתָם, מִסְתַּכֵּם בְּ-4.

אִם בִּרְצוֹנְכֶם לְבַצֵּעַ שִׁינּוּיִים קַרְיֵירִיסְטִיִּים, הֲרֵי שֶׁאַתֶּם נִמְצָאִים בִּתְקוּפָה טוֹבָה לְכָךְ.

סֶפְּטֶמְבֶּר עַד דֵּצֶמְבֶּר: סוֹדוֹת עִסְקִיִּים מִן הֶעָבָר עֲלוּלִים 'לְהִתְפּוֹצֵץ' בְּעִקְבוֹת מְרִיבָה קַטְנוּנִית.
הַכּוֹכָבִים צוֹפִים יְחָסִים יוֹצְאֵי דּוֹפֶן עִם יַלְדֵי מַזָּל בְּתוּלָה וְהִתְלַבְּטוּת בֵּין קֶשֶׁר יַצִּיב אַךְ שִׁגְרָתִי לְבֵין קֶשֶׁר מַסְעִיר אַךְ מְסוּכָּן.

מַצַּב הַכּוֹכָבִים רוֹמֵז עַל הַצְלָחוֹת עִסְקִיּוֹת וּפִינַנְסִיּוֹת, אַךְ כְּדַאי לְהִימָּנַע מִמַּצָּב אֲשֶׁר בּוֹ אַתֶּם עוֹשִׂים יָד אַחַת עִם אָדָם כָּלְשֶׁהוּ, נֶגֶד אָדָם אַחֵר.
בִּתְקוּפָה זוֹ אַתֶּם חוֹוִים יְחָסִים טוֹבִים וַחֲמִימִים עִם יְלִידֵי מַזַּל תְּאוֹמִים, כְּמוֹ גַּם עִם פַּרְטְנֶרִים אֲשֶׁר צֵירוּף אוֹתִיּוֹת שְׁמָם אוֹ סְכוּם סְפָרוֹת תַּאֲרִיךְ לֵידָתָם – מִסְתַּכֵּם בְּ-5.

הָאֵנֶרְגִּיּוֹת שֶׁל הַמִּסְפָּר חָמֵשׁ מְאַפְשְׁרוֹת קְפִיצַת מַדְרֵגָה עֲבוּר אֵלֶּה מִכֶּם הָעוֹסְקִים בְּתִקְשׁוֹרֶת, שִׁיוּוּק, פִּרְסוּם, יַחֲסֵי צִיבּוּר וּמִסְחָר אֶלֶקְטְרוֹנִי.

קְרִיסְטָל הַמַּזָּל שֶׁלְּךָ: אֶבֶן הַקְּרִיסְטָל הַשְּׁנָתִית שֶׁל מַזַּל טָלֶה הִיא אֶבֶן טוֹרְמֵלִין שָׁחוֹר.
(זוֹהִי אֶבֶן חֵן/ הַקְּרִיסְטָל שֶׁיְּחַזֵּק אוֹתְךָ וְיַעֲזוֹר לְהַצְלָחָה וְזִימּוּן הַשֶּׁפַע).
טוֹרְמֵלִין שָׁחוֹר הִיא אֶבֶן עוֹצְמָתִית לַהֲגָנָה מִפְּנֵי אֵנֶרְגִּיּוֹת שְׁלִילִיּוֹת (עַיִן הָרַע) וְזִימּוּן אֵנֶרְגִּיּוֹת חִיּוּבִיּוֹת.
טוֹרְמֵלִין שָׁחוֹר מְקַרְקַעַת וּמְחַזֶּקֶת אֶת יְכוֹלֶת לַעֲמוֹד מוּל קְשָׁיִים וְדִילֵמוֹת.
נִיתָּן לָלֶכֶת עִמָּהּ בַּתִּיק אוֹ כְּתַכְשִׁיט.
