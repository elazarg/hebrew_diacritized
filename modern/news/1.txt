בְּהֶתְאֵם לְצָו שֶׁעָלָיו חָתַם מַנְכַּ"ל מִשְׂרַד הַבְּרִיאוּת מֹשֶׁה בַּר סִימָן טוֹב בַּשָּׁבוּעַ הָאַחֲרוֹן, מִמָּחָר בְּ-07:00 תָּחוּל עַל כָּל הָאֶזְרָחִים חוֹבָה לַעֲטוֹת מַסֵּכָה בִּיצִיאָה מִמְּקוֹם מְגוּרֵיהֶם.
כַּוָּונַת מִשְׂרַד הַבְּרִיאוּת לְהָחִיל אֶת הַחוֹבָה פּוּרְסְמָה לָרִאשׁוֹנָה בְּ-ynet.
עִם זֹאת, יְצוּיַּן כִּי לְפִי שָׁעָה הַמִּשְׁטָרָה לֹא תִּרְשׁוֹם דו"חות וְתַטִּיל קְנָסוֹת עַל מִי שִׁיפֵּר אֶת הַהַנְחָיָה.

לְפִי הוֹרָאַת הַצַּו, "לֹא יִשְׁהֶה אָדָם בְּמָקוֹם שֶׁאֵינוֹ מְקוֹם מְגוּרָיו לְלֹא כִּיסּוּי הַפֶּה וְהָאַף בְּאֶמְצָעוּת מַסֵּכָה הַמְּיוֹעֶדֶת לְכָךְ אוֹ מַסֵּכָה בֵּיתִית בְּהֶתְאֵם לְהַנְחָיוֹת מִשְׂרַד הַבְּרִיאוּת.
מַסֵּיכַת פֶּה-אַף מַקְטִינָה מְאוֹד אֶת הַסִּיכּוּן לְהִידָּבֵק וּלְהַדְבִּיק.
הִיא מוֹנַעַת פְּלִיטַת נֶתֶז טִיפָּתִי וּמוֹנַעַת מִנֶּתֶז טִיפָּתִי לְהַגִּיעַ לְפֶה וְלָאַף".

עִם זֹאת, הַהוֹרָאוֹת אֵינָן תְּקֵפוֹת לְילָדִים מִתַּחַת לְגִיל 6 וְלַאֲנָשִׁים שֶׁמֵּחֲמַת מוּגְבָּלוּת נַפְשִׁית, שִׂכְלִית אוֹ רְפוּאִית מִתְקַשִּׁים בְּאוֹפֶן מַשְׁמָעוּתִי בְּשִׁימּוּשׁ בַּמַּסֵּכָה.
כְּמוֹ כֵן, אֲנָשִׁים הַשּׁוֹהִים בִּכְלִי רֶכֶב, בַּמִּבְנֶה אוֹ בַּחֶדֶר בַּמִּבְנֶה לְלֹא אָדָם נוֹסָף אוֹ אָדָם הַשּׁוֹהֶה בִּמְקוֹמוֹת כָּאָמוּר עִם אֲנָשִׁים הַגָּרִים בְּאוֹתוֹ מָקוֹם אִיתּוֹ פְּטוּרִים גַּם הֵם מֵעֲטִיַּית מַסֵּכָה.

עוֹד פְּטוּרִים בְּנֵי אָדָם שֶׁמִּשְׁתַּתְּפִים בְּשִׁידּוּר בְּאֶמְצָעִי תִּקְשׁוֹרֶת שֶׁנִּדְרָשִׁים לְדַבֵּר בַּשִּׁידּוּר וּבִלְבַד שֶׁשּׁוֹמְרִים עַל מֶרְחָק שֶׁל שְׁנֵי מְטָרִים מֵאָדָם אַחֵר, וּשְׁנֵי עוֹבְדִים הָעוֹבְדִים בִּקְבִיעוּת בְּחֶדֶר אֶחָד בִּמְקוֹם הָעֲבוֹדָה וּבִלְבַד שֶׁיִּשְׁמְרוּ עַל הַמֶּרְחָק הַנִּדְרָשׁ לְפִי הַהַנְחָיוֹת.

עִם זֹאת, לַמְרוֹת חוֹבַת עֲטִיַּית הַמַּסֵּכוֹת - הַמִּשְׁטָרָה לֹא תֶּאֱכוֹף אֶת הַצַּו בְּאֶמְצָעוּת קְנָסוֹת.
בַּמִּשְׁטָרָה יַנְחוּ אֶת הַשּׁוֹטְרִים לָגֶשֶׁת לָאֶזְרָחִים וּלְבַקֵּשׁ מֵהֶם לַעֲטוֹת מַסֵּכָה, מֵאַחַר שֶׁבִּידֵי הַשּׁוֹטְרִים אֵין כֵּלִים לֶאֱכוֹף אֶת הַהַנְחָיָה.
בֵּין הַיֶּתֶר, גַּם לֹא נִקְבַּע גּוֹבַהּ קְנָס לָאֶזְרָחִים שֶׁיָּפֵרוּ אֶת הַצַּו.

לְפִי שָׁעָה עוֹמֵד מִנְיַין הַנִּדְבָּקִים בַּנְּגִיף הַקּוֹרוֹנָה בְּיִשְׂרָאֵל עַל 10,525.
מַצָּבָם שֶׁל 180 מֵהֶם מוּגְדָּר קָשֶׁה, וּמִתּוֹךְ הַחוֹלִים הַקָּשִׁים - 132 בְּנֵי אָדָם מוּנְשָׁמִים.
