בְּכִירִים בַּהַיְיטֵק עוֹנִים לד"ר וַויְנְשְׁטֵיְין: "מַסְפִּיק בִּיקּוּר בְּבֵית חוֹלִים בִּכְדֵי לִרְאוֹת אֶת הַהַזְנָחָה"
אוֹרֶן קָנִיאֵל מַנְכַּ"ל אַפְּסְפְלַיֶיר הוּפְתַּע מְאוֹד לִקְרוֹא אֶת הַדְּבָרִים שֶׁאָמְרָה ד"ר אוֹרְלִי וַויְנְשְׁטֵיְין, הַנִּיצֶּבֶת בַּחֲזִית רָכַשׁ הַצִּיּוּד הָרְפוּאִי שֶׁל יִשְׂרָאֵל, הַבּוֹקֶר (ב') בְּכַלְכָּלִיסְט כְּשֶׁטָּעֲנָה שֶׁ"אַנְשֵׁי הַהַיְיטֵק הֵם עֲשִׁירִים שֶׁמְּבִיאִים לְמִשְׂרַד הַבְּרִיאוּת צִיּוּד לֹא תַּקִּין".

בַּפּוֹסְט שֶׁפִּירְסֵם בְּפֶיְיסְבּוּק זַעַם קָנִיאֵל: "ד"ר וַויְנְשְׁטֵיְין, בּוֹאִי אֲנִי אֲסַפֵּר לְךָ מָה זֶה אַנְשֵׁי הַיְיטֵק עֲשִׁירִים, כְּשֶׁהֵקַמְנוּ אֶת הַחֶבְרָה לִפְנֵי 9 שָׁנִים, עָבַדְנוּ לְלֹא מַשְׂכּוֹרֶת 3 שָׁנִים, וּלְאַחַר שֶׁהִשַּׂגְנוּ מִימּוּן מִינִימָלִי הִרְוַוחְנוּ שְׂכַר מִינִימוּם.
לָמַדְנוּ לִחְיוֹת כִּמְעַט בְּלִי כֶּסֶף, חִיסַּלְנוּ אֶת כָּל הַחֶסְכוֹנוֹת וְכָל קוּפּוֹת הַפִּיצּוּיִים שֶׁהָיָה אֶפְשָׁר לִפְתּוֹחַ.
כָּל מָה שֶׁהִשַּׂגְנוּ הָיָה בְּ-10 אֶצְבָּעוֹת וְלַמְרוֹת הַקְּשָׁיִים שֶׁהַמְּדִינָה הֶעֱרִימָה.
לְלֹא פְּרוֹטֶקְצִיּוֹת וּלְלֹא קְשָׁרִים.
אֲנִי מֵבִין שֶׁבְּמִשְׂרַד הַבְּרִיאוּת אוֹהֲבִים בְּעִיקָּר לְהַאֲשִׁים בַּמָּקוֹם לְהִסְתַּכֵּל עַל עַצְמָם בַּמַּרְאֶה.
לַמְרוֹת הַכֹּל אֲנַחְנוּ רוֹצִים לַעֲזוֹר.
תַּשְׁאִירוּ אֶת הָאֵגוֹ בַּצַּד".

בָּכִיר נוֹסָף בְּתַעֲשִׂיַּית הַהַיי טֵק, שַׁי וִינִינְגֶּר, מְיַיסֵּד פַיְיבֶּר וּלֶמוֹנַיְיד אוֹמֵר בִּתְגוּבָה לִדְבָרֶיהָ: "אֲמִירוֹתֶיהָ שֶׁל הַגַּב וַויְנְשְׁטֵיְין הֵן עֵדוּת נוֹסֶפֶת שֶׁרָאשֵׁי הַמַּעֲרֶכֶת אֵינָם מְצוּיִים בַּפְּרָטִים, וְאֵין לָהֶם הֲבָנָה עֲמוּקָּה שֶׁל הַמְּצִיאוּת בַּשֶּׁטַח.
זֶה לֹא מַפְתִּיעַ.
בְּמַעֲרֶכֶת כֹּה רִיכּוּזִית, מְיוּשֶּׁנֶת וּפוֹלִיטִית, לֹא נִיתָּן לְצַפּוֹת מִגּוֹרְמִים בַּשֶּׁטַח לְהָצִיף תְּמוּנַת אֱמֶת לֹא מַחְמִיאָה, שֶׁכֵּן, כָּל הַצִּיּוּד שֶׁעָזַרְנוּ לְהָבִיא לְיִשְׂרָאֵל אוּשַּׁר ע"י אַנְשֵׁי הַצֶּוֶות שֶׁל ד"ר וַויְנְשְׁטֵיְין מִמִּשְׂרַד הַבְּרִיאוּת בְּעַצְמוֹ, וּבְנוֹסָף ע"י הַ- FDA וַאֲחֵרִים".

וִינִינְגֶּר הִסְבִּיר שֶׁיֵּשׁ כָּאן שְׁתֵּי קְבוּצוֹת שׁוֹנוֹת בְּתַכְלִית - מְקַבְּלֵי הַחְלָטוֹת וּפוֹלִיטִיקָאִים, וּמֵהַצַּד הַשֵּׁנִי אַנְשֵׁי שֶׁטַח מֵעוּלִּים מִמַּעֲרֶכֶת הַבִּטָּחוֹן וְהַבְּרִיאוּת שֶׁעוֹשִׂים עֲבוֹדַת קוֹדֶשׁ יוֹם וָלַיְלָה בִּכְדֵי לַעֲזוֹר לַמַּאֲמָץ הַלְּאוּמִּי."מִנִּסְיוֹנִי בַּשָּׁבוּעוֹת הָאַחֲרוֹנִים, הַקְּבוּצָה הָרִאשׁוֹנָה בְּעִיקָּר מַפְרִיעָה לַקְּבוּצָה הַשְּׁנִיָּה לַעֲבוֹד.
הֲרֵי מַסְפִּיק בִּיקּוּר אֶחָד בְּבֵית חוֹלִים בִּכְדֵי לִרְאוֹת אֶת רָמַת הַהַזְנָחָה שֶׁהַמַּעֲרֶכֶת סָבְלָה מִמֶּנָּה בְּמֶשֶׁךְ שָׁנִים.
שִׂיחוֹת עִם רוֹפְאִים וַאֲחָיוֹת חוֹשְׂפוֹת מְצִיאוּת עֲגוּמָה וּמַפְחִידָה".

"דּוּגְמָה נוֹסֶפֶת הִיא הַהַזְנָחָה הַמְּקוֹמֶמֶת בְּטִיפּוּל בְּבָתֵּי הָאָבוֹת", אָמַר, "נוֹשֵׂא כֹּה קַל לְפִתְרוֹן, שֶׁהַמַּעֲרֶכֶת הָיְתָה צְרִיכָה לְהַקְדִּים וְלִפְעוֹל לִמְנִיעָה שָׁבוּעוֹת רַבִּים לִפְנֵי הַהִתְפָּרְצוּת.
בִּמְקוֹם זֹאת, מְבוּזְבָּזִים שָׁבוּעוֹת יְקָרִים עַל מַאַבְקֵי כּוֹחַ יַלְדוּתִיִּים בֵּין מַעֲרֶכֶת הַבְּרִיאוּת לְאִרְגּוּנֵי הַדִּיּוּר הַמּוּגָן וּבֵינְתַיִים, אֲנָשִׁים מְשַׁלְּמִים בְּחַיֵּיהֶם.
יַשְׂכִּילוּ קַבַּרְנִיטֵי הַבְּרִיאוּת אִם יַקְדִּישׁוּ אֶת זְמַנָּם לַעֲשִׂיָּיה שְׁקֵטָה וּמְמוּקֶּדֶת בִּמְקוֹם לְהִשְׁתַּמֵּשׁ בַּתִּקְשׁוֹרֶת בִּכְדֵי לָצֵאת נֶגֶד מִתְנַדְּבִים שֶׁמַּקְדִּישִׁים יָמִים כְּלֵילוֹת בְּנִסָּיוֹן לַעֲזוֹר - גַּם אִם עֶצֶם הַצּוֹרֶךְ בְּעֶזְרָה כַּזֹּאת חוֹשֵׂף אֶת חוּלְשָׁתָם וְחוּלְשַׁת הַמַּעֲרֶכֶת.
יִהְיֶה זְמַן לַעֲסוֹק בְּכָל זֹאת אַחַ"כּ".
