חֲמִישִּׁים גְּוָונִים שֶׁל אָפוֹר (בְּאַנְגְּלִית: Fifty Shades of Grey) הוּא רוֹמָן רוֹמַנְטִי אֵרוֹטִי מִשְּׁנַת 2011 שֶׁנִּכְתַּב עַל יְדֵי הַסּוֹפֶרֶת הַבְּרִיטִית אִי.אֶל. גֵ'יְימְס.

עֲלִילַת הַסִּיפּוּר מִתְרַחֶשֶׁת בְּעִיקָּר בְּסִיאַטְל.
זֶהוּ הַכֶּרֶךְ הָרִאשׁוֹן בַּטְּרִילוֹגְיָה, הָעוֹקֶבֶת אַחַר מַעֲרֶכֶת הַיְּחָסִים הָעֲמוּקָּה בֵּין בּוֹגֶרֶת קוֹלֵג', אָנַסְטַסְיָה "אָנָה" סְטִיל, וְאִישׁ עֲסָקִים מַצְלִיחַ, כְּרִיסְטְיָאן גְּרֶיי (Grey – מִכָּאן שֵׁם הַסֵּפֶר, הַמְּהַוֶּוה מִשְׂחַק מִילִּים שֶׁאָבַד בְּתַרְגּוּם לְעִבְרִית: חֲמִישִּׁים צְלָלִים לִגְרֶיי).
הַיְּצִירָה יְדוּעָה בְּשֶׁל הַסְּצֵנוֹת הָאֵירוֹטִיּוֹת הַמְּפוֹרָטוֹת בִּמְיוּחָד וּמְאוּפְיָינוֹת בְּאֵלֵמֶנְטִים שֶׁל תְּנוּחוֹת מִינִיּוֹת הַכּוֹלְלוֹת שִׁעְבּוּד/מִשְׁמַעַת, דּוֹמִינַנְטִיּוֹת/כְּנִיעָה, סָדִיזְם/מָזוֹכִיזְם.

הַכְּרָכִים הַשֵּׁנִי וְהַשְּׁלִישִׁי נִקְרָאִים בְּעִבְרִית: "חֲמִישִּׁים גְּוָונִים שֶׁל אוֹפֶל" וְ"חֲמִישִּׁים גְּוָונִים שֶׁל שִׁחְרוּר", בְּהַתְאָמָה.
סֵפֶר רְבִיעִי בַּסִּדְרָה, "GREY", מְסַפֵּר אֶת הַסִּיפּוּר מִנְּקוּדַּת מַבָּטוֹ שֶׁל גְּרֶיי.

הַטְּרִילוֹגְיָה, שֶׁפּוּרְסְמָה תְּחִילָּה בָּאִינְטֶרְנֵט, זָכְתָה לְהַצְלָחָה רַבָּה, כְּסֵפֶר אֵלֶקְטְרוֹנִי וּבְמַהֲדוּרַת דְּפוּס.
עַד מֵרְץ 2012 נִמְכְּרוּ יוֹתֵר מִ-250,000 עוֹתָקִים מִשְּׁלוֹשֶׁת סִפְרֵי הַטְּרִילוֹגְיָה, וְהַסֵּפֶר הִגִּיעַ לַמָּקוֹם הָרִאשׁוֹן בִּרְשִׁימַת רַבֵּי-הַמֶּכֶר שֶׁל נְיוּ יוֹרְק טַיְימְס וּלְמָקוֹם רִאשׁוֹן בְּאָמָזוֹן.
תּוֹךְ 11 שָׁבוּעוֹת נִמְכַּר הַסֵּפֶר בִּבְרִיטַנְיָה בְּיוֹתֵר מִמִּילְיוֹן עוֹתָקִים.
בְּרַחֲבֵי הָעוֹלָם הַסִּדְרָה נִמְכְּרָה בְּ-65 מִילְיוֹן עוֹתָקִים, עִם זְכוּיוֹת יוֹצְרִים שֶׁנִּמְכְּרוּ בְּ-37 אֲרָצוֹת, וְקָבְעוּ שִׂיא כַּסֵּפֶר הַמּוּדְפָּס שֶׁנִּמְכַּר בַּמְּהִירוּת הַגְּבוֹהָה בְּכָל הַזְּמַנִּים, תּוֹךְ עֲקִיפַת הַשִּׂיא הַקּוֹדֵם שֶׁהָיָה שַׁיָּיךְ לְסִדְרַת "הָארִי פּוֹטֵר".

בְּסֶפְּטֶמְבֶּר 2012 יָצְאָה לָאוֹר גִּרְסָה עִבְרִית שֶׁל הַסֵּפֶר בְּהוֹצָאַת יְדִיעוֹת סְפָרִים.
הַמַּהֲדוּרָה הָרִאשׁוֹנָה, שֶׁהוּדְפְּסָה בְּ-50,000 עוֹתָקִים, אָזְלָה מֵהַחֲנוּיוֹת בְּתוֹךְ מִסְפַּר יָמִים.
הַיְּצִירָה קִיבְּלָה בִּיקּוֹרֶת מְעוֹרֶבֶת עִם הַשָּׁקָתָהּ.

עַל פִּי הַסֵּפֶר הוּפַק סֶרֶט בְּאוֹתוֹ הַשֵּׁם, שֶׁיָּצָא לָאֶקְרָנִים בְּפֶבְּרוּאָר 2015 וְזָכָה לְהַצְלָחָה דּוֹמָה מִבְּחִינַת הַכְנָסוֹת (בְּיַחַס לְסֶרֶט לִמְבוּגָּרִים בִּלְבַד).

אָנַסְטַסְיָה "אָנָה" סְטִיל הִיא סְטוּדֶנְטִית בִּשְׁנָתָהּ הָאַחֲרוֹנָה בְּאוּנִיבֶרְסִיטַת ווֹשִׁינְגְּטוֹן סְטֵיְיט בְּוַונְקוּבֶר, ווֹשִׁינְגְּטוֹן, עִם חֲבֶרְתָהּ הַטּוֹבָה בְּיוֹתֵר, קָתְרִין "קֵיְיט" קָאוֹוֹאנָה, אֲשֶׁר כּוֹתֶבֶת עֲבוּר עִיתּוֹן הַסְּטוּדֶנְטִים שֶׁלָּהֶם.
עֵקֶב מַחֲלָה, קֵיְיט מְשַׁכְנַעַת אֶת אָנָה לְהַחֲלִיף אוֹתָהּ וּלְרַאֲיֵין אֶת כְּרִיסְטְיָאן גְּרֶיי בֶּן הַ-27, יַזָּם צָעִיר, עָשִׁיר וּמַצְלִיחַ לְהַפְלִיא.
אָנָה מִיָּד נִמְשֶׁכֶת אֶל כְּרִיסְטְיָאן, אַךְ גַּם מוֹצֵאת אוֹתוֹ מְאַיֵּים.
כְּתוֹצָאָה מִכָּךְ, הִיא אֵינָהּ מַצְלִיחָה לִשְׁאוֹל אֶת הַשְּׁאֵלוֹת הַנְּכוֹנוֹת וְיוֹצֵאת מִמִּשְׂרָדוֹ שֶׁל כְּרִיסְטְיָאן בִּתְחוּשָׁה שֶׁהָרֵיאָיוֹן הָיָה כּוֹשֵׁל.
אָנָה מְנַסָּה לְנַחֵם אֶת עַצְמָהּ בַּמַּחְשָׁבָה שֶׁשְּׁנֵיהֶם לֹא יִיפָּגְשׁוּ עוֹד לְעוֹלָם.
אוּלָם הִיא מוּפְתַּעַת כַּאֲשֶׁר כְּרִיסְטְיָאן מוֹפִיעַ בִּמְקוֹם הָעֲבוֹדָה שֶׁלָּהּ, הַחֲנוּת שֶׁל קְלַיְיטוֹן, הַחֲנוּת הָעַצְמָאִית הַגְּדוֹלָה בְּיוֹתֵר לְחוֹמְרֵי בִּנְיָין בְּכָל אֵזוֹר הָעִיר פּוֹרְטְלֵנְד.
בִּזְמַן שֶׁהוּא רוֹכֵשׁ פְּרִיטִים שׁוֹנִים הַכּוֹלְלִים אֲזִיקּוֹנִים, נְיַיר דֶּבֶק וְחֶבֶל, אָנָה מוֹדִיעָה לִכְּרִיסְטְיָאן שֶׁקֵיְיט תִּשְׂמַח לְצָרֵף תְּמוּנוֹת לַמַּאֲמָר שֶׁלָּהּ עָלָיו.
כְּרִיסְטְיָאן עוֹזֵב אֶת אָנָה עִם מִסְפַּר הַטֵּלֵפוֹן שֶׁלּוֹ.
מְאוּחָר יוֹתֵר בְּאוֹתוֹ הַיּוֹם, קֵיְיט דּוֹחֶקֶת בְּאָנָה לְהִתְקַשֵּׁר לִכְּרִיסְטְיָאן וּלְאַרְגֵּן יוֹם צִילּוּם עִם הַיָּדִיד הַצַּלָּם שֶׁלָּהֶן, חוֹזֵה רוֹדְרִיגֶז.

לְמָחֳרָת, חוֹזֶה, קֵיְיט וְאָנָה מַגִּיעִים לְיוֹם הַצִּילּוּמִים בַּמָּלוֹן בּוֹ מִתְאַכְסֵן כְּרִיסְטְיָאן, וּכְּרִיסְטְיָאן מַזְמִין אֶת אָנָה לִשְׁתּוֹת עִמּוֹ קָפֶה.
הַשְּׁנַיִים מְשׂוֹחֲחִים בֵּינֵיהֶם בְּבֵית קָפֶה, וּכְּרִיסְטְיָאן שׁוֹאֵל אֶת אָנָה אִם הִיא יוֹצֵאת עִם מִישֶׁהוּ, מַזְכִּיר בִּמְפוֹרָשׁ אֶת חוֹזֶה.
כְּשֶׁאָנָה מְשִׁיבָה לוֹ שֶׁהִיא אֵינָהּ יוֹצֵאת עִם אַף אֶחָד, כְּרִיסְטְיָאן מַתְחִיל לִשְׁאוֹל אוֹתָהּ עַל הַמִּשְׁפָּחָה שֶׁלָּהּ.
בְּמַהֲלַךְ הַשִּׂיחָה אָנָה מְגַלָּה שֶׁכְּרִיסְטְיָאן הוּא רַוּוֹק, אֲבָל הוּא לֹא "בָּחוּר בְּסִגְנוֹן לְבָבוֹת וּפְרָחִים".
אָנָה מִסְתַּקְרֶנֶת מִמֶּנּוּ, בִּמְיוּחָד אַחֲרֵי שֶׁהוּא מוֹשֵׁךְ אוֹתָהּ מֵהַדֶּרֶךְ כְּשֶׁרוֹכֵב אוֹפַנַּיִים מִתְקָרֵב אֵלֶיהָ בִּמְהִירוּת.
אוּלָם אָנָה חוֹשֶׁבֶת שֶׁהִיא לֹא מוֹשֶׁכֶת מַסְפִּיק בִּשְׁבִיל כְּרִיסְטְיָאן, לִרְבוֹת אַכְזָבָתָהּ שֶׁל קֵיְיט.

אַחֲרֵי שֶׁאָנָה מְסַיֶּימֶת אֶת הַבְּחִינוֹת שֶׁלָּהּ, הִיא מְקַבֶּלֶת חֲבִילָה מִכְּרִיסְטְיָאן הַמְּכִילָה מַהֲדוּרוֹת רִאשׁוֹנוֹת שֶׁל הַסֵּפֶר טֶס לְבֵית דַּרְבֶּרְוִויל, מָה שֶׁמַּדְהִים אוֹתָהּ.
מְאוּחָר יוֹתֵר בְּאוֹתוֹ לַיְלָה, אָנָה יוֹצֵאת לִשְׁתּוֹת עִם חֲבֵרֶיהָ, וּבְמַהֲלַךְ הַבִּילּוּי, הִיא מִתְקַשֶּׁרֶת לִכְּרִיסְטְיָאן בְּעוֹדָהּ שִׁיכּוֹרָה.
כְּרִיסְטְיָאן מוֹדִיעַ לָהּ שֶׁהוּא יָבוֹא לֶאֱסוֹף אוֹתָהּ מִפְּנֵי שֶׁהִיא שָׁתְתָה יוֹתֵר מִדַּי.
אָנָה יוֹצֵאת הַחוּצָה כְּדֵי לִנְשׁוֹם מְעַט אֲוִויר צַח, וְחוֹזֶה מְנַסֶּה לְנַשֵּׁק אוֹתָהּ, אַךְ נֶעֱצַר בִּידֵי כְּרִיסְטְיָאן שֶׁהִגִּיעַ לַמָּקוֹם.
אָנָה עוֹזֶבֶת יַחַד עִם כְּרִיסְטְיָאן, אֲבָל לֹא לִפְנֵי שֶׁהִיא מְגַלָּה שֶׁקֵיְיט הֵחֵלָּה לְפַלְרְטֵט עִם אָחִיו שֶׁל כְּרִיסְטְיָאן, אֵלִיּוֹט.
מְאוּחָר יוֹתֵר, אָנָה מִתְעוֹרֶרֶת וּמוֹצֵאת אֶת עַצְמָהּ בַּחֲדַר הַמָּלוֹן שֶׁל כְּרִיסְטְיָאן, שָׁם הוּא נוֹזֵף בָּהּ עַל כָּךְ שֶׁהִיא אֵינָהּ שׁוֹמֶרֶת עַל עַצְמָהּ.
לְאַחַר מִכֵּן כְּרִיסְטְיָאן חוֹשֵׂף בְּפָנֶיהָ שֶׁהוּא רוֹצֶה לִשְׁכַּב אִיתָּהּ.
תְּחִילָּה הוּא אוֹמֵר שֶׁאָנָה תִּצְטָרֵךְ לַחְתּוֹם עַל נַיֶּירֶת לִפְנֵי שֶׁיִּהְיֶה בֵּינֵיהֶם מַגָּע כָּלְשֶׁהוּ, אֲבָל הוּא לֹא עוֹמֵד בַּפִּיתּוּי וּמְנַשֵּׁק אוֹתָהּ בַּמַּעֲלִית.
