בִּתְחִילַּת 2004, קִיבְּלָה סָאסִיץ' אֶת אֶזְרָחוּתָהּ הַגֶּרְמָנִית וְהֵחֵלָּה לָקַחַת חֵלֶק בְּמִשְׂחֲקֵי הַנִּבְחָרוֹת הַצְּעִירוֹת.
בְּנוֹבֶמְבֶּר שֶׁל אוֹתָהּ שָׁנָה, הִיא הָיְיתָה חֵלֶק מֵהַסֶּגֶל הַגֶּרְמָנִי שֶׁזָּכָה בְּאַלִּיפוּת הָעוֹלָם עַד גִּיל 19 שֶׁנֶּעֶרְכָה בְּתַאיְלַנְד.
הִיא הָיְיתָה בַּת 17 בִּלְבַד בִּזְמַן הָאַלִּיפוּת, אַךְ בְּכָל זֹאת פָּתְחָה בַּהֶרְכֵּב הָרִאשׁוֹן וְאַף כָּבְשָׁה שְׁלוֹשָׁה שְׁעָרִים בְּמַהֲלַךְ הַטּוּרְנִיר.

בְּיָנוּאָר 2005, עֶרְכָּה סָאסִיץ' אֶת הוֹפָעַת הַבְּכוֹרָה שֶׁלָּהּ בַּנִּבְחֶרֶת הַבּוֹגֶרֶת שֶׁל גֶּרְמַנְיָה, בְּמִשְׂחָק נֶגֶד אוֹסְטְרַלְיָה.
שְׂעָרָהּ הָרִאשׁוֹן הִגִּיעַ בְּסֶפְּטֶמְבֶּר 2006, בְּנִיצָּחוֹן 3:4 עַל נִבְחֶרֶת קָנָדָה.
בְּמֵרְץ 2007, בְּמַהֲלַךְ מִשְׂחַק לִיגָה, שָׁבְרָה סָאסִיץ' אֶת הַשּׁוֹקָה, וְנֶאֶלְצָה לְהַחֲמִיץ אֶת גְּבִיעַ הָעוֹלָם שֶׁנֶּעֱרַךְ בְּאוֹתָהּ שָׁנָה.
שָׁנָה לְאַחַר מִכֵּן, הִיא הָיְיתָה חֵלֶק מִסֶּגֶל הַנִּבְחֶרֶת שֶׁזָּכָה בְּמֵדַלְיַית הָאָרָד בְּטוּרְנִיר הַכַּדּוּרֶגֶל שֶׁל אוֹלִימְפְּיָאדַת בֶּיְיגִ'ינְג.

בְּקַיִץ 2009, זָכְתָה סָאסִיץ' בְּתוֹאַר מַשְׁמָעוּתִי רִאשׁוֹן עִם הַנִּבְחֶרֶת הַבּוֹגֶרֶת – אַלִּיפוּת אֵירוֹפָּה.
בָּאַלִּיפוּת, שֶׁנֶּעֶרְכָה בְּפִינְלַנְד, נִיצְּחוּ הַגֶּרְמָנִיּוֹת בְּכָל שֵׁשֶׁת מִשְׂחֲקֵיהֶן, וּבַגְּמָר גָּבְרוּ 2:6 עַל אַנְגְּלִיָּה.
סָאסִיץ' כָּבְשָׁה שַׁעַר אֶחָד בְּמַהֲלַךְ הַטּוּרְנִיר (בַּחֲצִי הַגְּמָר נֶגֶד נוֹרְוֶוגְיָה), לְאַחַר שֶׁבְּרוּבּוֹ הִיא עָלְתָה כְּשֶׁמֵּחֲלִיפָה מֵהַסַּפְסָל.
שְׁנָתַיִים לְאַחַר מִכֵּן, בִּגְבִיעַ הָעוֹלָם 2011, סָאסִיץ' כְּבָר שִׁימְּשָׁה כְּשַׂחְקָנִית בַּהֶרְכֵּב הָרִאשׁוֹן.
הִיא כָּבְשָׁה שְׁנֵי שְׁעָרִים בְּמַהֲלַךְ שְׁלַב הַבָּתִּים שֶׁל הַטּוּרְנִיר, וְהוֹבִילָה אֶת הַגֶּרְמָנִיּוֹת לַמָּקוֹם הָרִאשׁוֹן בַּבַּיִת.
לַמְרוֹת זֹאת, בְּמִשְׂחַק רֶבַע הַגְּמָר נֶגֶד יַפָּן, הִפְסִידוּ הַגֶּרְמָנִיּוֹת וְהוּדְּחוּ מֵהַטּוּרְנִיר בְּשָׁלָב מוּקְדָּם מֵהַצָּפוּי.

בְּ-2012, הוֹבִילָה סָאסִיץ' אֶת הַנִּבְחֶרֶת הַגֶּרְמָנִית לִזְכִיָּיה בִּגְבִיעַ אַלְגַּארְבָּה, לְאַחַר שֶׁכָּבְשָׁה שְׁלוֹשַׁעַר בְּנִיצָּחוֹן 3:4 בַּגְּמָר עַל יַפָּן.
הִיא סִייְּמָה כְּמַלְכַּת הַשְּׁעָרִים שֶׁל הַטּוּרְנִיר, וּמְאוּחָר יוֹתֵר בְּאוֹתָהּ שָׁנָה קָטְפָה אֶת פְּרַס כַּדּוּרַגְלָנִית הַשָּׁנָה בְּגֶרְמַנְיָה.
בְּאַלִּיפוּת אֵירוֹפָּה שֶׁנֶּעֶרְכָה בִּשְׁנַת 2013 בִּשְׁוֶודְיָה, זָכוּ הַגֶּרְמָנִיּוֹת בַּפַּעַם הַשִּׁישִּׁית בִּרְצִיפוּת בַּמִּפְעָל.
סָאסִיץ' כָּבְשָׁה שְׁנֵי שְׁעָרִים בְּמַהֲלַךְ שְׁלַב הַבָּתִּים (צֶמֶד בְּנִיצָּחוֹן 0:3 עַל אִיסְלַנְד), וְאַף נִבְחֲרָה לְנִבְחֶרֶת הַטּוּרְנִיר בְּיַחַד עִם חָמֵשׁ שַׂחְקָנִיּוֹת גֶּרְמָנִיּוֹת נוֹסָפוֹת.

לִגְבִיעַ הָעוֹלָם שֶׁנֶּעֱרַךְ בְּקָנָדָה בְּ-2015, כְּבָר הִגִּיעָה סָאסִיץ' כַּכּוֹכֶבֶת הַבִּלְתִּי-מְעוּרְעֶרֶת שֶׁל הַנִּבְחֶרֶת הַגֶּרְמָנִית.
בְּמִשְׂחַק הַפְּתִיחָה שֶׁל הַנִּבְחֶרֶת נֶגֶד חוֹף הַשֶּׁנְהָב, הִיא כָּבְשָׁה שְׁלוֹשַׁעַר וְהוֹבִילָה אֶת גֶּרְמַנְיָה לְנִיצָּחוֹן 0:10.
בִּשְׁמִינִית הַגְּמָר נֶגֶד שְׁוֶודְיָה הִיא הוֹסִיפָה עוֹד צֶמֶד, וּבְרֶבַע הַגְּמָר כָּבְשָׁה אֶת הַשַּׁעַר הַמַּשְׁוֶוה מוּל צָרְפַת.
בְּמִשְׂחַק חֲצִי הַגְּמָר נֶגֶד אַרְצוֹת הַבְּרִית, בְּמַצָּב שֶׁל 0:0, הֶחְמִיצָה סָאסִיץ' פֶּנְדֵּל מַכְרִיעַ.
לְאַחַר הַחַטָּאת הַפֶּנְדֵּל, כָּבְשׁוּ הָאָמֵרִיקָאִיּוֹת שְׁנֵי שְׁעָרִים וְהִדִּיחוּ אֶת גֶּרְמַנְיָה מֵהַטּוּרְנִיר.
לַמְרוֹת הַהַחְמָצָה, נִבְחֲרָה סָאסִיץ' לְנִבְחֶרֶת הַטּוּרְנִיר שֶׁל פִיפָ"א, וְהָיְיתָה לְאַחַת מִשָּׁלוֹשׁ נְצִיגוֹתֶיהָ שֶׁל גֶּרְמַנְיָה בָּהּ (לְצִידָּן שֶׁל נָדִין אַנְּגְרָר וְאָנְיָה מִיטָאג).
בְּנוֹסָף, הִיא זָכְתָה בִּפְרַס נַעַל הַזָּהָב, לְאַחַר שֶׁסִּייְּמָה כְּמַלְכַּת הַשְּׁעָרִים שֶׁל הַטּוּרְנִיר עִם שִׁישָּׁה כִּיבּוּשִׁים.

בְּ-16 בְּיוּלִי 2015, מִיָּד בְּתוֹם גְּבִיעַ הָעוֹלָם, הוֹדִיעָה סָאסִיץ' עַל פְּרִישָׁתָהּ מִכַּדּוּרֶגֶל.
בְּעֶשֶׂר שָׁנִים בַּנִּבְחֶרֶת הַבּוֹגֶרֶת שֶׁל גֶּרְמַנְיָה, הִיא כָּבְשָׁה 63 שְׁעָרִים בְּ-111 מִשְׂחָקִים.
לְאַחַר פְּרִישָׁתָהּ, זָכְתָה סָאסִיץ' בִּפְרַס כַּדּוּרַגְלָנִית הַשָּׁנָה בְּגֶרְמַנְיָה (בַּפַּעַם הַשְּׁנִיָּיה) וּבִפְרַס שַׂחְקָנִית הַשָּׁנָה שֶׁל אופ"א.
