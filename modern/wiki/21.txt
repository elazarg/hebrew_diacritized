בִּשְׁנַת 2000, בְּגִיל 18, הֲלָכָה סִימֶרְיָה לְהִיבָּחֵן אֵצֶל מוֹרֶה לְמוּזִיקָה בְּשֵׁם פְרַנְק אַגְיַאר, אֲחוֹתָהּ סִימוֹנָה רָצְתָה לְהִיבָּחֵן גַּם כֵּן וְלָכֵן שְׁתֵּיהֶן הָלְכוּ לְהִיבָּחֵן, לְאַחַר שֶׁעָבְרוּ בְּהַצְלָחָה הֵן הִתְחִילוּ בַּקַּרְיֵירָה הַמּוּזִיקָלִית שֶׁלָּהֶן כְּצֶמֶד.
בְּמַקְבִּיל, בְּ-1 בְּאַפְּרִיל 2004, הַשְּׁתַּיִים הוֹצִיאוּ אֶת הָאַלְבּוֹם הָרִאשׁוֹן שֶׁלָּהֶן Nã, Nã, Nim Na Não, עַל יְדֵי תַּחַת חָסוּתָהּ שֶׁל אֶטְרְקוֹ רֵקוֹרְדְס, אַךְ הָאַלְבּוֹם נִכְשַׁל, וְלֹא זָכָה לְהַצְלָחָה שֶׁלָּהּ צִיפּוּ כּוּלָּם.
בִּשְׁנַת 2007, הָאֲחָיוֹת פֵּירְקוּ אֶת הַצֶּמֶד לְטוֹבַת בְּנִיַּית הַלַּהֲקָה שֶׁלָּהֶן, "Forró".
בְּמִסְגֶּרֶת הַלַּהֲקָה יָצְאוּ לָאוֹר 11 אַלְבּוֹמֵי אוּלְפָּן כְּשֶׁאַרְבָּעָה מֵהֶם הֵם אַלְבּוֹמֵי הוֹפָעָה.
כְּשֶׁהָיוּ בְּמִסְגֶּרֶת הַלַּהֲקָה הֵן הִשִּׂיגוּ אֶת הַהַכָּרָה הַגְּדוֹלָה בְּיוֹתֵר שֶׁלָּהֶן כַּאֲשֶׁר הֵן זָכוּ בִּפְרָס "Forozão", הַיּוּקְרָתִי בִּשְׁנַת 2011.
בִּשְׁנַת 2012, פֵּירְקוּ הָאֲחָיוֹת אֶת הַלַּהֲקָה לְטוֹבַת קַרְיֵירַת סוֹלוֹ שֶׁתִּתְמַקֵּד בִּשְׁתֵּיהֶן בִּלְבַד כְּצֶמֶד אֶחָד כְּמוֹ בַּעֲבָרָן.

עַד שְׁנַת 2014 הוֹצִיא הַצֶּמֶד חֲמִישָּׁה אַלְבּוֹמֵי אוּלְפָּן שֶׁזָּכוּ לְהַצְלָחָה רַבָּה בְּאָמֵרִיקָה הַלָּטִינִית, כְּשֶׁהַבּוֹלֵט מִבֵּינֵיהֶם הוּא As Coleguinhas - Vol.
4, שֶׁיָּצָא בִּשְׁנַת 2012, עִם אַלְבּוֹם אוּלְפָּן נוֹסָף.
לְאַחַר הַהַצְלָחָה שֶׁלָּהּ זָכָה הַצֶּמֶד עִם אַלְבּוֹם הַהוֹפָעָה הָרִאשׁוֹן שֶׁלָּהֶן, הוֹצִיא הַצֶּמֶד אֶת אַלְבּוֹם הַהוֹפָעָה הַשָּׁנִי שֶׁלָּהֶן Bar das Coleguinhas בִּשְׁנַת 2015, וּכְשָׁנָה לְאַחַר מִכֵּן יָצָא לְאוֹר אַלְבּוֹם הַהוֹפָעָה הַשְּׁלִישִׁי שֶׁלָּהֶן Live, שֶׁהָפַךְ לַמַּצְלִיחַ בְּיוֹתֵר שֶׁלָּהֶן מִבֵּין כָּל הַשְּׁלוֹשָׁה, וְהֵבִיא עִמּוֹ שְׁלוֹשָׁה סִינְגֵּלִים.
הַסִּינְגֶל הָרִאשׁוֹן שֶׁיָּצָא מִן אַלְבּוֹם הַהוֹפָעָה הָיָה "Amor Mal Resolvido".
הוּא זָכָה לְהַצְלָחָה בְּעִיקָּר בִּבְרָזִיל, וּבַמִּצְעָד הָרִשְׁמִי שֶׁל הַמְּדִינָה הִגִּיעַ לַמָּקוֹם הַ-22.
לְאַחַר מִכֵּן יָצָא הַסִּינְגֶל הַשָּׁנִי "126 Cabides", שֶׁהָפַךְ לְשִׁירָן הָרִאשׁוֹן שֶׁהִצְלִיחַ לִכְבּוֹשׁ אֶת פִּסְגַּת הַמִּצְעָד הָרִשְׁמִי שֶׁל בְּרָזִיל.

בִּשְׁנַת 2017, הוֹצִיאוּ הַצֶּמֶד אֶת שִׁירָן הַמַּצְלִיחַ בְּיוֹתֵר "Loka".
הַשִּׁיר קִיבֵּל בִּבְרָזִיל תּוֹאַר שֶׁל יַהֲלוֹם מִשָּׁלוֹשׁ, וּבְכָךְ הָפַךְ הַשִּׁיר לְאֶחָד מֵהַשִּׁירִים הַנָּשִׁיִּים הַמַּצְלִיחִים בְּיוֹתֵר בִּבְרָזִיל בְּ-2017.
הַשִּׁיר הָיָה בְּשִׁיתּוּף הַזַּמֶּרֶת אָנִיטָה, שֶׁבִּמְהֵרָה הָפַךְ הַשִּׁיר גַּם לַשִּׁיר הַמַּצְלִיחַ בְּיוֹתֵר בְּכָל הַקַּרְיֵירָה שֶׁלָּהּ.
הַוִּוידֵאוֹ קְלִיפּ הַנִּלְוֶוה שֶׁיָּצָא לַשִּׁיר זָכָה לְהַצְלָחָה עוֹלָמִית כַּאֲשֶׁר נָכוֹן לְאוֹגוּסְט 2017, צָבַר לְמַעְלָה מִ-440 מִילְיוֹן צְפִיּוֹת, וְהָפַךְ לְוִוידֵאוֹ קְלִיפּ הַנָּשִׁי הַנִּצְפֶּה בְּיוֹתֵר לִשְׁנַת 2017.
לְאַחַר הַהַצְלָחָה שֶׁלָּהּ זָכוּ עִם "Loka", בְּ-16 בְּיוּלִי 2017, יָצָא לְאוֹר הַסִּינְגֶל הַשְּׁלִישִׁי "Regime Fechado", מִתּוֹךְ אַלְבּוֹם הַהוֹפָעָה הַשְּׁלִישִׁי שֶׁלָּהֶן, Live.
בְּאוֹתָהּ שָׁנָה, נוֹדַע לַצִּיבּוּר כִּי הָאַלְבּוֹם Live, יוּעֲמַד בְּטֶקֶס פְּרָסֵי הַגְּרָאמִי הַלָּטִינִי הַ-18, בְּקָטֵגוֹרְיַית - "Mejor Álbum De Musica Sertaneja".

בִּתְחִילַּת 2018, הוֹצִיאוּ אֶת "Paga De Solteiro Feliz", שֶׁצָּבַר צְפִיּוֹת רַבּוֹת בְּסָמוּךְ לְכָךְ, וּלְאַחַר מִכֵּן, בְּ-30 בְּיָנוּאָר, הוֹצִיא הַצֶּמֶד שִׁיר בְּיַחַד עִם קָבִּינְיוֹ בְּשֵׁם, "Ta Tum Tum".
מְאוּחָר יוֹתֵר הוֹצִיא הַצֶּמֶד אֶת הַשִּׁיר "Novo" בְּיַחַד עִם הַזַּמֶּרֶת לָאוּרָה פָּאוּזִינִי, וְאָז הוֹצִיאוּ אֶת הַסִּינְגֶל הֲשִׁיוּוּקִי "Hasta La Vista", בְּיַחַד עִם פַּאבְּלוֹ וִיטָאָאר וּלְוָאן סַנְטָנָה.
בְּיוּלִי בְּאוֹתָהּ הַשָּׁנָה, יָצָא הַשִּׁיר "Um em Um Milhão".
