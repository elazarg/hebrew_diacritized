הֶקְטוֹר רָאוּל קוּפֶּר (סְפָרַדִּית: Héctor Raúl Cúper; נוֹלַד בְּ-16 בְּנוֹבֶמְבֶּר 1955 בְּצַ'בָּאס שֶׁבִּפְרוֹבִינְצְיַית סַנְטָה פֶה, אַרְגֶּנְטִינָה) הוּא כַּדּוּרַגְלָן עָבָר אַרְגֶּנְטִינָאִי וְכַיּוֹם מְאַמֵּן נִבְחֶרֶת אוּזְבֵּקִיסְטָן.
קוּפֶּר מְכוֹנָה בְּאַרְגֶּנְטִינָה אֶל קַאבְסוֹן ("רֹאשׁ גָּדוֹל" בִּסְפָרַדִּית).
רַבִּים רוֹאִים אֶת קוּפֶּר כְּ"לוּזֵר נִצְחִי" וְזֹאת בְּשֶׁל שְׁלוֹשָׁה הֶפְסֵדִים בְּמִשְׂחֲקֵי גְּמַר וּפִסְפוּס אַלִּיפוּת בְּעֵת שֶׁאִימֵּן אֶת אִינְטֶר מִילָאנוֹ בְּעוֹד שֶׁאֲחֵרִים רוֹאִים בּוֹ מְאַמֵּן אֲשֶׁר מְסוּגָּל לְהוֹצִיא אֶת הַמֵּיטָב מִשַּׂחְקָנָיו וּלְהוֹבִיל קְבוּצוֹת קְטַנּוֹת לְהֶישֵּׂגִים.

קוּפֶּר הֵחֵל לְשַׂחֵק בִּקְבוּצַת פְּרוֹ קָארִיל אוּאֶסְטָה בְּ-1976 וְשִׂיחֵק בָּהּ עַד 1988 לְמַעֵט שְׁנָתַיִים בָּהֶן עָבַר לְשַׂחֵק בִּקְבוּצַת אִינְפְדֵנַיְינְטָה רִיבֶּדֶבְיָה (בֵּין הַשָּׁנִים 1977 עַד 1978).
בְּ-1988 עָבַר לְשַׂחֵק בִּקְבוּצַת הוֹרְקָאן עַד פְּרִישָׁתוֹ בִּשְׁנַת 1992.
בְּמַדֵּי פְּרוֹ זָכָה קוּפֶּר בִּשְׁתֵּי אַלִּיפוּיוֹת (נַסְיוֹנָל 1982 וְנַסְיוֹנָל 1984).
בֵּין הַשָּׁנִים 1984 וְעַד 1986 הוֹפִיעַ קוּפֶּר גַּם בַּחֲמִישָּׁה מִשְׂחָקִים בְּמַדֵּי נִבְחֶרֶת אַרְגֶּנְטִינָה אִם כִּי לֹא זָכָה לְזִימּוּן לְמִשְׂחֲקֵי מוֹנְדְּיָאל 1986.

שָׁנָה אַחַת לְאַחַר פְּרִישָׁתוֹ מוּנָּה קוּפֶּר כִּמְאַמְּנָהּ שֶׁל הוֹרָאקֵן וְאִימֵּן אוֹתָהּ עַד 1995 אָז הִתְמַנָּה כִּמְאַמְּנָהּ שֶׁל לָנוּס אוֹתָהּ הוֹלִיךְ לִזְכִיָּיה בִּגְבִיעַ קוֹנְמַבּוּל.

בְּ-1997 מוּנָּה קוּפֶּר כִּמְאַמְּנָהּ שֶׁל רֵיאָל מָאיוֹרְקָה הַסְּפָרַדִּית בָּהּ הִתְפַּרְסֵם וְזֹאת לְאַחַר שֶׁהוֹבִיל אֶת הַקְּבוּצָה הַצְּנוּעָה לְהֶישֵּׂגִים נָאִים אֲשֶׁר כָּלְלוּ הַעְפָּלָה לִגְמַר גְּבִיעַ הַמֶּלֶךְ הַסְּפָרַדִּי (בַּפַּעַם הַשְּׁנִיָּיה בְּתוֹלְדוֹת הַמּוֹעֲדוֹן) וְלִגְמַר גְּבִיעַ הַמַּחֲזִיקוֹת הִיסְטוֹרִי בָּהֶם הִפְסִידָה הַקְּבוּצָה לְבַּרְצְלוֹנָה וְלַאצְיוֹ (בְּהַתְאָמָה) וּזְכִיָּיה בַּסּוּפֶּרְקַאפּ הַסְּפָרַדִּי אַחֲרֵי נִיצָּחוֹן עַל בַּרְצְלוֹנָה (הַזְּכִיָּיה הָרִאשׁוֹנָה וְהַיְּחִידָה שֶׁל הַקְּבוּצָה בַּתּוֹאַר).
בְּנוֹסָף, הוֹבִיל קוּפֶּר אֶת הַקְּבוּצָה בַּפַּעַם הָרִאשׁוֹנָה בְּתוֹלְדוֹתֶיהָ לַהִשְׁתַּתְּפוּת בְּלִיגַת הָאַלּוּפוֹת וְזֹאת אַחֲרֵי שֶׁהַקְּבוּצָה סִייְּמָה בַּמָּקוֹם הַשְּׁלִישִׁי בַּלִּיגָה, דֵּירוּג הַשִּׂיא בְּתוֹלְדוֹת הַמּוֹעֲדוֹן.

בְּ-1999 מוּנָּה קוּפֶּר כִּמְאַמְּנָהּ שֶׁל וָלֶנְסְיָה.
בְּעוֹנָתוֹ הָרִאשׁוֹנָה זָכָה קוּפֶּר בַּסּוּפֶּרְקְּאַפּ הַסְּפָרַדִּי הוֹבִיל אֶת הַקְּבוּצָה לַמָּקוֹם הַשְּׁלִישִׁי וּלְהַעְפָּלָה לִגְמַר לִיגַת הָאַלּוּפוֹת בַּפַּעַם הָרִאשׁוֹנָה בְּתוֹלְדוֹת הַמּוֹעֲדוֹן (הִפְסִידָה לְרֵיאָל מַדְרִיד).
בְּעוֹנָתוֹ הַשְּׁנִיָּיה וְהָאַחֲרוֹנָה הוֹבִיל קוּפֶּר אֶת הַקְּבוּצָה לְהוֹפָעָה שְׁנִיָּיה בִּגְמַר לִיגַת הָאַלּוּפוֹת בָּהּ הִפְסִידָה בְּפֶנְדֵּלִים לְבָּאיֶירְן מִינְכֶן וְלַמָּקוֹם הַחֲמִישִׁי בַּלִּיגָה.

בְּ-2001 מוּנָּה קוּפֶּר כִּמְאַמְּנָהּ שֶׁל אִינְטֶר מִילָאנוֹ.
מוֹעֲדוֹן הַפְּאֵר הָאִיטַלְקִי קִיוָוֹה לַחֲזוֹר לְצַמֶּרֶת הַלִּיגָה הָאִיטַלְקִית אַחֲרֵי שָׁנִים בָּהֶם הַהַשְׁקָעָה הַכַּסְפִּית הַגְּדוֹלָה לֹא הֵנִיבָה הַצְלָחוֹת מַרְשִׁימוֹת.
תַּחַת הַדְרָכָתוֹ שֶׁל קוּפֶּר חָזְרָה הַקְּבוּצָה לִהְיוֹת קְבוּצַת צַמֶּרֶת.
הַקְּבוּצָה סִייְּמָה בַּמָּקוֹם הַשְּׁלִישִׁי בְּעוֹנָתוֹ הָרִאשׁוֹנָה וּפִסְפְסָה זְכִיָּיה בְּאַלִּיפוּת בַּמַּחְזוֹר הָאַחֲרוֹן בְּעוֹנָתוֹ הַשְּׁנִיָּיה.
בְּנוֹסָף, הִגִּיעָה הַקְּבוּצָה עַד חֲצִי גְּמַר לִיגַת הָאַלּוּפוֹת בּוֹ הִפְסִידָה לַיְּרִיבָה הָעִירוֹנִית מִילָאן.
בְּמַהֲלַךְ תְּקוּפָתוֹ בְּאִינְטֶר הִסְתַּכְסֵךְ קוּפֶּר עִם כּוֹכָבָהּ הַפָּצוּעַ שֶׁל הַקְּבוּצָה בְּאוֹתָם יָמִים, הֶחָלוּץ הַבְּרָזִילָאִי רוֹנַלְדּוֹ.

בְּ-2003, אַחֲרֵי רֶצֶף מִשְׂחָקִים לֹא מְשַׁכְנְעִים, פּוּטַּר לְבַסּוֹף קוּפֶּר מֵאִימּוּן הַקְּבוּצָה.
הוּא חָזַר לְאַמֵּן כַּעֲבוֹר שָׁנָה לְאַחַר שֶׁמּוּנָּה שׁוּב כִּמְאַמְּנָהּ שֶׁל מָאיוֹרְקָה שֶׁהָיְיתָה שְׁרוּיָה בְּמַאַבְקֵי תַּחְתִּית.
קוּפֶּר הִצְלִיחַ לְהַשְׁאִיר אֶת הַקְּבוּצָה בַּלִּיגָה אַךְ עוֹנָה לְאַחַר מִכֵּן, בְּפֶבְּרוּאָר 2006 הִתְפַּטֵּר מִתַּפְקִידוֹ לְאַחַר שֶׁהַקְּבוּצָה כָּשְׁלָה בְּנִיסְּיוֹנָהּ לַחֲזוֹר לַצַּמֶּרֶת וְהִתְמוֹדְדָה בַּתַּחְתִּית.
בְּקַיִץ 2007 מוּנָּה קוּפֶּר כִּמְאַמְּנָהּ שֶׁל רֵיאָל בֶּטִיס אַךְ לְאַחַר מִסְפַּר מִשְׂחָקִים פּוּטַּר מִתַּפְקִידוֹ, בְּמֵרְץ 2008 מוּנָּה קוּפֶּר כִּמְאַמְּנָהּ שֶׁל פַּארְמָה וְזֹאת כְּדֵי לְהַשְׁאִיר אוֹתָהּ בַּלִּיגָה.
בְּ-12 בְּמַאי 2008, מַחְזוֹר אֶחָד לִפְנֵי סִיּוּם הָעוֹנָה, פּוּטַּר קוּפֶּר מֵאִימּוּן פַּארְמָה.
עִיתּוּי הַפִּיטּוּרִים הָיָה מַפְתִּיעַ, שֶׁכֵּן נוֹתַר מַחְזוֹר בּוֹדֵד לְסִיּוּם הָעוֹנָה.
לַמְרוֹת הַהַפְתָּעָה בָּעִיתּוּי, הַפִּיטּוּרִים בָּאוּ בְּעִקְבוֹת הַתּוֹצָאוֹת הַדַּלּוֹת שֶׁל פַּארְמָה.
תַּחַת הַדְרָכָתוֹ הַקְּבוּצָה נִיצְּחָה בִּשְׁנֵי מִשְׂחָקִים בִּלְבַד מִתּוֹךְ עֲשָׂרָה.

בְּאוֹגוּסְט 2008 מוּנָּה לִמְאַמֵּן נִבְחֶרֶת גֵּאוֹרְגִּיָּה, אוֹתָהּ הִדְרִיךְ בְּמַהֲלַךְ מוּקְדָּמוֹת מוֹנְדְּיָאל 2010 אוֹתוֹ סִייְּמָה הַנִּבְחֶרֶת בַּמָּקוֹם הָאַחֲרוֹן בַּבַּיִת.
עִם סִיּוּם חוֹזֵהוּ כִּמְאַמֵּן הַנִּבְחֶרֶת הַגֵּאוֹרְגִּית, מוּנָּה קוּפֶּר לִמְאַמֵּן אָרִיס סָלוֹנִיקִי הַיְּוָונִית, אוֹתָהּ הוֹבִיל לִגְמַר הַגָּבִיעַ הַיְּוָונִי בּוֹ הִפְסִידָה הַקְּבוּצָה לְפָנָאתִינַיְיקוֹס בְּתוֹם דּוּ-קְרָב בְּעִיטוֹת עוֹנְשִׁין.
בְּעוֹנַת 2010/2011 הוֹבִיל קוּפֶּר אֶת הַקְּבוּצָה לְשָׁלָב 32 הָאַחֲרוֹנוֹת בַּלִּיגָה הָאֵירוֹפִּית בּוֹ הוּדְּחָה הַקְּבוּצָה עַל יְדֵי צֶסְקָ"א מוֹסְקְבָה.
בְּיָנוּאָר 2011 הִתְפַּטֵּר קוּפֶּר מֵאִימּוּן הַקְּבוּצָה לְאַחַר תְּקוּפָה שֶׁל חוֹסֶר הַצְלָחָה בַּלִּיגָה הַמְּקוֹמִית.

בְּיוּנִי 2011 מוּנָּה קוּפֶּר לִמְאַמֵּן רָאסִינְג סַנְטַאנְדֵּר, אַךְ הִתְפַּטֵּר מֵאִימּוּן הַקְּבוּצָה לְאַחַר חֲמִישָּׁה חוֹדָשִׁים עֵת שֶׁהַקְּבוּצָה מוּקְּמָה בַּמָּקוֹם הָאַחֲרוֹן בַּלִּיגָה.
בְּדֵצֶמְבֶּר 2011 מוּנָּה לִמְאַמֵּן קְבוּצַת אוּרְדּוּסְפוּר הַטּוּרְקִית.
בְּ-13 בְּאַפְּרִיל 2013 עָזַב אֶת הַקְּבוּצָה.

בְּנוֹבֶמְבֶּר 2014 הִתְמַנָּה קוּפֶּר לְתַפְקִיד מְאַמֵּן אֶל-וָואסֶל מִלִּיגַת הָעַל שֶׁל אִיחוּד הָאֵמִירוּיוֹת הָעַרְבִיּוֹת.
בְּמֵרְץ 2014, לְאַחַר אַרְבָּעָה חוֹדָשִׁים בִּלְבַד בַּתַּפְקִיד וּבְעִקְבוֹת רֶצֶף תּוֹצָאוֹת מְאַכְזְבוֹת, פּוּטַּר קוּפֶּר מִתַּפְקִידוֹ.

בְּ-2 בְּמֵרְץ 2015 מוּנָּה קוּפֶּר לְתַפְקִיד מְאַמֵּן נִבְחֶרֶת מִצְרַיִם, עַל מְנַת לְהוֹבִיל אוֹתָהּ בְּמוּקְדְּמוֹת מוֹנְדְּיָאל 2018.
עִם נִבְחֶרֶת מִצְרַיִם קוּפֶּר רָשַׁם הַצְלָחָה כַּאֲשֶׁר הִגִּיעַ עִם הַנִּבְחֶרֶת לִגְמַר גְּבִיעַ אַפְרִיקָה לְאוּמּוֹת 2017.
בַּגְּמָר הִפְסִידָה הַנִּבְחֶרֶת לְקָמֵרוּן.
תַּחַת קוּפֶּר מִצְרַיִם הֶעְפִּילָה לְמוֹנְדְּיָאל 2018, הַמּוֹנְדְּיָאל הָרִאשׁוֹן שֶׁלָּהּ מִזֶּה 28 שָׁנָה.
בַּמּוֹנְדְּיָאל הַנִּבְחֶרֶת סִייְּמָה בַּמָּקוֹם הָאַחֲרוֹן בַּבַּיִת שֶׁלָּהּ עִם 3 הֶפְסֵדִים וְקוּפֶּר עָזַב אֶת תַּפְקִידוֹ אַחֲרֵי הַמּוֹנְדְּיָאל.

בְּ-1 בְּאוֹגוּסְט 2018 מוּנָּה קוּפֶּר לִמְאַמֵּן נִבְחֶרֶת אוּזְבֵּקִיסְטָן.
