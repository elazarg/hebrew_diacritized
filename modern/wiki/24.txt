רוּדִי בֶּיְילוֹר (מַאט דֶּיְימוֹן) הוּא מִשְׁפְּטָן צָעִיר שֶׁאַךְ עַתָּה סִייֵּם אֶת לִימּוּדֵי מִשְׁפָּט בְּאוּנִיבֶרְסִיטַת מֶמְפִיס.
הוּא מִתְקַבֵּל לַעֲבוֹדָה בְּמִשְׂרַד עוֹרְכֵי דִּין בְּנִיהוּלוֹ שֶׁל לֶיְימָּן סְטוֹן, עוֹרֵךְ דִּין שַׁחְצָן וּמוּשְׁחָת (מִיקִי רוֹרְק) שֶׁמַּצְמִיד לוֹ כְּחוֹנֵךְ אֶת דֶּק שִׁיפֶלְט (דֵּנִי דֶּה וִיטוֹ).
לְשִׁיפֶלְט אֵין זְכוּת לְהוֹפִיעַ בְּבֵית מִשְׁפָּט הֱיוֹת שֶׁלֹּא עָמַד בִּבְחִינוֹת הַהַסְמָכָה מִסְפַּר פְּעָמִים אֲבָל נִיסְּיוֹנוֹ הַגָּדוֹל בִּתְבִיעוֹת נְזִיקִין וְשִׁיוּוּקוֹ הָאַגְרֵסִיבִי מַקְנִים לוֹ מַעֲמָד שֶׁל חֲשִׁיבוּת בַּמִּשְׂרָד.

שִׁיפֶלְט לוֹקֵחַ אֶת רוּדִי לְבֵית הַחוֹלִים וּמַדְגִּים בְּפָנָיו אֶת שִׁיטַת הָעֲבוֹדָה שֶׁלּוֹ: בַּקַּבָּלָה הוּא מְקַבֵּל מֵידָע עַל תְּאוּנַת דְּרָכִים, מַגִּיעַ לַחֲדַר הַנִּפְגָּע הַמְּכוּסֶּה בְּגֶבֶס, דּוֹחֵף לְיָדוֹ אֶת כַּרְטִיס הַבִּיקּוּר שֶׁלּוֹ וּמְחִתִּים אוֹתוֹ עַל חוֹזֶה לְיִיצּוּג בִּלְעָדִי שֶׁלּוֹ כְּעוֹרֵךְ דִּינוֹ.
עַל אַף הַבְטָחָתוֹ שֶׁל שִׁיפֶלְט לָאֶפֶקְטִיבִיּוֹת שֶׁל הַשִּׁיטָה, רוּדִי נִרְתַּע מֵאוֹפֶן הַפְּעוּלָּה הַזֶּה.
בַּהֶמְשֵׁךְ נִשְׁלַח רוּדִי לְבַדּוֹ "לָדוּג" לָקוֹחוֹת פּוֹטֶנְצְיָאלִיִּים בְּאֵזוֹר בֵּית הַחוֹלִים, וּבְאַחַת הַפְּעָמִים הוּא מִתְוַודֵּעַ לְקֶלִי רִיקֶר (קְלֵיְיר דֶּיְינֶס), אִישָּׁה מוּכָּה, וּמִתְפַּתֵּחַ בֵּינֵיהֶם קֶשֶׁר שֶׁל חִיבָּה.

בְּמַקְבִּיל לַעֲבוֹדָתוֹ בְּמִשְׂרַד רוּדִי מְטַפֵּל בַּצַּוּוֹאָה שֶׁל אִישָּׁה קְשִׁישָׁה בְּשֵׁם מִיס בִּירְדֵי (תֵּרֵזָה רַיְיט), שֶׁבְּנוֹסָף מַשְׂכִּירָה לוֹ חֶדֶר בְּבֵיתָהּ.

לְאַחַר שֶׁסְּטוֹן נֶעֱצַר בַּחֲשָׁד לִשְׁחִיתוּת, פּוֹתְחִים רוּדִי וְשִׁיפֶלְט מִשְׂרָד קָטָן לַעֲרִיכַת דִּין כִּשְׂרוּדֵי מוֹפִיעַ בְּבֵית מִשְׁפָּט וְשִׁיפֶלְט עוֹזֵר לוֹ בְּיֶדַע וּבִרְכִישַׁת לָקוֹחוֹת.

הַתִּיק הָעִיקָּרִי שֶׁל רוּדִי הוּא תְּבִיעָה נֶגֶד חֶבְרַת בִּיטּוּחַ רְפוּאִי גְּדוֹלָה שֶׁמְּסָרֶבֶת לְמַמֵּן טִיפּוּל רְפוּאִי לַלָּקוֹחַ שֶׁלָּהּ, דּוֹנִי בְּלָאק, צָעִיר חוֹלֶה לוֹיְקֶמְיָה הַזָּקוּק לְנִיתּוּחַ הַשְׁתָּלַת מֹחַ עֶצֶם שֶׁיָּכוֹל לְהַצִּיל אֶת חַיָּיו.
לְחֶבְרַת הַבִּיטּוּחַ יֵשׁ סוֹלְלַת עוֹרְכֵי דִּין מְיוּמָּנִים וּבְרֹאשָׁם לֵיאוֹ דְּרֶמוֹנְד (גּ'וֹן וּוַיְיט) שֶׁמִּתְיַיחֵס לְרוּדִי הַטִּירוֹן בְּזִלְזוּל וּמַצִּיעַ הֶסְכֵּם פְּשָׁרָה שֶׁל 75,000 דּוֹלָר.
רוּדִי דּוֹחֶה אֶת הַהֶסְכֵּם עַל אַף שֶׁגַּם הַשּׁוֹפֵט הָרוִוי הַיְיל (דִּין סְטוֹקְווֹל) מְיַיעֵץ לוֹ לְקַבֵּל אוֹתוֹ, כְּשֶׁהוּא מוֹדֶה בְּגָלוּי שֶׁהוּא עָיֵיף מִכְּדֵי לַעֲסוֹק בְּנִיהוּל הַמַּאֲבָק הַמִּשְׁפָּטִי.

לְאַחַר שֶׁהַשּׁוֹפֵט הַיְיל לוֹקֶה בְּהֶתְקֵף לֵב הוּא מוּחְלָף בַּשּׁוֹפֵט טַיְירוֹן קִיפְּלֶר (דֵּנִי גְּלוֹבֶר), פָּעִיל לְמַעַן זְכוּיוֹת אָדָם שֶׁדּוֹחֶה אֶת בַּקָּשַׁת הַחֶבְרָה לְבִיטּוּל הַתְּבִיעָה וְהַמִּשְׁפָּט מִתְחַדֵּשׁ.
בֵּינְתַיִים נִפְטַר דּוֹנִי וְרוּדִי מַעֲלֶה אֶת סְכוּם הַתְּבִיעָה לַעֲשָׂרָה מִילְיוֹן דּוֹלָר.

בְּמַקְבִּיל לְהִתְנַהֲלוּת הַמִּשְׁפָּט מַמְשִׁיךְ רוּדִי לְהִיפָּגֵשׁ עִם קֶלִי, שֶׁבְּשָׁלָב מְסוּיָּם בּוֹחֶרֶת לַעֲזוֹב אֶת בַּעֲלָהּ הַמַּכָּה.
רוּדִי עוֹזֵר לְקֶלִי לֶאֱסוֹף אֶת חֲפָצֶיהָ מֵהַבַּיִת כְּשֶׁבַּעֲלָהּ מַגִּיעַ בְּמַפְתִּיעַ וּבֵינוֹ לְבֵין רוּדִי מִתְפַּתֵּחַ מַאֲבָק אַלִּים שֶׁמִּסְתַּיֵּים בְּמוֹתוֹ שֶׁל הַבַּעַל.
קֶלִי דּוֹחֶקֶת בְּרוּדִי לַעֲזוֹב אֶת הַמָּקוֹם לִפְנֵי שֶׁמַּגִּיעָה הַמִּשְׁטָרָה עַל מְנַת לְהָגֵן עָלָיו וְנִשְׁאֶרֶת לָשֵׂאת בְּאַשְׁמַת הַהֲרִיגָה.
לְאַחַר תְּקוּפַת מַעֲצָר קְצָרָה הִיא מְשׁוּחְרֶרֶת עַל סְמַךְ זְכוּתָהּ לַהֲגַנָּה עַצְמִית.
רוּדִי מוֹצִיא אוֹתָהּ מֵהַמַּעֲצָר וּמֵבִיא אוֹתָהּ אִיתּוֹ לְבֵיתָהּ שֶׁל בִּירְדִּי.

שִׁיפֶלְט עוֹזֵר לְרוּדִי לְהַשִּׂיג מִסְמָכִים וְעָדִים שֶׁהַחֶבְרָה מְבַקֶּשֶׁת לְהַסְתִּיר.
הוּא מוֹצֵא אֶת גֵ'קִי לְמַנְצִ'יק (וִירְגִ'ינְיָה מִדְּסָן) שֶׁהָיְיתָה אַחְרָאִית עַל הַתְּבִיעָה וּפוּטְּרָה מֵהַחֶבְרָה.
הִיא מְעִידָה עַל מְדִינִיּוּת הַחֶבְרָה שֶׁמְּבוּסֶּסֶת בְּעִיקָּר עַל דְּחִיָּיה חוֹזֶרֶת שֶׁל כָּל הַתְּבִיעוֹת, כָּךְ שֶׁרוֹב הַלָּקוֹחוֹת מִתְיָיאֲשִׁים.
עִם זֹאת, דְּרֵמוֹנְד מַצְלִיחַ לְהוֹכִיחַ שֶׁעֵדוּתָהּ בִּלְתִּי אֲמִינָה וְהִיא נִפְסֶלֶת.
אַף עַל פִּי כֵן, בַּחֲקִירָה נֶגְדִּית שֶׁל וִילְפוֹרְד קַיְלִי (רְוִי שַׁיְידֶּר), נְשִׂיא הַחֶבְרָה, רוּדִי מֵבִיא אוֹתוֹ לְהוֹדּוּת בְּנוֹגֵעַ לִמְדִינִיּוּת הַחֶבְרָה.
בְּסִיּוּם הַמִּשְׁפָּט רוּדִי מַקְרִין לַמּוּשְׁבָּעִים אֶת עֵדוּתוֹ הָאַחֲרוֹנָה שֶׁל דּוֹנִי בְּלָאק עַצְמוֹ, שֶׁצּוּלְּמָה לִפְנֵי מוֹתוֹ, וְהַמִּשְׁפָּט מוּכְרָע לְטוֹבַת מִשְׁפַּחְתּוֹ שֶׁל דּוֹנִי וְעַל הַחֶבְרָה לְשַׁלֵּם חֲמִישִּׁים מִילְיוֹן דּוֹלָר.

גְּזַר הַדִּין וְנִיצְחוֹנוֹ שֶׁל רוּדִי מִתְפַּרְסְמִים בְּרַחֲבֵי הַמְּדִינָה, אוּלָם הַחֶבְרָה מוֹדִיעָה עַל פְּשִׁיטַת רֶגֶל, מָה שֶׁמְּאַפְשֵׁר לָהּ לַחֲמוֹק מִתַּשְׁלוּם הַחוֹב לְמִשְׁפָּחָה, וְגַם רוּדִי וְשִׁיפֶלְט נִשְׁאָרִים לְלֹא שְׂכַר הָעֲמָלָה.
רוּדִי מַחֲלִיט שֶׁאֵינוֹ מַתְאִים לַפְּרַקְטִיקָה שֶׁל מַעֲרֶכֶת הַמִּשְׁפָּט וּמַחֲלִיט לְהִתְעַסֵּק בָּאָקָדֶמְיָה שֶׁל לִימּוּד מִשְׁפָּט עִם מִיקּוּד עַל אֶתִיקָה מִקְצוֹעִית.
כְּמוֹ כֵן הוּא רוֹצֶה לְוַודֵּא שֶׁיּוּכַל לִדְאוֹג לְקֶלִי כָּרָאוּי בַּתְּקוּפָה הַקְּרוֹבָה, וְהַשְּׁנַיִים עוֹזְבִים יַחַד אֶת הָעִיר.
