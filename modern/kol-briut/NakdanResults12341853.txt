נְכֵי הַמִּלְחָמָה בַּנָּאצִים
פּוֹרְטָל זֶה כּוֹלֵל אֶת הַזְּכוּיוֹת וְהַשֵּׁירוּתִים הַמּוֹעֳנָקִים לַאֲנָשִׁים שֶׁלָּקוּ בְּנָכוּת, בְּמַהֲלָךְ אוֹ עֵקֶב שֵׁירוּת בְּאֶחָד מִצִּבְאוֹת מַעַצְמוֹת הַבְּרִית (כּוֹלֵל הַצָּבָא הָאָדוֹם), אוֹ בַּמַּחְתֶּרֶת, אוֹ בַּפַּרְטִיזָנִים בַּמִּלְחָמָה נֶגֶד גֶּרְמַנְיָה וּבְנֵי בְּרִיתָהּ.
אִם אַתֶּם מוּגְדָּרִים כְּנִיצּוֹלֵי שׁוֹאָה אָנָּא לָחֲצוּ כָּאן.
אִם אֵינְכֶם נָכִים אֶלָּא מוּגְדָּרִים כְּוָתִיקֵי מִלְחֶמֶת הָעוֹלָם הַשְּׁנִיָּה (וֵטֵרָנִים) לָחֲצוּ כָּאן.
בְּיוֹם 09.06.2014 אִישְּׁרָה הַכְּנֶסֶת אֶת הַצָּעוֹת הַחוֹק הַמְּאַשְּׁרוֹת אֶת הַתָּכְנִית הַלְּאוּמִּית לְסִיּוּעַ לְנִיצּוֹלֵי שׁוֹאָה. לְמֵידָע עַל עִיקְּרֵי הַתָּכְנִית רָאוּ תּוֹכְנִית הַסִּיּוּעַ הַלְּאוּמִּית לְנִיצּוֹלֵי שׁוֹאָה 2014 - הַשְׁווֹאַת תְּנָאִים.

לְמֵידָע נוֹסָף רָאוּ אֶת חוֹבֶרֶת הַזְּכוּיוֹת הַמְּעוּדְכֶּנֶת לְחוֹדֶשׁ יוּלִי 2018 בַּאֲתַר הָרָשׁוּת לִזְכוּיוֹת נִיצּוֹלֵי הַשּׁוֹאָה.

נְכֵה הַמִּלְחָמָה בַּנָּאצִים
זְכוּיוֹת נִיצּוֹלֵי שׁוֹאָה
פּוֹרְטָל הַמֵּידָע הַלְּאוּמִּי לִזְכוּיוֹת נִיצּוֹלֵי הַשּׁוֹאָה בְּיִשְׂרָאֵל, מְהַוּוֹה שַׁעַר כְּנִיסָה לְכָל הַקָּשׁוּר בִּזְכוּיוֹת וּבְשֵׁירוּתִים לָהֶם זַכָּאִים נִיצּוֹלֵי הַשּׁוֹאָה וּבְנֵי מִשְׁפְּחוֹתֵיהֶם וְהַדְּרָכִים לְמִימּוּשָׁן.
מַטְּרַת הַפּוֹרְטָל לְהַגְדִּיל אֶת מִסְפַּר נִיצּוֹלֵי הַשּׁוֹאָה אֲשֶׁר יְמַמְּשׁוּ זְכוּיוֹת וְשֵׁירוּתִים מִתּוֹקֶף הַחוּקִּים וְהַנְּהָלִים הַקַּיָּימִים, וּמִתּוֹךְ הַשֵּׁירוּתִים הָרַבִּים הַמּוּצָּעִים עַל יְדֵי גּוֹרְמִים שׁוֹנִים לְנִיצּוֹלֵי שׁוֹאָה בְּיִשְׂרָאֵל. כְּדֵי לְהַכִּיר אֶת זְכוּיוֹת נִיצּוֹלֵי הַשּׁוֹאָה כְּדַאי לְהַכִּיר גַּם אֶת אֵת מַחְשְׁבוֹן הַזְּכוּיוֹת וְאֶת חוֹבְרוֹת הַזְּכוּיוֹת בַּאֲתַר הָרָשׁוּת לִזְכוּיוֹת נִיצּוֹלֵי הַשּׁוֹאָה.
