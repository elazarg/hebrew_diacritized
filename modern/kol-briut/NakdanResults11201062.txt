סְבִיבָה בְּטוּחָה וְהִתְנַהֲגוּת בְּרִיאָה
​
הַקְנָיַית הֶרְגֵּלֵי הִתְנַהֲגוּת בְּרִיאָה לַיֶּלֶד וִיצִירַת סְבִיבָה בְּטוּחָה לְמֶרְחַב פְּעוּלָּה עֲבוּרוֹ, מְאַפְשְׁרוֹת לִילָדִים לִגְדּוֹל וּלְהִתְפַּתֵּחַ בִּבְרִיאוּת וּבַבִּטְחָה.
בָּאֲתָר זֶה תִּמְצְאוּ הַמְלָצוֹת כֵּיצַד לִיצוֹר סְבִיבָה בְּטוּחָה עֲבוּר הַיְּלָדִים וְאֵיךְ לִשְׁמוֹר עֲלֵיהֶם מִפְּנֵי מִפְגָּעִים.

הַמְלָצוֹת לַקַּיִץ - חֲשִׂיפָה לְשַׁמֵּשׁ וּמְנִיעַת הִתְיַיבְּשׁוּת -  קַיִץ בְּיִשְׂרָאֵל מְזוֹהֶה אֵצֶל רוּבֵּינוּ עִם חוֹפֶשׁ, שֶׁמֶשׁ, יָם וְטִיּוּלִים.  בִּתְקוּפַת הַחוֹפֶשׁ הַגָּדוֹל, יְלָדִים נִמְצָאִים שָׁעוֹת אֲרוּכּוֹת מִחוּץ לַבַּיִת וְחָשׁוּב לְהַקְפִּיד לִשְׁמוֹר עַל מִסְפַּר כְּלָלִים בַּחֲשִׂיפָה לַשֶּׁמֶשׁ וּשְׁתִיַּית מַיִם.
 
בְּטִיחוּת יְלָדִים בַּחוֹרֶף - כְּלָלֵי זְהִירוּת לִבְטִיחוּת וּבְרִיאוּת הַיְּלָדִים בַּחוֹרֶף בַּבַּיִת וּבַחוּץ.
 
בְּטִיחוּת בְּרֶכֶב - הִיפָּגְעוּת שֶׁל יְלָדִים כְּתוֹצָאָה מֵהַשְׁאָרָתָם בְּרֶכֶב הִיא אַחַת מֵהַפְּגִיעוֹת הַבּוֹלְטוֹת בְּעוֹנַת הַקַּיִץ.  מֵידָע לְהוֹרִים כֵּיצַד נִיתָּן לִמְנוֹעַ שִׁכְחַת יְלָדִים בְּרֶכֶב.
 
בְּטִיחוּת יְלָדִים בְּחַגִּים - כְּלָלֵי בְּטִיחוּת חֲשׁוּבִים וּפְשׁוּטִים לִבְטִיחוּת יְלָדִים בַּחַגִּים, שֶׁיַּהַפְכוּ אֶת הַחַג לְבָטוּחַ וְשָׂמֵחַ.
 
הַדְרָכָה לְהוֹרֵי יְלָדִים בַּגִּיל הָרַךְ - כְּלָלֵי זְהִירוּת וְהַנְחָיוֹת לְגַבֵּי מָה שֶׁכְּדַאי לְאַפְשֵׁר לַיֶּלֶד בִּתְחוּמֵי הַנַּיָּידוּת, הַלְּמִידָה, הַמִּשְׂחָק, הָאֲכִילָה וְהַהִתְפַּתְּחוּת.
 
בְּדִיקוֹת הַכְּרוּכוֹת בִּקְרִינָה לַיְּלָדִים - בַּשָּׁנִים הָאַחֲרוֹנוֹת גּוֹבֶרֶת הַמּוֹדָעוֹת לַסִּיכּוּנִים הָאֶפְשָׁרִיִּים מִבְּדִיקוֹת רְפוּאִיּוֹת הַכְּרוּכוֹת בִּקְרִינָה מְיַינֶּנֶת, וּבִמְיוּחָד בִּילָדִים. כִּמְבוּגָּרִים הַדּוֹאֲגִים לִבְרִיאוּת יְלָדֵינוּ, בְּאַחְרָיוּתֵנוּ לִשְׁאוֹל אֶת הָרוֹפֵא הַמְּטַפֵּל אוֹדוֹת הַחֲלוּפוֹת הָאֶפְשָׁרִיּוֹת, לְוַודֵּא כִּי נַעֲשָׂה שִׁימּוּשׁ בְּאֶמְצְעֵי מִיגּוּן מַתְאִימִים בִּזְמַן הַבְּדִיקָה, אַךְ עָלֵינוּ לִזְכּוֹר לֹא לִמְנוֹעַ מִיְּלָדֵינוּ בְּדִיקָה רְפוּאִית חִיּוּנִית רַק בְּשֶׁל חֲשִׂיפָה אֶפְשָׁרִית לִקְרִינָה.
 
קְרִינָה סֵלוֹלָארִית - מִשְׂרַד הַבְּרִיאוּת מַמְלִיץ לְהוֹרִים לְצַמְצֵם בְּמִידַּת הָאֶפְשָׁר אֶת חֲשִׂיפַת הַיְּלָדִים לְטֵלֵפוֹנִים סֵלוֹלָרִיִּים, לִשְׁקוֹל אֶת גִּיל הַתְחָלַת הַשִּׁימּוּשׁ, לְצַמְצֵם אֶת מִידַּת הַשִּׁימּוּשׁ וּבְכָל מִקְרֶה לְהַקְפִּיד עַל שִׁימּוּשׁ בְּאוֹזְנִיּוֹת (לֹא אַלְחוּטִיּוֹת) אוֹ רַמְקוֹל בְּעֵת שִׁימּוּשׁ בַּסֵּלוֹלָרִי.

עֲשַׁן טַבָּק סְבִיבָתִי - הַסּוֹכְנוּת הַבֵּין-לְאוּמִּית לְחֵקֶר הַסַּרְטָן (IARC - International Agency for Research on Cancer) סִיוְּוגָה אֶת עָשָׁן הַטַּבָּק הַסְּבִיבָתִי כְּגוֹרֵם מְסַרְטֵן לִבְנֵי אָדָם. חֲשִׂיפַת יְלָדִים לְעָשָׁן טַבָּק סְבִיבָתִי עֲלוּלָה לִגְרוֹם לְהַשְׁפָּעוֹת הִתְפַּתְּחוּתִיּוֹת וּנְשִׁימָתִיּוֹת שׁוֹנוֹת

קִישּׁוּרִים רֵלֵוַונְטִים
