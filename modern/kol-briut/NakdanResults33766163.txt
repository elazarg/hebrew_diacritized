מְצַלְּמִים? מְשַׁתְּפִים? חוֹשְׁבִים! סוֹף שִׁיתּוּף בַּמַּחְשָׁבָה תְּחִילָּה – הַשָּׁבוּעַ הַלְּאוּמִּי לִגְלִישָׁה בְּטוּחָה 2020
הַשָּׁבוּעַ הַלְּאוּמִּי לִגְלִישָׁה בְּטוּחָה בְּרֶשֶׁת יְצוּיַּן הַשָּׁנָה בֵּין 9 לְ-14 בְּפֶבְּרוּאָר

אֲנַחְנוּ חַיִּים, נוֹשְׁמִים וְחוֹלְמִים בַּוִּויזוּאָלִית. הַטֵּלֵפוֹן הֶחָכָם מְצוּיָּד בְּמַצְלֵמָה שֶׁמִּשְׁתַּכְלֶלֶת תָּדִיר, אֶפְשָׁרוּת הַהַעֲבָרָה בְּוַואטְסְאַפּ זְמִינָה בִּלְחִיצַת כַּפְתּוֹר וְהָאִינְסְטַגְרָם לֹא יוֹדֵעַ שׂוֹבַע. הַמָּקוֹם הַמֶּרְכָּזִי שֶׁהַתְּמוּנוֹת וְהַסִּרְטוֹנִים תּוֹפְסִים בְּחַיֵּינוּ וְהַדִּינָמִיּוּת שֶׁלָּהֶם לִפְעָמִים מַשְׁכִּיחִים מֵאִיתָּנוּ שֶׁתְּמוּנָה עֲדַיִין שָׁוָוה אֶלֶף מִילִּים וְהַשְׁפָּעָתָהּ יְכוֹלָה לִהְיוֹת קָשָׁה מִנָּשׂוֹא. 

שִׁיתּוּף שֶׁל תְּמוּנוֹת וְסִרְטוֹנִים הוּא בִּיטּוּי לְצוֹרֶךְ טִבְעִי בְּשַׁיָּיכוּת וּבְנִרְאוּת וְאֵלֶּה מְעוֹדְדִים אֶת הָרָצוֹן לִצְבּוֹר לַיְיקִים וְעוֹקְבִים. הִתְנַהֲלוּת זוֹ עֲלוּלָה לְהוֹבִיל לְמַצְּבֵי סִיכּוּן וּפְגִיעָה. לָכֵן נָקֵל לִרְאוֹת שֶׁהַנּוֹשֵׂא הַשָּׁנָה נוֹגֵעַ בְּמֶרְחֲבֵי הַחַיִּים הַשּׁוֹנִים - מְנִיעַת פְּגִיעוֹת וּבִכְלָלָן פְּגִיעוֹת מִינִיּוֹת, שְׁמִירָה עַל הַפְּרָטִיּוּת וְעַל "הַתַּדְמִית הַדִּיגִיטָלִית", וְגַם הַהֶיבֵּט הַפְּלִילִי ("חוֹק הַסִּרְטוֹנִים"), שֶׁאֵינוֹ מוּכָּר דַּיּוֹ בְּקֶרֶב יְלָדִים וְהוֹרִים. 
הַשָּׁנָה שָׂמִים סוֹף לִפְגִיעוֹת מִתְּמוּנוֹת וְסִרְטוֹנִים:

מַטְּרוֹת הַשָּׁבוּעַ הַלְּאוּמִּי
מִתּוֹךְ הַכָּרָה בַּמָּקוֹם הַמֶּרְכָּזִי שֶׁתּוֹפֵס הָאִינְטֶרְנֶט בְּחַיֵּינוּ, קָבַע הָאִיחוּד הָאֵירוֹפֵּאִי אֶת יוֹם הָאִינְטֶרְנֶט הַבָּטוּחַ (Safer Internet Day) בְּכָל שָׁנָה בִּתְחִילַּת פֶבְּרוּאָר. יוֹם זֶה מוּקְדָּשׁ לִיצִירַת מֶרְחָב אִינְטֶרְנֵטִי בָּטוּחַ עֲבוּר כָּל הַמִּשְׁתַּמְּשִׁים, וּבְרֹאשָׁם הַיְּלָדִים. 
בְּיִשְׂרָאֵל מוּקְדָּשׁ שָׁבוּעַ שָׁלֵם לְנוֹשֵׂא הַגְּלִישָׁה הַבְּטוּחָה זוֹ הַשָּׁנָה הַ-14 בִּרְצִיפוּת. יִשְׂרָאֵל חוֹבֶרֶת בְּשָׁבוּעַ זֶה לְיוֹתֵר מִ-130 מְדִינוֹת שֶׁיְּצַיְּינוּ בְּ-11 בְּפֶבְּרוּאָר אֶת יוֹם הָאִינְטֶרְנֶט הַבָּטוּחַ הַבֵּין-לְאוּמִּי. 
יוֹם הָאִינְטֶרְנֶט הַבָּטוּחַ הַבֵּין-לְאוּמִּי וְהַשָּׁבוּעַ הַלְּאוּמִּי לִגְלִישָׁה בְּטוּחָה מוּקְדָּשִׁים מִדֵּי שָׁנָה לְהַעֲלָאַת הַמּוּדָעוּת בְּקֶרֶב יְלָדִים וְנוֹעַר, הוֹרִים וְאַנְשֵׁי מִקְצוֹעַ בָּאָרֶץ וּבָעוֹלָם לְשִׁימּוּשׁ נָבוֹן וְאַחְרַאי בְּרֶשֶׁת הָאִינְטֶרְנֶט וּלְהִתְנַהֲלוּת עֶרְכִּית, אַחְרָאִית וּמוּגֶנֶת בָּרֶשֶׁת. בְּשָׁבוּעַ זֶה עוֹסְקִים גּוֹרְמֵי הַמִּקְצוֹעַ בַּתְּחוּם בְּקִידּוּם הִתְנַהֲלוּת בְּטוּחָה וְעֶרְכִּית בָּרֶשֶׁת וּבְהַעֲלָאַת הַמּוֹדָעוֹת לְמַצְּבֵי הַסִּיכּוּן הַטְּמוּנִים בָּרֶשֶׁת, לִמְנִיעַת פְּגִיעָה וּלְהִתְמוֹדְדוּת עִימָּהּ, לַחֲשִׁיבוּת שֶׁל הֲגָנַת הַפְּרָטִיּוֹת שֶׁל הָאַחֵר וְלִשְׁמִירָה עַל צִנְעַת הַפְּרָט.

שׁוּתָּפִים
הַמַּטֶּה הַלְּאוּמִּי לַהֲגָנָה עַל יְלָדִים בָּרֶשֶׁת שָׂם לוֹ לְמַטָּרָה לְקַדֵּם אֶת נוֹשֵׂא הַגְּלִישָׁה הַבְּטוּחָה וְהַסּוֹבְלָנוּת בְּרֶשֶׁת בְּקֶרֶב יְלָדִים וּבְנֵי נוֹעַר. הַמַּטֶּה הַלְּאוּמִּי מוֹבִיל אֶת הַשָּׁבוּעַ הַלְּאוּמִּי לִגְלִישָׁה בְּטוּחָה בְּרֶשֶׁת בְּשִׁיתּוּף הַמִּשְׂרָד לְבִיטְחוֹן הַפְּנִים (בְּאֶמְצָעוּת הָרָשׁוּת לְמַאֲבָק בְּאַלִּימוּת, בְּסַמִּים וּבְאַלְכּוֹהוֹל – בִּיטָּחוֹן קְהִילָּתִי), מִשְׁטֶרֶת יִשְׂרָאֵל, מִשְׂרַד הַחִינּוּךְ, מִשְׂרַד הַבְּרִיאוּת, מִשְׂרַד הָעֲבוֹדָה, הָרְוָוחָה וְהַשֵּׁירוּתִים הַחֶבְרָתִיִּים וּמִשְׂרַד הַמִּשְׁפָּטִים וְחוֹבְרִים אֵלֵינוּ שׁוּתָּפִים נוֹסָפִים מֵהַמִּגְזָר הַצִּיבּוּרִי, הַמִּגְזָר הַשְּׁלִישִׁי, הַמִּגְזָר הַפְּרָטִי וּמֵהַמִּגְזָר הָאָקָדֵמִי.  

פְּעִילוּיוֹת
בְּמַהֲלַךְ הַשָּׁבוּעַ יִתְקַיְּימוּ בְּרַחֲבֵי הָאָרֶץ פְּעִילוּיוֹת רַבּוֹת וּמְגוּוֹנוֹת שֶׁיִּתְמַקְּדוּ בְּצִילּוּם, שִׁיתּוּף וַהֲפָצָה שֶׁל תְּמוּנוֹת וְסִרְטוֹנִים – פְּעוּלּוֹת שֶׁנּוֹתְנוֹת מִצַּד אֶחָד מַעֲנֶה לַצּוֹרֶךְ הַטִּבְעִי שֶׁל יְלָדִים וְנוֹעַר בְּשַׁיָּיכוּת וּבַנִּרְאוּת וּמִצַּד שֵׁנִי בְּהַעֲלָאַת הַמּוּדָעוּת לְמַצְּבֵי הַסִּיכּוּן שֶׁעֲלוּלִים לְהִיוָּוצֵר כְּשֶׁפְּעוּלּוֹת אֵלֶּה נַעֲשׂוֹת לְלֹא מַחְשָׁבָה תְּחִילָּה. 
יְלָדִים וּבְנֵי נוֹעַר יוּזְמְנוּ לִבְחוֹן אֶת פְּרַקְטִיקוֹת הַצִּילּוּם וְהַשִּׁיתּוּף שֶׁלָּהֶם, אֶת הַבְּעָיָיתִיּוּת שֶׁבַּהֲפָצָה לְלֹא שִׁיקּוּל דַּעַת, אֶת מְקוֹמָם שֶׁל הַתְּמוּנוֹת וְהַסִּרְטוֹנִים בְּהַבְנָיַית הַתַּדְמִית הַדִּיגִיטָלִית וְאֶת הַמִּגְבָּלוֹת הַחוּקִּיּוֹת עַל צִילּוּם, שִׁיתּוּף וַהֲפָצָה שֶׁל תְּכָנִים. בַּד בְּבַד הֵם יְלַמְּדוּ עַל הַהֶיבֵּטִים הַחִיּוּבִיִּים שֶׁבַּצִּילּוּם כִּכְלִי לַהַבָּעָה עַצְמִית וּבְשִׁיתּוּף כְּמַעֲשֵׂה חַבְרֵי וּמִקֶּרֶב. 
נוֹסָף עַל הַפְּעִילוּיוֹת לִילָדִים וְלִבְנֵי הַנּוֹעַר בַּמִּסְגְּרוֹת הַחִינּוּכִיּוֹת הַשּׁוֹנוֹת יוּצְּעוּ גַּם פְּעִילוּיוֹת רַבּוֹת לְהוֹרִים וּפְעִילוּיוֹת בַּקְּהִילָּה שֶׁמַּטְּרָתָן לְקַדֵּם מְעוֹרָבוּת הוֹרִים בְּחַיֵּי הָרֶשֶׁת שֶׁל יַלְדֵיהֶם וּלְעוֹדֵד שִׂיחַ פָּתוּחַ בֵּינָם לְבֵין הַיְּלָדִים. 

בְּמַהֲלַךְ הַשָּׁבוּעַ הַלְּאוּמִּי יַעַרְכוּ הַמַּטֶּה הַלְּאוּמִּי וְהַמִּשְׂרָדִים הַשּׁוּתָּפִים מִגְוַון פְּעִילוּיוֹת יִיעוּדִיּוֹת בְּדָגֵשׁ עַל נוֹשֵׂא הַצִּילּוּם וְהַהֲפָצָה שֶׁל תְּמוּנוֹת וְסִרְטוֹנִים:

הַמַּטֶּה הַלְּאוּמִּי לַהֲגָנָה עַל יְלָדִים בָּרֶשֶׁת  105
"הַכֵּה אֶת הַמַּטֶּה": פִּינָּה חֲדָשָׁה בַּעֲרוּץ הַהַסְבָּרָה שֶׁל הַמַּטֶּה הַלְּאוּמִּי בַּפֶיְיסְבּוּק. מוּמְחֶה מֵהַמַּטֶּה הַלְּאוּמִּי, מוּמְחֶה "מֵהַשֶּׁטַח" וְנַעַר אוֹ נַעֲרָה יַעֲנוּ לִשְׁאֵלוֹת הַגּוֹלְשִׁים בַּנּוֹשְׂאִים הַחַמִּים וְהַבּוֹעֲרִים בִּתְחוּם הַבִּיטָּחוֹן שֶׁל יְלָדִים וּבְנֵי נוֹעַר בָּרֶשֶׁת. הַפִּינָּה תַּעֲלֶה לָרִאשׁוֹנָה בְּמַהֲלַךְ הַשָּׁבוּעַ הַלְּאוּמִּי וּלְאַחַר מִכֵּן פַּעַם בְּחוֹדֶשׁ. 
פְּעִילוּת מְשׁוּתֶּפֶת עִם מוֹבִילֵי דַּעַת קָהָל: עֲבוֹדָה מְשׁוּתֶּפֶת עִם צְעִירִים מוֹבִילֵי דֵּעָה בְּקֶרֶב בְּנֵי נוֹעַר שֶׁמַּטְּרָתָהּ לְקַדֵּם אֶת מִסְרֵי הַשָּׁבוּעַ הַלְּאוּמִּי בְּקֶרֶב בְּנֵי הַנּוֹעַר. 
מִפְגָּשִׁים מְקוּוֹנִים אִינְטֶרְאַקְטִיבִיִּים בְּבָתֵּי הַסֵּפֶר: הַמַּטֶּה הַלְּאוּמִּי מִצְטָרֵף לְמֵיזָם "אָקָדֶמְיָה בָּרֶשֶׁת" וְיַעֲבִיר בְּמַהֲלַךְ הַשָּׁבוּעַ אַרְבָּעָה מַשְׁדְּרִים אִינְטֶרְאַקְטִיבִיִּים בְּשִׁידּוּר חַי – שְׁנַיִים לְיַלְדֵי יְסוֹדִי (אֶחָד בְּעִבְרִית וְאֶחָד בְּעַרְבִית) בְּנוֹשֵׂא שְׁמִירַת הַפְּרָטִיּוּת וּשְׁנַיִים לִבְנֵי נוֹעַר (אֶחָד בְּעִבְרִית וְאֶחָד בְּעַרְבִית) בְּנוֹשֵׂא צִילּוּם וַהֲפָצָה שֶׁל תְּכָנִים מִינִיִּים. בְּכָל מִשְׁדָּר יוֹפִיעוּ נָצִיג מִשְׂרַד הַחִינּוּךְ בְּמוֹקֵד 105 וְשׁוֹטֵר, שֶׁיַּעֲבִירוּ הַרְצָאָה קְצָרָה וְיַעֲנוּ בְּשִׁידּוּר חַי לִשְׁאֵלוֹת הַתַּלְמִידִים. מוּמְחֵי הַמַּטֶּה הַלְּאוּמִּי יִהְיוּ זְמִינִים גַּם בַּשָּׁעוֹת שֶׁלְּאַחַר מִכֵּן לַעֲנוֹת לִשְׁאֵלוֹת הַקָּהָל עַל גַּבֵּי דַּף הַפֶיְיסְבּוּק שֶׁל הַמַּטֶּה הַלְּאוּמִּי. 
פָּאנֵלִים מְקוּוֹנִים בְּמוֹסָדוֹת לְהַשְׂכָּלָה גְּבוֹהָה: פָּאנֵלִים בָּאוּנִיבֶרְסִיטָאוֹת וּבַמִּכְלָלוֹת שׁוֹנוֹת בָּאָרֶץ שֶׁיִּשְׁתַּתְּפוּ בָּהֶם נָצִיגֵי אָקָדֶמְיָה וְנָצִיגֵי הַמַּטֶּה הַלְּאוּמִּי. הַפָּאנֵלִים יְשׁוּדְּרוּ בְּשִׁידּוּר חַי בַּעֲרוּצֵי הָרַדְיוֹ שֶׁל הַמּוֹסָדוֹת הַמְּאָרְחִים.
הַרְצָאוֹת שֶׁל עוֹבְדֵי הַמַּטֶּה הַלְּאוּמִּי: בְּמַהֲלַךְ הַשָּׁבוּעַ הַלְּאוּמִּי לִגְלִישָׁה בְּטוּחָה יַעֲבִירוּ עוֹבְדֵי הַמַּטֶּה הַלְּאוּמִּי הַרְצָאוֹת לִילָדִים וּבְנֵי נוֹעַר מְאוּשְׁפָּזִים, לַחֲנִיכֵי פְּנִימִיּוֹת, לְהוֹרִים בִּמְקוֹמוֹת עֲבוֹדָה, בְּמִסְגְּרוֹת אָקָדֵמִיּוֹת, בְּאֵירוּעִים לְצִיּוּן הַשָּׁבוּעַ הַלְּאוּמִּי בָּרָשׁוּיוֹת הַמְּקוֹמִיּוֹת, וְעוֹד. 


הַמִּשְׂרָד לְבִיטְחוֹן הַפְּנִים – הָרָשׁוּת לְמַאֲבָק בְּאַלִּימוּת, בְּסַמִּים וּבְאַלְכּוֹהוֹל
בְּמִסְגֶּרֶת שְׁבוּעַ הַגְּלִישָׁה הַבְּטוּחָה תּוֹבִיל הָרָשׁוּת מִגְוַון פְּעִילוּיוֹת בָּרָשׁוּיוֹת הַמְּקוֹמִיּוֹת בְּרַחֲבֵי הָאָרֶץ הַכּוֹלְלוֹת הַרְצָאוֹת, סַדְנָאוֹת וְאֵירוּעִים שׁוֹנִים בְּדָגֵשׁ עַל פְּעִילוּת בַּקְּהִילָּה, בְּמִסְגְּרוֹת הַחִינּוּךְ הַבִּלְתִּי פוֹרְמָלִי וּבְקֶרֶב הַהוֹרִים בִּשְׁעוֹת אַחַר הַצָּהֳרַיִים וְהָעֶרֶב. 
הַפְּעִילוּיוֹת יְבוּצְּעוּ בְּתֵיאוּם וּבְשִׁיתּוּף כְּלָל הַגּוֹרְמִים הָרֵלֵוַונְטִיִּים בַּיִּישּׁוּב וְיִכְלְלוּ ימֵי עִיּוּן, הַרְצָאוֹת מוּמְחִים וְשׁוֹטְרִים, סַדְנָאוֹת מְנִיעָה, פָּאנֵלִים מִקְצוֹעִיִּים וּכְנָסִים יִישּׁוּבִיִּים, הַכְשָׁרוֹת וְהַדְרָכוֹת לְאַנְשֵׁי מִקְצוֹעַ וְ"שׁוֹמְרֵי סַף" נוֹסָפִים בַּקְּהִילָּה, הִשְׁתַּתְּפוּת מַדְרִיכֵי הַמּוּגַנּוֹת בִּפְעִילוּיוֹת "הַפְסָקָה פְּעִילָה" בְּמִסְגְּרוֹת הַחִינּוּךְ הַפוֹרְמָלִי (בְּהוֹבָלַת צִוְותֵי הַחִינּוּךְ), יִזּוּם פְּעִילוּיוֹת בִּתְנוּעוֹת הַנּוֹעַר, הַרְצָאוֹת לְהוֹרִים, (פְּעוּלּוֹת הוֹרִים יְזוּמוֹת בְּוַועֲדֵי הוֹרִים יִישּׁוּבִיִּים וּמִתְנַדְּבֵי סַיְּירוֹת הוֹרִים, פִּרְסוּמִים וּפְעוּלּוֹת הַסְבָּרָה בַּאֲתָרֵי הָרָשׁוּיוֹת הַמְּקוֹמִיּוֹת וּבַעֲרוּצֵי הַמֶּדְיָה הַחֶבְרָתִיִּים, יִזּוּם פְּעִילוּיוֹת בְּאֶמְצָעוּת קְבוּצוֹת מַנְהִיגוֹת נוֹעַר וְעוֹד. 

מִשְׁטֶרֶת יִשְׂרָאֵל

מֵאוֹת שׁוֹטְרִים מִמְּדוֹר נוֹעַר – חֲטִיבַת הַחֲקִירוֹת, מֵהַשִּׁיטּוּר הַקְּהִילָּתִי, מִיְּחִידַת הֲסַייְבֶּר וּמִיְּחִידָה 105 בַּלַּהַב 433 יַעֲבִירוּ הַרְצָאוֹת בְּבָתֵּי הַסֵּפֶר וּבַקְּהִילָּה. 
אֶת הַפְּעִילוּת מְתַכְלֵל מְדוֹר נוֹעַר בְּמִשְׁטֶרֶת יִשְׂרָאֵל וְהִיא מְבוּסֶּסֶת עַל מַצֶּגֶת הַסְבָּרָה בְּנוֹשֵׂא גְּלִישָׁה בְּטוּחָה שֶׁפִּיתֵּחַ מְדוֹר נוֹעַר בְּסִיּוּעַ תְּחוּם מִשְׂרַד הַחִינּוּךְ בַּמַּטֶּה הַלְּאוּמִּי לְשִׁימּוּשָׁם שֶׁל הַשּׁוֹטְרִים. 

מִשְׂרַד הַחִינּוּךְ
בַּשָּׁבוּעַ הַלְּאוּמִּי יְעוֹדֵד מִשְׂרַד הַחִינּוּךְ אֶת הַתַּלְמִידִים לְהַפְעִיל חֲשִׁיבָה בִּיקּוֹרְתִּית עַל אוֹדוֹת הַצּוֹרֶךְ שֶׁלָּהֶם לְצַלֵּם אֶת עַצְמָם וְאֶת חַבְרֵיהֶם וְלִבְחוֹן אֶת הַצּוֹרֶךְ בְּשִׁיתּוּף וּבַהֲפָצַת תְּכָנִים בְּדָגֵשׁ עַל תְּמוּנוֹת וְסִרְטוֹנִים. 
הַצְּוָותִים הַחִינּוּכִיִּים יַקְנוּ כֵּלִים וִיעוֹדְדוּ אֶת הַתַּלְמִידִים לְהִתְנַהֵל מִתּוֹךְ שִׁיקּוּל דַּעַת, בְּאוֹפֶן מְכַבֵּד וְרָגִישׁ כְּלַפֵּי עַצְמָם וּכְלַפֵּי אֲחֵרִים.
מִשְׂרַד הַחִינּוּךְ נֶעֱרַךְ בְּמִגְוַון רָחָב שֶׁל תְּכָנִים וּפְעִילוּיוֹת לְתַלְמִידִים, לְצִוְותֵי הַחִינּוּךְ וְלַהוֹרִים. כְּלַל הַתְּכָנִים יְפוּרְסְמוּ בְּעָלוֹן מְקוּוֹן לְצִוְותֵי הַחִינּוּךְ.
לַתַּלְמִידִים:
שִׁיעוּרֵי כִּישּׁוּרֵי חַיִּים לְתַלְמִידִים, מוּתְאָמִים לְשִׁכְבוֹת הַגִּיל הַשּׁוֹנוֹת וְלַחֲבֵרוֹת הַשּׁוֹנוֹת
אָקָדֶמְיָה בָּרֶשֶׁת – הַרְצָאוֹת מִקַּוּוָּנוּת לְתַלְמִידִים שֶׁיַּעֲבִירוּ נָצִיגֵי מִשְׂרַד הַחִינּוּךְ, נְצִיגֵי עֲמוּתּוֹת וַחֲבָרוֹת שׁוֹנוֹת
הַפְעָלַת קְבוּצוֹת שֶׁל מַנְהִיגֵי נוֹעַר וּמַנְהִיגוּת נוֹעַר לִגְלִישָׁה בְּטוּחָה בָּרֶשֶׁת
הַרְצָאוֹת בְּבָתֵּי הַסֵּפֶר שֶׁיַּעֲבִירוּ שׁוֹטְרִים 
הֲפָצַת עֲלוֹנֵי מֵידָע לְתַלְמִידִים
לְצִוְותֵי חִינּוּךְ:
הַכְשָׁרוֹת לְצִוְותֵי הַחִינּוּךְ בְּבָתֵּי הַסֵּפֶר וּכְנָסִים בִּמְחוֹזוֹת.
לְהוֹרִים:
מַעְגְּלֵי שִׂיחַ לְתַלְמִידִים, לְהוֹרִים וּלְצִוְותֵי חִינּוּךְ
הֲפָצַת עֲלוֹנֵי מֵידָע לְהוֹרִים
נִיתָּן לְהֵירָשֵׁם לִפְעִילוּיוֹת וּלְהִתְעַדְכֵּן בְּחוֹמְרֵי הַלִּימּוּד בֶּעָנָן הַחִינּוּכִי וּבַאֲתַר שפ"ינט.

מִשְׂרַד הַבְּרִיאוּת
מִשְׂרַד הַבְּרִיאוּת יִתְמַקֵּד בְּמַהֲלַךְ הַשָּׁבוּעַ בִּפְעִילוּיוֹת לִילָדִים מְאוּשְׁפָּזִים בְּבָתֵּי הַחוֹלִים, הַנִּמְצָאִים תְּקוּפוֹת מְמוּשָּׁכוֹת בְּנִיתּוּק גֵּיאוֹגְרָפִי מִקְּרוֹבֵיהֶם וְחַבְרֵיהֶם. 
הַפְּעִילוּיוֹת יִכְלְלוּ מַעְגְּלֵי שִׂיחַ מְשׁוּתָּפִים לִילָדִים וְהוֹרֵיהֶם בְּמַחְלְקוֹת הָאִשְׁפּוּז וּבְמִסְגֶּרֶת בֵּית הַסֵּפֶר שֶׁל בֵּית הַחוֹלִים וְגַם בִּיקּוּרִים אִישִׁיִּים לִילָדִים הַמְּרוּתָּקִים לַמִּיטָּה. 
בְּהֶמְשֵׁךְ הַחוֹדֶשׁ יִינָּתְנוּ הַרְצָאוֹת לְצִוְותֵי הַמַּחְלָקוֹת. כְּמוֹ כֵן יוּפְצוּ חוֹמְרֵי הַסְבָּרָה בַּאֲתַר הַמִּשְׂרָד. 

מִשְׂרַד הָעֲבוֹדָה, הָרְוָוחָה וְהַשֵּׁירוּתִים הַחֶבְרָתִיִּים
מִשְׂרַד הָרְוָוחָה יַעֲבִיר בְּמַהֲלַךְ הַשָּׁבוּעַ הַרְצָאוֹת בַּפְּנִימִיּוֹת הַטִּיפּוּלִיּוֹת שֶׁבְּאַחְרָיוּתוֹ וְיָפִיץ חוֹמְרֵי הַסְבָּרָה בְּמַחְלְקוֹת הָרְוָוחָה, בַּמִּסְגְּרוֹת בַּקְּהִילָּה וּבַאֲתַר הַמִּשְׂרָד.

הַמִּגְזָר הָעִסְקִי, חֶבְרוֹת טֶכְנוֹלוֹגְיָה, עֲמוּתּוֹת
נְצִיגִים שֶׁל אִרְגּוּנִים רַבִּים מֵהַמִּגְזָרִים הָעִסְקִי וְהַשְּׁלִישִׁי יַעֲבִירוּ הַרְצָאוֹת פְרוֹנְטָלִיּוֹת וְהַרְצָאוֹת מִקַּוּוָּנוּת בְּבָתֵּי הַסֵּפֶר וּבַקְּהִילָּה.
