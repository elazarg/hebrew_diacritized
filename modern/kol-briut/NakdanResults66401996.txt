בְּטִיחוּת יְלָדִים בְּמַיִם וּמְנִיעַת טְבִיעָה

טְבִיעָה מִתְרַחֶשֶׁת בִּמְהִירוּת וּבִדְמָמָה. לָרוֹב, בְּעֵת טְבִיעָה לֹא שׁוֹמְעִים בְּכִי אוֹ קוֹלוֹת מְצוּקָה. טְבִיעָה יְכוֹלָה לְהִתְרַחֵשׁ תּוֹךְ שְׁנִיּוֹת וּבְדֶרֶךְ כְּלָל מִתְרַחֶשֶׁת בְּהֶעְדֵּר הַשְׁגָּחָה אוֹ בְּעֵת נֶתֶק קָצָר בְּהַשְׁגָּחָה.

​​טְבִיעָה הִינָּהּ סִיבַּת הַמָּוֶות הַמּוֹבִילָה בַּקַּיִץ בְּקֶרֶב הַיְּלָדִים.
תִּינוֹקוֹת וִילָדִים קְטַנִּים נִמְשָׁכִים לְמִשְׂחָק בְּמַיִם וַעֲלוּלִים לִטְבּוֹעַ אֲפִילּוּ בְּמַיִם שֶׁעוֹמְקָם 10 ס"מ. זֹאת, מִכֵּיוֹוֹן שֶׁרֹאשָׁם כָּבֵד בְּיַחַס לְמִשְׁקָל גּוּפָם, וְהֵם עֲלוּלִים לְהִתְהַפֵּךְ לְתוֹךְ דַּלֵּי מַיִם, אַמְבַּטְיָה אוֹ גִּיגִית.
יְלָדִים הַשּׂוֹרְדִים טְבִיעָה וְנוֹתְרִים לְלֹא מוּגְבָּלוּיוֹת נוֹיְרוֹלוֹגִיּוֹת הֵם אֵלּוּ שֶׁנִּמְשׁוּ מֵהַמַּיִם תּוֹךְ 2 דַּקּוֹת. כַּעֲבוֹר 2 דַּקּוֹת מֵרֶגַע הִטְבִּיעָה מִתְרַחֵשׁ אִיבּוּד הַכָּרָה וְכַעֲבוֹר 4-6 דַּקּוֹת נִגְרָם נֶזֶק מוֹחִי בִּלְתִּי הָפִיךְ. 
 
נְתוּנִים 
מֵאָז שְׁנַת 2008 מֵתוּ 137 יְלָדִים בְּיִשְׂרָאֵל עֵקֶב טְבִיעָה.
60% מֵהַטְּבִיעוֹת הִתְרַחֲשׁוּ בְּחוֹדְשֵׁי יוּנִי - יוּלִי - אוֹגוּסְט.
40% מֵהַיְּלָדִים טָבְעוּ בַּיָּם. 
כִּשְׁלִישׁ מִמִּקְרֵי הַטְּבִיעָה שֶׁהִסְתַּיְּימוּ בְּמָוֶות, הָיוּ שֶׁל יְלָדִים בְּגִילִי 0-4. בְּ-41% מֵהַמִּקְרִים, טָבַע הַיֶּלֶד בְּמֶרְחַב הַבַּיִת.
51% מֵהַטּוֹבְעִים הָיוּ מֵהַמִּגְזָר הַיְּהוּדִי וְ-40% מֵהַמִּגְזָר הָעַרְבִי (הַשְּׁאָר הָיוּ תַּיָּירִים, מְהַגְּרֵי עֲבוֹדָה אוֹ לֹא יָדוּעַ). 

כְּלָלִים לִמְנִיעַת טְבִיעָה
מַשְׁגִּיחִים תָּמִיד וּבְכָל גִּיל!  
עִם יְלָדִים שֶׁאֵינָם יוֹדְעִים לִשְׂחוֹת אוֹ קְטַנִּים מִגִּיל 5 - מַשְׁגִּיחִים בְּאוֹפֶן צָמוּד וּפָעִיל עַל יְדֵי מְבוּגָּר.
הַשְׁגָּחָה צְמוּדָה וּפְעִילָה הִיא הַשְׁגָּחָה בַּמַּיִם עִם הַיֶּלֶד, בְּמֶרְחַק נְגִיעָה מִמֶּנּוּ וּבְקֶשֶׁר עַיִן עִימּוֹ, תּוֹךְ הִתְפַּנּוּת מִכָּל עִיסּוּק אַחֵר (כְּמוֹ שִׂיחָה, קְרִיאָה אוֹ דִּיבּוּר בַּטֵּלֵפוֹן). 
כַּאֲשֶׁר הַבִּילּוּי הוּא חֶבְרָתִי עִם מִשְׁפָּחוֹת וּמְבוּגָּרִים אֲחֵרִים, נִיתָּן לְבַצֵּעַ תּוֹרָנוּת "מַשְׁגִּיחִי מַיִם", בְּמִשְׁמָרוֹת שֶׁל כְּ-15 דַּקּוֹת - כָּל פַּעַם מְבוּגָּר אַחַר יִקַּח אֶת הָאַחְרָיוּת לְהַשְׁגָּחָה מְלֵאָה עַל הַיְּלָדִים וְנִיתּוּק מֵחֶבְרַת הַמְּבוּגָּרִים. 
עִם יְלָדִים שֶׁלָּמְדוּ שְׂחִיָּיה וְשׁוֹחִים - מַשְׁגִּיחִים בְּמַהֲלַךְ כָּל הַשְּׁהִיָּיה בְּמַיִם. 
מִשְׁתַּמְּשִׁים בִּמְצוֹפִים וְגַלְגַּלֵּי יָם, אַךְ לֹא כְּתַחֲלִיף לְהַשְׁגָּחָה פְּעִילָה. 
מְלַמְּדִים יְלָדִים שְׂחִיָּיה הֵחֵל מִגִּיל 4, בְּהֶתְאֵם לִיכוֹלוֹת הַיֶּלֶד.
שׂוֹחִים בַּיָּם וּבַבְּרֵיכָה בְּנוֹכְחוּת מַצִּיל וְנִשְׁמָעִים לְהוֹרָאוֹתָיו.
בַּבַּיִת - מַקְפִּידִים לְרוֹקֵן אֶת הַמַּיִם מִכָּל מִיכַל (דְּלִי, אַמְבָּט, גִּיגִית וּבְרֵיכָה מִתְנַפַּחַת) כְּשֶׁהֵם לֹא בְּשִׁימּוּשׁ, וּמוֹנְעִים גִּישָׁה לְמַאַגְרֵי מַיִם שֶׁלֹּא נִיתָּן לְרוֹקְנָם (כְּמוֹ בְּרֵיכָה פְּרָטִית, מַאַגְרֵי מַיִם בַּטֶּבַע, בּוֹרוֹת מַיִם). 

הַמֵּידָע בַּאֲדִיבוּת אִרְגּוּן "בְּטֶרֶם" לִבְטִיחוּת יְלָדִים.
