בִּיקּוֹרֶת סִפְרוּתִית עַל הַפֶּלֶא שֶׁל אָנָה מֵאֵת אִמָּהּ דּוֹנַהְיוֹ
סֵפֶר בְּסֵדֶר דֵּירוּג שֶׁל שְׁלוֹשָׁה כּוֹכָבִים הַבִּיקּוֹרֶת נִכְתְּבָה בְּיוֹם שֵׁנִי, 1 בְּיוּנִי, 2020
ע"י נִצְחִיָּה

פַּעַם, לִפְנֵי דֵּי הַרְבֵּה שָׁנִים, הִדְרַכְתִּי בַּפְּנִימִיָּה. לְמִסְגֶּרֶת הַתַּפְקִידִים הַשּׁוֹנִים שֶׁהָיוּ עָלַי כְּמַדְרִיכָה - הַשְׁכָּמָה בַּבּוֹקֶר, כִּיבּוּי אוֹרוֹת בִּשְׁעוֹת הָעֶרֶב, פְּעִילוּיוֹת חֶבְרָתִיּוֹת וְלִיוּוּי תַּלְמִידוֹת חוֹלוֹת, נוֹסָף גַּם צַד שֶׁלֹּא חָשַׁבְתִּי מֵרֹאשׁ שֶׁיִּהְיֶה חֵלֶק מֵהַתַּפְקִידִים שֶׁלִּי, אֲבָל הוּא הָיָה. הָיִיתִי צְרִיכָה לְהַשְׁגִּיחַ עַל תַּלְמִידוֹת מְסוּיָּמוֹת וְלִרְאוֹת שֶׁהֵן אוֹכְלוֹת. וּכְמוֹ כֵן לְהַשְׁגִּיחַ בְּאוֹפֶן כְּלָלִי עַל הַתַּלְמִידוֹת שֶׁבְּאַחְרָיוּתִי וְלִבְדּוֹק שֶׁאַף אַחַת מֵהֶן לֹא מַפְסִיקָה לֶאֱכוֹל פִּתְאוֹם, כִּי הָרָצוֹן הַזֶּה מְדַבֵּק.

אָנוֹרֶקְסִיָּה, הַרְעָבָה עַצְמִית, נִרְאֵית לָנוּ מַחֲלָה שֶׁל חֶבְרַת הַשֶּׁפַע שֶׁל הַמֵּאָה הָעֶשְׂרִים, כְּמַעֲנֶה לְתַכְתִּיבֵי אוֹפְנָה וּלְתִקְשׁוֹרֶת הַהֲמוֹנִים הַיּוֹצֶרֶת נוֹרְמוֹת רָזוֹן מְעַוְּותוֹת. אֲבָל הִיא לֹא. זוֹ מַחֲלָה שֶׁיֵּשׁ לָהּ פָּנִים שׁוֹנוֹת וְסִיבּוֹת שׁוֹנוֹת, אֲבָל הִיא לֹא בְּהֶכְרֵחַ מוֹדֶרְנִית. הַסֵּפֶר הַזֶּה מְדַבֵּר עַל אָנוֹרְקְסִיָּה קְדוּמָה, כָּזֹאת עִם אִיצְטְלָה דָּתִית.

לִיב רַייְט הִיא אֲחוֹת אַנְגְּלִיָּה, תַּלְמִידָה שֶׁל פְלוֹרֶנְס נַיְיטִינְגֶּל הַמְּפוּרְסֶמֶת, אֲבָל גַּם בּוֹדֵדָה וּנְטוּלַת מִשְׁפָּחָה. הִיא מְקַבֶּלֶת הַצָּעַת עֲבוֹדָה מֵעֵבֶר לַיָּם, בִּכְפַר אִירִי נִידָּח, וְהִיא כָּל כָּךְ נוֹאֶשֶׁת מִבְּחִינָה מִשְׁפַּחְתִּית וּמִבְּחִינָה כַּלְכָּלִית עַד שֶׁהִיא מַסְכִּימָה לְקַבֵּל אוֹתָהּ. הִיא מִתְקַשָּׁה לְהַשְׁגִּיחַ עַל אָנָה, יַלְדָּה בַּת 11 שֶׁהִפְסִיקָה לֶאֱכוֹל. לִיב מוּזְמֶנֶת בְּשֵׁם וַעֲדָה שֶׁתַּפְקִידָהּ לְוַודֵּא שֶׁהַצּוֹם אֲמִיתִּי, וְלִגְרוֹם לְכָךְ שֶׁהַיַּלְדָּה תֵּיחָשֵׁב קְדוֹשָׁה. הַפְּעָרִים גְּדוֹלִים, מִבְּחִינָה דָּתִית, תַּרְבּוּתִית, סוֹצְיוֹ-אֶקוֹנוֹמִית, וּמִכָּל הֶיבֵּט שֶׁקַּיָּים, וְלִיב מִתְקַשָּׁה לְגַשֵּׁר עַל הַפְּעָרִים. מָה גַּם שֶׁאֲנָשִׁים שׁוֹנִים מִתְעַקְּשִׁים לְהַפְרִיעַ לָהּ בְּתַפְקִידָהּ.

אָז אָנָה אָנוֹרֶקְטִית, אֲבָל הַסִּיפּוּר הוּא לֹא עָלֶיהָ. הַסִּיפּוּר הוּא עַל לִיב, עַל שִׁינּוּי הָעֲמָדוֹת שֶׁהִיא חוֹוָה, עַל הַתְּחוּשׁוֹת הַשּׁוֹנוֹת הָעוֹלוֹת בָּהּ מוּל הַתּוֹפָעוֹת הַמְּשׁוּנּוֹת שֶׁהִיא נִתְקֶלֶת בָּהֶן, וְגַם עַל הַקּוֹשִׁי לִרְאוֹת בִּבְחִינָה מִקָּרוֹב וּבְהַשְׁגָּחָה יוֹמְיוֹמִית סִימָנִים מִתְרַבִּים וְהוֹלְכִים שֶׁל הִדַּרְדְּרוּת. הַסּוֹף טוֹב בְּאוֹפֶן בִּלְתִּי אֶפְשָׁרִי, וְכוֹלֵל סִיפּוּר אַהֲבָה שֶׁהוּא גַּם כֵּן בִּלְתִּי אֶפְשָׁרִי, אֲבָל אִם מִתְעַלְּמִים מֵהֶם, אֶפְשָׁר לִרְאוֹת כָּאן תַּעֲלוּמָה פְּסִיכוֹלוֹגִית מְעַנְיֶינֶת.
