בְּמַעֲלֵה הַכְּבִישׁ הַמּוֹבִיל לִמְלוֹן כְּרָמִים, בּוֹאֲכָה הֲרֵי יְרוּשָׁלַיִם, נִזְכַּרְתִּי בְּסַבְתָּא שֶׁלִּי.
מִי שֶׁהָיְיתָה יְלִידַת תֵּל אָבִיב וְעָבְדָה רוֹב חַיֶּיהָ כְּמַזְכִּירָה בְּמַעֲרֶכֶת עִיתּוֹן "דָּבָר", הִתְרַפְּקָה מֶשֶׁךְ שָׁנִים עַל הַחוּפְשׁוֹת שֶׁנָּהֲגָה לְבַלּוֹת בְּבָתֵּי הַהַבְרָאָה שֶׁל הַהִסְתַּדְּרוּת, חוּפְשׁוֹת פְּשׁוּטוֹת וּצְנוּעוֹת שֶׁכָּל קִסְמָן טָמוּן בֵּין כָּרֵי הַדֶּשֶׁא הַמּוֹרִיקִים לְצֵל הָעֵצִים וְשִׂיאַן טְבִילָה וְשִׁכְשׁוּךְ בִּבְרֵיכָה קְרִירָה וּבְעִיקָּר אֲוִויר הָרִים צָלוּל כַּיַּיִן וּזְמַן חוֹפְשִׁי מְלוֹא הַחוֹפֶן לִקְרוֹא סְפָרִים מוּל הַנּוֹף.

בְּיִשְׂרוֹטֶל לָקְחוּ פִּיסַּת הִיסְטוֹרְיָה וְהֵפִיחוּ בָּהּ חַיִּים חֲדָשִׁים, מְלֵאֵי צֶבַע וְטַעַם.
עַל אַדְמַת בֵּית הַהַבְרָאָה שֶׁל קִיבּוּץ קִרְיַית עֲנָבִים, הֵקִימוּ אֶת הַמָּלוֹן הֶחָדָשׁ שֶׁלָּהֶם, יַהֲלוֹם מְלוּטָּשׁ בְּקוֹלֶקְצְיַית מְלוֹנוֹת הַיּוּקְרָה שֶׁל הָרֶשֶׁת - מְלוֹן כְּרָמִים, שֶׁבְּלִיבּוֹ פּוֹעֵם חָזָק וְסָגוֹל קוֹנְסֵפְּט הַיַּיִן וְהַטְּרוּאָר הָאֲזוֹרִי.

תַּיָּירוּת יַיִן הִיא דָּבָר מְבוּסָּס בְּעוֹלָם, שֶׁהוֹלֵךְ וְתוֹפֵס תְּאוּצָה בַּאֲזוֹרֵי יַיִן "יְשָׁנִים" כְּמוֹ צָרְפַת וְאִיטַלְיָה אֲבָל גַּם בִּנְסִיכוּיוֹת עוֹלַם הַיַּיִן הֶחָדָשׁ, דּוּגְמַת קָלִיפוֹרְנִיָּה אוֹ אוֹסְטְרַלְיָה.
זֶהוּ מְלוֹן הַיַּיִן הָרִאשׁוֹן בְּיִשְׂרָאֵל, וְהוּא נִבְנָה וְתוּכְנַן בְּהַשְׁרָאַת מְלוֹנוֹת בּוּטִיק שֶׁמּוּקְדָּשִׁים לְיַיִן וּמִתְכַּתְּבִים עִם הַסְּבִיבָה הָאֲזוֹרִית שֶׁמַּקִּיפָה אוֹתָם.

כְּבָר בַּכְּנִיסָה לַמָּלוֹן מַגִּישִׁים תִּירוֹשׁ עֲנָבִים טָרִי, קַר וּמְרַעֲנֵן וּלְצִידּוֹ מַמְתַּקִּים שֶׁמְּשׁוּלָּבִים בָּהֶם עֲנָבִים: מַרְמֵלָדוֹת קְטַנּוֹת וְכֵן צ'וֹרְצַ'חלָה, קִינּוּחַ גֵּיאוֹרְגִּי מָסוֹרְתִּי עָשׂוּי עֲנָבִים וֶאֱגוֹזִים (קוֹנְדִּיטוֹרִית הַמָּלוֹן הִיא מִמּוֹצָא גֵּיאוֹרְגִּי - אַל תְּפַסְפְסוּ אֶת הָאוֹצָרוֹת הַקְּטַנִּים שֶׁיּוֹצְאִים לָהּ תַּחַת הַיָּדַיִים).
הַשְּׁטָחִים הַצִּיבּוּרִיִּים מַרְגִּיעִים וְאַוְרִירִיִּים, הַחֲדָרִים בְּהִירִים וּנְעִימִים וְחַדְרֵי הָרַחֲצָה עֲנָקִיִּים מַמָּשׁ, מַזְמִינִים אֶתְכֶם לִשְׁקוֹעַ בְּכִמְעַט-סְפָּא פְּרָטִי וּמְפַנֵּק, שָׁם אֲפִילּוּ מוּצְרֵי הָאַמְבַּטְיָה מְכִילִים זַרְעֵי עֲנָבִים, אֲבָל הַיי, עִם כָּל הַיּוֹפִי הַזֶּה, אֲנַחְנוּ כָּאן בִּשְׁבִיל הַיַּיִן.

בַּמָּלוֹן סַדְנָאוֹת יַיִן שׁוֹוֹת שֶׁמַּצִּיעוֹת טְעִימוֹת רוֹחַב מִיִּקְבֵי הָאֵזוֹר, הֶסְבֵּרִים עַל תַּהֲלִיךְ הֲפָקַת הַיַּיִן, הַטְּרוּאָר וְיִיחוּדִיּוּת הַזָּנִים.
בְּכָל יוֹם מוּצָּעוֹת 3 סַדְנָאוֹת יַיִן לְפָחוֹת, חֶלְקָן מְשַׁלְּבוֹת טְעִימָה עִם שׁוֹקוֹלָד, אֲחֵרוֹת מַצִּיעוֹת טְעִימַת אֲדוּמִּים מִתְיַישְּׁנִים אוֹ יֵינוֹת קִינּוּחַ מְתוּקִים וְהַשִּׂמְחָה רַבָּה.

מִדֵּי פַּעַם מִתְאָרֵחַ בַּמָּלוֹן יֵינָן שֶׁמַּצִּיעַ לִטְעִימָה מִיֵּינוֹתָיו וְיֵינוֹת אֲחֵרִים שֶׁהוּא אוֹהֵב.
אִם בְּסַדְנָאוֹת הַיַּיִן שֶׁנֶּעֱרָכוֹת בַּמָּלוֹן תִּטְעֲמוּ רַק יֵינוֹת כְּשֵׁרִים, בְּעִיקָּר מֵאֵזוֹר הֲרֵי יְרוּשָׁלַיִם (עָגוּר הַנִּפְלָא, צִרְעָה, קַסְטֵל, טָפֶּרְבֵּרְג, עֵמֶק הָאֵלָה, יֶקֶב הַר הֲדַר הַמְּעוּלֶּה מַמָּשׁ, צוּבָּה וַאֲחֵרִים) הֲרֵי שֶׁבְּסִיּוּרֵי הַיַּיִן שֶׁנֶּעֱרָכִים מִחוּץ לַמָּלוֹן (צֶוֶות הַמָּלוֹן יִשְׂמַח לְתָאֵם עֲבוּרְכֶם בִּיקּוּר מוּדְרָךְ בְּיִקְבֵי הָאֵיזוֹר וּטְעִימוֹת), תּוּזְמְנוּ גַּם לִיקָבִים מַסְעִירִים אֲחֵרִים בָּאֵזוֹר דּוּגְמַת סוּסוֹן יָם, יֶקֶב שׂוֹרֵק, יֶקֶב כַּץ וְרַבִּים אֲחֵרִים.

סַדְנָאוֹת הַיַּיִן קוֹלְחוֹת וּנְעִימוֹת וְהֵן יְדַבְּרוּ גַּם אֵל מִי שֶׁלֹּא מַחֲשִׁיב אֶת עַצְמוֹ חוֹבֵב יַיִן גָּדוֹל.
לְהֵיפֶךְ.
בִּמְיוּחָד אִם אַתֶּם כָּאֵלֶּה, אַל תְּוַותְּרוּ עַל בִּיקּוּר בַּסַּדְנָאוֹת וְעַל הַטְּעִימוֹת שֶׁמַּעֲבִיר צֶוֶות הַיַּיִן.
תּוּפְתְּעוּ לְגַלּוֹת כַּמָּה טָעִים וְנָעִים יָכוֹל הָעֶרֶב שֶׁלָּכֶם לִהְיוֹת וּלְאָן מִתְפַּתְּחוֹת הַסַּדְנָאוֹת הַלָּלוּ, שֶׁהֵן קוֹמוּנִיקָטִיבִיּוֹת וּפְתוּחוֹת.

הַסַּדְנָאוֹת נֶעֱרָכוֹת בְּבוּטִיק הַיַּיִן הַמּוּשְׁקָע, שֶׁשָּׁווֹה לַעֲצוֹר בּוֹ בַּדֶּרֶךְ הַחוּצָה וּלְהִצְטַיֵּיד בְּיֵינוֹת בּוּטִיק מְקוֹמִיִּים מִזַּנִּים לֹא שִׁגְרָתִיִּים שֶׁהֵיטִיבוּ לְהִיקָּלֵט דַּוְוקָא בָּאֵזוֹר הָרֵי יְרוּשָׁלַיִם: פֵּטִי וַרְדּוֹ, קַבָּרְנֶה פְרַנְק, גֶּרְנָאשׁ, קָרִינְיָאן וְסִירָה מְחוּסְפֶּסֶת וּמְלֵאַת אוֹפִי, לֹא דּוֹמָה בִּכְלוּם לַבִּיטּוּי הָאֵלֵגַנְטִי שֶׁל הַזַּן בְּרָמַת הַגּוֹלָן לְמָשָׁל.
 
הַמְּחִירִים בַּמָּלוֹן נוֹחִים וְהֵם יְאַפְשְׁרוּ לָכֶם לִטְעוֹם גַּם מִבְחַר יֵינוֹת גָּדוֹל בְּבַר הַיַּיִן הַיָּפֶה שֶׁל הַמָּלוֹן (הַתַּפְרִיט הַקּוֹלֵחַ יֵילֵךְ לִקְרַאתְכֶם עַל גַּבֵּי מָסַךְ הַאַיְיפֶּד וְהוּא נָגִישׁ וּבָהִיר, נִיכָּר שֶׁהוּשְׁקְעָה בּוֹ מַחְשָׁבָה רַבָּה וִיצִירָתִית), לְצַד טַפָּאסִים וּפִינְגֶּר פּוּד צִבְעוֹנִי וִיצִירָתִי בְּסִגְנוֹן פְיוּזְ'ן אַסְיָאתִי שֶׁמִּתְכַּתֵּב עִם הַיַּיִן.
הַיַּיִן מִתְשַׁלֵּב גַּם בְּאוֹכֶל שֶׁמּוּגָּשׁ בַּמָּלוֹן.
בֵּין שַׁקְשׁוּקַת הַתֶּרֶד הַמְּצוּיֶּנֶת, קַדַאִיף גְּבִינַת עִיזִּים וּמַאֲפִים מְדוּיָּיקִים שֶׁל גְּבִינוֹת וִירָקוֹת, תּוּכְלוּ לִמְזוֹג לְעַצְמְכֶם מִתּוֹךְ הַשֶּׁמְפֶּנְיָירָה מוֹסְקְטוֹ אָדוֹם אוֹ לָבָן שֶׁל יֶקֶב טָפֶּרְבֵּרְג, יַיִן מְבַעְבֵּעַ קַלִּיל וּמְתַקְתַּק, שֶׁעוֹשֶׂה נָעִים וּפוֹתֵחַ אֶת הַבּוֹקֶר בְּכֵּיף.

לְיַד הַבְּרֵיכוֹת – יֵשׁ חִיצוֹנִית וּפְנִימִית - הִצִּיבוּ בַּר מַשְׁקָאוֹת עָשִׁיר שֶׁבְּעִיקָּר נִמְזָגִים בּוֹ לְבָנִים קְרִירִים וְרוֹזֶה לְכוֹסוֹת, בִּמְחִירִים שֶׁמַּתְחִילִים בְּ- 24 שְׁקָלִים, שִׂיחוּק קַיִץ אַדִּיר.
בַּמָּקוֹם סְפָּא עֲנָק שֶׁמִּשְׂתָּרֵעַ עַל 2 קוֹמוֹת וּבוֹ 23 חַדְרֵי טִיפּוּלִים, שֶׁמַּצִּיעִים, לְצַד סְאוּנוֹת, חָמָאם טוּרְקִי יְפֵהפֶה, סוֹלַרְיוּם, חֲדַר כּוֹשֶׁר מְשׁוּכְלָל וּפִינּוֹת לָאוֹנְג' רוֹגְעוֹת, גַּם טִיפּוּלִים קְלָאסִיִּים וְגַם טִיפּוּלֵי וִינוֹ-תֶּרָאפְּיָה, שָׁם יַעַטְפוּ אֶתְכֶם בַּעֲטִיפוֹת זַרְעֵי עֲנָבִים, יְקַרְצְפוּ עִם פִּילִינְג עֲנָבִים וְיוֹצִיאוּ אֶתְכֶם רַעֲנַנִּים וּוְרַדְרַדִּים כְּמוֹ בּוּז'וֹלֶה.

מוּמְלָץ לִמְחַפְּשֵׂי נוֹפֶשׁ רוֹמַנְטִי בַּאֲוִירָה בְּהִירָה, לִמְסִיבּוֹת רַוַּוקוֹת וְגַם לִימֵי גִּיבּוּשׁ וְכֵּיף עִם הַבּוֹס שֶׁלָּכֶם.
קְצָת יַיִן מוּל הַנּוֹף הַיָּפֶה וְתִרְאוּ גַּם הוּא הוֹלֵךְ וְרָפֶה.
