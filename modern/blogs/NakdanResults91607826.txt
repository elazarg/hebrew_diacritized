נִצְחִיָּה 
הַתְּלַת רַגְלִים - גּ'וֹן כְּרִיסְטוֹפְר
בִּיקּוֹרֶת סִפְרוּתִית עַל הַתְּלַת רַגְלַיִם מֵאֵת גּ'וֹן כְּרִיסְטוֹפְר
סֵפֶר טוֹב דֵּירוּג שֶׁל אַרְבָּעָה כּוֹכָבִים הַבִּיקּוֹרֶת נִכְתְּבָה בְּיוֹם רִאשׁוֹן, 2 בְּאוֹגוּסְט, 2020
ע"י נְצָחִיהָ

פַּעַם, יְלָדִים, לֹא הָיָה נֶטְפְלִיקְס. לֹא הָיוּ אֲתָרִים לִצְפִיָּה יְשִׁירָה. לֹא הָיוּ עֲרוּצֵי טֵלֵוִויזְיָה עִם תּוֹכְנִיּוֹת מוּקְלָטוֹת מְסוּדָּרוֹת בָּאֲתָר כְּדֵי שֶׁיִּהְיֶה אֶפְשָׁר לִצְפּוֹת בַּזְּמַן הַנּוֹחַ. הָיָה עָרוּץ אֶחָד, וְעוֹד עֲרוּץ שֵׁנִי עִם שִׁידּוּרֵי נִיסָּיוֹן שֶׁלֹּא קָלְטוּ בְּכָל הָאָרֶץ. וְהָיוּ עֲרוּצֵי כְּבָלִים פִּירָטִיִּים שֶׁהָיָה אֶפְשָׁר לִרְאוֹת בָּהֶם תּוֹכְנִיּוֹת, אֲבָל אֵצֶל הַשְּׁכֵנִים, כִּי הַהוֹרִים שֶׁלְּךָ יְשָׁרִים כְּמוֹ סַרְגֵּל וְלֹא רוֹאִים צוֹרֶךְ לְשֵׁם פִּירָטִי, מָה גַּם שֶׁמִּמֵּילָא הַטֵּלֵוִויזְיָה הַזֹּאת מְמַכֶּרֶת אֶתְכֶם, עוֹשָׂה אֶתְכֶם זוֹמְבִּים וּמַפְרִיעָה לִקְרִיאַת סְפָרִים. אוּלַי נוֹצִיא אוֹתָהּ מֵהַבַּיִת בִּכְלָל. בֶּעָרוּץ הַיָּחִיד שֶׁהָיָה הָיוּ שִׁידּוּרֵי טֵלֵוִויזְיָה חִינּוּכִית בִּשְׁעוֹת הַבּוֹקֶר, שֶׁזּוֹ סוּגְיָה נִפְרֶדֶת כִּי בֶּאֱמֶת צִיפּוּ שֶׁתַּלְמִידִים בַּכִּיתָּה יִצְפּוּ בַּסְּרָטִים שֶׁיִּלְמְדוּ אוֹתָם אַנְגְּלִית וּמָתֵמָטִיקָה. אַחַר הַצָּהֳרַיִם הָיוּ תּוֹכְנִיּוֹת שׁוֹנוֹת, וְגַם סְדָרוֹת. וְהָיָה לוּחַ שִׁידּוּרִים מוּדְפָּס בָּעִיתּוֹן שֶׁל שַׁבָּת, כְּדֵי שֶׁנֵּדַע לְהִתְכּוֹנֵן , אֲבָל אֵיכְשֶׁהוּ אִיבַּדְנוּ אוֹתוֹ תָּמִיד, אוֹ סְתָם לֹא הִתְיַחַסְנוּ אֵלָיו.

אָז בַּטֵּלֵוִיזְיָה הַהִיא, בֶּעָרוּץ הַיָּחִיד שֶׁהָיָה, עִם הַזְּמַנִּים הַקְּשִׁיחִים וּמִשְׁבְּצוֹת הַשִּׁידּוּר, בְּלִי הִתְחַיְּיבוּת לְשִׁידּוּר חוֹזֵר וּבְלִי שֶׁיָּדַעְנוּ שֶׁאֵי פַּעַם עֲתִידִים לְהַמְצִיא אֶת הַמּוּנָּח "בִּינְגְ'" רָאִיתִי תְּלַת רֶגֶל לָרִאשׁוֹנָה. זוֹ הָיְתָה סִדְרָה שֶׁרָאִיתִי בְּהֶבְזֵקִים וּבְהַבְלָחוֹת, כְּשֶׁזָּכַרְתִּי לִפְתּוֹחַ אֶת הַטֵּלֵוִיזְיָה, וְלָכֵן לֹא הִצְלַחְתִּי לְיַיצֵּר לִי רֶצֶף הֶגְיוֹנִי שֶׁל הִתְרַחֲשׁוּיוֹת. בַּהַתְחָלָה הָיָה עוֹלָם שֶׁנִּרְאָה יְמֵי-בֵּינַיְמִי, עִם אִיכָּרִים בַּשָּׂדֶה וְטֶכְנוֹלוֹגְיָה מְיוּשֶּׁנֶת. אַחַר כָּךְ הִתְבָּרֵר שֶׁזֶּה לֹא סְצֵנָה מֵהֶעָבָר, אֶלָּא מֵהֶעָתִיד. הַגִּיבּוֹרִים עָבְרוּ בְּחוֹרְבוֹת שֶׁל עִיר שֶׁנִּרְאֵית כְּמוֹ פָּרִיז (זִיהִיתִי בִּגְלַל מִגְדַּל אַיְיפֶל הָרוּס) הוֹלְכִים בְּתַחֲנוֹת רַכֶּבֶת תַּחְתִּית וּמִתְעַסְּקִים בְּצוּרָה חַסְרַת זְהִירוּת עִם רִימּוֹנִים. אַחַר כָּךְ הָיוּ קְרָבוֹת אַבִּירִים, נַעֲרָה יָפָה שֶׁהוּכְתְּרָה לְמַלְכַּת הַתַּחֲרוּת וְאָז מְכוֹנַת עֲנָק הָעוֹמֶדֶת עַל שָׁלוֹשׁ רַגְלַיִם גְּדוֹלוֹת לוֹקַחַת אֵלֶיהָ אֶת הַמַּלְכָּה הַזֹּאת. עוֹד הָיְתָה סְצֵנַת בְּרִיחָה מִמְּכוֹנוֹת תְּלַת רַגְלִיּוֹת, חִיפּוּשׂ הָרִים לְבָנִים, עוֹד תַּחֲרוּיוֹת, וְעִיר וּבָהּ מִפְלָצוֹת מְעוּוָּתוֹת.

כָּל הַסְּצֵנוֹת הָאֵלֶּה לֹא הִתְחַבְּרוּ לִי בְּצוּרָה הֶגְיוֹנִית, וְלֹא הִצְלַחְתִּי לְהָבִין מָה הַסִּיפּוּר וּמָה הַסּוֹף. לָכֵן מְאוֹד שָׂמַחְתִּי לְגַלּוֹת סֵפֶר בְּשֵׁם דּוֹמָה בַּחֲנוּת יָד שְׁנִיָּה, וְלִקְרוֹא פַּעַם אַחַת וּלְתָמִיד אֶת הַכֹּל בְּצוּרָה מְסוּדֶּרֶת. מַעֲשֶׂה בְּעוֹלָם עֲתִידָנִי שֶׁסּוֹבֵל מִפְּלִישַׁת חַיְיזָרִים אֵלָיו. הַחַיְיזָרִים הַנִּקְרָאִים תְּלַת רַגְלַיִים מַשְׁמִידִים אֶת רוֹב הָאוּכְלוּסִיָּה הָאֱנוֹשִׁית וְשׁוֹלְטִים בְּאֵלֶּה שֶׁנִּשְׁאֲרוּ בְּעֶזְרַת הֶתְקֵן טֵלֶפָּתִי עַל הָרֹאשׁ הַנִּקְרָא כִּיפָּה וּמוּתְקָן עֲלֵיהֶם בְּטֶקֶס חֲגִיגִי בְּגִיל הַהִתְבַּגְּרוּת. גִּיבּוֹר הַסֵּפֶר הוּא וַיְל, נַעַר אַנְגְּלִי שֶׁחַי בְּהוֹוֶה עֲתִידִי כָּלְשֶׁהוּ. לְוַיְל יֵשׁ חָבֵר טוֹב אֶחָד בַּכְּפָר, וְהוּא מְבוּגָּר מִמֶּנּוּ בְּשָׁנָה. הֶחָבֵר, גֵּ'ק, עוֹבֵר אֶת טֶקֶס הַכִּיפּוּי וּמַשְׁאִיר אֶת גֵּ'ק לְהִתְמוֹדֵד עִם בֶּן דּוֹדוֹ הַמֵּצִיק הֶנְרִי. לְאַחַר פְּגִישָׁה אַקְרָאִית עִם נַוָּד תִּמְהוֹנִי וַיְל וְהֶנְרִי בּוֹרְחִים מֵהַכְּפָר כְּדֵי לְהִימָּלֵט מֵהַכִּיפּוּי הַצָּפוּי לָהֶם. הֵם חוֹבְרִים לְנַעַר חְנוּן צָרְפָתִי בְּשֵׁם זָ'אן פּוֹל וְיַחַד הוֹלְכִים בְּאֵירוֹפָּה הַשּׁוֹמֵמָה וְהַמּוּשְׁמֶדֶת עַד לְ"הָרִים הַלְּבָנִים" שָׁם מִתְאַרְגֵּן גַּרְעִין שֶׁל אַנְשֵׁי חוֹפֶשׁ הַמּוֹרֵד בַּתְּלַת רַגְלַיִים. הֵם עוֹמְדִים מוּל הַדִּילֶמָה הָעוֹסֶקֶת בִּמְחִירָהּ שֶׁל הַחֵירוּת תַּחַת אוֹשֶׁר הָאָדָם הַפְּרָטִי. וַיְל וַחֲבֵרָיו מוּדָעִים לְכָךְ שֶׁהַחֶבְרָה הָאֱנוֹשִׁית שֶׁקָּדְמָה לָהֶם הָיְתָה מִתְקַדֶּמֶת טֶכְנוֹלוֹגִית יוֹתֵר מִזּוֹ שֶׁהֵם מַכִּירִים, אֲבָל גַּם יָדְעָה יוֹתֵר סֵבֶל, מַחֲלוֹת וּמִלְחָמוֹת. אֶת כָּל אֵלֶּה פָּתַר הַשִּׁלְטוֹן הַחַיְיזָרִי.

בִּקְרִיאָה שְׁנִיָּה, אַחֲרֵי שָׁנִים, הַסֵּפֶר אָמְנָם עָמוּס עֲלִילָה וּמְעַנְיֵין, אֲבָל הַדְּבָרִים נִרְאִים מִסְתַּדְּרִים קְצָת יוֹתֵר מִדַּי טוֹב. גַּם בָּעֲלִילָה עַצְמָהּ, דְּבָרִים בָּאִים עַל פִּתְרוֹנוֹת בְּצוּרָה קַלָּה מִדַּי, אֲבָל גַּם בִּפְרָטֵי הַתַּרְבּוּת הַחַיְיזָרִית הַיְּרוּקָּה. לְכוּלָּם שָׁלוֹשׁ רַגְלַיִים, חַיִּים בַּפִּירָמִידוֹת וּמִתְנַיְּידִים בִּרְכָבִים בַּעֲלֵי שָׁלוֹשׁ רַגְלַיִים וְכָךְ הָלְאָה. מַשֶּׁהוּ שָׁם הַרְבֵּה יוֹתֵר מִדַּי מְתוּכְנָן וּקְצָת נִרְאֶה תָּפוּר מִדַּי. אֲבָל נֶחְמָד.

וְהַסִּדְרָה שֶׁרָאִיתִי? מִסְתַּבֵּר שֶׁזֶּה לֹא אֲנִי, זֶה הִיא. מִסְתַּבֵּר שֶׁבְּשֶׁל מִגְבְּלוֹת תַּקְצִיב מֵעוֹלָם לֹא סִייְּמוּ לְהָפִיק אוֹתָהּ
