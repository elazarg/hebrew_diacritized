לְשִׁילֶר, שֶׁאִמִּי הִכִּירָה לִי

בַּלַּיְלָה, כְּשֶׁעֵרֶב־רַב שֶׁל חַיּוֹת מְמַלֵּא אֶת הַחֶדֶר, נוֹהֶגֶת אִמּוֹ לוֹמַר: "תְּלַמֵּד לְסַלֵּק אוֹתָן, עַל הָאֲדָמָה אֵין מָקוֹם לְכוּלָּם."
הוּא שׁוֹכֵב בַּמִּיטָּה, מֵאִיר אוֹתָן בַּפָּנָס הָאָדוֹם הַקָּטָן, לוֹחֵשׁ: "אֲנִי מְפַחֵד."
"הַמִּיטָּה הִיא מְקוֹר הַחַיִּים שֶׁל הַחֲלוֹמוֹת," אוֹמֶרֶת אִמּוֹ, "אֵין אֶפְשָׁרוּת אַחֶרֶת, תָּאִיר בַּפָּנָס אֶת הַדֶּרֶךְ וְאַל תָּמָרָה אֶת פִּיהָ, הִיא תּוֹבִיל אוֹתְךָ."
אֲבָל סוֹלְלַת הַפָּנָס הִתְרוֹקְנָה, וְהוּא מוֹצֵא עַצְמוֹ שָׁכוּב עַל הַמִּיטָּה, בִּרְכָּיו אֲסוּפּוֹת, חֲבוּקוֹת בְּיָדָיו הַלַּחוּת מִזֵּיעָה קָרָה, מִסְתַּכֵּל בַּחַיּוֹת שֶׁמְּמַלְּאוֹת אֶת חַדְרוֹ: אֲרָיוֹת, זְאֵבִים, עַקְרַבִּים וּנְחָשִׁים. הוּא צוֹעֵק: "לֹא מַגִּיעַ לִי לָמוּת כָּכָה, אִמָּא! תַּגִּידִי לָהֶם שֶׁלֹּא יִטְרְפוּ אוֹתִי!"
אִמּוֹ אוֹמֶרֶת: "אַל תְּאַבֵּד אֶת הַדֶּרֶךְ," פָּנֶיהָ הָאֲפוֹרִים מַאֲבָק מַזִּיעִים בְּחוֹם, הַטִּיפּוֹת הַגְּדוֹלוֹת נוֹטְפוֹת עַל שִׂמְלָתָהּ הַשְּׁחוֹרָה הַרְחָבָה.
אֲבָל הוּא גָּדֵל וּמִתְעַקֵּשׁ לָלֶכֶת בְּדֶרֶךְ שֶׁאֵינֶנָּה הַדֶּרֶךְ הַהִיא שֶׁלָּהּ, אֶלָּא בָּזוֹ שֶׁמּוֹבִילָה לְמָקוֹם שֶׁהִיא קוֹרֵאת לוֹ: "לְאִיבּוּד".
מָקוֹם שֶׁשָּׁמָיו גְּבוֹהִים מְאוֹד, וְאַדְמָתוֹ נֶעֱלֶמֶת וְנִגְלֵית לְסֵירוּגִין, וּכְשֶׁאָדָם מְבַקֵּשׁ לִמְצוֹא אֶת עַצְמוֹ, הוּא מַצְבִּיעַ לִמְקוֹמוֹת שׁוֹנִים וְאוֹמֵר: "שָׁם אֲנִי לֹא." כָּל כָּךְ לֹא, שֶׁשּׁוּם חַיָּה טוֹרֶפֶת לֹא יְכוֹלָה לִמְצוֹא אוֹתוֹ.
"בֶּן," אִמּוֹ מִתְעַקֶּשֶׁת, "כְּשֶׁהוֹלְכִים לְאִיבּוּד, הוֹלְכִים בֶּאֱמֶת לְאִיבּוּד."
רוּחַ נוֹשֶׁבֶת מֵהַבָּתִּים לְמַטָּה, מְבִיאָה לְרֶגַע הֲמוּלַּת קוֹלוֹת, יוֹצֶרֶת הֶמְיָה הַדּוֹמָה לְרַעַשׁ הַגּוֹבֵר שֶׁל הַמַּיִם, כַּאֲשֶׁר הֵם מִתְגַּלְגְּלִים בְּאַדְמַת הַטְּרָשִׁים שֶׁלָּהֶם, כְּמוֹ נַעֲנִים לְאֵיזֶה סֵדֶר.
"אֲנִי זוֹכֶרֶת שֶׁנּוֹלַדְתָּ. כְּבָר אָז הָיִיתָ כָּזֶה. הִתְעוֹרַרְתָּ רָעָב וְאָכַלְתָּ כְּדֵי לַחֲזוֹר לִישׁוֹן. נֶאֱלַצְתִּי לָתֵת לְךָ מַיִם, כִּי חִיסַּלְתָּ אֶת כָּל הֶחָלָב שֶׁהָיָה לִי. הִרְגַּשְׁתִּי שֶׁבָּגַדְתָּ בִּי עִם הַמַּיִם... עָלָה בְּדַעְתִּי שֶׁעִם הַזְּמַן זֶרַע הַבְּגִידָה הַזֶּה יִצְמַח בְּרֹאשְׁךָ."
הוּא מִסְתַּכֵּל לְעֵבֶר גַּגּוֹת בָּתֵּי הַיִּישּׁוּב הַבּוֹהֲקִים בְּאוֹר הַיָּרֵחַ. יֵשׁ לוֹ תְּחוּשָׁה שֶׁמִּשְׁקַל גּוּפָהּ שֶׁל אִמּוֹ מוֹחֵץ אוֹתוֹ, שֶׁבִּרְכָּיו מִתְפּוֹקְקוֹת מֵהַמַּאֲמָץ. הוּא תּוֹלֵשׁ מֵעַצְמוֹ אֶת גּוּפָהּ הַמְּדוּלְדָּל וְהוֹלֵךְ לְאִיבּוּד.

כָּךְ מָצָא אֶת שִׁילֶר, שֶׁגַּם הִיא הָלְכָה לְמָקוֹם שֶׁנִּקְרָא "לְאִיבּוּד". הֵם נָהֲגוּ לִשְׁכַּב בַּמִּיטָּה, לָקַחַת כַּדּוּרִים וּלְהִסְתַּכֵּל לִחְיוֹת בָּעֵינַיִים.
"קָשֶׁה לוֹמַר שֶׁצַּ'ארְלְס דַּרְוִוין הִסְבִּיר אֶת הָעוֹלָם," הוּא אוֹמֵר לְשִׁילֶר בְּעוֹדָהּ שׁוֹבֶרֶת אֶת לִבּוֹ כִּי הִיא מִתְאַהֶבֶת בְּמִישֶׁהוּ אַחֵר, "אִם הָיִיתִי חַיּוֹת, הָיִיתִי יוֹדֵעַ לְהַבְדִּיל בֵּין יֵיאוּשׁ לְפַחֵד. הֵן נִכְנָעוֹת מִתּוֹךְ יֵיאוּשׁ, וְנָסוֹת מִתּוֹךְ פַּחַד, וְהַחַיָּה 'אָדָם', לֹא."
"יֵשׁ יִתְרוֹן לִחְיוֹת," שִׁילֶר אוֹמֶרֶת, "אֵין לָהֶן מוּסַר כְּלָיוֹת."
"אַתְּ מְדַבֶּרֶת!" הוּא צוֹרֵחַ: "גַּם לַאֲנָשִׁים לֹא מַמָּשׁ."

כְּשֶׁהוּא מְסַפֵּר לְאִמּוֹ עַל שִׁילֶר, הִיא אוֹמֶרֶת, "הַכּוּשִׁי עָשָׂה אֶת שֶׁלּוֹ, הַכּוּשִׁי יָכוֹל לָלֶכֶת, הַכְּנֶסֶת רֹאשׁ בָּרִיא לַמִּיטָּה חוֹלֶה." מְנִידָה בְּרֹאשָׁהּ מִצַּד לְצַד וּמְסַפֶּרֶת אֵיךְ הָיָה קוֹשֵׁר חוּט לְרֶגֶל אַחַת שֶׁל חִיפּוּשִׁית יְרוּקָּה כָּךְ שֶׁלֹּא תּוּכַל לְהַרְחִיק עוֹף וְלִבְרוֹחַ לוֹ, לַמְרוֹת שֶׁהָיוּ אֶצְלָם הָמוֹן חִיפּוּשִׁיּוֹת יְרוּקּוֹת, שֶׁאוּלַי כֵּן הָיוּ קְשׁוּרוֹת לָאוֹפֶן שֶׁבּוֹ הִסְבִּיר צַ'ארְלְס דַּרְוִוין אֶת הָעוֹלָם.
הוּא חוֹשֵׁב, אוּלַי אִמִּי צוֹדֶקֶת, עוּבְדָּה שֶׁלַּאֲנָשִׁים שׁוֹנִים קוֹרְאִים בְּאוֹתוֹ שָׁם, וְאוּלַי בֶּאֱמֶת אֵין הֶבְדֵּל גָּדוֹל אִם שִׁילֶר אַחַת בָּאָה לַמִּיטָּה בִּמְקוֹמָהּ שֶׁל שִׁילֶר אַחֶרֶת. אוֹ שֶׁשִּׁילֶר אֶחָד בָּא לַמִּיטָּה בִּמְקוֹם שִׁילֶר אַחֵר.
"בְּלַיְלָה קַוֵּוי הַמְּתָאֵר מִיטַּשְׁטְשִׁים," אִמּוֹ מְנַפְנֶפֶת בְּיָדָהּ.
הוּא מַבִּיט בַּיָּד הַמִּתְנַפְנֶפֶת, חוֹשֵׁב עַל הַשִּׁילֵרִיּוֹת שֶׁבָּאוֹת וְהוֹלְכוֹת אֶל הַמִּיטָּה וּמִמֶּנָּה בַּלַּיְלָה עַד שֶׁהִיא (הַמִּיטָּה) נִהְיֵית חוֹלֶה. אוֹר כּוֹכָבִים מִתְרַסֵּק עַל רִצְפַּת הַחֶדֶר. הוּא זוֹכֵר שֶׁאִמּוֹ אָמְרָה: "שָׁם, עַל תִּקְרַת אוֹהֶל הַכּוֹכָבִים, רָשׁוּם שְׁמָהּ שֶׁל הָאַחַת. אִם תִּישָּׂא עֵינַיִים תּוּכַל לְפַעְנֵחַ."
הוּא רוֹצֶה לָצֵאת מֵהַמִּיטָּה, לָגֶשֶׁת לַחַלּוֹן לְהַבִּיט בַּכּוֹכָבִים, לְפַעְנֵחַ. אֲבָל קָשֶׁה לָצֵאת מֵהַמִּיטָּה הַחוֹלֶה, קַל יוֹתֵר לִיפּוֹל. הוּא נוֹפֵל. לְמַטָּה. לְמַטָּה לְגַמְרֵי — נֶחְבַּט בָּעוֹלָם.
הוּא מִתְעוֹרֵר, שְׁאֵרִיּוֹת שֶׁל אוֹכֶל וְאֶמְצְעֵי מְנִיעָה וּבְגָדִים, לִכְלוּךְ־בָּלָגָן שֶׁעֶרֶב־רַב שֶׁל חַיּוֹת סוֹבֵב סְבִיבוֹ, הָאֲרָיוֹת, הַזְּאֵבִים, הַנְּחָשִׁים וְהָעַקְרַבִּים.
שִׁילֶר אַחַת עוֹמֶדֶת לְיַד הַחַלּוֹן, מַבִּיטָה בַּשָּׁמַיִם, אוֹמֶרֶת: "רוֹאֶה, כָּתַב הַכּוֹכָבִים הוּא חֲרִישִׁי וְקָדוּם מְאוֹד, אִי־אֶפְשָׁר לִקְרוֹא אוֹתוֹ, אֲבָל כְּתוּבוֹת בּוֹ אֲמִיתּוֹת נִצְחִיּוֹת." הִיא מְרִימָה אֶת הַתַּחְתּוֹנִים וְאֶת הֶחָזִיָּיה, מַכְנִיסָה אוֹתָם לְתַרְמִיל הַגַּב הַקָּטָן שֶׁלָּהּ. לוֹבֶשֶׁת אֶת הַחֲצָאִית הָאֲדוּמָּה, וּמְנַעֶרֶת אֶת חוּלְצַת הַטְּרִיקוֹ שֶׁעֶרֶב־רַב שֶׁל חַיּוֹת רְקוּמוֹת עָלֶיהָ, מְשַׁחְרֶרֶת אוֹתָן שֶׁיְּזַנְּקוּ לְעֵבֶר הַמִּיטָּה שֶׁלּוֹ, "רוֹצָה לְשַׂחֵק אִיתִּי בִּלְהִיטָּרֵף?" הִיא שׁוֹאֶלֶת.
