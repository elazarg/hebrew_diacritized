1

יַלְדָּה קְטַנָּה לֹא חָשָׁה בְּטוֹב, וְדוֹקְטוֹר מִיכָאִיל פֶּטְרוֹבִיץ', שֶׁאוֹתוֹ הִיא מַכִּירָה מִיָּמִים יָמִימָה, בָּא אֵלֶיהָ כָּל יוֹם. לְעִתִּים רְחוֹקוֹת הוּא מֵבִיא אִתּוֹ שְׁנֵי רוֹפְאִים נוֹסָפִים, לֹא מַכָּרִים. הֵם הוֹפְכִים אֶת הַיַּלְדָּה עַל הַגַּב וְעַל הַבֶּטֶן, מַצְמִידִים אֶת אָזְנָם אֶל גּוּפָהּ וּמַקְשִׁיבִים לְמַשֶּׁהוּ, מוֹשְׁכִים מַטֵּה אֶת הָעַפְעַפְף הַתַּחְתּוֹן שֶׁלָּהּ וּמַבִּיטִים. בְּעוֹדָם עוֹשִׂים כָּךְ, הֵם מִתְנַשְּׁמִים בַּאֲרֶשֶׁת חֲשִׁיבוּת, פְּנֵיהֶם חֲמוּרִים, וּמְדַבְּרִים בֵּינֵיהֶם בְּשָׂפָה לֹא מוּבֶנֶת.
לְאַחַר מִכֵּן הֵם עוֹבְרִים מֵחֲדַר הַיְּלָדִים לַחֲדַר הָאוֹרְחִים, שֶׁבּוֹ אִמָּא מְחַכָּה לָהֶם. הָרוֹפֵא הַבָּכִיר בְּיוֹתֵר – אִישׁ גָּבוֹהַּ עִם שַׁעַר שֵׂיבָה שֶׁמַּרְכִּיב מְשַׁקְּפִים מִזְהָבִים – מְסַפֵּר לָהּ מַשֶּׁהוּ רְצִינִי וְאֶרֶךְ. הַדֶּלֶת לֹא סְגוּרָה, וּמִמַּטָּתָהּ הַיַּלְדָּה רוֹאָה וְשׁוֹמַעַת הַכֹּל. הִיא אֵינֶנָּה מְבִינָה רַבִּים מִן הַדְּבָרִים, אַךְ הִיא יוֹדַעַת שֶׁמְּדַבְּרִים עָלֶיהָ.
אַמָּה מַבִּיטָה בְּרוֹפֵא בְּעֵינַיִם גְּדוֹלוֹת עִיפוֹת וּמִכְּתַמּוֹת בִּדְמָעוֹת. כְּשֶׁהָרוֹפֵא הַבָּכִיר נִפְרָד לְשָׁלוֹם, הוּא אוֹמֵר בְּקוֹל רָם:
"הַדָּבָר הֶחָשׁוּב בְּיוֹתֵר – אַל תִּתְּנוּ לָהּ לְהִשְׁתַּעֲמֵם. מָלְאוּ אַחֵר כָּל גְּחַמּוֹתֶיהָ."
"הוֹ, דּוֹקְטוֹר, אֲבָל הִיא לֹא רוֹצָה כְּלוּם."
"אִם כֵּן, אֲנִי לֹא יוֹדֵעַ... תְּנַסּוּ לְהִזָּכֵר מָה הִיא אַהֲבָה פַּעַם, לִפְנֵי הַמַּחֲלָה. צַעֲצוּעִים... מַמְתַּקִּים כָּלְשֶׁהֵם..."
"לֹא, לֹא, דּוֹקְטוֹר, הִיא לֹא רוֹצָה כְּלוּם..."
"אָז תְּנַסּוּ לְשַׁעֲשֵׁעַ אוֹתָהּ בְּכָל דֶּרֶךְ שֶׁהִיא... תִּהְיֶה אֲשֶׁר תִּהְיֶה... אֲנִי נוֹתֵן לָכֶם אֶת מִלָּתִי: אִם תַּצְלִיחוּ לְהַצְחִיק וּלְשַׂמֵּחַ אוֹתָהּ, זֹאת תִּהְיֶה הַתְּרוּפָה הַטּוֹבָה בְּיוֹתֵר. הַבֵּינִי, הַיַּלְדָּה שֶׁלָּכֶם חוֹלָה בַּאֲדִישׁוּת כְּלַפֵּי הַחַיִּים וְתוּ לֹא... לְהִתְרָאוֹת, גְּבִרְתִּי!"

2

" נַדְיָה מְתוּקָה, יַלְדָּה שֶׁלִּי יִקְרֶה," אוֹמֶרֶת אִמָּא, "אוּלַי אַתְּ רוֹצָה מַשֶּׁהוּ?"
"לֹא, אִמָּא, אֲנִי לֹא רוֹצֶה כְּלוּם."
"אוֹשִׁיב עַל הַמַּטֶּה שֶׁלָּךְ אֶת כָּל הַבָּבוֹת שֶׁלְּךָ, רוֹצֶה? נָשִׁים כֻּרְסָאוֹת, סַפָּה, שֻׁלְחַן וּמַעֲרֶכֶת תֶּה. הַבָּבוֹת יִשְׁתּוּ תֶּה וִידַבְּרוּ עַל מֶזֶג הָאֲוִיר וְעַל בְּרִיאוּת הַיְּלָדִים שֶׁלָּהֶן."
"תּוֹדָה, אִמָּא... אֲנִי רוֹצֶה... מְשַׁעֲמֵם לִי..."
"טוֹב, יַלְדָּה שֶׁלִּי, לֹא צָרִיךְ בָּבוֹת. אוּלַי אֶקְרָא לְקַטְיָה אוֹ לִזְ'נִיצַ'קִה שֶׁיָּבוֹאוּ אֵלֶיךָ? הֲרֵי אַתְּ אוֹהֶבֶת אוֹתָן כָּל כָּךְ."
"לֹא צָרִיךְ, אִמָּא. בֶּאֱמֶת, לֹא צָרִיךְ. אֲנִי לֹא רוֹצֶה כְּלוּם. כָּל כָּךְ מְשַׁעֲמֵם לִי!"
"רוֹצָה שֶׁאָבִיא לְךָ שׁוֹקוֹלָד?"
אֲבָל הַיַּלְדָּה לֹא עוֹנָה וּבוֹהָה בְּעֶצֶב בַּתִּקְרָה. שׁוּם דָּבָר לֹא כּוֹאֵב לָהּ, וְגַם אֵין לָהּ חַם. אֲבָל הִיא מַרְזָה וְנֶחֱלֶשֶׁת מִיּוֹם לְיוֹם. דָּבָר מֵהַנַּעֲשֶׂה לֹא מְעַנְיֵן אוֹתָהּ, וְאֵין דָּבָר שֶׁהִיא רוֹצָה. כָּךְ הִיא שׁוֹכֶבֶת לֵילוֹת כְּיָמִים, שְׁקֵטָה וַעֲצוּבָה. לִפְעָמִים הִיא נִרְדֶּמֶת לִכְמַחֲצִית הַשָּׁעָה, אֲבָל גַּם בַּחֲלוֹם הִיא רוֹאָה מַשֶּׁהוּ אֵפֶר, אֶרֶךְ, מְשַׁעֲמֵם, כְּמוֹ גֶּשֶׁם סְתָוִי.
כְּשֶׁדֶּלֶת חֲדַר הַיְּלָדִים פְּתוּחָה לַחֲדַר הָאוֹרְחִים וּמֵחֲדַר הָאוֹרְחִים הָלְאָה, לַמִּשְׂרָד שֶׁל אַבָּא, אָז הַיַּלְדָּה רוֹאָה אֶת אָבִיהָ. אַבָּא מִתְהַלֵּךְ בִּמְהִירוּת מִפְנֶה לְפָנָהּ וּמְעַשֵּׁן סִיגַרְיָה אַחֲרֵי סִיגַרְיָה. לִפְעָמִים הוּא נִכְנָס לַחֲדַר הַיְּלָדִים, מִתְיַשֵּׁב עַל קְצֵה הַמַּטֶּה וּמְלַטֵּף אֶת רַגְלֶיהָ שֶׁל נַדְיָה בַּעֲדִינוּת. אָז הוּא נֶעֱמַד פִּתְאוֹם וּפוֹסֵעַ אֶל הַחַלּוֹן. הוּא שׂוֹרֵק לְעַצְמוֹ בְּעוֹדוֹ מַבִּיט אֶל הָרְחוֹב, אַךְ כְּתֵפָיו רוֹעֲדוֹת. לְאַחַר מִכֵּן הוּא מַצְמִיד בַּחֲטָף מִטְפַּחַת לְעַיִן אַחַת, לָעַיִן הַשְּׁנִיָּה, וְחוֹזֵר לַמִּשְׂרָד שֶׁלּוֹ, הַבָּעַת פָּנָיו כְּמוֹ כְּעוּסָה. שָׁם הוּא שׁוּב רָץ מַפְנֶה לְפָנָהּ וּמְעַשֵּׁן סִיגַרְיָה אַחֲרֵי סִיגַרְיָה... וְהַמִּשְׂרָד שֶׁלּוֹ נַעֲשָׂה כָּלוּ כָּחָל מֵרַב עָשָׁן.

3

בָּקָר אֶחָד הַיַּלְדָּה הַקְּטַנָּה מִתְעוֹרֶרֶת נִמְרֶצֶת קְצָת יוֹתֵר מִתָּמִיד. הִיא רָאֲתָה מַשֶּׁהוּ בַּחֲלוֹם, אֲבָל לֹא מַצְלִיחָה לְהִזָּכֵר מָה בְּדִיּוּק, וּמַבִּיטָה בַּאֲרִיכוּת וּבִתְשׂוּמֶת לֵב לְתוֹךְ עֵינֵי אִמָּהּ.
"אַתְּ רוֹצָה מַשֶּׁהוּ?" שׁוֹאֶלֶת אִמָּא.
אֲבָל הַיַּלְדָּה נִזְכֶּרֶת בַּחֲלוֹמָהּ וְאוֹמֶרֶת בְּלַחַשׁ, כָּאֵלּוּ מְדַבֵּר בְּסוֹד:
"אִמָּא... אֶפְשָׁר שֶׁיִּהְיֶה לִי... פִּיל? אֲבָל לֹא כָּזֶה שֶׁמִּצִּיר בַּתְּמוּנָה... אֶפְשָׁר?"
"כַּמּוּבָן, יַלְדָּה שֶׁלִּי, כַּמּוּבָן, אֶפְשָׁר."
הִיא הוֹלֶכֶת לַמִּשְׂרָד וְאוֹמֶרֶת לְאַבָּא שֶׁהַיַּלְדָּה רוֹצָה פִּיל. אַבָּא לוֹבֵשׁ מִיָּד מְעִיל, חוֹבֵשׁ כּוֹבַע וְנוֹסֵעַ לְאָנְשֶׁהוּ. לְאַחַר מַחֲצִית הַשָּׁעָה הוּא חוֹזֵר עִם צַעֲצוּעַ יָקָר וְיָפֶה. זֶהוּ פִּיל אֵפֶר גָּדוֹל שֶׁיּוֹדֵעַ לְהַנְהֵן בְּרֹאשׁוֹ וּלְנוֹפֵף בִּזְנָבוֹ; עַל גַּב הַפִּיל אָכַף אָדָם, וְעַל הָאֶכֶף אֹהֶל מִזָּהָב, וְיוֹשְׁבִים בּוֹ שְׁלוֹשָׁה אֲנָשִׁים קְטַנִּים.
אַךְ הַיַּלְדָּה מַבִּיטָה בְּצַעֲצוּעַ בְּאוֹתָהּ אֲדִישׁוּת שֶׁבָּהּ הִיא מַבִּיטָה בַּתִּקְרָה וּבַקִּירוֹת וְאוֹמֶרֶת בְּרִפְיוֹן:
"לֹא. זֶה בִּכְלָל לֹא זֶה. רָצִיתִי פִּיל אֲמִתִּי, פִּיל חַי, זֶה מֵת."
"רַק תִּסְתַּכְּלִי, נַדְיָה," אוֹמֵר אַבָּא. "עַכְשָׁו נִמְתָּח אֶת הַקְּפִיץ, וְהוּא יִהְיֶה מַמָּשׁ-מַמָּשׁ כְּמוֹ חַי."
הוּא מוֹתֵחַ אֶת הַקְּפִיץ, וְהִפִּיל מֵנִיד בְּרֹאשׁוֹ, מְנוֹפֵף בִּזְנָבוֹ, מַתְחִיל לָנוּעַ עַל רַגְלָיו וְהוֹלֵךְ בְּאִטִּיּוּת לְאֹרֶךְ הַשֻּׁלְחָן. זֶה לֹא מְעַנְיֵן כְּלָל אֶת הַיַּלְדָּה. לְמַעֲשֶׂה, זֶה מְשַׁעֲמֵם אוֹתָהּ, אֲבָל כְּדֵי לֹא לְצַעֵר אֶת אָבִיהָ הִיא לוֹחֶשֶׁת בְּהַכְנָעָה:
"אֲנִי מְאוֹד-מְאוֹד מוֹדֶה לְךָ, אַבָּא יָקָר. אֲנִי חוֹשֶׁבֶת שֶׁלְּאַף אֶחָד אֵין צַעֲצוּעַ יָפֶה כָּזֶה... רַק ש... זוֹכֵר... מִזְּמַן-מִזְּמַן הַבְטָחַת לָקַחַת אוֹתִי לְקִרְקָס לִרְאוֹת פִּיל אֲמִתִּי... וְלֹא לָקַחַת מֵעוֹלָם..."
"אֲבָל, יָלַדְתִּי הַמְּתוּקָה, הַבֵּינִי, זֶה בִּלְתִּי אֶפְשָׁרִי. הַפִּיל גָּדוֹל מְאוֹד, הוּא מַגִּיעַ עַד הַתִּקְרָה, הוּא לֹא יְכַנֵּס לַבַּיִת שֶׁלָּנוּ... וְחוּץ מִזֶּה, הֵיכָן אַשִּׂיג פִּיל?"
"הוֹ, אַבָּא, הֲרֵי אֲנִי לֹא צְרִיכָה כָּזֶה גָּדוֹל... תָּבִיא לִי אֶחָד קָטָן, רַק שֶׁיִּהְיֶה חַי. אָפְלוּ מַמָּשׁ קָטָן... אָפְלוּ פִּיל פִּילוֹן."
"יַלְדָּתִי הַיְּקָרָה, אֶעֱשֶׂה כָּל דָּבָר בִּשְׁבִילְךָ, אֲבָל אֶת זֶה אֲנִי לֹא יָכוֹל לַעֲשׂוֹת. הֲרֵי זֶה כָּאֵלּוּ הָיִיתָ אוֹמֶרֶת לִי פִּתְאוֹם: 'אַבָּא, תּוֹרִיד לִי אֶת הַשֶּׁמֶשׁ מֵהַשָּׁמַיִם.'"
הַיַּלְדָּה מְחַיֶּכֶת בְּעַצְבוּת.
"אֵיזֶה טָפַשׁ אַתָּה, אַבָּא. הֲרֵי אֲנִי יוֹדַעַת שֶׁאֶת הַשֶּׁמֶשׁ אִי אֶפְשָׁר לְהוֹרִיד מִפְּנֵי שֶׁהִיא שׂוֹרֶפֶת. וְשֶׁגַּם אֶת הַיָּרֵחַ אִי אֶפְשָׁר. לֹא, רַק רָצִיתִי שֶׁיִּהְיֶה לִי פִּיל תִּינוֹק... אֲמִתִּי."
וְהִיא עוֹצֶמֶת אֶת עֵינֶיהָ וְלוֹחֶשֶׁת: "אֲנִי עָיְפָה... תִּסְלַח לִי, אַבָּא..."
אָבִיהָ תּוֹפֵס בְּשַׁעֲרוֹ וְרָץ לַמִּשְׂרָד שֶׁלּוֹ. הִיא רוֹאָה אוֹתוֹ צוֹעֵד אָנֶה וָאָנָה בְּמֶשֶׁךְ זְמַן-מָה. לְאַחַר מִכֵּן הוּא זוֹרֵק בְּהֶחְלֵטִיּוּת עַל הָרִצְפָּה סִיגַרְיָה שֶׁלֹּא גָּמַר לְעַשֵּׁן (עַל זֶה הוּא תָּמִיד חוֹטֵף מֵאִמָּא) וְצוֹעֵק לִמְשָׁרֶתֶת:
"אוֹלְגָּה, מְעִיל וְכוֹבַע!"
אִשְׁתּוֹ מוֹפִיעָה בַּמְּבוֹאָה.
"לְאָן זֶה, סָשָׁה?" הִיא שׁוֹאֶלֶת.
הוּא נוֹשֵׁם בִּכְבֵדוּת בְּעוֹד הוּא מְכַפְתֵּר אֶת כַּפְתּוֹרֵי הַמְּעִיל.
"אֲנִי בְּעַצְמִי לֹא יוֹדֵעַ, מַאֲשִׁינְקָה... רַק שֶׁנִּדְמֶה לִי שֶׁהַיּוֹם לִקְרַאת עֶרֶב בֶּאֱמֶת אָבִיא אֵלֵינוּ לְכָאן פִּיל אֲמִתִּי."
אִשְׁתּוֹ מַבִּיטָה בּוֹ בִּדְאָגָה.
"יַקִּירֵי, אַתָּה חָשׁ בְּטוֹב? לֹא כּוֹאֵב לָךְ הָרֹאשׁ? אוּלַי לֹא יָשַׁנְתְּ טוֹב?"
"לֹא יָשַׁנְתִּי בִּכְלָל," הוּא עוֹנֶה בִּרְגַז. "אֲנִי רוֹאֶה שֶׁאַתְּ רוֹצָה לִשְׁאֹל אוֹתִי אִם הִשְׁתַּגַּעְתִּי. בֵּינְתַיִם לֹא. לְהִתְרָאוֹת! הַכֹּל יִתְבַּהֵר בָּעֶרֶב."
הוּא טוּרֶק בַּחֲזָקָה אֶת דֶּלֶת הַכְּנִיסָה וְנֶעֱלָם.

4

כְּעָבָר שֶׁעִתִּים הוּא מוֹצֵא אֶת עַצְמוֹ בַּקִּרְקָס, יָשׁוּב בַּשּׁוּרָה הָרִאשׁוֹנָה. הוּא רוֹאֶה חַיּוֹת מְאַלְּפוֹת מְבַצְּעוֹת תַּרְגִּילִים שׁוֹנִים. כְּלָבִים חֲכָמִים מְקַפְּצִים, מִתְגַּלְגְּלִים, רוֹקְדִים, מִיְּלֵלִים לִצְלִילֵי מוּזִיקָה, מְחַבְּרִים מִלִּים מֵאוֹתִיּוֹת גְּדוֹלוֹת עֲשׂוּיוֹת קַרְטוֹן. הַקּוֹפִים – כַּמָּה מֵהֶם בַּחֲצָאִיּוֹת אֲדָמוֹת, אֲחֵרִים בַּמִּכְנָסִים כָּחֳלָיִם – הוֹלְכִים עַל חֲבָל וְרוֹכְבִים עַל גַּבֵּי פּוּדֶל גָּדוֹל. אֲרָיוֹת גִּ'ינְגִּ'יִּים עֲנָקִיִּים קוֹפְצִים מִבַּעַד לַחֲשׁוּקִים בּוֹעֲרִים. כֶּלֶב-יָם גְּמָלוּנִי יוֹרֶה בְּאֶקְדָּח. לִקְרַאת סוֹף הַמּוֹפָע מוֹבִילִים פְּנִימָה אֶת הַפִּילִים. יֵשׁ שְׁלוֹשָׁה: אֶחָד גָּדוֹל וְשָׁנִים מַמָּשׁ קְטַנִּים, נַנָּסִים, אֲבָל בְּכָל זֹאת גָּדְלָם רַב בְּהַרְבֵּה מִשֶּׁל סוּס. מוּזָר לִרְאוֹת כֵּיצַד הַחַיּוֹת הָעֲנָקִיּוֹת הָאֵלֶּה, הַמְּגַשְּׁמוֹת כָּל כָּךְ לְמַרְאֶה, מְבַצְּעוֹת אֶת הַלַּהֲטוּטִים הַמֶּרְכָּבִים בְּיוֹתֵר, שֶׁגַּם אָדָם מֵימָן וְזָרִיז מְאוֹד לֹא יַצְלִיחַ לְבַצֵּעַ. הַפִּיל הַגָּדוֹל בְּיוֹתֵר מַרְשִׁים בְּמֵיחַד. הוּא נֶעֱמָד עַל רַגְלָיו הָאֲחוֹרִיּוֹת, מִתְיַשֵּׁב, נֶעֱמָד עַל רֹאשׁוֹ, רַגְלִים מַעְלָה, הוֹלֵךְ עַל בַּקְבּוּקֵי עֵץ, עַל חָבִית מִתְגַּלְגֶּלֶת, הוֹפֵךְ דַּפִּים בְּסֵפֶר קַרְטוֹן גָּדוֹל בְּעֶזְרַת הַחֵדֶק וּלְבַסּוֹף מִתְיַשֵּׁב לְשֻׁלְחַן, כּוֹרֵךְ מַפִּית וְסוֹעֵד מַמָּשׁ כְּמוֹ יֶלֶד מְחַנֵּךְ.
הַמּוֹפָע נִגְמַר. הַקָּהָל מִתְפַּזֵּר. אָבִיהָ שֶׁל נַדְיָה נָגַשׂ לַגֶּרְמָנִי גָּדַל גּוּף, הַבְּעָלִים שֶׁל הַקִּרְקָס. הַבְּעָלִים עוֹמֵד מֵאֲחוֹרֵי מַחֲצַת קְרָשִׁים וּבְפִיו סִיגָר שַׁחַר גָּדוֹל.
"סְלַח לִי בְּבַקָּשָׁה," אוֹמֵר אָבִיהָ שֶׁל נַדְיָה, "תַּסְכִּים אוּלַי לְשַׁחְרֵר אֶת הַפִּיל שֶׁלָּךְ אֵלַי הַבַּיְתָה לִזְמַן-מָה?"
מֵרַב הַפְתָּעָה הַגֶּרְמָנִי פּוֹעֵר לְרוּחָהּ אֶת עֵינָיו וְאָפְלוּ אֶת פִּיו, וְהַסִּיגָר נוֹפֵל אַרְצָה. הוּא מִתְכּוֹפֵף בִּגְנִיחָה, מֵרִים אוֹתוֹ, מַכְנִיס אוֹתוֹ בַּחֲזָרָה לְפִיו וְרַק אָז אוֹמֵר:
"לְשַׁחְרֵר? אֶת הַפִּיל? הַבַּיְתָה? אֲנִי לֹא מֵבִין."
בְּעֵינַיִם שֶׁל הַגֶּרְמָנִי רוֹאִים שֶׁגַּם הוּא רוֹצֶה לְשֶׁאַל אֶת אַבָּא שֶׁל נַדְיָה אִם כּוֹאֵב לוֹ הָרֹאשׁ... אַךְ הָאָב מַסְבִּיר בִּמְהֵרָה בַּמֶּה הָעִנְיָן: בִּתּוֹ הַיְּחִידָה, נַדְיָה, חוֹלָה בְּאֵיזוֹ מַחֲלָה מְשַׁנֶּה שֶׁאָפְלוּ הָרוֹפְאִים לֹא מְבִינִים אוֹתָהּ עַד תַּם. הִיא שׁוֹכֶבֶת זֶה חָדָשׁ בְּמִטָּתָהּ, מַרְזֶה וְנֶחֱלֶשֶׁת מִיּוֹם לְיוֹם, לֹא מִתְעַנְיֶנֶת בַּדָּבָר, מִשְׁתַּעֲמֶמֶת וְדוֹעֶכֶת לְאִטָּהּ. הָרוֹפְאִים מְצֻוִּים לְשַׁעֲשֵׁעַ אוֹתָהּ, אֲבָל דָּבָר לֹא מוֹצֵא חֵן בְּעֵינֶיהָ; הָרוֹפְאִים מְצֻוִּים לְהַגְשִׁים אֶת כָּל מִשְׁאֲלוֹתֶיהָ, אֲבָל אֵין לָהּ כָּל מִשְׁאֲלוֹת. הַיּוֹם הִיא הִבִּיעָה רָצוֹן לִרְאוֹת פִּיל חַי. הֲיִתָּכֵן שֶׁאֵין כָּל אֶפְשָׁרוּת לְהַגְשִׁים אֶת מִשְׁאַלְתָּהּ?
הוּא אוֹחֵז אֶת הַגֶּרְמָנִי בְּכַפְתּוֹר מְעִילוֹ וּמוֹסִיף בְּקוֹל רוֹעֵד :"וּבְכֵן... אֲנִי, מְקַוֶּה, כַּמּוּבָן, שֶׁיָּלַדְתִּי תַּבְרִיא. אֲבָל... אִם חָס וְחָלִילָה... מָה אִם מַחֲלָתָהּ תִּגָּמֵר רַע... מָה אִם הַיַּלְדָּה תָּמוּת פִּתְאוֹם?... רַק תֵּחָשֵׁב: הֲרֵי הַמַּחֲשָׁבָה שֶׁלֹּא מִלֵּאתִי אַחֵר מִשְׁאַלְתָּהּ הָאַחֲרוֹנָה תַּעֲנֶה אוֹתִי כָּל חַיֵּי!"
הַגֶּרְמָנִי מִתְקַדֵּר וּבְזָרָתוֹ מְגָרֵד אֶת הַגַּבָּה הַשְּׂמָאלִית בְּהִרְהוּר. לְבַסּוֹף הוּא שׁוֹאֵל: "אהממ... בַּת כַּמָּה הַבַּת שֶׁלְּךָ?"
"בַּת שֵׁשׁ."
"אהממ... גַּם לִיזָה שֶׁלִּי בַּת שֵׁשׁ. אהממ... אֲבָל דַּע לְךָ שֶׁזֶּה יַעֲלֶה לְךָ בִּיקֵּר. אֶת הַפִּיל נִצְטָרֵךְ לְהָבִיא בַּלַּיְלָה וְלָקַחַת בַּחֲזָרָה רַק בַּלַּיְלָה שֶׁלְּמָחֳרָת. בְּמַהֲלַךְ הַיּוֹם אִי אֶפְשָׁר. יִתְאַסֵּף קָהָל, וְתִהְיֶה שַׁעֲרוּרִיָּה אַחַת גְּדוֹלָה... וְאִם כָּךְ, יֵצֵא שֶׁאֲנִי אַפְסִיד יוֹם שָׁלֵם, וְתִצְטָרֵךְ לְכַסּוֹת עַל הַהֶפְסֵד."
"הוֹ, בֶּטַח, בֶּטַח... אַל תִּדְאַג בְּנוֹגֵעַ לָזֶה."
"וְחוּץ מִזֶּה, הַאִם הַמִּשְׁטָרָה תַּרְשֶׁה לְהַכְנִיס אֶת הַפִּיל לְבֵית?"
"אֲנִי אֲאַרְגֵּן אֶת זֶה. הִיא תַּרְשֶׁה."
"עוֹד שְׁאֵלָה אַחַת: הַאִם בַּעַל הַבַּיִת יַרְשֶׁה לְהַכְנִיס אֶת הַפִּיל לַבַּיִת שֶׁלּוֹ?
"יָרְשָׁה. אֲנִי הוּא בַּעַל הַבַּיִת."
"אֲהָהּ! זֶה עוֹד יוֹתֵר טוֹב. ו... כֵּן,  עוֹד שְׁאֵלָה אַחַת: בְּאֵיזוֹ קוֹמָה אַתֶּם גָּרִים?"
"בְּקוֹמָה שְׁנִיָּה."
"אהממ... זֶה כְּבָר לֹא כָּל כָּךְ טוֹב... הַאִם יֵשׁ בַּבַּיִת שֶׁלָּכֶם גֶּרֶם מַדְרֵגוֹת רָחָב, תִּקְרָה גְּבוֹהָה, חֶדֶר גָּדוֹל, דְּלָתוֹת רְחָבוֹת וְרִצְפָּה חֲזָקָה מְאוֹד? מִפְּנֵי שֶׁהַגַּבָּה שֶׁל טוֹמִי שֶׁלִּי הוּא שְׁלוֹשָׁה אָרְשִׂין וָחֵצִי, וְהָרָחָב שֶׁלּוֹ הוּא אַרְבָּעָה אָרְשִׂין. חוּץ מִזֶּה, הוּא שׁוֹקֵל מֵאָה וְעֶשְׂרִים טוֹן."
אַבָּא שֶׁל נַדְיָה מְהַרְהֵר רֶגַע.
"אַתָּה יוֹדֵעַ מָה?" הוּא אוֹמֵר. "בּוֹא נָסַע עַכְשָׁו אֵלַי וְנִבְחַן הַכֹּל בַּמָּקוֹם. אִם יֵשׁ צָרַךְ, אֲנִי אֶתֵּן הוֹרָאָה לְהַרְחִיב אֶת הַדְּלָתוֹת."
"טוֹב מְאוֹד!" מַסְכִּים הַבְּעָלִים שֶׁל הַקִּרְקָס.

5

בַּלַּיְלָה מוֹבִילִים אֶת הַפִּיל לְבֵיתָהּ שֶׁל הַיַּלְדָּה הַחוֹלֶה. הוּא צוֹעֵד בַּאֲרֶשֶׁת חֲשִׁיבוּת תַּחַת כְּסוּי לָבָן בְּאֶמְצַע הָרְחוֹב, מְנַעְנֵעַ אֶת רֹאשׁוֹ וּמְקַפֵּל וּמֵישָׁר לְסֵרוּגִין אֶת הַחֵדֶק שֶׁלּוֹ. סְבִיבוֹ, לַמְרוֹת הַשָּׁעָה הַמְּאַחֶרֶת, הִתְקַהֲלוּת גְּדוֹלָה. אַךְ הַפִּיל לֹא שָׂם לֵב אֵלֶיהָ כְּלָל: הוּא רוֹאֶה מֵאוֹת אֲנָשִׁים בַּקִּרְקָס בְּכָל יוֹם. רַק פַּעַם אַחַת קָרָה שֶׁהִתְרַגֵּז קַלּוֹת. פִּרְחָח אֶחָד קְרָב לְרַגְלָיו וְהִתְחִיל לְהִשְׁתּוֹלֵל כְּדֵי לְשַׁעֲשֵׁעַ אֶת הָאֲסַפְסוּף. שֶׁלּוֹ וְרָגוּעַ הֵסִיר הַפִּיל אֶת כּוֹבָעוֹ שֶׁל הַפִּרְחָח מֵרֹאשׁוֹ בְּעֶזְרַת הַחֵדֶק וְזָרַק אוֹתוֹ מַעֲבָר לְגֶדֶר הַשְּׁכֵנָה שֶׁהָיְתָה מְכַסָּה בְּמַסְמְרִים.
שׁוֹטֵר עוֹבֵר בַּקָּהָל וּמְנַסֶּה לְדַבֵּר אֶל לִבּוֹ:
"רַבּוֹתַי, אֲנִי מְבַקֵּשׁ מִכֶּם לְהִתְפַּזֵּר. מָה כָּל כָּךְ מְעַנְיֵן פֹּה? אֲנִי מִתְפַּלֵּא עֲלֵיכֶם! כָּאֵלּוּ שֶׁאַף פַּעַם לֹא רְאִיתֶם בִּרְחוֹבוֹת פִּיל חַי."
הֵם מִתְקָרְבִים אֶל הַבַּיִת. כָּל הַדְּלָתוֹת הַמּוֹבִילוֹת לַחֶדֶר הָאֹכֶל עַצְמוֹ, הֵחֵל בַּדֶּלֶת הָרָאשִׁית, פְּתוּחוֹת לְרוּחָהּ, כִּי לִכְבוֹד הַפִּיל הוֹרִידוּ אֶת כָּל הַבְּרִיחִים מֵהַדְּלָתוֹת בְּפַטִּישׁ. כָּךְ כְּבָר עָשׂוּ פַּעַם, כְּשֶׁהִכְנִיסוּ לְבֵית צַלְמִית פִּלְאִית גְּדוֹלָה.
אֲבָל הַפִּיל עוֹצֵר לִפְנֵי הַמַּדְרֵגוֹת. הוּא נֶעֱמָד שָׁם חֲסַר מְנוּחָה, לֹא מוּכָן לָזוּז מִמְּקוֹמוֹ.
"אַתָּה צָרִיךְ לָתֵת לוֹ מַשֶּׁהוּ מָתוֹק," אוֹמֵר הַגֶּרְמָנִי. "לַחְמָנִיָּה מְתוּקָה אוֹ מַשֶּׁהוּ... נוּ כְּבָר, טוֹמִי! הֵי, יַלְדּוֹן!"
אַבָּא שֶׁל נַדְיָה רָץ לַמַּאֲפִיָּה הַקְּרוֹבָה וְקוֹנֶה עוּגַת פִּיסְטוֹק גְּדוֹלָה עֲגָלָה. הַפִּיל מַבִּיעַ רָצוֹן לְבַלֵּעַ אֶת כָּל כַּלָּה, עִם קֻפְסַת הַקַּרְטוֹן, אֲבָל הַגֶּרְמָנִי נוֹתֵן לוֹ רֶבַע מִמֶּנָּה. הָעוּגָה עֲרָבָה לְחִכּוֹ שֶׁל טוֹמִי, וְהוּא מוֹשִׁיט אֶת הַחֵדֶק לִפְרוּסָה נוֹסֶפֶת. אַךְ הַגֶּרְמָנִי מִתְחַכֵּם מִמֶּנּוּ. הוּא מַחֲזִיק אֶת הָעוּגָה בְּיָדוֹ, וְהִפִּיל עוֹלֶה לְמַעְלָה מַדְרֵגָה אַחֲרֵי מַדְרֵגָה. הַפִּיל פּוֹשֵׁט אֶת הַחֵדֶק, פּוֹרֵשׂ אֶת הָאֲזֵנִים וְעוֹקֵב אַחֲרָיו עַל אַף רְצוֹנוֹ. בְּמַעֲלֵה הַמַּדְרֵגוֹת טוֹמִי מְקַבֵּל פְּרוּסָה נוֹסֶפֶת.
כָּךְ מוֹבִילִים אוֹתוֹ לַחֲדַר הָאֹכֶל, שֶׁכָּל הָרָהִיטִים הוּצְאוּ מִמֶּנּוּ מִבְּעוֹד מוֹעֵד וְהָרִצְפָּה בּוֹ כַּלָּה כָּסְתָה בַּקֵּשׁ. אֶת הַפִּיל קוֹשְׁרִים בְּרַגְלוֹ לַטַּבַּעַת שֶׁמַּבְרֵגַת לָרִצְפָּה. מַנִּיחִים לְפָנָיו גֶּזֶר, כְּרוּב וְלָפַת טָרִיִּים. הַגֶּרְמָנִי מִתְמַקֵּם עַל הַסַּפָּה לְיָדוֹ. מְכַבִּים אֶת הָאוֹרוֹת, וְכֻלָּם הוֹלְכִים לִישֹׁן.

6

לְמָחֳרָת הַיַּלְדָּה מִתְעוֹרֶרֶת לִפְנֵי שֶׁהָאוֹר הָרִאשׁוֹן מַפְצִיעַ. הַדָּבָר הָרִאשׁוֹן שֶׁהִיא שׁוֹאֶלֶת הוּא:
"מָה עִם הַפִּיל? הוּא בָּא?
"הוּא בָּא," עוֹנֶה אִמָּא. "אֲבָל הוּא אָמַר שֶׁקָּדַם כָּל נַדְיָה תִּתְרַחֵץ וְאַחַר כָּךְ תֹּאכַל בֵּיצָה רַכָּה וְתִשְׁתֶּה חָלָב חַם."
"הוּא טוֹב לֵב?"
"הוּא טוֹב לֵב. תֹּאכְלִי, יַקִּירָתִי. תֶּכֶף נֵלֵךְ אֵלָיו."
"הוּא נִרְאֶה מַצְחִיק?"
"קְצָת. תִּלְבְּשִׁי אֶת הַסּוּדָר הַחַם שֶׁלָּךְ."
הַבֵּיצָה נֶאֱכֶלֶת מַהֵר, כּוֹס הֶחָלָב מִתְרוֹקֶנֶת. מוֹשִׁיבִים אֶת נַדְיָה בְּאוֹתָהּ עֲגָלָה שֶׁהִשְׁתַּמְּשָׁה בָּהּ כְּשֶׁעוֹד הָיְתָה קְטַנָּה מִכְּדֵי לָלֶכֶת וְלוֹקְחִים אוֹתָהּ לַחֲדַר הָאֹכֶל.
מִתְבָּרֵר שֶׁהִפִּיל הַרְבֵּה יוֹתֵר גָּדוֹל מִכְּפִי שֶׁנַּדְיָה דְּמֶינָה כְּשֶׁרָאֲתָה אֶת הַתְּמוּנוֹת. גָּבְהוּ רַק מְעַט נָמוּךְ יוֹתֵר מֵהַדֶּלֶת, וְהוּא תּוֹפֵס חֲצִי מֵאַרְכּוֹ שֶׁל חֲדַר הָאֹכֶל. עוֹרוֹ גַּס וּמְלֵא קְפָלִים כְּבֵדִים. רַגְלָיו עָבוֹת כְּמוֹ עַמּוּדִים. זְנָבוֹ אֶרֶךְ וּמַשֶּׁהוּ דְּמוּי מִבְרֶשֶׁת בְּקָצֵהוּ. רֹאשׁוֹ מָלֵא גַּבְשׁוּשִׁיּוֹת גְּדוֹלוֹת. אָזְנָיו גְּדוֹלוֹת כְּמוֹ מְפָרְשִׁים וְשֶׁמּוֹטוֹת. עֵינָיו זַעֲרוּרִיּוֹת מַמָּשׁ, אַךְ חֲכָמוֹת וְטוֹבוֹת. הַחַטִּים כְּרוּתִים. הַחֵדֶק נִרְאֶה כְּמוֹ נָחָשׁ אָרַךְ וּמְסַתֵּים בִּנְחִירִים וּבֵינֵיהֶם אֶצְבַּע פְּעִילָה גְּמִישָׁה. לוֹ מֶתַח הִפִּיל אֶת הַחֵדֶק לְכָל אָרְכוּ, בְּוַדַּאי הָיָה מַגִּיעַ עַד הַחַלּוֹן.
הַיַּלְדָּה אֵינָהּ מִבֶּהָלַת כְּלָל. הִיא רַק מֻפְתַּעַת קְצָת מִגְדָּלָהּ הָאֵימְתָנִי שֶׁל הַחַיָּה. אֲבָל הָאוֹמֶנֶת, פּוֹלָה בַּת הַשֵּׁשׁ-עֶשְׂרֵה, מַתְחִילָה לָצוֹחַ מְפַחֵד.
הַבְּעָלִים שֶׁל הַפִּיל, הַגֶּרְמָנִי, נָגַשׂ אֶל הָעֲגָלָה וְאוֹמֵר:
"בֵּקֶר טוֹב, עַלְמָתִי. אָנָּא אַל תִּפְחֲדִי. טוֹמִי מְאוֹד נֶחְמָד וְאוֹהֵב יְלָדִים."
הַיַּלְדָּה מוֹשִׁיטָה לוֹ אֶת יָדָהּ הַקְּטַנָּה הַחוֹרֵת.
"שָׁלוֹם," הִיא עוֹנָה, "אֲנִי לֹא מְפַחֶדֶת בִּכְלָל, אָפְלוּ לֹא טַפָּהּ. אֵיךְ קוֹרְאִים לוֹ?"
"טוֹמִי."
"שָׁלוֹם, טוֹמִי," אוֹמֶרֶת הַיַּלְדָּה וּמְנִידָה בְּרֹאשָׁהּ.
הַפִּיל גָּדוֹל כָּל כָּךְ, שֶׁהִיא מַחֲלִיטָה לִהְיוֹת מְאוֹד מְנַמֶּסֶת.
"אֵיךְ יָשַׁנְתְּ הַלַּיְלָה?"
הִיא גַּם מוֹשִׁיטָה לוֹ אֶת יָדָהּ. הַפִּיל לוֹקֵחַ אוֹתָהּ בַּעֲדִינוּת וְלוֹחֵץ אֶת אֶצְבְּעוֹתֶיהָ הַדְּקִיקוֹת בְּאֶצְבָּעוֹ הַחֲזָקָה הַפְּעִילָה. הוּא עוֹשֶׂה זֹאת בַּעֲדִינוּת רַבָּה יוֹתֵר מֵהָרוֹפֵא שֶׁלָּהּ, מִיכָאִיל פֶּטְרוֹבִיץ'. הִפִּיל מְנַעְנֵעַ אֶת רֹאשׁוֹ, וְעֵינָיו הַקְּטַנּוֹת מִצְטַמְצְמוֹת כָּאֵלּוּ שֶׁהֵן צוֹחֲקוֹת.
"הוּא מֵבִין הַכֹּל, נָכוֹן?" שׁוֹאֶלֶת הַיַּלְדָּה אֶת הַגֶּרְמָנִי.
"הוֹ, לְגַמְרֵי הַכֹּל, עַלְמָתִי!"
"רַק שֶׁהוּא לֹא מְדַבֵּר?"
"כֵּן, בְּדִיּוּק כָּךְ. גַּם לִי יֵשׁ, אַתְּ יוֹדַעַת, בַּת אַחַת, קְטַנָּה כָּמוֹךָ. קוֹרְאִים לָהּ לִיזָה. טוֹמִי וְהִיא חֲבֵרִים מְאוֹד-מְאוֹד טוֹבִים."
"טוֹמִי, כְּבָר שָׁתִיתָ תֶּה הַיּוֹם?" שׁוֹאֶלֶת הַיַּלְדָּה אֶת הַפִּיל.
הַפִּיל שׁוּב מוֹתֵחַ אֶת הַחֵדֶק וְנוֹשֵׁף מַשַּׁב אֲוִיר חַם וְחָזָק לְתוֹךְ פָּנֶיהָ שֶׁל הַיַּלְדָּה. כְּתוֹצָאָה מִכָּךְ הַשְׁעָרוֹת הֲקַלּוֹת שֶׁעַל רֹאשׁ הַיַּלְדָּה מִתְבַּדְּרוֹת וּמִתְפַּזְּרוֹת לְכָל הַכַּוּוֹנִים.
נַדְיָה מִתְגַּלְגֶּלֶת מִצְּחוֹק וּמוֹחֵאת כַּפַּיִם. הַגֶּרְמָנִי צוֹחֵק בְּקוֹל עָבֶה וְסָמִיךְ. הוּא עַצְמוֹ כָּל כָּךְ גָּדוֹל, שֶׁמֶן וְנֶחְמָד, כְּמוֹ הַפִּיל, וּלְנֵדֶיהָ נִדְמֶה שֶׁהֵם דּוֹמִים. אוּלַי הֵם קְרוֹבֵי מִשְׁפָּחָה?
"לֹא, עַלְמָתִי, הוּא עָדִין לֹא שָׁתָה תֶּה. אֲבָל הוּא שׁוֹתֶה בַּהֲנָאָה רַבָּה מִי סֶכֶר. הוּא גַּם אוֹהֵב לַחְמָנִיּוֹת."
מְבִיאִים מַגַּשׁ לַחְמָנִיּוֹת. הַיַּלְדָּה מַצִּיעָה אַחַת לַפִּיל. הוּא תּוֹפֵס אֶת הַלַּחְמָנִיָּה בִּמְיֻמָּנוּת בְּאֶצְבָּעוֹ, מַעְגָּל אֶת הַחֵדֶק, מַחְבִּיא אוֹתָהּ אֵיפְשֶׁהוּ לְמַטָּה מִתַּחַת לְרֹאשׁ, בַּמָּקוֹם שֶׁבּוֹ נִמְצֵאת הַשָּׂפָה הַתַּחְתּוֹנָה הַפֵּרוֹתִית הַמְּשַׁלֶּשֶׁת הַמַּצְחִיקָה שֶׁלּוֹ. שׁוֹמְעִים אֶת הַלַּחְמָנִיָּה מְרַשְׁרֶשֶׁת עַל עוֹרוֹ הַיָּבֵשׁ. אֶת אוֹתוֹ הַדָּבָר טוֹמִי עוֹשֶׂה עִם הַלַּחְמָנִיָּה הַשְּׁנִיָּה, הַשְּׁלִישִׁית, הָרְבִיעִית וְהַחֲמִישִׁית, וּלְאוֹת תּוֹדָה הוּא מֵנִיד בְּרֹאשׁוֹ, וְעֵינָיו הַקְּטַנּוֹת מִצְטַמְצְמוֹת עוֹד יוֹתֵר בִּשְׂבִיעוֹת רָצוֹן. הַיַּלְדָּה צוֹחֶקֶת בְּשִׂמְחָה.
לְאַחַר שֶׁכָּל הַלַּחְמָנִיּוֹת נֶאֶכְלוּ, נַדְיָה עוֹרֶכֶת הֶכֵּרוּת בֵּין בָּבוֹתֶיהָ לַפִּיל:
"תִּרְאֶה, טוֹמִי, הַבַּבָּה הַמְּגַנְדֶּרֶת הַזֹּאת הִיא סוֹנְיָה. הִיא יַלְדָּה טוֹבָה מְאוֹד, אֲבָל קְצָת מְפַנֶּקֶת וְלֹא רוֹצָה לְאַכֵּל מַרְק. זֹאת נָטְשָׁה, הַבַּת שֶׁל סוֹנְיָה. הִיא כְּבָר הִתְחִילָה לְלַמֵּד וּמְכִירָה כִּמְעַט אֶת כָּל הָאוֹתִיּוֹת. וְזֹאת מַטְרִיּוֹשְׁקָה. הִיא הַבָּבָה הֲכִי רִאשׁוֹנָה שֶׁלִּי. אַתָּה רוֹאֶה, אֵין לָהּ אַף, וְהָרֹאשׁ שֶׁלָּהּ מְדַבֵּק, וְאֵין לָהּ שַׁעַר יוֹתֵר. אֲבָל בְּכָל זֹאת אִי אֶפְשָׁר לְגָרֵשׁ מֵהַבַּיִת אֶת הַקְּשִׁישָׁה. נָכוֹן, טוֹמִי? פַּעַם הִיא הָיְתָה אִמָּא שֶׁל סוֹנְיָה, וְעַכְשָׁו הִיא הַטַּבָּחִית שֶׁלָּנוּ. אָז בּוֹא נִשְׁחַק, טוֹמִי: אַתָּה תִּהְיֶה אַבָּא, אֲנִי אִמָּא, וְאֵלֶּה יִהְיוּ הַיְּלָדִים שֶׁלָּנוּ."
טוֹמִי מָסַכִּים. הוּא צוֹחֵק, לוֹקֵחַ אֶת מַטְרִיּוֹשְׁקָה בְּצַוָּארָהּ וְתוֹחֵב אוֹתָהּ לְפִיו. אֲבָל זֶה רַק בִּצְחוֹק. אַחֲרֵי שֶׁהוּא לוֹעֵס מְעַט אֶת הַבַּבָּה, הוּא מַנִּיחַ אוֹתָהּ עַל בִּרְכֶּיהָ שֶׁל הַיַּלְדָּה, אַף שֶׁהִיא קְצָת רָטְבָה וּמְקַמֶּטֶת.
אַחֵר כָּךְ נַדְיָה מַרְאֶה לוֹ סֵפֶר גָּדוֹל עִם תְּמוּנוֹת וּמַסְבִּירָה:
"זֶה סוּס, זֹאת קַנְרִי, זֶה רוֹבֶה... הִנֵּה כְּלוּב עִם צִפּוֹר, מַרְאֶה, תַּנּוּר, אַתְּ, עוֹרֵב... וְהִנֵּה, תִּרְאֶה, זֶה פִּיל! הוּא בִּכְלָל לֹא נִרְאֶה כָּזֶה, נָכוֹן? פִּילִים יְכוֹלִים לִהְיוֹת כָּל כָּךְ קְטַנִּים, טוֹמִי?"
טוֹמִי לֹא חוֹשֵׁב שֶׁיֵּשׁ בָּעוֹלָם פִּילִים קְטַנִּים כָּל כָּךְ. לְמַעֲשֶׂה, הַתְּמוּנָה לֹא מוֹצֵאת חֵן בְּעֵינָיו בִּכְלָל. הוּא תּוֹפֵס בְּאֶצְבָּעוֹ אֶת קְצֵה הַדַּף וּמַעֲבִיר עַמּוּד.
מַגִּיעָה שְׁעַת אֲרוּחַת הַצָּהֳרַיִם, אֲבָל אִי אֶפְשָׁר לְהַפְרִיד אֶת הַיַּלְדָּה מֵהַפִּיל. הַגֶּרְמָנִי נֶחֱלַץ לְעֶזְרָה:
"הַרְשׁוּ לִי לְסֵדֶר הַכֹּל. הֵם אוֹכְלִים יַחַד."
הוּא מִצְוָה עַל הַפִּיל לָשֶׁבֶת. הַפִּיל מִתְיַשֵּׁב בְּצַיְתָנוּת, וְהָרִצְפָּה בְּכָל הַבַּיִת רוֹעֶדֶת, הַכֵּלִים בָּאָרוֹן מִצְטַלְצְלִים, וְאֵצֶל הַשְּׁכֵנִים מִלְּמַטָּה נוֹשֵׁר הֵטִיחַ מֵהַתִּקְרָה. הַיַּלְדָּה מִתְיַשֶּׁבֶת מוּלוֹ. בֵּינֵיהֶם מַצִּיבִים שֻׁלְחָן. מִפֹּה נִקְשֶׁרֶת סְבִיב צַוָּארוֹ שֶׁל הַפִּיל, וְהַחֲבֵרִים הַחֲדָשִׁים מַתְחִילִים לְאַכֵּל. הַיַּלְדָּה אוֹכֶלֶת מַרְק עוֹף וּקְצִיצָה, וְהִפִּיל – יְרָקוֹת שׁוֹנִים וְסָלָט. הַיַּלְדָּה מְקַבֶּלֶת כּוֹסִית קְטַנְטַנָּה שֶׁל שָׂרִי, וְהִפִּיל – מַיִם חַמִּים עִם כּוֹס רוּם. הוּא שׁוֹאֵב אֶת הַמַּשְׁקֶה הַזֶּה בַּחֶדֶק שֶׁלּוֹ מֵהַקְּעָרָה בַּהֲנָאָה. לְאַחַר מִכֵּן הֵם מְקַבְּלִים קִנּוּחַ: כּוֹס קָקָאוֹ לְיַלְדָּה וָחֵצִי עוּגָה לַפִּיל. הַפַּעַם זוֹ עוּגַת אֱגוֹזִים. כָּל אוֹתוֹ הַזְּמַן הַגֶּרְמָנִי יוֹשֵׁב עִם אַבָּא בַּחֲדַר הָאוֹרְחִים, וּבַהֲנָאָה כְּשֶׁל הַפִּיל שׁוֹתֶה בִּירָה, רַק בְּכַמּוּת גְּדוֹלָה יוֹתֵר.
אַחַר הַצָּהֳרַיִם מַגִּיעִים אִי-אֵלּוּ מַכָּרִים שֶׁל אַבָּא. עוֹד בַּמְּבוֹאָה מְסַפְּרִים לָהֶם עַל הַפִּיל כְּדֵי שֶׁלֹּא יַבְהִלוּ. בַּהַתְחָלָה הֵם לֹא מַאֲמִינִים, וְאַחַר כָּךְ, כְּשֶׁהֵם רוֹאִים אֶת טוֹמִי, הֵם נְסוֹגִים אֶל הַדֶּלֶת.
"אַל תִּפְחֲדוּ, הוּא נֶחְמָד!" מַרְגִּיעָה אוֹתָם הַיַּלְדָּה.
אַךְ הַמַּכָּרִים הוֹלְכִים בְּחִפָּזוֹן לַחֲדַר הָאוֹרְחִים, וְלֹא עוֹבְרוֹת חָמֵשׁ דַּקּוֹת וְהֵם קָמִים וְהוֹלְכִים.
הָעֶרֶב יוֹרֵד. מֵאַחַר. הִגִּיעָה שְׁעַת הַשָּׁנָה שֶׁל הַיַּלְדָּה. אֲבָל אִי אֶפְשָׁר לְהַפְרִיד אוֹתָהּ מֵהַפִּיל. הִיא נִרְדֶּמֶת כָּכָה לְיָדוֹ, וְלוֹקְחִים אוֹתָהּ יְשֵׁנָה לְמֶחֱצָה לַחֲדַר הַיְּלָדִים. הִיא אָפְלוּ לֹא שׁוֹמַעַת אֵיךְ מַפְשִׁיטִים אוֹתָהּ.
בְּאוֹתוֹ הַלַּיְלָה נַדְיָה רוֹאָה בַּחֲלוֹמָהּ שֶׁהִיא הִתְחַתְּנָה עִם טוֹמִי, וְשֶׁיֵּשׁ לָהֶם הַרְבֵּה יְלָדִים, פִּילוֹנִים קְטַנִּים וְעַלִּיזִים. הַפִּיל, שֶׁלָּקְחוּ אוֹתוֹ בַּלַּיְלָה לְקִרְקָס, רוֹאֶה גַּם הוּא בַּחֲלוֹמוֹ יַלְדָּה חֲמוּדָה וּמְתוּקָה. חוּץ מִזֶּה, הוּא חוֹלֵם עַל עוּגוֹת גְּדוֹלוֹת, עוּגוֹת אֱגוֹזִים וּפִיסְטוּקִים, בְּגָדֵל שֶׁל שַׁעַר...
לְמָחֳרָת בְּבֹקֶר הַיַּלְדָּה מִתְעוֹרֶרֶת רַעֲנָנָה וְעֵרָנִית, כְּמוֹ בַּיָּמִים הַטּוֹבִים כְּשֶׁעוֹד הָיְתָה בְּרִיאָה, וְצוֹעֶקֶת בְּקוֹל רָם וּבִקְצַר רוּחַ:
"ח-ל-בִּי!"
כְּשֶׁאִמָּהּ שׁוֹמַעַת אֶת הַצְּעָקָה הַזֹּאת, הִיא מִצְטַלֶּבֶת בְּשִׂמְחָה בַּחֲדַר הַשָּׁנָה שֶׁלָּהּ.
אֲבָל הַיַּלְדָּה נִזְכֶּרֶת חִישׁ מַהֵר בְּאֵרוּעֵי אֶמֶשׁ וְשׁוֹאֶלֶת:
"הַפִּיל?"
מַסְבִּירִים לָהּ שֶׁהִפִּיל הָלַךְ הַבַּיְתָה לַעֲסוּקָיו, שֶׁיֵּשׁ לוֹ יְלָדִים שֶׁאָסוּר לְהַשְׁאִיר לְבַד, שֶׁהוּא מָסַר לְנֵדֶיהָ דְּרִישַׁת שָׁלוֹם וּמְחַכֶּה לַבִּקּוּר שֶׁלָּהּ לִכְשֶׁתַּבְרִיא.
הַיַּלְדָּה מְחַיֶּכֶת חִיּוּךְ עַרְמוּמִי וְאוֹמֶרֶת:
"תִּמְסְרוּ לְטוֹמִי שֶׁאֲנִי כְּבָר לְגַמְרֵי בְּרִיאָה!"
